124    private String getFileContent(final File file) {
125        FileInputStream in = null;
126        try {
127            if (file.exists() && file.canRead()) {
128                final StringBuilder sb = new StringBuilder(128);
129                final byte[] b = new byte[512];
130                int read;
131                in = new FileInputStream(file);
132                while ((read = in.read(b)) != -1) {
133                    sb.append(new String(b, 0, read, "UTF-8"));
134                }
135                return sb.toString();
136            }
137            throw new IllegalArgumentException("File does not exist or cannot be read: "
138                    + file.getCanonicalPath());
139        } catch (IOException ioe) {
140            throw new IllegalStateException(ioe);
141        } finally {
142            if (in != null) {
143                try {
144                    in.close();
145                } catch (IOException ignored) {
146                }
147            }
148        }
149
150    }

207    private class RedirectHandler extends AbstractHandler {
208        public void handle(String s,
209                           Request r,
210                           HttpServletRequest request,
211                           HttpServletResponse response) throws IOException, ServletException {
212
213            System.err.println("redirecthandler");
214            System.err.println("request: " + request.getRequestURI());
215            if ("/uff".equals(request.getRequestURI())) {
216
217                System.err.println("redirect to /bla");
218                response.setStatus(302);
219                response.setHeader("Location", "/bla");
220                response.getOutputStream().flush();
221                response.getOutputStream().close();
222
223                return;
224
225            } else {
226                System.err.println("got redirected" + request.getRequestURI());
227                response.addHeader("X-Auth", request.getHeader("Authorization"));
228                response.addHeader("X-Content-Length", String.valueOf(request.getContentLength()));
229                response.setStatus(200);
230                response.getOutputStream().write("content".getBytes("UTF-8"));
231                response.getOutputStream().flush();
232                response.getOutputStream().close();
233            }
234        }
235    }

237    private class SimpleHandler extends AbstractHandler {
238        public void handle(String s,
239                           Request r,
240                           HttpServletRequest request,
241                           HttpServletResponse response) throws IOException, ServletException {
242
243            if (request.getHeader("X-401") != null) {
244                response.setStatus(401);
245                response.getOutputStream().flush();
246                response.getOutputStream().close();
247
248                return;
249            }
250            response.addHeader("X-Auth", request.getHeader("Authorization"));
251            response.addHeader("X-Content-Length", String.valueOf(request.getContentLength()));
252            response.setStatus(200);
253
254
255            int size = 10 * 1024;
256            if (request.getContentLength() > 0) {
257                size = request.getContentLength();
258            }
259            byte[] bytes = new byte[size];
260            if (bytes.length > 0) {
261                int read = request.getInputStream().read(bytes);
262                if (read > 0) {
263                    response.getOutputStream().write(bytes, 0, read);
264                }
265            }
266            response.getOutputStream().flush();
267            response.getOutputStream().close();
268        }
269    }

316    @Test(groups = {"standalone", "default_provider"})
317    public void basic401Test() throws IOException, ExecutionException, TimeoutException, InterruptedException {
318        AsyncHttpClient client = getAsyncHttpClient(null);
319        AsyncHttpClient.BoundRequestBuilder r = client.prepareGet(getTargetUrl())
320                .setHeader("X-401", "401").setRealm((new Realm.RealmBuilder()).setPrincipal(user).setPassword(admin).build());
321
322        Future<Integer> f = r.execute(new AsyncHandler<Integer>() {
323
324            private HttpResponseStatus status;
325
326
327            public void onThrowable(Throwable t) {
328
329            }
330
331            public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
332                return STATE.CONTINUE;
333            }
334
335            public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
336                this.status = responseStatus;
337
338                if (status.getStatusCode() != 200) {
339                    return STATE.ABORT;
340                }
341                return STATE.CONTINUE;
342            }
343
344            public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
345                return STATE.CONTINUE;
346            }
347
348            public Integer onCompleted() throws Exception {
349                return status.getStatusCode();
350            }
351        });
352        Integer statusCode = f.get(10, TimeUnit.SECONDS);
353        assertNotNull(statusCode);
354        assertEquals(statusCode.intValue(), 401);
355        client.close();
356    }
