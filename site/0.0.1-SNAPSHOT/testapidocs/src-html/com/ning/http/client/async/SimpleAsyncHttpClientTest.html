161    @Test(groups = { "standalone", "default_provider" })
162    public void testSimpleTransferListener() throws Exception {
163
164        SimpleAHCTransferListener listener = new SimpleAHCTransferListener() {
165
166            public void onStatus(String url, int statusCode, String statusText) {
167                assertEquals(statusCode, 200);
168                assertEquals(url, getTargetUrl());
169            }
170
171            public void onHeaders(String url, HeaderMap headers) {
172                assertEquals(url, getTargetUrl());
173                assertNotNull(headers);
174                assertTrue(!headers.isEmpty());
175                assertEquals(headers.getFirstValue("X-Custom"), "custom");
176            }
177
178            public void onCompleted(String url, int statusCode, String statusText) {
179                assertEquals(statusCode, 200);
180                assertEquals(url, getTargetUrl());
181            }
182
183            public void onBytesSent(String url, long amount, long current, long total) {
184                assertEquals(url, getTargetUrl());
185                assertEquals(total, MY_MESSAGE.getBytes().length);
186            }
187
188            public void onBytesReceived(String url, long amount, long current, long total) {
189                assertEquals(url, getTargetUrl());
190                assertEquals(total, -1);
191            }
192        };
193
194        SimpleAsyncHttpClient client = new SimpleAsyncHttpClient.Builder().setUrl(getTargetUrl()).setHeader("Custom", "custom").setListener(listener).build();
195        ByteArrayOutputStream o = new ByteArrayOutputStream(10);
196
197        InputStreamBodyGenerator generator = new InputStreamBodyGenerator(new ByteArrayInputStream(MY_MESSAGE.getBytes()));
198        OutputStreamBodyConsumer consumer = new OutputStreamBodyConsumer(o);
199
200        Future<Response> future = client.post(generator, consumer);
201
202        Response response = future.get();
203        client.close();
204        assertEquals(response.getStatusCode(), 200);
205        assertEquals(o.toString(), MY_MESSAGE);
206    }

251    @Test(groups = { "standalone", "default_provider" })
252    public void testMultiPartPut() throws Exception {
253        SimpleAsyncHttpClient client = new SimpleAsyncHttpClient.Builder().setUrl(getTargetUrl() + "/multipart").build();
254        
255        Response response = client.put(new ByteArrayPart("baPart", "fileName", "testMultiPart".getBytes("utf-8"), "application/test", "utf-8")).get();
256        
257        String body = response.getResponseBody();
258        String contentType = response.getHeader("X-Content-Type");
259        
260        assertTrue(contentType.contains("multipart/form-data"));
261        
262        String boundary = contentType.substring(contentType.lastIndexOf("=") + 1);
263        
264        assertTrue(body.startsWith("--" + boundary));
265        assertTrue(body.trim().endsWith("--" + boundary + "--"));
266        assertTrue(body.contains("Content-Disposition:"));
267        assertTrue(body.contains("Content-Type: application/test"));
268        assertTrue(body.contains("name=\"baPart"));
269        assertTrue(body.contains("filename=\"fileName"));
270        
271    }
