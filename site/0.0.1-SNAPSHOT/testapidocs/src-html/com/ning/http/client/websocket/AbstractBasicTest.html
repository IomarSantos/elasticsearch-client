013package com.ning.http.client.websocket;
014
015import com.ning.http.client.AsyncHttpClient;
016import com.ning.http.client.AsyncHttpClientConfig;
017import org.eclipse.jetty.server.Request;
018import org.eclipse.jetty.server.Server;
019import org.eclipse.jetty.server.handler.HandlerWrapper;
020import org.eclipse.jetty.server.nio.SelectChannelConnector;
021import org.eclipse.jetty.websocket.WebSocketFactory;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024import org.testng.annotations.AfterClass;
025import org.testng.annotations.BeforeClass;
026
027import javax.servlet.ServletException;
028import javax.servlet.http.HttpServletRequest;
029import javax.servlet.http.HttpServletResponse;
030import java.io.IOException;
031import java.net.ServerSocket;
032
033public abstract class AbstractBasicTest extends Server {
034
035    public abstract class WebSocketHandler extends HandlerWrapper implements WebSocketFactory.Acceptor {
036        private final WebSocketFactory _webSocketFactory = new WebSocketFactory(this, 32 * 1024);
037
038        public WebSocketHandler(){
039            _webSocketFactory.setMaxIdleTime(10000);
040        }
041
042        public WebSocketFactory getWebSocketFactory() {
043            return _webSocketFactory;
044        }
045
046        /* ------------------------------------------------------------ */
047        @Override
048        public void handle(String target, Request baseRequest, HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
049            if (_webSocketFactory.acceptWebSocket(request, response) || response.isCommitted())
050                return;
051            super.handle(target, baseRequest, request, response);
052        }
053
054        /* ------------------------------------------------------------ */
055        public boolean checkOrigin(HttpServletRequest request, String origin) {
056            return true;
057        }
058
059    }
060
061    protected final Logger log = LoggerFactory.getLogger(AbstractBasicTest.class);
062    protected int port1;
063    SelectChannelConnector _connector;
064
065    @AfterClass(alwaysRun = true)
066    public void tearDownGlobal() throws Exception {
067        stop();
068    }
069
070    protected int findFreePort() throws IOException {
071        ServerSocket socket = null;
072
073        try {
074            socket = new ServerSocket(0);
075
076            return socket.getLocalPort();
077        } finally {
078            if (socket != null) {
079                socket.close();
080            }
081        }
082    }
083
084    protected String getTargetUrl() {
085        return String.format("ws://127.0.0.1:%d/", port1);
086    }
087
088    @BeforeClass(alwaysRun = true)
089    public void setUpGlobal() throws Exception {
090        port1 = findFreePort();
091        _connector = new SelectChannelConnector();
092        _connector.setPort(port1);
093
094        addConnector(_connector);
095        WebSocketHandler _wsHandler = getWebSocketHandler();
096
097        setHandler(_wsHandler);
098
099        start();
100        log.info("Local HTTP server started successfully");
101    }
102
103    public abstract WebSocketHandler getWebSocketHandler() ;
104
105    public abstract AsyncHttpClient getAsyncHttpClient(AsyncHttpClientConfig config);
106
107}
