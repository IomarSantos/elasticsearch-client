016package com.ning.http.client.async;
017
018import com.ning.http.client.AsyncHandler;
019import com.ning.http.client.AsyncHttpClient;
020import com.ning.http.client.AsyncHttpClientConfig;
021import com.ning.http.client.HttpResponseBodyPart;
022import com.ning.http.client.HttpResponseHeaders;
023import com.ning.http.client.HttpResponseStatus;
024import org.testng.annotations.Test;
025
026import java.io.IOException;
027import java.util.concurrent.CountDownLatch;
028import java.util.concurrent.ExecutionException;
029import java.util.concurrent.ExecutorService;
030import java.util.concurrent.Executors;
031import java.util.concurrent.TimeoutException;
032
033
034/**
035 * Simple stress test for exercising the follow redirect.
036 */
037public abstract class FollowingThreadTest extends AbstractBasicTest {
038
039    private final static int COUNT = 10;
040
041    @Test(timeOut = 30 * 1000, groups = {"online", "default_provider", "scalability"})
042    public void testFollowRedirect() throws IOException, ExecutionException, TimeoutException, InterruptedException {
043
044        final CountDownLatch countDown = new CountDownLatch(COUNT);
045        ExecutorService pool = Executors.newCachedThreadPool();
046        for (int i = 0; i < COUNT; i++) {
047            pool.submit(new Runnable() {
048
049                private int status;
050
051                public void run() {
052                    final CountDownLatch l = new CountDownLatch(1);
053                    final AsyncHttpClient ahc = getAsyncHttpClient(
054                            new AsyncHttpClientConfig.Builder().setFollowRedirects(true).build());
055                    try {
056                        ahc.prepareGet("http://www.google.com/").execute(new AsyncHandler<Integer>() {
057
058                            public void onThrowable(Throwable t) {
059                                t.printStackTrace();
060                            }
061
062                            public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
063                                System.out.println(new String(bodyPart.getBodyPartBytes()));
064                                return STATE.CONTINUE;
065                            }
066
067                            public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
068                                status = responseStatus.getStatusCode();
069                                System.out.println(responseStatus.getStatusText());
070                                return STATE.CONTINUE;
071                            }
072
073                            public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
074                                return STATE.CONTINUE;
075                            }
076
077                            public Integer onCompleted() throws Exception {
078                                l.countDown();
079                                return status;
080                            }
081                        });
082
083                        l.await();
084                    } catch (Exception e) {
085                        e.printStackTrace();
086                    } finally {
087                        ahc.close();
088                        countDown.countDown();
089                    }
090                }
091            });
092        }
093        countDown.await();
094    }
095
096}
