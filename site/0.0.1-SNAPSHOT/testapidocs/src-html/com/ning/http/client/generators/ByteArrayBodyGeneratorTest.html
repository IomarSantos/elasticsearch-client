014package com.ning.http.client.generators;
015
016import com.ning.http.client.Body;
017
018import org.testng.annotations.Test;
019
020import java.io.IOException;
021import java.nio.ByteBuffer;
022import java.util.Random;
023
024import static org.testng.Assert.assertEquals;
025import static org.testng.Assert.assertTrue;
026
027/**
028 * @author Bryan Davis bpd@keynetics.com
029 */
030public class ByteArrayBodyGeneratorTest {
031
032    private final Random random = new Random();
033    private final int chunkSize = 1024 * 8;
034
035    @Test(groups = "standalone")
036    public void testSingleRead() throws IOException {
037        final int srcArraySize = chunkSize - 1;
038        final byte[] srcArray = new byte[srcArraySize];
039        random.nextBytes(srcArray);
040
041        final ByteArrayBodyGenerator babGen =
042            new ByteArrayBodyGenerator(srcArray);
043        final Body body = babGen.createBody();
044
045        final ByteBuffer chunkBuffer = ByteBuffer.allocate(chunkSize);
046
047        // should take 1 read to get through the srcArray
048        assertEquals(body.read(chunkBuffer), srcArraySize);
049        assertEquals(chunkBuffer.position(), srcArraySize, "bytes read");
050        chunkBuffer.clear();
051
052        assertEquals(body.read(chunkBuffer), -1, "body at EOF");
053    }
054
055    @Test(groups = "standalone")
056    public void testMultipleReads() throws IOException {
057        final int srcArraySize = (3 * chunkSize) + 42;
058        final byte[] srcArray = new byte[srcArraySize];
059        random.nextBytes(srcArray);
060
061        final ByteArrayBodyGenerator babGen =
062            new ByteArrayBodyGenerator(srcArray);
063        final Body body = babGen.createBody();
064
065        final ByteBuffer chunkBuffer = ByteBuffer.allocate(chunkSize);
066
067        int reads = 0;
068        int bytesRead = 0;
069        while (body.read(chunkBuffer) != -1) {
070          reads += 1;
071          bytesRead += chunkBuffer.position();
072          chunkBuffer.clear();
073        }
074        assertEquals(reads, 4, "reads to drain generator");
075        assertEquals(bytesRead, srcArraySize, "bytes read");
076    }
077
078}
