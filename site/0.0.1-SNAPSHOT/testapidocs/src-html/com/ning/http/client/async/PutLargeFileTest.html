025import javax.servlet.ServletException;
026import javax.servlet.ServletInputStream;
027import javax.servlet.http.HttpServletRequest;
028import javax.servlet.http.HttpServletResponse;
029import java.io.File;
030import java.io.FileOutputStream;
031import java.io.IOException;
032import java.util.UUID;
033
034/**
035 * @author Benjamin Hanzelmann
036 */
037public abstract class PutLargeFileTest
038        extends AbstractBasicTest {
039
040    private File largeFile;
041
042    @Test(groups = {"standalone", "default_provider"}, enabled = true)
043    public void testPutLargeFile()
044            throws Exception {
045        byte[] bytes = "RatherLargeFileRatherLargeFileRatherLargeFileRatherLargeFile".getBytes("UTF-16");
046        long repeats = (1024 * 1024 * 100 / bytes.length) + 1;
047        largeFile = createTempFile(bytes, (int) repeats);
048        int timeout = (int) (largeFile.length() / 1000);
049        AsyncHttpClientConfig config = new AsyncHttpClientConfig.Builder().setConnectionTimeoutInMs(timeout).build();
050        AsyncHttpClient client = getAsyncHttpClient(config);
051        BoundRequestBuilder rb = client.preparePut(getTargetUrl());
052
053        rb.setBody(largeFile);
054
055        Response response = rb.execute().get();
056        Assert.assertEquals(200, response.getStatusCode());
057        client.close();
058    }
059
060    @Test(groups = {"standalone", "default_provider"})
061    public void testPutSmallFile()
062            throws Exception {
063        byte[] bytes = "RatherLargeFileRatherLargeFileRatherLargeFileRatherLargeFile".getBytes("UTF-16");
064        long repeats = (1024 / bytes.length) + 1;
065//        int timeout = (5000);
066        largeFile = createTempFile(bytes, (int) repeats);
067
068        AsyncHttpClientConfig config = new AsyncHttpClientConfig.Builder().build();
069        AsyncHttpClient client = getAsyncHttpClient(config);
070        BoundRequestBuilder rb = client.preparePut(getTargetUrl());
071
072        rb.setBody(largeFile);
073
074        Response response = rb.execute().get();
075        Assert.assertEquals(200, response.getStatusCode());
076        client.close();
077    }
078
079    @AfterMethod
080    public void after() {
081        largeFile.delete();
082    }
083
084    @Override
085    public AbstractHandler configureHandler()
086            throws Exception {
087        return new AbstractHandler() {
088
089            public void handle(String arg0, Request arg1, HttpServletRequest req, HttpServletResponse resp)
090                    throws IOException, ServletException {
091
092                ServletInputStream in = req.getInputStream();
093                byte[] b = new byte[8092];
094
095                int count = -1;
096                int total = 0;
097                while ((count = in.read(b)) != -1) {
098                    total += count;
099                }
100
101                System.err.println("consumed " + total + " bytes.");
102
103                resp.setStatus(200);
104                resp.getOutputStream().flush();
105                resp.getOutputStream().close();
106
107                arg1.setHandled(true);
108
109            }
110        };
111    }
112
113    private static final File TMP = new File(System.getProperty("java.io.tmpdir"), "ahc-tests-"
114            + UUID.randomUUID().toString().substring(0, 8));
115
116    public static File createTempFile(byte[] pattern, int repeat)
117            throws IOException {
118        TMP.mkdirs();
119        TMP.deleteOnExit();
120        File tmpFile = File.createTempFile("tmpfile-", ".data", TMP);
121        tmpFile.deleteOnExit();
122        write(pattern, repeat, tmpFile);
123
124        return tmpFile;
125    }
126
127    public static void write(byte[] pattern, int repeat, File file)
128            throws IOException {
129        file.deleteOnExit();
130        file.getParentFile().mkdirs();
131        FileOutputStream out = null;
132        try {
133            out = new FileOutputStream(file);
134            for (int i = 0; i < repeat; i++) {
135                out.write(pattern);
136            }
137        }
138        finally {
139            if (out != null) {
140                out.close();
141            }
142        }
143    }
