016package com.ning.http.client.async;
017
018import com.ning.http.client.AsyncHttpClient;
019import com.ning.http.client.Response;
020import org.eclipse.jetty.server.Request;
021import org.eclipse.jetty.server.handler.AbstractHandler;
022import org.testng.annotations.Test;
023
024import javax.servlet.ServletException;
025import javax.servlet.http.HttpServletRequest;
026import javax.servlet.http.HttpServletResponse;
027import java.io.File;
028import java.io.IOException;
029import java.net.URL;
030import java.util.concurrent.Future;
031
032import static org.testng.Assert.assertEquals;
033import static org.testng.Assert.assertNotNull;
034
035/**
036 * Test the Expect: 100-Continue.
037 */
038public abstract class Expect100ContinueTest extends AbstractBasicTest {
039
040    private class ZeroCopyHandler extends AbstractHandler {
041        public void handle(String s,
042                           Request r,
043                           HttpServletRequest httpRequest,
044                           HttpServletResponse httpResponse) throws IOException, ServletException {
045
046            int size = 10 * 1024;
047            if (httpRequest.getContentLength() > 0) {
048                size = httpRequest.getContentLength();
049            }
050            byte[] bytes = new byte[size];
051            if (bytes.length > 0) {
052                final int read = httpRequest.getInputStream().read(bytes);
053                httpResponse.getOutputStream().write(bytes, 0, read);
054            }
055
056            httpResponse.setStatus(200);
057            httpResponse.getOutputStream().flush();
058        }
059    }
060
061    @Test(groups = {"standalone", "default_provider"})
062    public void Expect100Continue() throws Throwable {
063        AsyncHttpClient client = getAsyncHttpClient(null);
064
065        ClassLoader cl = getClass().getClassLoader();
066        URL url = cl.getResource("SimpleTextFile.txt");
067        File file = new File(url.toURI());
068
069        Future<Response> f = client.preparePut("http://127.0.0.1:" + port1 + "/").setHeader("Expect", "100-continue").setBody(file).execute();
070        Response resp = f.get();
071        assertNotNull(resp);
072        assertEquals(resp.getStatusCode(), HttpServletResponse.SC_OK);
073        assertEquals(resp.getResponseBody(), "This is a simple test file");
074        client.close();
075
076    }
077
078    @Override
079    public AbstractHandler configureHandler() throws Exception {
080        return new ZeroCopyHandler();
081    }
082
083}
