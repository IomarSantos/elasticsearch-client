028    private final class EchoByteWebSocket implements org.eclipse.jetty.websocket.WebSocket, org.eclipse.jetty.websocket.WebSocket.OnBinaryMessage {
029
030        private Connection connection;
031
032        @Override
033        public void onOpen(Connection connection) {
034            this.connection = connection;
035            connection.setMaxBinaryMessageSize(1000);
036        }
037
038        @Override
039        public void onClose(int i, String s) {
040            connection.close();
041        }
042
043        @Override
044        public void onMessage(byte[] bytes, int i, int i1) {
045            try {
046                connection.sendMessage(bytes, i, i1);
047            } catch (IOException e) {
048                try {
049                    connection.sendMessage("FAIL");
050                } catch (IOException e1) {
051                    e1.printStackTrace();
052                }
053            }
054        }
055    }

059        return new WebSocketHandler() {
060            @Override
061            public org.eclipse.jetty.websocket.WebSocket doWebSocketConnect(HttpServletRequest httpServletRequest, String s) {
062                return new EchoByteWebSocket();
063            }
064        };

067    @Test
068    public void echoByte() throws Throwable {
069        AsyncHttpClient c = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().build());
070        final CountDownLatch latch = new CountDownLatch(1);
071        final AtomicReference<byte[]> text = new AtomicReference<byte[]>(new byte[0]);
072
073        WebSocket websocket = c.prepareGet(getTargetUrl())
074                .execute(new WebSocketUpgradeHandler.Builder().addWebSocketListener(new WebSocketByteListener() {
075
076                    @Override
077                    public void onOpen(WebSocket websocket) {
078                    }
079
080                    @Override
081                    public void onClose(WebSocket websocket) {
082                        latch.countDown();
083                    }
084
085                    @Override
086                    public void onError(Throwable t) {
087                        t.printStackTrace();
088                        latch.countDown();
089                    }
090
091                    @Override
092                    public void onMessage(byte[] message) {
093                        text.set(message);
094                        latch.countDown();
095                    }
096
097                    @Override
098                    public void onFragment(byte[] fragment, boolean last) {
099                    }
100                }).build()).get();
101
102        websocket.sendMessage("ECHO".getBytes());
103
104        latch.await();
105        assertEquals(text.get(), "ECHO".getBytes());
106    }

108    @Test
109    public void echoTwoMessagesTest() throws Throwable {
110        AsyncHttpClient c = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().build());
111        final CountDownLatch latch = new CountDownLatch(2);
112        final AtomicReference<byte[]> text = new AtomicReference<byte[]>(null);
113
114        WebSocket websocket = c.prepareGet(getTargetUrl())
115                .execute(new WebSocketUpgradeHandler.Builder().addWebSocketListener(new WebSocketByteListener() {
116
117                    @Override
118                    public void onOpen(WebSocket websocket) {
119                    }
120
121                    @Override
122                    public void onClose(WebSocket websocket) {
123                        latch.countDown();
124                    }
125
126                    @Override
127                    public void onError(Throwable t) {
128                        t.printStackTrace();
129                        latch.countDown();
130                    }
131
132                    @Override
133                    public void onMessage(byte[] message) {
134                        if (text.get() == null) {
135                            text.set(message);
136                        } else {
137                            byte[] n = new byte[text.get().length + message.length];
138                            System.arraycopy(text.get(), 0, n, 0, text.get().length);
139                            System.arraycopy(message, 0, n, text.get().length, message.length);
140                            text.set(n);
141                        }
142                        latch.countDown();
143                    }
144
145                    @Override
146                    public void onFragment(byte[] fragment, boolean last) {
147                    }
148                }).build()).get();
149
150        websocket.sendMessage("ECHO".getBytes()).sendMessage("ECHO".getBytes());
151
152        latch.await();
153        assertEquals(text.get(), "ECHOECHO".getBytes());
154    }
