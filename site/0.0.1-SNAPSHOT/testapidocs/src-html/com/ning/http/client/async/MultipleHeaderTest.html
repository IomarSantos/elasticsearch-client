051    @Test(groups = {"standalone", "default_provider"})
052    public void testMultipleOtherHeaders()
053            throws IOException, ExecutionException, TimeoutException, InterruptedException {
054        final String[] xffHeaders = new String[]{null, null};
055
056        AsyncHttpClient ahc = getAsyncHttpClient(null);
057        Request req = new RequestBuilder("GET").setUrl("http://localhost:" + port1 + "/MultiOther").build();
058        final CountDownLatch latch = new CountDownLatch(1);
059        ahc.executeRequest(req, new AsyncHandler<Void>() {
060            public void onThrowable(Throwable t) {
061                t.printStackTrace(System.out);
062            }
063
064            public STATE onBodyPartReceived(HttpResponseBodyPart objectHttpResponseBodyPart) throws Exception {
065                return STATE.CONTINUE;
066            }
067
068            public STATE onStatusReceived(HttpResponseStatus objectHttpResponseStatus) throws Exception {
069                return STATE.CONTINUE;
070            }
071
072            public STATE onHeadersReceived(HttpResponseHeaders response) throws Exception {
073                int i = 0;
074                for (String header : response.getHeaders().get("X-Forwarded-For")) {
075                    xffHeaders[i++] = header;
076                }
077                latch.countDown();
078                return STATE.CONTINUE;
079            }
080
081            public Void onCompleted() throws Exception {
082                return null;
083            }
084        }).get(3, TimeUnit.SECONDS);
085
086        if (!latch.await(2, TimeUnit.SECONDS)) {
087            Assert.fail("Time out");
088        }
089        Assert.assertNotNull(xffHeaders[0]);
090        Assert.assertNotNull(xffHeaders[1]);
091        try {
092            Assert.assertEquals(xffHeaders[0], "abc");
093            Assert.assertEquals(xffHeaders[1], "def");
094        } catch (AssertionError ex) {
095            Assert.assertEquals(xffHeaders[1], "abc");
096            Assert.assertEquals(xffHeaders[0], "def");
097        }
098        ahc.close();
099    }

102    @Test(groups = {"standalone", "default_provider"})
103    public void testMultipleEntityHeaders()
104            throws IOException, ExecutionException, TimeoutException, InterruptedException {
105        final String[] clHeaders = new String[]{null, null};
106
107        AsyncHttpClient ahc = getAsyncHttpClient(null);
108        Request req = new RequestBuilder("GET").setUrl("http://localhost:" + port1 + "/MultiEnt").build();
109        final CountDownLatch latch = new CountDownLatch(1);
110        ahc.executeRequest(req, new AsyncHandler<Void>() {
111            public void onThrowable(Throwable t) {
112                t.printStackTrace(System.out);
113            }
114
115            public STATE onBodyPartReceived(HttpResponseBodyPart objectHttpResponseBodyPart) throws Exception {
116                return STATE.CONTINUE;
117            }
118
119            public STATE onStatusReceived(HttpResponseStatus objectHttpResponseStatus) throws Exception {
120                return STATE.CONTINUE;
121            }
122
123            public STATE onHeadersReceived(HttpResponseHeaders response) throws Exception {
124                try {
125                    int i = 0;
126                    for (String header : response.getHeaders().get("Content-Length")) {
127                        clHeaders[i++] = header;
128                    }
129                } finally {
130                    latch.countDown();
131                }
132                return STATE.CONTINUE;
133            }
134
135            public Void onCompleted() throws Exception {
136                return null;
137            }
138        }).get(3, TimeUnit.SECONDS);
139
140        if (!latch.await(2, TimeUnit.SECONDS)) {
141            Assert.fail("Time out");
142        }
143        Assert.assertNotNull(clHeaders[0]);
144        Assert.assertNotNull(clHeaders[1]);
145
146        // We can predict the order
147        try {
148            Assert.assertEquals(clHeaders[0], "2");
149            Assert.assertEquals(clHeaders[1], "1");
150        } catch (Throwable ex) {
151            Assert.assertEquals(clHeaders[0], "1");
152            Assert.assertEquals(clHeaders[1], "2");
153        }
154        ahc.close();
155
156    }

158    @BeforeClass(alwaysRun = true)
159    public void setUpGlobal() throws Exception {
160        port1 = findFreePort();
161
162        serverSocket = new ServerSocket(port1);
163        executorService = Executors.newFixedThreadPool(1);
164        voidFuture = executorService.submit(new Callable<Void>() {
165            public Void call() throws Exception {
166                Socket socket;
167                while ((socket = serverSocket.accept()) != null) {
168                    InputStream inputStream = socket.getInputStream();
169                    BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
170                    String req = reader.readLine().split(" ")[1];
171                    int i = inputStream.available();
172                    long l = inputStream.skip(i);
173                    Assert.assertEquals(l, i);
174                    socket.shutdownInput();
175                    if (req.endsWith("MultiEnt")) {
176                        OutputStreamWriter outputStreamWriter = new OutputStreamWriter(socket.getOutputStream());
177                        outputStreamWriter.append("HTTP/1.0 200 OK\n" +
178                                "Connection: close\n" +
179                                "Content-Type: text/plain; charset=iso-8859-1\n" +
180                                "Content-Length: 2\n" +
181                                "Content-Length: 1\n" +
182                                "\n0\n");
183                        outputStreamWriter.flush();
184                        socket.shutdownOutput();
185                    } else if (req.endsWith("MultiOther")) {
186                        OutputStreamWriter outputStreamWriter = new OutputStreamWriter(socket.getOutputStream());
187                        outputStreamWriter.append("HTTP/1.0 200 OK\n" +
188                                "Connection: close\n" +
189                                "Content-Type: text/plain; charset=iso-8859-1\n" +
190                                "Content-Length: 1\n" +
191                                "X-Forwarded-For: abc\n" +
192                                "X-Forwarded-For: def\n" +
193                                "\n0\n");
194                        outputStreamWriter.flush();
195                        socket.shutdownOutput();
196                    }
197                }
198                return null;
199            }
200        });
201    }
202
203    @AfterClass(alwaysRun = true)
204    public void tearDownGlobal() throws Exception {
205        voidFuture.cancel(true);
206        executorService.shutdownNow();
207        serverSocket.close();
208    }
