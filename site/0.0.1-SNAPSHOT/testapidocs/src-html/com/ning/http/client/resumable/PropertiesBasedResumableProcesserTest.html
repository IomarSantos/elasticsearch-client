016import org.testng.annotations.Test;
017
018import java.util.Map;
019
020import static org.testng.Assert.assertEquals;
021
022/**
023 * @author Benjamin Hanzelmann
024 */
025public class PropertiesBasedResumableProcesserTest {
026    @Test (enabled = false)
027    public void testSaveLoad()
028            throws Exception {
029        PropertiesBasedResumableProcessor p = new PropertiesBasedResumableProcessor();
030        p.put("http://localhost/test.url", 15L);
031        p.put("http://localhost/test2.url", 50L);
032        p.save(null);
033        p = new PropertiesBasedResumableProcessor();
034        Map<String, Long> m = p.load();
035        assertEquals(m.size(), 2);
036        assertEquals(m.get("http://localhost/test.url"), Long.valueOf(15L));
037        assertEquals(m.get("http://localhost/test2.url"), Long.valueOf(50L));
038    }
039
040}
