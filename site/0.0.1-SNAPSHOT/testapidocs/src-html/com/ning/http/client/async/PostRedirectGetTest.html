078    private void doTestNegative(final int status, boolean strict) throws Exception {
079        AsyncHttpClient p = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().setFollowRedirects(true).
080                setStrict302Handling(strict).
081                addResponseFilter(new ResponseFilter() {
082                    @Override
083                    public <T> FilterContext<T> filter(FilterContext<T> ctx) throws FilterException {
084                        // pass on the x-expect-get and remove the x-redirect
085                        // headers if found in the response
086                        ctx.getResponseHeaders().getHeaders().get("x-expect-post");
087                        ctx.getRequest().getHeaders().add("x-expect-post", "true");
088                        ctx.getRequest().getHeaders().remove("x-redirect");
089                        return ctx;
090                    }
091                }).build());
092        Request request = new RequestBuilder("POST").setUrl(getTargetUrl())
093                .addParameter("q", "a b")
094                .addHeader("x-redirect", +status + "@" + "http://localhost:" + port1 + "/foo/bar/baz")
095                .addHeader("x-negative", "true")
096                .build();
097        Future<Integer> responseFuture = p.executeRequest(request, new AsyncCompletionHandler<Integer>() {
098
099            @Override
100            public Integer onCompleted(Response response) throws Exception {
101                return response.getStatusCode();
102            }
103
104            /* @Override */
105            public void onThrowable(Throwable t) {
106                t.printStackTrace();
107                Assert.fail("Unexpected exception: " + t.getMessage(), t);
108            }
109
110        });
111        int statusCode = responseFuture.get();
112        Assert.assertEquals(statusCode, 200);
113        p.close();
114    }

157    public static class PostRedirectGetHandler extends AbstractHandler {
158
159        final AtomicInteger counter = new AtomicInteger();
160
161        /* @Override */
162        public void handle(String pathInContext,
163                           org.eclipse.jetty.server.Request request,
164                           HttpServletRequest httpRequest,
165                           HttpServletResponse httpResponse) throws IOException, ServletException {
166
167            final boolean expectGet = (httpRequest.getHeader("x-expect-get") != null);
168            final boolean expectPost = (httpRequest.getHeader("x-expect-post") != null);
169            if (expectGet) {
170                final String method = request.getMethod();
171                if (!"GET".equals(method)) {
172                    httpResponse.sendError(500, "Incorrect method.  Expected GET, received " + method);
173                    return;
174                }
175                httpResponse.setStatus(200);
176                httpResponse.getOutputStream().write("OK".getBytes());
177                httpResponse.getOutputStream().flush();
178                return;
179            } else if (expectPost) {
180                final String method = request.getMethod();
181                if (!"POST".equals(method)) {
182                    httpResponse.sendError(500, "Incorrect method.  Expected POST, received " + method);
183                    return;
184                }
185                httpResponse.setStatus(200);
186                httpResponse.getOutputStream().write("OK".getBytes());
187                httpResponse.getOutputStream().flush();
188                return;
189            }
190
191            String header = httpRequest.getHeader("x-redirect");
192            if (header != null) {
193                // format for header is <status code>|<location url>
194                String[] parts = header.split("@");
195                int redirectCode;
196                try {
197                    redirectCode = Integer.parseInt(parts[0]);
198                } catch (Exception ex) {
199                    ex.printStackTrace();
200                    httpResponse.sendError(500, "Unable to parse redirect code");
201                    return;
202                }
203                httpResponse.setStatus(redirectCode);
204                if (httpRequest.getHeader("x-negative") == null) {
205                    httpResponse.addHeader("x-expect-get", "true");
206                } else {
207                    httpResponse.addHeader("x-expect-post", "true");
208                }
209                httpResponse.setContentLength(0);
210                httpResponse.addHeader("Location", parts[1] + counter.getAndIncrement());
211                httpResponse.getOutputStream().flush();
212                return;
213            }
214
215
216            httpResponse.sendError(500);
217            httpResponse.getOutputStream().flush();
218            httpResponse.getOutputStream().close();
219        }
220    }
