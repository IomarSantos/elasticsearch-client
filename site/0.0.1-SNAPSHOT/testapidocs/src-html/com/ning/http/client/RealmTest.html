013package com.ning.http.client;
014
015import com.ning.http.client.Realm.AuthScheme;
016import com.ning.http.client.Realm.RealmBuilder;
017import org.testng.Assert;
018import java.math.BigInteger;
019import java.security.MessageDigest;
020
021import org.testng.annotations.Test;
022
023public class RealmTest {
024    @Test(groups = "fast")
025    public void testClone() {
026        RealmBuilder builder = new RealmBuilder();
027        builder.setPrincipal( "user" ).setPassword( "pass" );
028        builder.setEnconding( "enc" ).setUsePreemptiveAuth( true );
029        builder.setRealmName( "realm" ).setAlgorithm( "algo" );
030        builder.setScheme( AuthScheme.BASIC );
031        Realm orig = builder.build();
032        
033        Realm clone = new RealmBuilder().clone( orig ).build();
034        Assert.assertEquals( clone.getPrincipal(), orig.getPrincipal() );
035        Assert.assertEquals( clone.getPassword(), orig.getPassword() );
036        Assert.assertEquals( clone.getEncoding(), orig.getEncoding() );
037        Assert.assertEquals( clone.getUsePreemptiveAuth(), orig.getUsePreemptiveAuth() );
038        Assert.assertEquals( clone.getRealmName(), orig.getRealmName() );
039        Assert.assertEquals( clone.getAlgorithm(), orig.getAlgorithm() );
040        Assert.assertEquals( clone.getAuthScheme(), orig.getAuthScheme() );
041    }
042    @Test(groups = "fast")
043    public void testOldDigestEmptyString() {
044        String qop="";
045        testOldDigest(qop);
046    }
047    @Test(groups = "fast")
048    public void testOldDigestNull() {
049        String qop=null;
050        testOldDigest(qop);
051    }
052
053    private void testOldDigest(String qop){
054        String user="user";
055        String pass="pass";
056        String realm="realm";
057        String nonce="nonce";
058        String method="GET";
059        String uri="/foo";
060        RealmBuilder builder = new RealmBuilder();
061        builder.setPrincipal( user ).setPassword( pass );
062        builder.setNonce( nonce );
063        builder.setUri( uri );
064        builder.setMethodName(method);
065        builder.setRealmName( realm );
066        builder.setQop(qop);
067        builder.setScheme( AuthScheme.DIGEST );
068        Realm orig = builder.build();
069
070        String ha1=getMd5(user +":" + realm +":"+pass);
071        String ha2=getMd5(method +":"+ uri);
072        String expectedResponse=getMd5(ha1 +":" + nonce +":"  + ha2);
073
074        Assert.assertEquals(expectedResponse,orig.getResponse());
075    }
076
077    @Test(groups = "fast")
078    public void testStrongDigest() {
079        String user="user";
080        String pass="pass";
081        String realm="realm";
082        String nonce="nonce";
083        String method="GET";
084        String uri="/foo";
085        String qop="auth";
086        RealmBuilder builder = new RealmBuilder();
087        builder.setPrincipal( user ).setPassword( pass );
088        builder.setNonce( nonce );
089        builder.setUri( uri );
090        builder.setMethodName(method);
091        builder.setRealmName( realm );
092        builder.setQop(qop);
093        builder.setScheme( AuthScheme.DIGEST );
094        Realm orig = builder.build();
095
096        String nc = orig.getNc();
097        String cnonce = orig.getCnonce();
098        String ha1=getMd5(user +":" + realm +":"+pass);
099        String ha2=getMd5(method +":"+ uri);
100        String expectedResponse=getMd5(ha1 +":" + nonce +":" + nc + ":" + cnonce +":" + qop + ":" + ha2);
101
102        Assert.assertEquals(expectedResponse,orig.getResponse());
103    }
104
105    private String getMd5(String what){
106            try {
107            MessageDigest md = MessageDigest.getInstance("MD5");
108                md.update(what.getBytes("ISO-8859-1"));
109                byte[] hash  = md.digest();
110                BigInteger bi = new BigInteger(1, hash);
111                String result = bi.toString(16);
112                if (result.length() % 2 != 0) {
113                    return "0" + result;
114                }
115                return result;
116            } catch (Exception e) {
117                throw new RuntimeException(e);
118            }
119    }
