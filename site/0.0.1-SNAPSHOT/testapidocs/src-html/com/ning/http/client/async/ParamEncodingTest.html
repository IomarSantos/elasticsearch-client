016package com.ning.http.client.async;
017
018import com.ning.http.client.AsyncHttpClient;
019import com.ning.http.client.Response;
020import org.eclipse.jetty.server.Request;
021import org.eclipse.jetty.server.handler.AbstractHandler;
022import org.testng.annotations.Test;
023
024import javax.servlet.ServletException;
025import javax.servlet.http.HttpServletRequest;
026import javax.servlet.http.HttpServletResponse;
027import java.io.IOException;
028import java.util.concurrent.ExecutionException;
029import java.util.concurrent.Future;
030import java.util.concurrent.TimeUnit;
031import java.util.concurrent.TimeoutException;
032
033import static org.testng.Assert.assertEquals;
034import static org.testng.Assert.assertNotNull;
035
036public abstract class ParamEncodingTest extends AbstractBasicTest {
037
038    private class ParamEncoding extends AbstractHandler {
039        public void handle(String s,
040                           Request r,
041                           HttpServletRequest request,
042                           HttpServletResponse response) throws IOException, ServletException {
043            if ("POST".equalsIgnoreCase(request.getMethod())) {
044                String p = request.getParameter("test");
045                if (p != null && !p.equals("")) {
046                    response.setStatus(HttpServletResponse.SC_OK);
047                    response.addHeader("X-Param", p);
048                } else {
049                    response.sendError(HttpServletResponse.SC_NOT_ACCEPTABLE);
050                }
051            } else { // this handler is to handle POST request
052                response.sendError(HttpServletResponse.SC_FORBIDDEN);
053            }
054            response.getOutputStream().flush();
055            response.getOutputStream().close();
056        }
057    }
058
059    @Test(groups = {"standalone", "default_provider"})
060    public void testParameters() throws IOException, ExecutionException, TimeoutException, InterruptedException {
061
062        String value = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKQLMNOPQRSTUVWXYZ1234567809`~!@#$%^&*()_+-=,.<>/?;:'\"[]{}\\| ";
063        AsyncHttpClient client = getAsyncHttpClient(null);
064        Future<Response> f = client
065                .preparePost("http://127.0.0.1:" + port1)
066                .addParameter("test", value)
067                .execute();
068        Response resp = f.get(10, TimeUnit.SECONDS);
069        assertNotNull(resp);
070        assertEquals(resp.getStatusCode(), HttpServletResponse.SC_OK);
071        assertEquals(resp.getHeader("X-Param"), value.trim());
072        client.close();
073    }
074
075    @Override
076    public AbstractHandler configureHandler() throws Exception {
077        return new ParamEncoding();
078    }
079}
