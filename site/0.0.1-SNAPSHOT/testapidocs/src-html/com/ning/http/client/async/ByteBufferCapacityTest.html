021import javax.servlet.ServletException;
022import javax.servlet.http.HttpServletRequest;
023import javax.servlet.http.HttpServletResponse;

025import java.io.FileOutputStream;

028import java.io.OutputStream;
029import java.util.Enumeration;
030import java.util.UUID;
031import java.util.concurrent.atomic.AtomicInteger;
032
033import static org.testng.Assert.*;
034
035public abstract class ByteBufferCapacityTest extends AbstractBasicTest {
036    private static final File TMP = new File(System.getProperty("java.io.tmpdir"), "ahc-tests-"
037            + UUID.randomUUID().toString().substring(0, 8));
038
039    private class BasicHandler extends AbstractHandler {
040
041        public void handle(String s,
042                           org.eclipse.jetty.server.Request r,
043                           HttpServletRequest httpRequest,
044                           HttpServletResponse httpResponse) throws IOException, ServletException {
045
046            Enumeration<?> e = httpRequest.getHeaderNames();
047            String param;
048            while (e.hasMoreElements()) {
049                param = e.nextElement().toString();
050                httpResponse.addHeader("X-" + param, httpRequest.getHeader(param));
051            }
052
053            int size = 10 * 1024;
054            if (httpRequest.getContentLength() > 0) {
055                size = httpRequest.getContentLength();
056            }
057            byte[] bytes = new byte[size];
058            if (bytes.length > 0) {
059                final InputStream in = httpRequest.getInputStream();
060                final OutputStream out = httpResponse.getOutputStream();
061                int read;
062                while ((read = in.read(bytes)) != -1) {
063                    out.write(bytes, 0, read);
064                }
065            }
066
067            httpResponse.setStatus(200);
068            httpResponse.getOutputStream().flush();
069            httpResponse.getOutputStream().close();
070        }
071    }
072
073    @Override
074    public AbstractHandler configureHandler() throws Exception {
075        return new BasicHandler();
076    }
077
078    @Test(groups = {"standalone", "default_provider"})
079    public void basicByteBufferTest() throws Throwable {
080        AsyncHttpClient c = getAsyncHttpClient(null);
081
082        byte[] bytes = "RatherLargeFileRatherLargeFileRatherLargeFileRatherLargeFile".getBytes("UTF-16");
083        long repeats = (1024 * 100 * 10 / bytes.length) + 1;
084        File largeFile = createTempFile(bytes, (int) repeats);
085        final AtomicInteger byteReceived = new AtomicInteger();
086
087        try {
088            Response response = c.preparePut(getTargetUrl()).setBody(largeFile)
089                    .execute(new AsyncCompletionHandlerAdapter() {
090                        /* @Override */
091                        public STATE onBodyPartReceived(final HttpResponseBodyPart content) throws Exception {
092                            byteReceived.addAndGet(content.getBodyByteBuffer().capacity());
093                            return super.onBodyPartReceived(content);
094                        }
095
096                    }).get();
097
098            assertNotNull(response);
099            assertEquals(response.getStatusCode(), 200);
100            assertEquals(byteReceived.get(), largeFile.length());
101            assertEquals(response.getResponseBody().length(), largeFile.length());
102
103        } catch (IOException ex) {
104            fail("Should have timed out");
105        }
106        c.close();
107    }
108
109    public String getTargetUrl() {
110        return String.format("http://127.0.0.1:%d/foo/test", port1);
111    }
112
113    public static File createTempFile(byte[] pattern, int repeat)
114            throws IOException {
115        TMP.mkdirs();
116        TMP.deleteOnExit();
117        File tmpFile = File.createTempFile("tmpfile-", ".data", TMP);
118        write(pattern, repeat, tmpFile);
119
120        return tmpFile;
121    }
122
123    public static void write(byte[] pattern, int repeat, File file)
124            throws IOException {
125        file.deleteOnExit();
126        file.getParentFile().mkdirs();
127        FileOutputStream out = null;
128        try {
129            out = new FileOutputStream(file);
130            for (int i = 0; i < repeat; i++) {
131                out.write(pattern);
132            }
133        }
134        finally {
135            if (out != null) {
136                out.close();
137            }
138        }
139    }
140}
