016package com.ning.http.client.async;
017
018import com.ning.http.client.AsyncCompletionHandlerBase;
019import com.ning.http.client.AsyncHttpClient;
020import com.ning.http.client.Request;
021import com.ning.http.client.RequestBuilder;
022import com.ning.http.client.Response;
023import org.eclipse.jetty.server.handler.AbstractHandler;
024import org.testng.Assert;
025import org.testng.annotations.Test;
026
027import javax.servlet.ServletException;
028import javax.servlet.http.HttpServletRequest;
029import javax.servlet.http.HttpServletResponse;
030import java.io.IOException;
031import java.util.concurrent.BrokenBarrierException;
032import java.util.concurrent.CountDownLatch;
033import java.util.concurrent.ExecutionException;
034import java.util.concurrent.TimeUnit;
035import java.util.concurrent.TimeoutException;
036
037/**
038 * Tests HEAD request that gets 302 response.
039 *
040 * @author Hubert Iwaniuk
041 */
042public abstract class Head302Test extends AbstractBasicTest {
043    /**
044     * Handler that does Found (302) in response to HEAD method.
045     */
046    private class Head302handler extends AbstractHandler {
047        public void handle(String s,
048                           org.eclipse.jetty.server.Request r,
049                           HttpServletRequest request,
050                           HttpServletResponse response) throws IOException, ServletException {
051            if ("HEAD".equalsIgnoreCase(request.getMethod())) {
052                if (request.getPathInfo().endsWith("_moved")) {
053                    response.setStatus(HttpServletResponse.SC_OK);
054                } else {
055                    response.setStatus(HttpServletResponse.SC_FOUND); // 302
056                    response.setHeader("Location", request.getPathInfo() + "_moved");
057                }
058            } else { // this handler is to handle HEAD reqeust
059                response.setStatus(HttpServletResponse.SC_FORBIDDEN);
060            }
061        }
062    }
063
064    @Test(groups = {"standalone", "default_provider"})
065    public void testHEAD302() throws IOException, BrokenBarrierException, InterruptedException, ExecutionException, TimeoutException {
066        AsyncHttpClient client = getAsyncHttpClient(null);
067        final CountDownLatch l = new CountDownLatch(1);
068        Request request = new RequestBuilder("HEAD").setUrl("http://127.0.0.1:" + port1 + "/Test").build();
069
070        client.executeRequest(request, new AsyncCompletionHandlerBase() {
071            @Override
072            public Response onCompleted(Response response) throws Exception {
073                l.countDown();
074                return super.onCompleted(response);
075            }
076        }).get(3, TimeUnit.SECONDS);
077
078        if (!l.await(TIMEOUT, TimeUnit.SECONDS)) {
079            Assert.fail("Timeout out");
080        }
081        client.close();
082    }
083
084    @Override
085    public AbstractHandler configureHandler() throws Exception {
086        return new Head302handler();
087    }
088}
