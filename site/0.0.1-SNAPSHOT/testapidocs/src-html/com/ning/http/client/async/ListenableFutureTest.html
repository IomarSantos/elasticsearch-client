013package com.ning.http.client.async;
014
015import com.ning.http.client.AsyncHttpClient;
016import com.ning.http.client.ListenableFuture;
017import com.ning.http.client.Response;
018import org.testng.annotations.Test;
019
020import java.util.concurrent.CountDownLatch;
021import java.util.concurrent.ExecutionException;
022import java.util.concurrent.Executors;
023import java.util.concurrent.TimeUnit;
024import java.util.concurrent.atomic.AtomicInteger;
025
026import static org.testng.Assert.assertEquals;
027
028public abstract class ListenableFutureTest extends AbstractBasicTest {
029
030    @Test(groups = {"standalone", "default_provider"})
031    public void testListenableFuture() throws Throwable {
032        final AtomicInteger statusCode = new AtomicInteger(500);
033        AsyncHttpClient ahc = getAsyncHttpClient(null);
034        final CountDownLatch latch = new CountDownLatch(1);
035        final ListenableFuture<Response> future = ahc.prepareGet(getTargetUrl()).execute();
036        future.addListener(new Runnable(){
037
038            public void run() {
039                try {
040                    statusCode.set(future.get().getStatusCode());
041                    latch.countDown();
042                } catch (InterruptedException e) {
043                    e.printStackTrace();
044                } catch (ExecutionException e) {
045                    e.printStackTrace();
046                }
047            }
048        }, Executors.newFixedThreadPool(1));
049
050        latch.await(10, TimeUnit.SECONDS);
051        assertEquals(statusCode.get(), 200);
052        ahc.close();
053    }
054}
