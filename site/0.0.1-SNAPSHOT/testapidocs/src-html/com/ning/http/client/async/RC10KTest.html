047/**
048 * Reverse C10K Problem test.
049 *
050 * @author Hubert Iwaniuk
051 */
052public abstract class RC10KTest extends AbstractBasicTest {
053    private static final int C10K = 1000;
054    private static final String ARG_HEADER = "Arg";
055    private static final int SRV_COUNT = 10;
056    protected List<Server> servers = new ArrayList<Server>(SRV_COUNT);
057    private int[] ports;
058
059    @BeforeClass(alwaysRun = true)
060    public void setUpGlobal() throws Exception {
061        ports = new int[SRV_COUNT];
062        for (int i = 0; i < SRV_COUNT; i++) {
063            ports[i] = createServer();
064        }
065        log.info("Local HTTP servers started successfully");
066    }
067
068    @AfterClass(alwaysRun = true)
069    public void tearDownGlobal() throws Exception {
070        for (Server srv : servers) {
071            srv.stop();
072        }
073    }
074
075    private int createServer() throws Exception {
076        Server srv = new Server();
077        Connector listener = new SelectChannelConnector();
078        listener.setHost("127.0.0.1");
079        int port = findFreePort();
080        listener.setPort(port);
081        srv.addConnector(listener);
082        srv.setHandler(configureHandler());
083        srv.start();
084        servers.add(srv);
085        return port;
086    }
087
088    @Override
089    public AbstractHandler configureHandler() throws Exception {
090        return new AbstractHandler() {
091            public void handle(String s, Request r, HttpServletRequest req, HttpServletResponse resp) throws IOException, ServletException {
092                resp.setContentType("text/pain");
093                String arg = s.substring(1);
094                resp.setHeader(ARG_HEADER, arg);
095                resp.setStatus(200);
096                resp.getOutputStream().print(arg);
097                resp.getOutputStream().flush();
098                resp.getOutputStream().close();
099            }
100        };
101    }
102
103    @Test(timeOut = 10 * 60 * 1000, groups = "scalability")
104    public void rc10kProblem() throws IOException, ExecutionException, TimeoutException, InterruptedException {
105        AsyncHttpClient ahc = getAsyncHttpClient(
106                new AsyncHttpClientConfig.Builder().setMaximumConnectionsPerHost(C10K).setAllowPoolingConnection(true).build());
107        List<Future<Integer>> resps = new ArrayList<Future<Integer>>(C10K);
108        int i = 0;
109        while (i < C10K) {
110            resps.add(ahc.prepareGet(String.format("http://127.0.0.1:%d/%d", ports[i % SRV_COUNT], i)).execute(new MyAsyncHandler(i++)));
111        }
112        i = 0;
113        for (Future<Integer> fResp : resps) {
114            Integer resp = fResp.get();
115            assertNotNull(resp);
116            assertEquals(resp.intValue(), i++);
117        }
118        ahc.close();
119    }
120
121    private class MyAsyncHandler implements AsyncHandler<Integer> {
122        private String arg;
123        private AtomicInteger result = new AtomicInteger(-1);
124
125        public MyAsyncHandler(int i) {
126            arg = String.format("%d", i);
127        }
128
129        public void onThrowable(Throwable t) {
130            log.warn("onThrowable called.", t);
131        }
132
133        public STATE onBodyPartReceived(HttpResponseBodyPart event) throws Exception {
134            String s = new String(event.getBodyPartBytes());
135            result.compareAndSet(-1, new Integer(s.trim().equals("") ? "-1" : s));
136            return STATE.CONTINUE;
137        }
138
139        public STATE onStatusReceived(HttpResponseStatus event) throws Exception {
140            assertEquals(event.getStatusCode(), 200);
141            return STATE.CONTINUE;
142        }
143
144        public STATE onHeadersReceived(HttpResponseHeaders event) throws Exception {
145            assertEquals(event.getHeaders().getJoinedValue(ARG_HEADER, ", "), arg);
146            return STATE.CONTINUE;
147        }
148
149        public Integer onCompleted() throws Exception {
150            return result.get();
151        }
152    }
153}
