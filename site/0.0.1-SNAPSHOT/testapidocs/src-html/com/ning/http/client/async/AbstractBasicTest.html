052    public static class EchoHandler extends AbstractHandler {
053
054        /* @Override */
055        public void handle(String pathInContext,
056                           Request request,
057                           HttpServletRequest httpRequest,
058                           HttpServletResponse httpResponse) throws IOException, ServletException {
059
060            if (httpRequest.getHeader("X-HEAD") != null) {
061                httpResponse.setContentLength(1);
062            }
063
064            if (httpRequest.getHeader("X-ISO") != null) {
065                httpResponse.setContentType("text/html; charset=ISO-8859-1");
066            } else {
067                httpResponse.setContentType("text/html; charset=utf-8");
068            }
069
070            if (request.getMethod().equalsIgnoreCase("OPTIONS")) {
071                httpResponse.addHeader("Allow","GET,HEAD,POST,OPTIONS,TRACE");
072            };
073
074            Enumeration<?> e = httpRequest.getHeaderNames();
075            String param;
076            while (e.hasMoreElements()) {
077                param = e.nextElement().toString();
078
079                if (param.startsWith("LockThread")) {
080                    try {
081                        Thread.sleep(40 * 1000);
082                    } catch (InterruptedException ex) {
083                    }
084                }
085
086                if (param.startsWith("X-redirect")) {
087                    httpResponse.sendRedirect(httpRequest.getHeader("X-redirect"));
088                    return;
089                }
090                httpResponse.addHeader("X-" + param, httpRequest.getHeader(param));
091            }
092
093            Enumeration<?> i = httpRequest.getParameterNames();
094
095            StringBuilder requestBody = new StringBuilder();
096            while (i.hasMoreElements()) {
097                param = i.nextElement().toString();
098                httpResponse.addHeader("X-" + param, httpRequest.getParameter(param));
099                requestBody.append(param);
100                requestBody.append("_");
101            }
102
103            String pathInfo = httpRequest.getPathInfo();
104            if (pathInfo != null)
105                httpResponse.addHeader("X-pathInfo", pathInfo);
106
107            String queryString = httpRequest.getQueryString();
108            if (queryString != null)
109                httpResponse.addHeader("X-queryString", queryString);
110
111            httpResponse.addHeader("X-KEEP-ALIVE", httpRequest.getRemoteAddr() + ":" + httpRequest.getRemotePort());
112
113            javax.servlet.http.Cookie[] cs = httpRequest.getCookies();
114            if (cs != null) {
115                for (javax.servlet.http.Cookie c : cs) {
116                    httpResponse.addCookie(c);
117                }
118            }
119
120            if (requestBody.length() > 0) {
121                httpResponse.getOutputStream().write(requestBody.toString().getBytes());
122            }
123
124            int size = 16384;
125            if (httpRequest.getContentLength() > 0) {
126                size = httpRequest.getContentLength();
127            }
128            byte[] bytes = new byte[size];
129            if (bytes.length > 0) {
130                int read = 0;
131                while (read > -1) {
132                    read = httpRequest.getInputStream().read(bytes);
133                    if (read > 0) {
134                        httpResponse.getOutputStream().write(bytes, 0, read);
135                    }
136                }
137            }
138
139            httpResponse.setStatus(200);
140            httpResponse.getOutputStream().flush();
141            httpResponse.getOutputStream().close();
142        }
143    }

145    @AfterClass(alwaysRun = true)
146    public void tearDownGlobal() throws Exception {
147        server.stop();
148    }
149
150    protected int findFreePort() throws IOException {
151        ServerSocket socket = null;
152
153        try {
154            socket = new ServerSocket(0);
155
156            return socket.getLocalPort();
157        }
158        finally {
159            if (socket != null) {
160                socket.close();
161            }
162        }
163    }
164
165    protected String getTargetUrl() {
166        return String.format("http://127.0.0.1:%d/foo/test", port1);
167    }
168
169    protected String getTargetUrl2() {
170        return String.format("https://127.0.0.1:%d/foo/test", port2);
171    }
172
173    public AbstractHandler configureHandler() throws Exception {
174        return new EchoHandler();
175    }
176
177    @BeforeClass(alwaysRun = true)
178    public void setUpGlobal() throws Exception {
179        server = new Server();
180
181        port1 = findFreePort();
182        port2 = findFreePort();
183
184        Connector listener = new SelectChannelConnector();
185
186        listener.setHost("127.0.0.1");
187        listener.setPort(port1);
188
189        server.addConnector(listener);
190
191        listener = new SelectChannelConnector();
192        listener.setHost("127.0.0.1");
193        listener.setPort(port2);
194
195        server.addConnector(listener);
196
197        server.setHandler(configureHandler());
198        server.start();
199        log.info("Local HTTP server started successfully");
200    }
