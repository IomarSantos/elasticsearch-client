049    private class ZeroCopyHandler extends AbstractHandler {
050        public void handle(String s,
051                           Request r,
052                           HttpServletRequest httpRequest,
053                           HttpServletResponse httpResponse) throws IOException, ServletException {
054
055            int size = 10 * 1024;
056            if (httpRequest.getContentLength() > 0) {
057                size = httpRequest.getContentLength();
058            }
059            byte[] bytes = new byte[size];
060            if (bytes.length > 0) {
061                httpRequest.getInputStream().read(bytes);
062                httpResponse.getOutputStream().write(bytes);
063            }
064
065            httpResponse.setStatus(200);
066            httpResponse.getOutputStream().flush();
067        }
068    }

070    @Test(groups = {"standalone", "default_provider"})
071    public void zeroCopyPostTest() throws IOException, ExecutionException, TimeoutException, InterruptedException, URISyntaxException {
072        AsyncHttpClient client = getAsyncHttpClient(null);
073
074        ClassLoader cl = getClass().getClassLoader();
075        // override system properties
076        URL url = cl.getResource("SimpleTextFile.txt");
077        File file = new File(url.toURI());
078        final AtomicBoolean headerSent = new AtomicBoolean(false);
079        final AtomicBoolean operationCompleted = new AtomicBoolean(false);
080
081        Future<Response> f = client.preparePost("http://127.0.0.1:" + port1 + "/").setBody(file).execute(new AsyncCompletionHandler<Response>() {
082
083            public STATE onHeaderWriteCompleted() {
084                headerSent.set(true);
085                return STATE.CONTINUE;
086            }
087
088            public STATE onContentWriteCompleted() {
089                operationCompleted.set(true);
090                return STATE.CONTINUE;
091            }
092
093            @Override
094            public Response onCompleted(Response response) throws Exception {
095                return response;
096            }
097        });
098        Response resp = f.get();
099        assertNotNull(resp);
100        assertEquals(resp.getStatusCode(), HttpServletResponse.SC_OK);
101        assertEquals(resp.getResponseBody(), "This is a simple test file");
102        assertTrue(operationCompleted.get());
103        assertTrue(headerSent.get());
104        client.close();
105    }

130    @Test(groups = {"standalone", "default_provider"})
131    public void zeroCopyFileTest() throws IOException, ExecutionException, TimeoutException, InterruptedException, URISyntaxException {
132        AsyncHttpClient client = getAsyncHttpClient(null);
133
134        ClassLoader cl = getClass().getClassLoader();
135        // override system properties
136        URL url = cl.getResource("SimpleTextFile.txt");
137        File file = new File(url.toURI());
138
139        File tmp = new File(System.getProperty("java.io.tmpdir") + File.separator + "zeroCopy.txt");
140        tmp.deleteOnExit();
141        final FileOutputStream stream = new FileOutputStream(tmp);
142        Future<Response> f = client.preparePost("http://127.0.0.1:" + port1 + "/").setBody(file).execute(new AsyncHandler<Response>() {
143            public void onThrowable(Throwable t) {
144            }
145
146            public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
147                bodyPart.writeTo(stream);
148                return STATE.CONTINUE;
149            }
150
151            public STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
152                return STATE.CONTINUE;
153            }
154
155            public STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
156                return STATE.CONTINUE;
157            }
158
159            public Response onCompleted() throws Exception {
160                return null;
161            }
162        });
163        Response resp = f.get();
164        stream.close();
165        assertNull(resp);
166        assertEquals(file.length(), tmp.length());
167        client.close();
168
169    }
