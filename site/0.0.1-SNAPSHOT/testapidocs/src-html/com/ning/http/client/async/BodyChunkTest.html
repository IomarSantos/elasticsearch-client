016package com.ning.http.client.async;
017
018import com.ning.http.client.AsyncHttpClient;
019import com.ning.http.client.AsyncHttpClientConfig;
020import com.ning.http.client.RequestBuilder;
021import com.ning.http.client.Response;
022import com.ning.http.client.generators.InputStreamBodyGenerator;
023import org.testng.annotations.Test;
024
025import java.io.ByteArrayInputStream;
026import java.util.concurrent.Future;
027
028import static org.testng.Assert.assertEquals;
029
030public abstract class BodyChunkTest extends AbstractBasicTest {
031
032    private final static String MY_MESSAGE = "my message";
033
034    @Test(groups = {"standalone", "default_provider"})
035    public void negativeContentTypeTest() throws Throwable {
036
037        AsyncHttpClientConfig.Builder confbuilder = new AsyncHttpClientConfig.Builder();
038        confbuilder = confbuilder.setConnectionTimeoutInMs(100);
039        confbuilder = confbuilder.setMaximumConnectionsTotal(50);
040        confbuilder = confbuilder.setRequestTimeoutInMs(5 * 60 * 1000); // 5 minutes
041
042        // Create client
043        AsyncHttpClient client = getAsyncHttpClient(confbuilder.build());
044
045        RequestBuilder requestBuilder = new RequestBuilder("POST")
046                .setUrl(getTargetUrl())
047                .setHeader("Content-Type", "message/rfc822");
048
049        requestBuilder.setBody(new InputStreamBodyGenerator(new ByteArrayInputStream(MY_MESSAGE.getBytes())));
050
051        Future<Response> future = client.executeRequest(requestBuilder.build());
052
053        System.out.println("waiting for response");
054        Response response = future.get();
055        assertEquals(response.getStatusCode(), 200);
056        assertEquals(response.getResponseBody(), MY_MESSAGE);
057        
058        client.close();
059    }
060
061}
