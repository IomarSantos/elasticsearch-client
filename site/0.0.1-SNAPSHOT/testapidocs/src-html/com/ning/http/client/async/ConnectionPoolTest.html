099    @Test(groups = {"standalone", "default_provider", "async"}, enabled = true, invocationCount = 10, alwaysRun = true)
100    public void asyncDoGetKeepAliveHandlerTest_channelClosedDoesNotFail() throws Throwable {
101        AsyncHttpClient client = getAsyncHttpClient(null);
102
103        // Use a l in case the assert fail
104        final CountDownLatch l = new CountDownLatch(2);
105
106        final Map<String, Boolean> remoteAddresses = new
107                ConcurrentHashMap<String, Boolean>();
108
109        AsyncCompletionHandler<Response> handler = new
110                AsyncCompletionHandlerAdapter() {
111
112                    @Override
113                    public Response onCompleted(Response response) throws
114                            Exception {
115                        System.out.println("ON COMPLETED INVOKED " +
116                                response.getHeader("X-KEEP-ALIVE"));
117                        try {
118                            assertEquals(response.getStatusCode(), 200);
119                            remoteAddresses.put(response.getHeader("X-KEEP-ALIVE"), true);
120                        } finally {
121                            l.countDown();
122                        }
123                        return response;
124                    }
125                };
126
127        client.prepareGet(getTargetUrl()).execute(handler).get();
128        server.stop();
129        server.start();
130        client.prepareGet(getTargetUrl()).execute(handler);
131
132
133        if (!l.await(TIMEOUT, TimeUnit.SECONDS)) {
134            Assert.fail("Timed out");
135        }
136
137        assertEquals(remoteAddresses.size(), 2);
138        client.close();
139    }

206    /**
207     * This test just make sure the hack used to catch disconnected channel under win7 doesn't throw any exception.
208     * The onComplete method must be only called once.
209     *
210     * @throws Throwable if something wrong happens.
211     */
212    @Test(groups = {"standalone", "default_provider"})
213    public void win7DisconnectTest() throws Throwable {
214        final AtomicInteger count = new AtomicInteger(0);
215
216        AsyncHttpClient client = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().build());
217        AsyncCompletionHandler<Response> handler = new
218                AsyncCompletionHandlerAdapter() {
219
220                    @Override
221                    public Response onCompleted(Response response) throws
222                            Exception {
223
224                        count.incrementAndGet();
225                        StackTraceElement e = new StackTraceElement("sun.nio.ch.SocketDispatcher", "read0", null, -1);
226                        IOException t = new IOException();
227                        t.setStackTrace(new StackTraceElement[]{e});
228                        throw t;
229                    }
230                };
231
232        try {
233            client.prepareGet(getTargetUrl()).execute(handler).get();
234            fail("Must have received an exception");
235        } catch (ExecutionException ex) {
236            assertNotNull(ex);
237            assertNotNull(ex.getCause());
238            assertEquals(ex.getCause().getCause().getClass(), IOException.class);
239            assertEquals(count.get(), 1);
240        }
241        client.close();
242    }

244    @Test(groups = {"standalone", "default_provider"})
245    public void asyncHandlerOnThrowableTest() throws Throwable {
246        AsyncHttpClient client = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().build());
247        final AtomicInteger count = new AtomicInteger();
248        final String THIS_IS_NOT_FOR_YOU = "This is not for you";
249        final CountDownLatch latch = new CountDownLatch(16);
250        for (int i = 0; i < 16; i++) {
251            client.prepareGet(getTargetUrl()).execute(new AsyncCompletionHandlerBase() {
252                @Override
253                public Response onCompleted(Response response) throws Exception {
254                    throw new Exception(THIS_IS_NOT_FOR_YOU);
255                }
256            });
257
258            client.prepareGet(getTargetUrl()).execute(new AsyncCompletionHandlerBase() {
259                /* @Override */
260                public void onThrowable(Throwable t) {
261                    if (t.getMessage() != null && t.getMessage().equalsIgnoreCase(THIS_IS_NOT_FOR_YOU)) {
262                        count.incrementAndGet();
263                    }
264                }
265
266                @Override
267                public Response onCompleted(Response response) throws Exception {
268                    latch.countDown();
269                    return response;
270                }
271            });
272        }
273        latch.await(TIMEOUT, TimeUnit.SECONDS);
274        assertEquals(count.get(), 0);
275        client.close();
276    }
