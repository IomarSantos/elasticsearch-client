045/**
046 * Test for multithreaded url fetcher calls that use two separate
047 * sets of ssl certificates.  This then tests that the certificate
048 * settings do not clash (override each other), resulting in the
049 * peer not authenticated exception
050 *
051 * @author dominict
052 */
053public abstract class RedirectConnectionUsageTest extends AbstractBasicTest{
054    private String BASE_URL;
055
056    private String servletEndpointRedirectUrl;
057
058    @BeforeClass
059    public void setUp() throws Exception {
060        server = new Server();
061
062        port1 = findFreePort();
063
064        Connector listener = new SelectChannelConnector();
065        listener.setHost("localhost");
066        listener.setPort(port1);
067
068        server.addConnector(listener);
069
070
071        ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
072
073        context.setContextPath("/");
074        server.setHandler(context);
075
076        context.addServlet(new ServletHolder(new MockRedirectHttpServlet()), "/redirect/*");
077        context.addServlet(new ServletHolder(new MockFullResponseHttpServlet()), "/*");
078
079        server.start();
080
081        BASE_URL = "http://localhost" + ":" + port1;
082        servletEndpointRedirectUrl = BASE_URL + "/redirect";
083
084    }
085
086    @AfterClass
087    public void tearDown() {
088        try {
089            if (server != null) {
090                server.stop();
091            }
092
093        } catch (Exception e) {
094            System.err.print("Error stopping servlet tester");
095            e.printStackTrace();
096        }
097    }
098
099    /**
100     * Tests that after a redirect the final url in the response
101     * reflect the redirect
102     */
103    @Test
104    public void testGetRedirectFinalUrl() {
105
106        AsyncHttpClient c = null;
107        try {
108            AsyncHttpClientConfig.Builder bc =
109                    new AsyncHttpClientConfig.Builder();
110
111            bc.setAllowPoolingConnection(true);
112            bc.setMaximumConnectionsPerHost(1);
113            bc.setMaximumConnectionsTotal(1);
114            bc.setConnectionTimeoutInMs(1000);
115            bc.setRequestTimeoutInMs(1000);
116            bc.setFollowRedirects(true);
117
118            c = getAsyncHttpClient(bc.build());
119
120            RequestBuilder builder = new RequestBuilder("GET");
121            builder.setUrl(servletEndpointRedirectUrl);
122
123            com.ning.http.client.Request r = builder.build();
124
125            try {
126                ListenableFuture<Response> response = c.executeRequest(r);
127                Response res = null;
128                res = response.get();
129                assertNotNull(res.getResponseBody());
130                assertEquals(BASE_URL + "/overthere", BASE_URL + "/overthere", res.getUri().toString());
131
132            } catch (Exception e) {
133                System.err.print("============");
134                e.printStackTrace();
135                System.err.print("============");
136                System.err.flush();
137                fail("Should not get here, The request threw an exception");
138            }
139
140
141        }
142        finally {
143            // can hang here
144            if (c != null) c.close();
145        }
146
147
148    }
149
150    protected abstract AsyncHttpProviderConfig getProviderConfig();
151
152    @SuppressWarnings("serial")
153    class MockRedirectHttpServlet extends HttpServlet {
154        public void service(HttpServletRequest req, HttpServletResponse res)
155                throws ServletException, IOException {
156            res.sendRedirect("/overthere");
157        }
158    }
159
160    @SuppressWarnings("serial")
161    class MockFullResponseHttpServlet extends HttpServlet {
162
163        private static final String contentType = "text/xml";
164        private static final String xml = "<?xml version=\"1.0\"?><hello date=\"%s\"></hello>";
165
166        public void service(HttpServletRequest req, HttpServletResponse res) throws ServletException, IOException {
167            String xmlToReturn = String.format(xml, new Object[]{new Date().toString()});
168
169            res.setStatus(200, "Complete, XML Being Returned");
170            res.addHeader("Content-Type", contentType);
171            res.addHeader("X-Method", req.getMethod());
172            res.addHeader("MultiValue", "1");
173            res.addHeader("MultiValue", "2");
174            res.addHeader("MultiValue", "3");
175
176            OutputStream os = res.getOutputStream();
177
178            byte[] retVal = xmlToReturn.getBytes();
179            res.setContentLength(retVal.length);
180            os.write(retVal);
181            os.close();
182        }
183    }
