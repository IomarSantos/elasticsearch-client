013package com.ning.http.client.async;
014
015import com.ning.http.client.AsyncHttpClient;
016import com.ning.http.client.AsyncHttpClientConfig;
017import org.eclipse.jetty.server.Request;
018import org.eclipse.jetty.server.handler.AbstractHandler;
019import org.testng.annotations.Test;
020
021import javax.servlet.ServletException;
022import javax.servlet.http.HttpServletRequest;
023import javax.servlet.http.HttpServletResponse;
024import java.io.IOException;
025import java.io.OutputStream;
026
027import static org.testng.Assert.*;
028
029public abstract class RetryRequestTest extends AbstractBasicTest {
030    public static class SlowAndBigHandler extends AbstractHandler {
031
032        public void handle(String pathInContext, Request request,
033                           HttpServletRequest httpRequest, HttpServletResponse httpResponse)
034                throws IOException, ServletException {
035
036            int load = 100;
037            httpResponse.setStatus(200);
038            httpResponse.setContentLength(load);
039            httpResponse.setContentType("application/octet-stream");
040
041            httpResponse.flushBuffer();
042
043
044            OutputStream os = httpResponse.getOutputStream();
045            for (int i = 0; i < load; i++) {
046                os.write(i % 255);
047
048                try {
049                    Thread.sleep(300);
050                } catch (InterruptedException ex) {
051                    // nuku
052                }
053
054
055                if (i > load / 10) {
056                    httpResponse.sendError(500);
057                }
058            }
059
060            httpResponse.getOutputStream().flush();
061            httpResponse.getOutputStream().close();
062        }
063    }
064
065    protected String getTargetUrl() {
066        return String.format("http://127.0.0.1:%d/", port1);
067    }
068
069    @Override
070    public AbstractHandler configureHandler() throws Exception {
071        return new SlowAndBigHandler();
072    }
073
074
075    @Test(groups = {"standalone", "default_provider"})
076    public void testMaxRetry() throws Throwable {
077        AsyncHttpClient ahc = getAsyncHttpClient(new AsyncHttpClientConfig.Builder().setMaxRequestRetry(0).build());
078        try {
079            ahc.executeRequest(ahc.prepareGet(getTargetUrl()).build()).get();
080            fail();
081        } catch (Exception t) {
082            assertNotNull(t.getCause());
083            assertEquals(t.getCause().getClass(), IOException.class);
084            if (!t.getCause().getMessage().startsWith("Remotely Closed")) {
085                fail();
086            }
087        }
088
089        ahc.close();
090    }
091}
