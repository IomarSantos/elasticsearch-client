001package org.elasticsearch.client.http;
002
003import java.net.ConnectException;
004import org.elasticsearch.action.get.GetResponse;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007import org.elasticsearch.common.util.concurrent.UncategorizedExecutionException;
008import org.testng.annotations.Test;
009
010public class GetTests {
011
012    private final ESLogger logger = ESLoggerFactory.getLogger(getClass().getName());
013
014    @Test
015    public void testGet() throws Exception {
016        try {
017            HttpIngestClient client = new HttpIngestClient();
018            
019            GetResponse response = client.prepareGet("test", "type1", "1").execute().actionGet();
020            
021            logger.info("success, got response, _source = {}", 
022                    response.getSourceAsString());
023            
024            
025            
026        } catch (UncategorizedExecutionException e) {
027            if (!(e.getRootCause() instanceof ConnectException)) {
028                throw e;
029            } else {
030                logger.warn("oops, test skipped, no node found to connect to: " + e.getRootCause().getMessage());
031            }
032        }
033    }
034}
