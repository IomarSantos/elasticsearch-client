001package org.elasticsearch.client.http;
002
003import java.net.ConnectException;
004import org.elasticsearch.action.bulk.BulkItemResponse;
005import org.elasticsearch.action.bulk.BulkResponse;
006import static org.elasticsearch.client.IngestRequests.indexRequest;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.ESLoggerFactory;
009import org.elasticsearch.common.util.concurrent.UncategorizedExecutionException;
010import org.testng.annotations.Test;
011
012public class BulkTests {
013
014    private final ESLogger logger = ESLoggerFactory.getLogger(getClass().getName());
015
016    @Test
017    public void testBulkIndex() throws Exception {
018        try {
019            HttpIngestClient client = new HttpIngestClient();
020            
021            BulkResponse response = client.prepareBulk()
022                    .add(indexRequest().index("test").type("type1").id("2").source("Hello", "World"))
023                    .add(indexRequest().index("test").type("type1").id("3").source("Welcome", "to Elasticsearch"))
024                    .execute().actionGet();
025            
026            for (BulkItemResponse r : response.items()) {
027                if (r.failed()) {
028                    logger.error("error: {}", r.failureMessage());
029                } else {
030                    logger.info("info: {}/{}/{}", r.getIndex(), r.getType(), r.getId());
031                }
032            }
033            
034        } catch (UncategorizedExecutionException e) {
035            if (!(e.getRootCause() instanceof ConnectException)) {
036                throw e;
037            } else {
038                logger.warn("oops, test skipped, no node found to connect to: " + e.getRootCause().getMessage());
039            }
040        }
041    }
042}
