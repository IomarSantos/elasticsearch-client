049    @Test
050    public void compareParsingTokens() throws IOException {
051        FastByteArrayOutputStream xsonOs = new FastByteArrayOutputStream();
052        XContentGenerator xsonGen = XContentFactory.xContent(XContentType.SMILE).createGenerator(xsonOs);
053
054        FastByteArrayOutputStream jsonOs = new FastByteArrayOutputStream();
055        XContentGenerator jsonGen = XContentFactory.xContent(XContentType.JSON).createGenerator(jsonOs);
056
057        xsonGen.writeStartObject();
058        jsonGen.writeStartObject();
059
060        xsonGen.writeStringField("test", "value");
061        jsonGen.writeStringField("test", "value");
062
063        xsonGen.writeArrayFieldStart("arr");
064        jsonGen.writeArrayFieldStart("arr");
065        xsonGen.writeNumber(1);
066        jsonGen.writeNumber(1);
067        xsonGen.writeNull();
068        jsonGen.writeNull();
069        xsonGen.writeEndArray();
070        jsonGen.writeEndArray();
071
072        xsonGen.writeEndObject();
073        jsonGen.writeEndObject();
074
075        xsonGen.close();
076        jsonGen.close();
077
078        verifySameTokens(XContentFactory.xContent(XContentType.JSON).createParser(jsonOs.bytes().toBytes()), XContentFactory.xContent(XContentType.SMILE).createParser(xsonOs.bytes().toBytes()));
079    }
080
081    private void verifySameTokens(XContentParser parser1, XContentParser parser2) throws IOException {
082        while (true) {
083            XContentParser.Token token1 = parser1.nextToken();
084            XContentParser.Token token2 = parser2.nextToken();
085            if (token1 == null) {
086                assertThat(token2, nullValue());
087                return;
088            }
089            assertThat(token1, equalTo(token2));
090            switch (token1) {
091                case FIELD_NAME:
092                    assertThat(parser1.currentName(), equalTo(parser2.currentName()));
093                    break;
094                case VALUE_STRING:
095                    assertThat(parser1.text(), equalTo(parser2.text()));
096                    break;
097                case VALUE_NUMBER:
098                    assertThat(parser1.numberType(), equalTo(parser2.numberType()));
099                    assertThat(parser1.numberValue(), equalTo(parser2.numberValue()));
100                    break;
101            }
102        }
103    }
