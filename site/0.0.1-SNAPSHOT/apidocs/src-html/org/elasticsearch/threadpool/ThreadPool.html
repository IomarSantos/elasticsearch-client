001package org.elasticsearch.threadpool;
002
003import java.util.concurrent.Executor;
004import java.util.concurrent.ScheduledExecutorService;
005import java.util.concurrent.ScheduledFuture;
006import java.util.concurrent.TimeUnit;
007import org.elasticsearch.common.unit.TimeValue;
008
009public interface ThreadPool {
010
011    interface Names {
012        String SAME = "same";
013        String GENERIC = "generic";
014        String GET = "get";
015        String INDEX = "index";
016        String BULK = "bulk";
017        String SEARCH = "search";
018        String PERCOLATE = "percolate";
019        String MANAGEMENT = "management";
020        String FLUSH = "flush";
021        String MERGE = "merge";
022        String CACHE = "cache";
023        String REFRESH = "refresh";
024        String SNAPSHOT = "snapshot";
025    }    
026    
027    ThreadPoolInfo info();
028    
029    ThreadPoolStats stats();
030    
031    Executor generic();
032    
033    long estimatedTimeInMillis();
034    
035    Executor executor(String name);
036    
037    ScheduledExecutorService scheduler();
038    
039    ScheduledFuture<?> scheduleWithFixedDelay(Runnable command, TimeValue interval);
040    
041    ScheduledFuture<?> schedule(TimeValue delay, String name, Runnable command);
042    
043    void shutdown();
044    
045    void shutdownNow();
046    
047    boolean awaitTermination(long timeout, TimeUnit unit) throws InterruptedException;
048}
