020package org.elasticsearch.common.lucene.search;
021
022import org.apache.lucene.index.IndexReader;
023import org.apache.lucene.search.BooleanClause;
024import org.apache.lucene.search.DocIdSet;
025import org.apache.lucene.search.Filter;
026import org.apache.lucene.search.FilterClause;
027import org.apache.lucene.util.FixedBitSet;
028import org.elasticsearch.common.lucene.docset.DocSet;
029import org.elasticsearch.common.lucene.docset.DocSets;
030
031import java.io.IOException;
032import java.util.ArrayList;
033import java.util.List;
034
035/**
036 *
037 */
038// LUCENE MONITOR: added to take into account DocSet that wraps OpenBitSet when optimizing or/and/...
039public class XBooleanFilter extends Filter {
040    ArrayList<Filter> shouldFilters = null;
041    ArrayList<Filter> notFilters = null;
042    ArrayList<Filter> mustFilters = null;
043
044    private DocIdSet getDISI(ArrayList<Filter> filters, int index, IndexReader reader)
045            throws IOException {
046        DocIdSet docIdSet = filters.get(index).getDocIdSet(reader);
047        if (docIdSet == DocIdSet.EMPTY_DOCIDSET || docIdSet == DocSet.EMPTY_DOC_SET) {
048            return null;
049        }
050        return docIdSet;
051    }
052
053    public List<Filter> getShouldFilters() {
054        return this.shouldFilters;
055    }
056
057    public List<Filter> getMustFilters() {
058        return this.mustFilters;
059    }
060
061    public List<Filter> getNotFilters() {
062        return this.notFilters;
063    }
064
065    /**
066     * Returns the a DocIdSetIterator representing the Boolean composition
067     * of the filters that have been added.
068     */
069    @Override
070    public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
071        FixedBitSet res = null;
072
073        if (mustFilters == null && notFilters == null && shouldFilters != null && shouldFilters.size() == 1) {
074            return shouldFilters.get(0).getDocIdSet(reader);
075        }
076
077        if (shouldFilters == null && notFilters == null && mustFilters != null && mustFilters.size() == 1) {
078            return mustFilters.get(0).getDocIdSet(reader);
079        }
080
081        if (shouldFilters != null) {
082            for (int i = 0; i < shouldFilters.size(); i++) {
083                final DocIdSet disi = getDISI(shouldFilters, i, reader);
084                if (disi == null) continue;
085                if (res == null) {
086                    res = new FixedBitSet(reader.maxDoc());
087                }
088                DocSets.or(res, disi);
089            }
090
091            // if no should clauses match, return null (act as min_should_match set to 1)
092            if (res == null && !shouldFilters.isEmpty()) {
093                return null;
094            }
095        }
096
097
098        if (notFilters != null) {
099            for (int i = 0; i < notFilters.size(); i++) {
100                if (res == null) {
101                    res = new FixedBitSet(reader.maxDoc());
102                    res.set(0, reader.maxDoc()); // NOTE: may set bits on deleted docs
103                }
104                final DocIdSet disi = getDISI(notFilters, i, reader);
105                if (disi != null) {
106                    DocSets.andNot(res, disi);
107                }
108            }
109        }
110
111        if (mustFilters != null) {
112            for (int i = 0; i < mustFilters.size(); i++) {
113                final DocIdSet disi = getDISI(mustFilters, i, reader);
114                if (disi == null) {
115                    return null;
116                }
117                if (res == null) {
118                    res = new FixedBitSet(reader.maxDoc());
119                    DocSets.or(res, disi);
120                } else {
121                    DocSets.and(res, disi);
122                }
123            }
124        }
125
126        return res;
127    }
128
129    /**
130     * Adds a new FilterClause to the Boolean Filter container
131     *
132     * @param filterClause A FilterClause object containing a Filter and an Occur parameter
133     */
134
135    public void add(FilterClause filterClause) {
136        if (filterClause.getOccur().equals(BooleanClause.Occur.MUST)) {
137            if (mustFilters == null) {
138                mustFilters = new ArrayList<Filter>();
139            }
140            mustFilters.add(filterClause.getFilter());
141        }
142        if (filterClause.getOccur().equals(BooleanClause.Occur.SHOULD)) {
143            if (shouldFilters == null) {
144                shouldFilters = new ArrayList<Filter>();
145            }
146            shouldFilters.add(filterClause.getFilter());
147        }
148        if (filterClause.getOccur().equals(BooleanClause.Occur.MUST_NOT)) {
149            if (notFilters == null) {
150                notFilters = new ArrayList<Filter>();
151            }
152            notFilters.add(filterClause.getFilter());
153        }
154    }
155
156    public void addMust(Filter filter) {
157        if (mustFilters == null) {
158            mustFilters = new ArrayList<Filter>();
159        }
160        mustFilters.add(filter);
161    }
162
163    public void addShould(Filter filter) {
164        if (shouldFilters == null) {
165            shouldFilters = new ArrayList<Filter>();
166        }
167        shouldFilters.add(filter);
168    }
169
170    public void addNot(Filter filter) {
171        if (notFilters == null) {
172            notFilters = new ArrayList<Filter>();
173        }
174        notFilters.add(filter);
175    }
176
177
178    private boolean equalFilters(ArrayList<Filter> filters1, ArrayList<Filter> filters2) {
179        return (filters1 == filters2) ||
180                ((filters1 != null) && filters1.equals(filters2));
181    }
182
183    @Override
184    public boolean equals(Object obj) {
185        if (this == obj)
186            return true;
187
188        if ((obj == null) || (obj.getClass() != this.getClass()))
189            return false;
190
191        XBooleanFilter other = (XBooleanFilter) obj;
192        return equalFilters(notFilters, other.notFilters)
193                && equalFilters(mustFilters, other.mustFilters)
194                && equalFilters(shouldFilters, other.shouldFilters);
195    }
196
197    @Override
198    public int hashCode() {
199        int hash = 7;
200        hash = 31 * hash + (null == mustFilters ? 0 : mustFilters.hashCode());
201        hash = 31 * hash + (null == notFilters ? 0 : notFilters.hashCode());
202        hash = 31 * hash + (null == shouldFilters ? 0 : shouldFilters.hashCode());
203        return hash;
204    }
205
206    /**
207     * Prints a user-readable version of this query.
208     */
209    @Override
210    public String toString() {
211        StringBuilder buffer = new StringBuilder();
212        buffer.append("BooleanFilter(");
213        appendFilters(shouldFilters, "", buffer);
214        appendFilters(mustFilters, "+", buffer);
215        appendFilters(notFilters, "-", buffer);
216        buffer.append(")");
217        return buffer.toString();
218    }
219
220    private void appendFilters(ArrayList<Filter> filters, String occurString, StringBuilder buffer) {
221        if (filters != null) {
222            for (int i = 0; i < filters.size(); i++) {
223                buffer.append(' ');
224                buffer.append(occurString);
225                buffer.append(filters.get(i).toString());
226            }
227        }
228    }
