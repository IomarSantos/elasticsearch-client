020package org.elasticsearch.common.lucene.search;
021
022import org.apache.lucene.analysis.Analyzer;
023import org.apache.lucene.index.IndexReader;
024import org.apache.lucene.search.*;
025import org.apache.lucene.search.similar.MoreLikeThis;
026import org.elasticsearch.common.io.FastStringReader;
027
028import java.io.IOException;
029import java.util.Set;
030
031/**
032 *
033 */
034public class MoreLikeThisQuery extends Query {
035
036    public static final float DEFAULT_PERCENT_TERMS_TO_MATCH = 0.3f;
037
038    private Similarity similarity;
039
040    private String likeText;
041    private String[] moreLikeFields;
042    private Analyzer analyzer;
043    private float percentTermsToMatch = DEFAULT_PERCENT_TERMS_TO_MATCH;
044    private int minTermFrequency = MoreLikeThis.DEFAULT_MIN_TERM_FREQ;
045    private int maxQueryTerms = MoreLikeThis.DEFAULT_MAX_QUERY_TERMS;
046    private Set<?> stopWords = MoreLikeThis.DEFAULT_STOP_WORDS;
047    private int minDocFreq = MoreLikeThis.DEFAULT_MIN_DOC_FREQ;
048    private int maxDocFreq = MoreLikeThis.DEFAULT_MAX_DOC_FREQ;
049    private int minWordLen = MoreLikeThis.DEFAULT_MIN_WORD_LENGTH;
050    private int maxWordLen = MoreLikeThis.DEFAULT_MAX_WORD_LENGTH;
051    private boolean boostTerms = MoreLikeThis.DEFAULT_BOOST;
052    private float boostTermsFactor = 1;
053
054
055    public MoreLikeThisQuery() {
056
057    }
058
059    public MoreLikeThisQuery(String likeText, String[] moreLikeFields, Analyzer analyzer) {
060        this.likeText = likeText;
061        this.moreLikeFields = moreLikeFields;
062        this.analyzer = analyzer;
063    }
064
065    @Override
066    public Query rewrite(IndexReader reader) throws IOException {
067        MoreLikeThis mlt = new MoreLikeThis(reader, similarity == null ? new DefaultSimilarity() : similarity);
068
069        mlt.setFieldNames(moreLikeFields);
070        mlt.setAnalyzer(analyzer);
071        mlt.setMinTermFreq(minTermFrequency);
072        mlt.setMinDocFreq(minDocFreq);
073        mlt.setMaxDocFreq(maxDocFreq);
074        mlt.setMaxQueryTerms(maxQueryTerms);
075        mlt.setMinWordLen(minWordLen);
076        mlt.setMaxWordLen(maxWordLen);
077        mlt.setStopWords(stopWords);
078        mlt.setBoost(boostTerms);
079        mlt.setBoostFactor(boostTermsFactor);
080        BooleanQuery bq = (BooleanQuery) mlt.like(new FastStringReader(likeText));
081        BooleanClause[] clauses = bq.getClauses();
082
083        bq.setMinimumNumberShouldMatch((int) (clauses.length * percentTermsToMatch));
084
085        bq.setBoost(getBoost());
086        return bq;
087    }
088
089    @Override
090    public String toString(String field) {
091        return "like:" + likeText;
092    }
093
094    public String getLikeText() {
095        return likeText;
096    }
097
098    public void setLikeText(String likeText) {
099        this.likeText = likeText;
100    }
101
102    public String[] getMoreLikeFields() {
103        return moreLikeFields;
104    }
105
106    public void setMoreLikeFields(String[] moreLikeFields) {
107        this.moreLikeFields = moreLikeFields;
108    }
109
110    public Similarity getSimilarity() {
111        return similarity;
112    }
113
114    public void setSimilarity(Similarity similarity) {
115        this.similarity = similarity;
116    }
117
118    public Analyzer getAnalyzer() {
119        return analyzer;
120    }
121
122    public void setAnalyzer(Analyzer analyzer) {
123        this.analyzer = analyzer;
124    }
125
126    public float getPercentTermsToMatch() {
127        return percentTermsToMatch;
128    }
129
130    public void setPercentTermsToMatch(float percentTermsToMatch) {
131        this.percentTermsToMatch = percentTermsToMatch;
132    }
133
134    public int getMinTermFrequency() {
135        return minTermFrequency;
136    }
137
138    public void setMinTermFrequency(int minTermFrequency) {
139        this.minTermFrequency = minTermFrequency;
140    }
141
142    public int getMaxQueryTerms() {
143        return maxQueryTerms;
144    }
145
146    public void setMaxQueryTerms(int maxQueryTerms) {
147        this.maxQueryTerms = maxQueryTerms;
148    }
149
150    public Set<?> getStopWords() {
151        return stopWords;
152    }
153
154    public void setStopWords(Set<?> stopWords) {
155        this.stopWords = stopWords;
156    }
157
158    public int getMinDocFreq() {
159        return minDocFreq;
160    }
161
162    public void setMinDocFreq(int minDocFreq) {
163        this.minDocFreq = minDocFreq;
164    }
165
166    public int getMaxDocFreq() {
167        return maxDocFreq;
168    }
169
170    public void setMaxDocFreq(int maxDocFreq) {
171        this.maxDocFreq = maxDocFreq;
172    }
173
174    public int getMinWordLen() {
175        return minWordLen;
176    }
177
178    public void setMinWordLen(int minWordLen) {
179        this.minWordLen = minWordLen;
180    }
181
182    public int getMaxWordLen() {
183        return maxWordLen;
184    }
185
186    public void setMaxWordLen(int maxWordLen) {
187        this.maxWordLen = maxWordLen;
188    }
189
190    public boolean isBoostTerms() {
191        return boostTerms;
192    }
193
194    public void setBoostTerms(boolean boostTerms) {
195        this.boostTerms = boostTerms;
196    }
197
198    public float getBoostTermsFactor() {
199        return boostTermsFactor;
200    }
201
202    public void setBoostTermsFactor(float boostTermsFactor) {
203        this.boostTermsFactor = boostTermsFactor;
204    }
205}
