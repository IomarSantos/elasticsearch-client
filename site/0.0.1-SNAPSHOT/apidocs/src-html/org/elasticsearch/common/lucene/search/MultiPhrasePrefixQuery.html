020package org.elasticsearch.common.lucene.search;
021
022import org.apache.lucene.index.IndexReader;
023import org.apache.lucene.index.Term;
024import org.apache.lucene.index.TermEnum;
025import org.apache.lucene.search.MultiPhraseQuery;
026import org.apache.lucene.search.Query;
027import org.apache.lucene.util.ToStringUtils;
028
029import java.io.IOException;
030import java.util.*;
031
032public class MultiPhrasePrefixQuery extends Query {
033
034    private String field;
035    private ArrayList<Term[]> termArrays = new ArrayList<Term[]>();
036    private ArrayList<Integer> positions = new ArrayList<Integer>();
037    private int maxExpansions = Integer.MAX_VALUE;
038
039    private int slop = 0;
040
041    /**
042     * Sets the phrase slop for this query.
043     *
044     * @see org.apache.lucene.search.PhraseQuery#setSlop(int)
045     */
046    public void setSlop(int s) {
047        slop = s;
048    }
049
050    public void setMaxExpansions(int maxExpansions) {
051        this.maxExpansions = maxExpansions;
052    }
053
054    /**
055     * Sets the phrase slop for this query.
056     *
057     * @see org.apache.lucene.search.PhraseQuery#getSlop()
058     */
059    public int getSlop() {
060        return slop;
061    }
062
063    /**
064     * Add a single term at the next position in the phrase.
065     *
066     * @see org.apache.lucene.search.PhraseQuery#add(Term)
067     */
068    public void add(Term term) {
069        add(new Term[]{term});
070    }
071
072    /**
073     * Add multiple terms at the next position in the phrase.  Any of the terms
074     * may match.
075     *
076     * @see org.apache.lucene.search.PhraseQuery#add(Term)
077     */
078    public void add(Term[] terms) {
079        int position = 0;
080        if (positions.size() > 0)
081            position = positions.get(positions.size() - 1).intValue() + 1;
082
083        add(terms, position);
084    }
085
086    /**
087     * Allows to specify the relative position of terms within the phrase.
088     *
089     * @param terms
090     * @param position
091     * @see org.apache.lucene.search.PhraseQuery#add(Term, int)
092     */
093    public void add(Term[] terms, int position) {
094        if (termArrays.size() == 0)
095            field = terms[0].field();
096
097        for (int i = 0; i < terms.length; i++) {
098            if (terms[i].field() != field) {
099                throw new IllegalArgumentException(
100                        "All phrase terms must be in the same field (" + field + "): "
101                                + terms[i]);
102            }
103        }
104
105        termArrays.add(terms);
106        positions.add(Integer.valueOf(position));
107    }
108
109    /**
110     * Returns a List of the terms in the multiphrase.
111     * Do not modify the List or its contents.
112     */
113    public List<Term[]> getTermArrays() {
114        return Collections.unmodifiableList(termArrays);
115    }
116
117    /**
118     * Returns the relative positions of terms in this phrase.
119     */
120    public int[] getPositions() {
121        int[] result = new int[positions.size()];
122        for (int i = 0; i < positions.size(); i++)
123            result[i] = positions.get(i).intValue();
124        return result;
125    }
126
127    @Override
128    public Query rewrite(IndexReader reader) throws IOException {
129        if (termArrays.isEmpty()) {
130            return MatchNoDocsQuery.INSTANCE;
131        }
132        MultiPhraseQuery query = new MultiPhraseQuery();
133        query.setSlop(slop);
134        int sizeMinus1 = termArrays.size() - 1;
135        for (int i = 0; i < sizeMinus1; i++) {
136            query.add(termArrays.get(i), positions.get(i));
137        }
138        Term[] suffixTerms = termArrays.get(sizeMinus1);
139        int position = positions.get(sizeMinus1);
140        List<Term> terms = new ArrayList<Term>();
141        for (Term term : suffixTerms) {
142            getPrefixTerms(terms, term, reader);
143            if (terms.size() > maxExpansions) {
144                break;
145            }
146        }
147        if (terms.isEmpty()) {
148            return MatchNoDocsQuery.INSTANCE;
149        }
150        query.add(terms.toArray(new Term[terms.size()]), position);
151        return query.rewrite(reader);
152    }
153
154    private void getPrefixTerms(List<Term> terms, final Term prefix, final IndexReader reader) throws IOException {
155        TermEnum enumerator = reader.terms(prefix);
156        try {
157            do {
158                Term term = enumerator.term();
159                if (term != null
160                        && term.text().startsWith(prefix.text())
161                        && term.field().equals(field)) {
162                    terms.add(term);
163                } else {
164                    break;
165                }
166                if (terms.size() >= maxExpansions) {
167                    break;
168                }
169            } while (enumerator.next());
170        } finally {
171            enumerator.close();
172        }
173    }
174
175    @Override
176    public final String toString(String f) {
177        StringBuilder buffer = new StringBuilder();
178        if (field == null || !field.equals(f)) {
179            buffer.append(field);
180            buffer.append(":");
181        }
182
183        buffer.append("\"");
184        Iterator<Term[]> i = termArrays.iterator();
185        while (i.hasNext()) {
186            Term[] terms = i.next();
187            if (terms.length > 1) {
188                buffer.append("(");
189                for (int j = 0; j < terms.length; j++) {
190                    buffer.append(terms[j].text());
191                    if (j < terms.length - 1)
192                        buffer.append(" ");
193                }
194                buffer.append(")");
195            } else {
196                buffer.append(terms[0].text());
197            }
198            if (i.hasNext())
199                buffer.append(" ");
200        }
201        buffer.append("\"");
202
203        if (slop != 0) {
204            buffer.append("~");
205            buffer.append(slop);
206        }
207
208        buffer.append(ToStringUtils.boost(getBoost()));
209
210        return buffer.toString();
211    }
212
213    /**
214     * Returns true if <code>o</code> is equal to this.
215     */
216    @Override
217    public boolean equals(Object o) {
218        if (!(o instanceof MultiPhrasePrefixQuery)) return false;
219        MultiPhrasePrefixQuery other = (MultiPhrasePrefixQuery) o;
220        return this.getBoost() == other.getBoost()
221                && this.slop == other.slop
222                && termArraysEquals(this.termArrays, other.termArrays)
223                && this.positions.equals(other.positions);
224    }
225
226    /**
227     * Returns a hash code value for this object.
228     */
229    @Override
230    public int hashCode() {
231        return Float.floatToIntBits(getBoost())
232                ^ slop
233                ^ termArraysHashCode()
234                ^ positions.hashCode()
235                ^ 0x4AC65113;
236    }
237
238    // Breakout calculation of the termArrays hashcode
239    private int termArraysHashCode() {
240        int hashCode = 1;
241        for (final Term[] termArray : termArrays) {
242            hashCode = 31 * hashCode
243                    + (termArray == null ? 0 : Arrays.hashCode(termArray));
244        }
245        return hashCode;
246    }
247
248    // Breakout calculation of the termArrays equals
249    private boolean termArraysEquals(List<Term[]> termArrays1, List<Term[]> termArrays2) {
250        if (termArrays1.size() != termArrays2.size()) {
251            return false;
252        }
253        ListIterator<Term[]> iterator1 = termArrays1.listIterator();
254        ListIterator<Term[]> iterator2 = termArrays2.listIterator();
255        while (iterator1.hasNext()) {
256            Term[] termArray1 = iterator1.next();
257            Term[] termArray2 = iterator2.next();
258            if (!(termArray1 == null ? termArray2 == null : Arrays.equals(termArray1,
259                    termArray2))) {
260                return false;
261            }
262        }
263        return true;
264    }
