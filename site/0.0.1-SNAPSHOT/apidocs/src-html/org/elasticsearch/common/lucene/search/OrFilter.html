020package org.elasticsearch.common.lucene.search;
021
022import com.google.common.collect.Lists;
023import org.apache.lucene.index.IndexReader;
024import org.apache.lucene.search.DocIdSet;
025import org.apache.lucene.search.Filter;
026import org.elasticsearch.common.lucene.docset.DocSet;
027import org.elasticsearch.common.lucene.docset.OrDocIdSet;
028import org.elasticsearch.common.lucene.docset.OrDocSet;
029
030import java.io.IOException;
031import java.util.List;
032
033/**
034 *
035 */
036public class OrFilter extends Filter {
037
038    private final List<? extends Filter> filters;
039
040    public OrFilter(List<? extends Filter> filters) {
041        this.filters = filters;
042    }
043
044    public List<? extends Filter> filters() {
045        return filters;
046    }
047
048    @Override
049    public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
050        if (filters.size() == 1) {
051            return filters.get(0).getDocIdSet(reader);
052        }
053        List sets = Lists.newArrayListWithExpectedSize(filters.size());
054        boolean allAreDocSet = true;
055        for (Filter filter : filters) {
056            DocIdSet set = filter.getDocIdSet(reader);
057            if (set == null) { // none matching for this filter, continue
058                continue;
059            }
060            if (!(set instanceof DocSet)) {
061                allAreDocSet = false;
062            }
063            sets.add(set);
064        }
065        if (sets.size() == 0) {
066            return DocSet.EMPTY_DOC_SET;
067        }
068        if (sets.size() == 1) {
069            return (DocIdSet) sets.get(0);
070        }
071        if (allAreDocSet) {
072            return new OrDocSet(sets);
073        }
074        return new OrDocIdSet(sets);
075    }
076
077    @Override
078    public int hashCode() {
079        int hash = 7;
080        hash = 31 * hash + (null == filters ? 0 : filters.hashCode());
081        return hash;
082    }
083
084    @Override
085    public boolean equals(Object obj) {
086        if (this == obj)
087            return true;
088
089        if ((obj == null) || (obj.getClass() != this.getClass()))
090            return false;
091
092        OrFilter other = (OrFilter) obj;
093        return equalFilters(filters, other.filters);
094    }
095
096    @Override
097    public String toString() {
098        StringBuilder builder = new StringBuilder();
099        for(Filter filter: filters) {
100            if(builder.length() > 0) {
101                builder.append(' ');
102            }
103            builder.append(filter);
104        }
105        return builder.toString();
106    }
107
108    private boolean equalFilters(List<? extends Filter> filters1, List<? extends Filter> filters2) {
109        return (filters1 == filters2) || ((filters1 != null) && filters1.equals(filters2));
110    }
111}
