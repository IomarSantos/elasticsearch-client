020package org.elasticsearch.common.lucene.docset;
021
022import org.apache.lucene.index.IndexReader;
023import org.apache.lucene.search.DocIdSet;
024import org.apache.lucene.search.DocIdSetIterator;
025import org.apache.lucene.util.FixedBitSet;
026import org.apache.lucene.util.OpenBitSet;
027import org.elasticsearch.common.Nullable;
028
029import java.io.IOException;
030
031/**
032 */
033public class DocSets {
034
035    public static FixedBitSet createFixedBitSet(DocIdSetIterator disi, int numBits) throws IOException {
036        FixedBitSet set = new FixedBitSet(numBits);
037        int doc;
038        while ((doc = disi.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
039            set.set(doc);
040        }
041        return set;
042    }
043
044    public static void or(FixedBitSet into, DocIdSet other) throws IOException {
045        if (other == null) {
046            return;
047        }
048        if (other instanceof FixedBitSet) {
049            into.or((FixedBitSet) other);
050        } else if (other instanceof FixedBitDocSet) {
051            into.or(((FixedBitDocSet) other).set());
052        } else {
053            DocIdSetIterator disi = other.iterator();
054            if (disi != null) {
055                into.or(disi);
056            }
057        }
058    }
059
060    public static void and(FixedBitSet into, DocIdSet other) throws IOException {
061        if (other instanceof FixedBitDocSet) {
062            other = ((FixedBitDocSet) other).set();
063        }
064        if (other instanceof FixedBitSet) {
065            into.and((FixedBitSet) other);
066        } else {
067            if (other == null) {
068                into.clear(0, into.length());
069            } else {
070                // copied from OpenBitSetDISI#inPlaceAnd
071                DocIdSetIterator disi = other.iterator();
072                if (disi == null) {
073                    into.clear(0, into.length());
074                } else {
075                    into.and(disi);
076                }
077            }
078        }
079    }
080
081    public static void andNot(FixedBitSet into, DocIdSet other) throws IOException {
082        if (other == null) {
083            return;
084        }
085        if (other instanceof FixedBitDocSet) {
086            other = ((FixedBitDocSet) other).set();
087        }
088        if (other instanceof FixedBitSet) {
089            into.andNot((FixedBitSet) other);
090        } else {
091            // copied from OpenBitSetDISI#inPlaceNot
092            DocIdSetIterator disi = other.iterator();
093            if (disi != null) {
094                into.andNot(disi);
095            }
096        }
097    }
098
099    public static DocSet convert(IndexReader reader, DocIdSet docIdSet) throws IOException {
100        if (docIdSet == null) {
101            return DocSet.EMPTY_DOC_SET;
102        } else if (docIdSet instanceof DocSet) {
103            return (DocSet) docIdSet;
104        } else if (docIdSet instanceof FixedBitSet) {
105            return new FixedBitDocSet((FixedBitSet) docIdSet);
106        } else if (docIdSet instanceof OpenBitSet) {
107            return new OpenBitDocSet((OpenBitSet) docIdSet);
108        } else {
109            final DocIdSetIterator it = docIdSet.iterator();
110            // null is allowed to be returned by iterator(),
111            // in this case we wrap with the empty set,
112            // which is cacheable.
113            return (it == null) ? DocSet.EMPTY_DOC_SET : new FixedBitDocSet(createFixedBitSet(it, reader.maxDoc()));
114        }
115    }
116
117    /**
118     * Returns a cacheable version of the doc id set (might be the same instance provided as a parameter).
119     */
120    public static DocSet cacheable(IndexReader reader, @Nullable DocIdSet set) throws IOException {
121        if (set == null) {
122            return DocSet.EMPTY_DOC_SET;
123        }
124        if (set == DocIdSet.EMPTY_DOCIDSET) {
125            return DocSet.EMPTY_DOC_SET;
126        }
127
128        DocIdSetIterator it = set.iterator();
129        if (it == null) {
130            return DocSet.EMPTY_DOC_SET;
131        }
132        int doc = it.nextDoc();
133        if (doc == DocIdSetIterator.NO_MORE_DOCS) {
134            return DocSet.EMPTY_DOC_SET;
135        }
136
137        if (set.isCacheable() && (set instanceof DocSet)) {
138            return (DocSet) set;
139        }
140        if (set instanceof FixedBitSet) {
141            return new FixedBitDocSet((FixedBitSet) set);
142        }
143        if (set instanceof OpenBitSet) {
144            return new OpenBitDocSet((OpenBitSet) set);
145        }
146
147        // work with the iterator...
148        FixedBitSet fixedBitSet = new FixedBitSet(reader.maxDoc());
149        fixedBitSet.set(doc);
150        while ((doc = it.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
151            fixedBitSet.set(doc);
152        }
153        return new FixedBitDocSet(fixedBitSet);
154    }

157
158    }
