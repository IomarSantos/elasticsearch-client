022import org.apache.lucene.search.DocIdSet;
023import org.apache.lucene.search.DocIdSetIterator;
024
025import java.io.IOException;
026import java.util.List;

028/**
029 *
030 */
031public class OrDocIdSet extends DocIdSet {
032
033    private final List<DocIdSet> sets;
034
035    public OrDocIdSet(List<DocIdSet> sets) {
036        this.sets = sets;
037    }
038
039    @Override
040    public boolean isCacheable() {
041        // not cacheable, the reason is that by default, when constructing the filter, it is not cacheable,
042        // so if someone wants it to be cacheable, we might as well construct a cached version of the result
043        return false;

050    }
051
052    @Override
053    public DocIdSetIterator iterator() throws IOException {
054        return new OrDocIdSetIterator();
055    }
056
057    public class OrDocIdSetIterator extends DocIdSetIterator {
058
059        private final class Item {
060            public final DocIdSetIterator iter;
061            public int doc;
062
063            public Item(DocIdSetIterator iter) {
064                this.iter = iter;
065                this.doc = -1;
066            }
067        }
068
069        private int _curDoc;
070        private final Item[] _heap;
071        private int _size;
072
073        OrDocIdSetIterator() throws IOException {
074            _curDoc = -1;
075            _heap = new Item[sets.size()];
076            _size = 0;
077            for (DocIdSet set : sets) {
078                DocIdSetIterator iterator = set.iterator();
079                if (iterator != null) {
080                    _heap[_size++] = new Item(iterator);
081                }
082            }
083            if (_size == 0) _curDoc = DocIdSetIterator.NO_MORE_DOCS;
084        }
085
086        @Override
087        public final int docID() {
088            return _curDoc;
089        }
090
091        @Override
092        public final int nextDoc() throws IOException {
093            if (_curDoc == DocIdSetIterator.NO_MORE_DOCS) return DocIdSetIterator.NO_MORE_DOCS;
094
095            Item top = _heap[0];
096            while (true) {
097                DocIdSetIterator topIter = top.iter;
098                int docid;
099                if ((docid = topIter.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
100                    top.doc = docid;
101                    heapAdjust();
102                } else {
103                    heapRemoveRoot();
104                    if (_size == 0) return (_curDoc = DocIdSetIterator.NO_MORE_DOCS);
105                }
106                top = _heap[0];
107                int topDoc = top.doc;
108                if (topDoc > _curDoc) {
109                    return (_curDoc = topDoc);
110                }
111            }
112        }
113
114        @Override
115        public final int advance(int target) throws IOException {
116            if (_curDoc == DocIdSetIterator.NO_MORE_DOCS) return DocIdSetIterator.NO_MORE_DOCS;
117
118            if (target <= _curDoc) target = _curDoc + 1;
119
120            Item top = _heap[0];
121            while (true) {
122                DocIdSetIterator topIter = top.iter;
123                int docid;
124                if ((docid = topIter.advance(target)) != DocIdSetIterator.NO_MORE_DOCS) {
125                    top.doc = docid;
126                    heapAdjust();
127                } else {
128                    heapRemoveRoot();
129                    if (_size == 0) return (_curDoc = DocIdSetIterator.NO_MORE_DOCS);
130                }
131                top = _heap[0];
132                int topDoc = top.doc;
133                if (topDoc >= target) {
134                    return (_curDoc = topDoc);
135                }
136            }
137        }
138
139// Organize subScorers into a min heap with scorers generating the earlest document on top.
140        /*
141        private final void heapify() {
142            int size = _size;
143            for (int i=(size>>1)-1; i>=0; i--)
144                heapAdjust(i);
145        }
146        */
147        /* The subtree of subScorers at root is a min heap except possibly for its root element.
148        * Bubble the root down as required to make the subtree a heap.
149        */
150
151        private final void heapAdjust() {
152            final Item[] heap = _heap;
153            final Item top = heap[0];
154            final int doc = top.doc;
155            final int size = _size;
156            int i = 0;
157
158            while (true) {
159                int lchild = (i << 1) + 1;
160                if (lchild >= size) break;
161
162                Item left = heap[lchild];
163                int ldoc = left.doc;
164
165                int rchild = lchild + 1;
166                if (rchild < size) {
167                    Item right = heap[rchild];
168                    int rdoc = right.doc;
169
170                    if (rdoc <= ldoc) {
171                        if (doc <= rdoc) break;
172
173                        heap[i] = right;
174                        i = rchild;
175                        continue;
176                    }
177                }
178
179                if (doc <= ldoc) break;
180
181                heap[i] = left;
182                i = lchild;
183            }
184            heap[i] = top;
185        }
186
187        // Remove the root Scorer from subScorers and re-establish it as a heap
188
189        private void heapRemoveRoot() {
190            _size--;
191            if (_size > 0) {
192                Item tmp = _heap[0];
193                _heap[0] = _heap[_size];
194                _heap[_size] = tmp; // keep the finished iterator at the end for debugging
195                heapAdjust();
196            }
197        }
198
199    }
200}
