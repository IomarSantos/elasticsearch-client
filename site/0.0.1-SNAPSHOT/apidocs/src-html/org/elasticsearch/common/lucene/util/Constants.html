068  
069  /** True iff running on a 64bit JVM */
070  public static final boolean JRE_IS_64BIT;
071  
072  static {
073    boolean is64Bit = false;
074    try {
075      final Class<?> unsafeClass = Class.forName("sun.misc.Unsafe");
076      final Field unsafeField = unsafeClass.getDeclaredField("theUnsafe");
077      unsafeField.setAccessible(true);
078      final Object unsafe = unsafeField.get(null);
079      final int addressSize = ((Number) unsafeClass.getMethod("addressSize")
080        .invoke(unsafe)).intValue();
081      //System.out.println("Address size: " + addressSize);
082      is64Bit = addressSize >= 8;
083    } catch (Exception e) {
084      final String x = System.getProperty("sun.arch.data.model");
085      if (x != null) {
086        is64Bit = x.indexOf("64") != -1;
087      } else {
088        if (OS_ARCH != null && OS_ARCH.indexOf("64") != -1) {
089          is64Bit = true;
090        } else {
091          is64Bit = false;
092        }
093      }
094    }
095    JRE_IS_64BIT = is64Bit;
096
097    // this method only exists in Java 6:
098    boolean v6 = true;
099    try {
100      String.class.getMethod("isEmpty");
101    } catch (NoSuchMethodException nsme) {
102      v6 = false;
103    }
104    JRE_IS_MINIMUM_JAVA6 = v6;
105    
106    // this method only exists in Java 7:
107    boolean v7 = true;
108    try {
109      Throwable.class.getMethod("getSuppressed");
110    } catch (NoSuchMethodException nsme) {
111      v7 = false;
112    }
113    JRE_IS_MINIMUM_JAVA7 = v7;
114  }
115
116  // this method prevents inlining the final version constant in compiled classes,
117  // see: http://www.javaworld.com/community/node/3400
118  private static String ident(final String s) {
119    return s.toString();
120  }
