153  /** Encode characters from a char[] source, starting at
154   *  offset for length chars.  Returns a hash of the resulting bytes.  After encoding, result.offset will always be 0. */
155  // TODO: broken if incoming result.offset != 0
156  public static int UTF16toUTF8WithHash(final char[] source, final int offset, final int length, BytesRef result) {
157    int hash = 0;
158    int upto = 0;
159    int i = offset;
160    final int end = offset + length;
161    byte[] out = result.bytes;
162    // Pre-allocate for worst case 4-for-1
163    final int maxLen = length * 4;
164    if (out.length < maxLen)
165      out = result.bytes = new byte[ArrayUtil.oversize(maxLen, 1)];
166    result.offset = 0;
167
168    while(i < end) {
169      
170      final int code = (int) source[i++];
171
172      if (code < 0x80) {
173        hash = 31*hash + (out[upto++] = (byte) code);
174      } else if (code < 0x800) {
175        hash = 31*hash + (out[upto++] = (byte) (0xC0 | (code >> 6)));
176        hash = 31*hash + (out[upto++] = (byte)(0x80 | (code & 0x3F)));
177      } else if (code < 0xD800 || code > 0xDFFF) {
178        hash = 31*hash + (out[upto++] = (byte)(0xE0 | (code >> 12)));
179        hash = 31*hash + (out[upto++] = (byte)(0x80 | ((code >> 6) & 0x3F)));
180        hash = 31*hash + (out[upto++] = (byte)(0x80 | (code & 0x3F)));
181      } else {
182        // surrogate pair
183        // confirm valid high surrogate
184        if (code < 0xDC00 && i < end) {
185          int utf32 = (int) source[i];
186          // confirm valid low surrogate and write pair
187          if (utf32 >= 0xDC00 && utf32 <= 0xDFFF) { 
188            utf32 = (code << 10) + utf32 + SURROGATE_OFFSET;
189            i++;
190            hash = 31*hash + (out[upto++] = (byte)(0xF0 | (utf32 >> 18)));
191            hash = 31*hash + (out[upto++] = (byte)(0x80 | ((utf32 >> 12) & 0x3F)));
192            hash = 31*hash + (out[upto++] = (byte)(0x80 | ((utf32 >> 6) & 0x3F)));
193            hash = 31*hash + (out[upto++] = (byte)(0x80 | (utf32 & 0x3F)));
194            continue;
195          }
196        }
197        // replace unpaired surrogate or out-of-order low surrogate
198        // with substitution character
199        hash = 31*hash + (out[upto++] = (byte) 0xEF);
200        hash = 31*hash + (out[upto++] = (byte) 0xBF);
201        hash = 31*hash + (out[upto++] = (byte) 0xBD);
202      }
203    }
204    //assert matches(source, offset, length, out, upto);
205    result.length = upto;
206    return hash;
207  }

209  /** Encode characters from a char[] source, starting at
210   *  offset and stopping when the character 0xffff is seen.
211   *  Returns the number of bytes written to bytesOut. */
212  public static void UTF16toUTF8(final char[] source, final int offset, UTF8Result result) {
213
214    int upto = 0;
215    int i = offset;
216    byte[] out = result.result;
217
218    while(true) {
219      
220      final int code = (int) source[i++];
221
222      if (upto+4 > out.length) {
223        out = result.result = ArrayUtil.grow(out, upto+4);
224      }
225      if (code < 0x80)
226        out[upto++] = (byte) code;
227      else if (code < 0x800) {
228        out[upto++] = (byte) (0xC0 | (code >> 6));
229        out[upto++] = (byte)(0x80 | (code & 0x3F));
230      } else if (code < 0xD800 || code > 0xDFFF) {
231        if (code == 0xffff)
232          // END
233          break;
234        out[upto++] = (byte)(0xE0 | (code >> 12));
235        out[upto++] = (byte)(0x80 | ((code >> 6) & 0x3F));
236        out[upto++] = (byte)(0x80 | (code & 0x3F));
237      } else {
238        // surrogate pair
239        // confirm valid high surrogate
240        if (code < 0xDC00 && source[i] != 0xffff) {
241          int utf32 = (int) source[i];
242          // confirm valid low surrogate and write pair
243          if (utf32 >= 0xDC00 && utf32 <= 0xDFFF) { 
244            utf32 = ((code - 0xD7C0) << 10) + (utf32 & 0x3FF);
245            i++;
246            out[upto++] = (byte)(0xF0 | (utf32 >> 18));
247            out[upto++] = (byte)(0x80 | ((utf32 >> 12) & 0x3F));
248            out[upto++] = (byte)(0x80 | ((utf32 >> 6) & 0x3F));
249            out[upto++] = (byte)(0x80 | (utf32 & 0x3F));
250            continue;
251          }
252        }
253        // replace unpaired surrogate or out-of-order low surrogate
254        // with substitution character
255        out[upto++] = (byte) 0xEF;
256        out[upto++] = (byte) 0xBF;
257        out[upto++] = (byte) 0xBD;
258      }
259    }
260    //assert matches(source, offset, i-offset-1, out, upto);
261    result.length = upto;
262  }

318   *  for length characters.  Returns the number of bytes
319   *  written to bytesOut. */
320  public static void UTF16toUTF8(final String s, final int offset, final int length, UTF8Result result) {
321    final int end = offset + length;
322
323    byte[] out = result.result;
324
325    int upto = 0;
326    for(int i=offset;i<end;i++) {
327      final int code = (int) s.charAt(i);
328
329      if (upto+4 > out.length) {
330        out = result.result = ArrayUtil.grow(out, upto+4);
331      }
332      if (code < 0x80)
333        out[upto++] = (byte) code;
334      else if (code < 0x800) {
335        out[upto++] = (byte) (0xC0 | (code >> 6));
336        out[upto++] = (byte)(0x80 | (code & 0x3F));
337      } else if (code < 0xD800 || code > 0xDFFF) {

339        out[upto++] = (byte)(0x80 | ((code >> 6) & 0x3F));
340        out[upto++] = (byte)(0x80 | (code & 0x3F));
341      } else {
342        // surrogate pair
343        // confirm valid high surrogate
344        if (code < 0xDC00 && (i < end-1)) {
345          int utf32 = (int) s.charAt(i+1);
346          // confirm valid low surrogate and write pair
347          if (utf32 >= 0xDC00 && utf32 <= 0xDFFF) {

365    result.length = upto;
366  }

478  /** Convert UTF8 bytes into UTF16 characters.  If offset
479   *  is non-zero, conversion starts at that starting point
480   *  in utf8, re-using the results from the previous call
481   *  up until offset. */
482  public static void UTF8toUTF16(final byte[] utf8, final int offset, final int length, final UTF16Result result) {
483
484    final int end = offset + length;
485    char[] out = result.result;
486    if (result.offsets.length <= end) {
487      result.offsets = ArrayUtil.grow(result.offsets, end+1);
488    }
489    final int[] offsets = result.offsets;
490
491    // If incremental decoding fell in the middle of a
492    // single unicode character, rollback to its start:
493    int upto = offset;
494    while(offsets[upto] == -1)
495      upto--;
496
497    int outUpto = offsets[upto];
498
499    // Pre-allocate for worst case 1-for-1
500    if (outUpto+length >= out.length) {
501      out = result.result = ArrayUtil.grow(out, outUpto+length+1);
502    }
503
504    while (upto < end) {
505
506      final int b = utf8[upto]&0xff;
507      final int ch;
508
509      offsets[upto++] = outUpto;
510
511      if (b < 0xc0) {
512        assert b < 0x80;
513        ch = b;
514      } else if (b < 0xe0) {
515        ch = ((b&0x1f)<<6) + (utf8[upto]&0x3f);
516        offsets[upto++] = -1;
517      } else if (b < 0xf0) {
518        ch = ((b&0xf)<<12) + ((utf8[upto]&0x3f)<<6) + (utf8[upto+1]&0x3f);
519        offsets[upto++] = -1;
520        offsets[upto++] = -1;
521      } else {
522        assert b < 0xf8;
523        ch = ((b&0x7)<<18) + ((utf8[upto]&0x3f)<<12) + ((utf8[upto+1]&0x3f)<<6) + (utf8[upto+2]&0x3f);
524        offsets[upto++] = -1;
525        offsets[upto++] = -1;
526        offsets[upto++] = -1;
527      }
528
529      if (ch <= UNI_MAX_BMP) {
530        // target is a character <= 0xFFFF
531        out[outUpto++] = (char) ch;
532      } else {
533        // target is a character in range 0xFFFF - 0x10FFFF
534        final int chHalf = ch - HALF_BASE;
535        out[outUpto++] = (char) ((chHalf >> HALF_SHIFT) + UNI_SUR_HIGH_START);
536        out[outUpto++] = (char) ((chHalf & HALF_MASK) + UNI_SUR_LOW_START);
537      }
538    }
539
540    offsets[upto] = outUpto;
541    result.length = outUpto;
542  }

645  /** Shift value for lead surrogate to form a supplementary character. */
646  private static final int LEAD_SURROGATE_SHIFT_ = 10;
647  /** Mask to retrieve the significant value from a trail surrogate.*/
648  private static final int TRAIL_SURROGATE_MASK_ = 0x3FF;
649  /** Trail surrogate minimum value */
650  private static final int TRAIL_SURROGATE_MIN_VALUE = 0xDC00;
651  /** Lead surrogate minimum value */
652  private static final int LEAD_SURROGATE_MIN_VALUE = 0xD800;
653  /** The minimum value for Supplementary code points */
654  private static final int SUPPLEMENTARY_MIN_VALUE = 0x10000;
655  /** Value that all lead surrogate starts with */
656  private static final int LEAD_SURROGATE_OFFSET_ = LEAD_SURROGATE_MIN_VALUE
657          - (SUPPLEMENTARY_MIN_VALUE >> LEAD_SURROGATE_SHIFT_);

659  /**
660   * Cover JDK 1.5 API. Create a String from an array of codePoints.
661   *
662   * @param codePoints The code array
663   * @param offset The start of the text in the code point array
664   * @param count The number of code points
665   * @return a String representing the code points between offset and count
666   * @throws IllegalArgumentException If an invalid code point is encountered
667   * @throws IndexOutOfBoundsException If the offset or count are out of bounds.
668   */
669  public static String newString(int[] codePoints, int offset, int count) {
670      if (count < 0) {
671          throw new IllegalArgumentException();
672      }
673      char[] chars = new char[count];
674      int w = 0;
675      for (int r = offset, e = offset + count; r < e; ++r) {
676          int cp = codePoints[r];
677          if (cp < 0 || cp > 0x10ffff) {
678              throw new IllegalArgumentException();
679          }
680          while (true) {
681              try {
682                  if (cp < 0x010000) {
683                      chars[w] = (char) cp;
684                      w++;
685                  } else {
686                      chars[w] = (char) (LEAD_SURROGATE_OFFSET_ + (cp >> LEAD_SURROGATE_SHIFT_));
687                      chars[w + 1] = (char) (TRAIL_SURROGATE_MIN_VALUE + (cp & TRAIL_SURROGATE_MASK_));
688                      w += 2;
689                  }
690                  break;
691              } catch (IndexOutOfBoundsException ex) {
692                  int newlen = (int) (Math.ceil((double) codePoints.length * (w + 2)
693                          / (r - offset + 1)));
694                  char[] temp = new char[newlen];
695                  System.arraycopy(chars, 0, temp, 0, w);
696                  chars = temp;
697              }
698          }
699      }
700      return new String(chars, 0, w);
701  }

703  /**
704   * Interprets the given byte array as UTF-8 and converts to UTF-16. The {@link CharsRef} will be extended if 
705   * it doesn't provide enough space to hold the worst case of each byte becoming a UTF-16 codepoint.
706   * <p>
707   * NOTE: Full characters are read, even if this reads past the length passed (and
708   * can result in an ArrayOutOfBoundsException if invalid UTF-8 is passed).
709   * Explicit checks for valid UTF-8 are not performed. 
710   */
711  // TODO: broken if incoming result.offset != 0
712  public static void UTF8toUTF16(byte[] utf8, int offset, int length, CharsRef chars) {
713    int out_offset = chars.offset = 0;
714    final char[] out = chars.chars =  ArrayUtil.grow(chars.chars, length);
715    final int limit = offset + length;
716    while (offset < limit) {
717      int b = utf8[offset++]&0xff;
718      if (b < 0xc0) {
719        assert b < 0x80;
720        out[out_offset++] = (char)b;
721      } else if (b < 0xe0) {
722        out[out_offset++] = (char)(((b&0x1f)<<6) + (utf8[offset++]&0x3f));
723      } else if (b < 0xf0) {
724        out[out_offset++] = (char)(((b&0xf)<<12) + ((utf8[offset]&0x3f)<<6) + (utf8[offset+1]&0x3f));
725        offset += 2;
726      } else {
727        assert b < 0xf8: "b=" + b;
728        int ch = ((b&0x7)<<18) + ((utf8[offset]&0x3f)<<12) + ((utf8[offset+1]&0x3f)<<6) + (utf8[offset+2]&0x3f);
729        offset += 3;
730        if (ch < UNI_MAX_BMP) {
731          out[out_offset++] = (char)ch;
732        } else {
733          int chHalf = ch - 0x0010000;
734          out[out_offset++] = (char) ((chHalf >> 10) + 0xD800);
735          out[out_offset++] = (char) ((chHalf & HALF_MASK) + 0xDC00);          
736        }
737      }
738    }
739    chars.length = out_offset - chars.offset;
740  }
