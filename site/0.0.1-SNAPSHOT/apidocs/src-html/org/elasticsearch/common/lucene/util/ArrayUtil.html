070  /**
071   * Parses the string argument as if it was an int value and returns the
072   * result. Throws NumberFormatException if the string does not represent an
073   * int quantity. The second argument specifies the radix to use when parsing
074   * the value.
075   *
076   * @param chars a string representation of an int quantity.
077   * @param radix the base to use for conversion.
078   * @return int the value represented by the argument
079   * @throws NumberFormatException if the argument could not be parsed as an int quantity.
080   */
081  public static int parseInt(char[] chars, int offset, int len, int radix)
082          throws NumberFormatException {
083    if (chars == null || radix < Character.MIN_RADIX
084            || radix > Character.MAX_RADIX) {
085      throw new NumberFormatException();
086    }
087    int  i = 0;
088    if (len == 0) {
089      throw new NumberFormatException("chars length is 0");
090    }
091    boolean negative = chars[offset + i] == '-';
092    if (negative && ++i == len) {
093      throw new NumberFormatException("can't convert to an int");
094    }
095    if (negative == true){
096      offset++;
097      len--;
098    }
099    return parse(chars, offset, len, radix, negative);
100  }
101
102
103  private static int parse(char[] chars, int offset, int len, int radix,
104                           boolean negative) throws NumberFormatException {
105    int max = Integer.MIN_VALUE / radix;
106    int result = 0;
107    for (int i = 0; i < len; i++){
108      int digit = Character.digit(chars[i + offset], radix);
109      if (digit == -1) {
110        throw new NumberFormatException("Unable to parse");
111      }
112      if (max > result) {
113        throw new NumberFormatException("Unable to parse");
114      }
115      int next = result * radix - digit;
116      if (next > result) {
117        throw new NumberFormatException("Unable to parse");
118      }
119      result = next;
120    }
121    /*while (offset < len) {
122
123    }*/
124    if (!negative) {
125      result = -result;
126      if (result < 0) {
127        throw new NumberFormatException("Unable to parse");
128      }
129    }
130    return result;
131  }

139  /** Returns an array size >= minTargetSize, generally
140   *  over-allocating exponentially to achieve amortized
141   *  linear-time cost as the array grows.
142   *
143   *  NOTE: this was originally borrowed from Python 2.4.2
144   *  listobject.c sources (attribution in LICENSE.txt), but
145   *  has now been substantially changed based on
146   *  discussions from java-dev thread with subject "Dynamic
147   *  array reallocation algorithms", started on Jan 12
148   *  2010.
149   *
150   * @param minTargetSize Minimum required value to be returned.
151   * @param bytesPerElement Bytes used by each element of
152   * the array.  See constants in {@link RamUsageEstimator}.
153   *
154   */
155
156  public static int oversize(int minTargetSize, int bytesPerElement) {
157
158    if (minTargetSize < 0) {
159      // catch usage that accidentally overflows int
160      throw new IllegalArgumentException("invalid array size " + minTargetSize);
161    }
162
163    if (minTargetSize == 0) {
164      // wait until at least one element is requested
165      return 0;
166    }
167
168    // asymptotic exponential growth by 1/8th, favors
169    // spending a bit more CPU to not tie up too much wasted
170    // RAM:
171    int extra = minTargetSize >> 3;
172
173    if (extra < 3) {
174      // for very small arrays, where constant overhead of
175      // realloc is presumably relatively high, we grow
176      // faster
177      extra = 3;
178    }
179
180    int newSize = minTargetSize + extra;
181
182    // add 7 to allow for worst case byte alignment addition below:
183    if (newSize+7 < 0) {
184      // int overflowed -- return max allowed array size
185      return Integer.MAX_VALUE;
186    }
187
188    if (Constants.JRE_IS_64BIT) {
189      // round up to 8 byte alignment in 64bit env
190      switch(bytesPerElement) {
191      case 4:
192        // round up to multiple of 2
193        return (newSize + 1) & 0x7ffffffe;
194      case 2:
195        // round up to multiple of 4
196        return (newSize + 3) & 0x7ffffffc;
197      case 1:
198        // round up to multiple of 8
199        return (newSize + 7) & 0x7ffffff8;
200      case 8:
201        // no rounding
202      default:
203        // odd (invalid?) size
204        return newSize;
205      }
206    } else {
207      // round up to 4 byte alignment in 64bit env
208      switch(bytesPerElement) {
209      case 2:
210        // round up to multiple of 2
211        return (newSize + 1) & 0x7ffffffe;
212      case 1:
213        // round up to multiple of 4
214        return (newSize + 3) & 0x7ffffffc;
215      case 4:
216      case 8:
217        // no rounding
218      default:
219        // odd (invalid?) size
220        return newSize;
221      }
222    }
223  }

225  public static int getShrinkSize(int currentSize, int targetSize, int bytesPerElement) {
226    final int newSize = oversize(targetSize, bytesPerElement);
227    // Only reallocate if we are "substantially" smaller.
228    // This saves us from "running hot" (constantly making a
229    // bit bigger then a bit smaller, over and over):
230    if (newSize < currentSize / 2)
231      return newSize;
232    else
233      return currentSize;
234  }

492  /**
493   * See if two array slices are the same.
494   *
495   * @param left        The left array to compare
496   * @param offsetLeft  The offset into the array.  Must be positive
497   * @param right       The right array to compare
498   * @param offsetRight the offset into the right array.  Must be positive
499   * @param length      The length of the section of the array to compare
500   * @return true if the two arrays, starting at their respective offsets, are equal
501   * 
502   * @see java.util.Arrays#equals(char[], char[])
503   */
504  public static boolean equals(char[] left, int offsetLeft, char[] right, int offsetRight, int length) {
505    if ((offsetLeft + length <= left.length) && (offsetRight + length <= right.length)) {
506      for (int i = 0; i < length; i++) {
507        if (left[offsetLeft + i] != right[offsetRight + i]) {
508          return false;
509        }
510
511      }
512      return true;
513    }
514    return false;
515  }

557  /** SorterTemplate with custom {@link Comparator} */
558  private static <T> SorterTemplate getSorter(final T[] a, final Comparator<? super T> comp) {
559    return new SorterTemplate() {
560      @Override
561      protected void swap(int i, int j) {
562        final T o = a[i];
563        a[i] = a[j];
564        a[j] = o;
565      }
566      
567      @Override
568      protected int compare(int i, int j) {
569        return comp.compare(a[i], a[j]);
570      }
571
572      @Override
573      protected void setPivot(int i) {
574        pivot = a[i];
575      }
576  
577      @Override
578      protected int comparePivot(int j) {
579        return comp.compare(pivot, a[j]);
580      }
581      
582      private T pivot;
583    };
584  }

588    return new SorterTemplate() {

591        final T o = a[i];
592        a[i] = a[j];
593        a[j] = o;
594      }
595      
596      @Override
597      protected int compare(int i, int j) {
598        return a[i].compareTo(a[j]);
599      }
600
601      @Override
602      protected void setPivot(int i) {
603        pivot = a[i];
604      }
605  
606      @Override
607      protected int comparePivot(int j) {
608        return pivot.compareTo(a[j]);
609      }
610      
611      private T pivot;
612    };

700   * algorithm. It is only recommended to use this algorithm for partially sorted small arrays!
701   * @param fromIndex start index (inclusive)
702   * @param toIndex end index (exclusive)
703   */
704  public static <T> void insertionSort(T[] a, int fromIndex, int toIndex, Comparator<? super T> comp) {
705    if (toIndex-fromIndex <= 1) return;
706    getSorter(a, comp).insertionSort(fromIndex, toIndex-1);
