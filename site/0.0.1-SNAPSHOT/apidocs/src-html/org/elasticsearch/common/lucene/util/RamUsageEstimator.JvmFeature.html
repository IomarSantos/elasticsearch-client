026/**
027 * Estimates the size (memory representation) of Java objects.
028 * 
029 * <p>NOTE: Starting with Lucene 3.6, creating instances of this class
030 * is deprecated. If you still do this, please note, that instances of
031 * {@code RamUsageEstimator} are not thread-safe!
032 * It is also deprecated to enable checking of String intern-ness,
033 * the new static method no longer allow to do this. Interned strings
034 * will be counted as any other object and count for memory usage.
035 * 
036 * <p>In Lucene 3.6, custom {@code MemoryModel}s were completely
037 * removed. The new implementation is now using Hotspot&trade; internals
038 * to get the correct scale factors and offsets for calculating
039 * memory usage.
040 * 
041 * @see #sizeOf(Object)
042 * @see #shallowSizeOf(Object)
043 * @see #shallowSizeOfInstance(Class)
044 * 
045 */
046public final class RamUsageEstimator {
047  /**
048   * JVM diagnostic features.
049   */
050  public static enum JvmFeature {
051    OBJECT_REFERENCE_SIZE("Object reference size estimated using array index scale"),
052    ARRAY_HEADER_SIZE("Array header size estimated using array based offset"),
053    FIELD_OFFSETS("Shallow instance size based on field offsets"),
054    OBJECT_ALIGNMENT("Object alignment retrieved from HotSpotDiagnostic MX bean");
055
056    public final String description;
057
058    private JvmFeature(String description) {
059      this.description = description;
060    }
061    
062    @Override
063    public String toString() {
064      return super.name() + " (" + description + ")";
065    }
066  }

141  /**
142   * Initialize constants and try to collect information about the JVM internals. 
143   */
144  static {
145    // Initialize empirically measured defaults. We'll modify them to the current
146    // JVM settings later on if possible.
147    int referenceSize = Constants.JRE_IS_64BIT ? 8 : 4;
148    int objectHeader = Constants.JRE_IS_64BIT ? 16 : 8;
149    // The following is objectHeader + NUM_BYTES_INT, but aligned (object alignment)
150    // so on 64 bit JVMs it'll be align(16 + 4, @8) = 24.
151    int arrayHeader = Constants.JRE_IS_64BIT ? 24 : 12;
152
153    supportedFeatures = EnumSet.noneOf(JvmFeature.class);
154
155    Class<?> unsafeClass = null;
156    Object tempTheUnsafe = null;
157    try {
158      unsafeClass = Class.forName("sun.misc.Unsafe");
159      final Field unsafeField = unsafeClass.getDeclaredField("theUnsafe");
160      unsafeField.setAccessible(true);
161      tempTheUnsafe = unsafeField.get(null);
162    } catch (Exception e) {
163      // Ignore.
164    }
165    theUnsafe = tempTheUnsafe;
166
167    // get object reference size by getting scale factor of Object[] arrays:
168    try {
169      final Method arrayIndexScaleM = unsafeClass.getMethod("arrayIndexScale", Class.class);
170      referenceSize = ((Number) arrayIndexScaleM.invoke(theUnsafe, Object[].class)).intValue();
171      supportedFeatures.add(JvmFeature.OBJECT_REFERENCE_SIZE);
172    } catch (Exception e) {
173      // ignore.
174    }
175
176    // "best guess" based on reference size. We will attempt to modify
177    // these to exact values if there is supported infrastructure.
178    objectHeader = Constants.JRE_IS_64BIT ? (8 + referenceSize) : 8;
179    arrayHeader =  Constants.JRE_IS_64BIT ? (8 + 2 * referenceSize) : 12;
180
181    // get the object header size:
182    // - first try out if the field offsets are not scaled (see warning in Unsafe docs)
183    // - get the object header size by getting the field offset of the first field of a dummy object
184    // If the scaling is byte-wise and unsafe is available, enable dynamic size measurement for
185    // estimateRamUsage().
186    Method tempObjectFieldOffsetMethod = null;
187    try {
188      final Method objectFieldOffsetM = unsafeClass.getMethod("objectFieldOffset", Field.class);
189      final Field dummy1Field = DummyTwoLongObject.class.getDeclaredField("dummy1");
190      final int ofs1 = ((Number) objectFieldOffsetM.invoke(theUnsafe, dummy1Field)).intValue();
191      final Field dummy2Field = DummyTwoLongObject.class.getDeclaredField("dummy2");
192      final int ofs2 = ((Number) objectFieldOffsetM.invoke(theUnsafe, dummy2Field)).intValue();
193      if (Math.abs(ofs2 - ofs1) == NUM_BYTES_LONG) {
194        final Field baseField = DummyOneFieldObject.class.getDeclaredField("base");
195        objectHeader = ((Number) objectFieldOffsetM.invoke(theUnsafe, baseField)).intValue();
196        supportedFeatures.add(JvmFeature.FIELD_OFFSETS);
197        tempObjectFieldOffsetMethod = objectFieldOffsetM;
198      }
199    } catch (Exception e) {
200      // Ignore.
201    }
202    objectFieldOffsetMethod = tempObjectFieldOffsetMethod;
203
204    // Get the array header size by retrieving the array base offset
205    // (offset of the first element of an array).
206    try {
207      final Method arrayBaseOffsetM = unsafeClass.getMethod("arrayBaseOffset", Class.class);
208      // we calculate that only for byte[] arrays, it's actually the same for all types:
209      arrayHeader = ((Number) arrayBaseOffsetM.invoke(theUnsafe, byte[].class)).intValue();
210      supportedFeatures.add(JvmFeature.ARRAY_HEADER_SIZE);
211    } catch (Exception e) {
212      // Ignore.
213    }
214
215    NUM_BYTES_OBJECT_REF = referenceSize;
216    NUM_BYTES_OBJECT_HEADER = objectHeader;
217    NUM_BYTES_ARRAY_HEADER = arrayHeader;
218    
219    // Try to get the object alignment (the default seems to be 8 on Hotspot, 
220    // regardless of the architecture). Retrieval only works with Java 6.
221    int objectAlignment = 8;
222    try {
223      final Class<?> beanClazz = Class.forName("com.sun.management.HotSpotDiagnosticMXBean");
224      final Object hotSpotBean = ManagementFactory.newPlatformMXBeanProxy(
225        ManagementFactory.getPlatformMBeanServer(),
226        "com.sun.management:type=HotSpotDiagnostic",
227        beanClazz
228      );
229      final Method getVMOptionMethod = beanClazz.getMethod("getVMOption", String.class);
230      final Object vmOption = getVMOptionMethod.invoke(hotSpotBean, "ObjectAlignmentInBytes");
231      objectAlignment = Integer.parseInt(
232          vmOption.getClass().getMethod("getValue").invoke(vmOption).toString()
233      );
234      supportedFeatures.add(JvmFeature.OBJECT_ALIGNMENT);
235    } catch (Exception e) {
236      // Ignore.
237    }
238
239    NUM_BYTES_OBJECT_ALIGNMENT = objectAlignment;
240
241    JVM_INFO_STRING = "[JVM: " +
242        Constants.JVM_NAME + ", " + Constants.JVM_VERSION + ", " + Constants.JVM_VENDOR + ", " + 
243        Constants.JAVA_VENDOR + ", " + Constants.JAVA_VERSION + "]";
244  }

283  /** 
284   * Aligns an object size to be the next multiple of {@link #NUM_BYTES_OBJECT_ALIGNMENT}. 
285   */
286  public static long alignObjectSize(long size) {
287    size += (long) NUM_BYTES_OBJECT_ALIGNMENT - 1L;
288    return size - (size % NUM_BYTES_OBJECT_ALIGNMENT);
289  }

331  /** 
332   * Estimates the RAM usage by the given object. It will
333   * walk the object tree and sum up all referenced objects.
334   * 
335   * <p><b>Resource Usage:</b> This method internally uses a set of
336   * every object seen during traversals so it does allocate memory
337   * (it isn't side-effect free). After the method exits, this memory
338   * should be GCed.</p>
339   */
340  public static long sizeOf(Object obj) {
341    return measureObjectSize(obj, false);
342  }
343
344  /** 
345   * Estimates a "shallow" memory usage of the given object. For arrays, this will be the
346   * memory taken by array storage (no subreferences will be followed). For objects, this
347   * will be the memory taken by the fields.
348   * 
349   * JVM object alignments are also applied.
350   */
351  public static long shallowSizeOf(Object obj) {
352    if (obj == null) return 0;
353    final Class<?> clz = obj.getClass();
354    if (clz.isArray()) {
355      return shallowSizeOfArray(obj);
356    } else {
357      return shallowSizeOfInstance(clz);
358    }
359  }
360
361  /**
362   * Returns the shallow instance size in bytes an instance of the given class would occupy.
363   * This works with all conventional classes and primitive types, but not with arrays
364   * (the size then depends on the number of elements and varies from object to object).
365   * 
366   * @see #shallowSizeOf(Object)
367   * @throws IllegalArgumentException if {@code clazz} is an array class. 
368   */
369  public static long shallowSizeOfInstance(Class<?> clazz) {
370    if (clazz.isArray())
371      throw new IllegalArgumentException("This method does not work with array classes.");
372    if (clazz.isPrimitive())
373      return primitiveSizes.get(clazz);
374    
375    long size = NUM_BYTES_OBJECT_HEADER;
376
377    // Walk type hierarchy
378    for (;clazz != null; clazz = clazz.getSuperclass()) {
379      final Field[] fields = clazz.getDeclaredFields();
380      for (Field f : fields) {
381        if (!Modifier.isStatic(f.getModifiers())) {
382          size = adjustForField(size, f);
383        }
384      }
385    }
386    return alignObjectSize(size);    
387  }
388
389  /**
390   * Return shallow size of any <code>array</code>.
391   */
392  private static long shallowSizeOfArray(Object array) {
393    long size = NUM_BYTES_ARRAY_HEADER;
394    final int len = Array.getLength(array);
395    if (len > 0) {
396      Class<?> arrayElementClazz = array.getClass().getComponentType();
397      if (arrayElementClazz.isPrimitive()) {
398        size += (long) len * primitiveSizes.get(arrayElementClazz);
399      } else {
400        size += (long) NUM_BYTES_OBJECT_REF * len;
401      }
402    }
403    return alignObjectSize(size);
404  }

406  /*
407   * Non-recursive version of object descend. This consumes more memory than recursive in-depth 
408   * traversal but prevents stack overflows on long chains of objects
409   * or complex graphs (a max. recursion depth on my machine was ~5000 objects linked in a chain
410   * so not too much).  
411   */
412  private static long measureObjectSize(Object root, boolean checkInterned) {
413    // Objects seen so far.
414    final IdentityHashSet<Object> seen = new IdentityHashSet<Object>();
415    // Class cache with reference Field and precalculated shallow size. 
416    final IdentityHashMap<Class<?>, ClassCache> classCache = new IdentityHashMap<Class<?>, ClassCache>();
417    // Stack of objects pending traversal. Recursion caused stack overflows. 
418    final ArrayList<Object> stack = new ArrayList<Object>();
419    stack.add(root);
420
421    long totalSize = 0;
422    while (!stack.isEmpty()) {
423      final Object ob = stack.remove(stack.size() - 1);
424
425      if (ob == null || seen.contains(ob)) {
426        continue;
427      }
428      seen.add(ob);
429
430      // *** BEGIN deprecation
431      if (checkInterned && ob instanceof String && ob == ((String) ob).intern()) {
432        continue;
433      }
434      // *** END deprecation
435      
436      final Class<?> obClazz = ob.getClass();
437      if (obClazz.isArray()) {
438        /*
439         * Consider an array, possibly of primitive types. Push any of its references to
440         * the processing stack and accumulate this array's shallow size. 
441         */
442        long size = NUM_BYTES_ARRAY_HEADER;
443        final int len = Array.getLength(ob);
444        if (len > 0) {
445          Class<?> componentClazz = obClazz.getComponentType();
446          if (componentClazz.isPrimitive()) {
447            size += (long) len * primitiveSizes.get(componentClazz);
448          } else {
449            size += (long) NUM_BYTES_OBJECT_REF * len;
450
451            // Push refs for traversal later.
452            for (int i = len; --i >= 0 ;) {
453              final Object o = Array.get(ob, i);
454              if (o != null && !seen.contains(o)) {
455                stack.add(o);
456              }
457            }            
458          }
459        }
460        totalSize += alignObjectSize(size);
461      } else {
462        /*
463         * Consider an object. Push any references it has to the processing stack
464         * and accumulate this object's shallow size. 
465         */
466        try {
467          ClassCache cachedInfo = classCache.get(obClazz);
468          if (cachedInfo == null) {
469            classCache.put(obClazz, cachedInfo = createCacheEntry(obClazz));
470          }
471
472          for (Field f : cachedInfo.referenceFields) {
473            // Fast path to eliminate redundancies.
474            final Object o = f.get(ob);
475            if (o != null && !seen.contains(o)) {
476              stack.add(o);
477            }
478          }
479
480          totalSize += cachedInfo.alignedShallowInstanceSize;
481        } catch (IllegalAccessException e) {
482          // this should never happen as we enabled setAccessible().
483          throw new RuntimeException("Reflective field access failed?", e);
484        }
485      }
486    }
487
488    // Help the GC (?).
489    seen.clear();
490    stack.clear();
491    classCache.clear();
492
493    return totalSize;
494  }

496  /**
497   * Create a cached information about shallow size and reference fields for 
498   * a given class.
499   */
500  private static ClassCache createCacheEntry(final Class<?> clazz) {
501    ClassCache cachedInfo;
502    long shallowInstanceSize = NUM_BYTES_OBJECT_HEADER;
503    final ArrayList<Field> referenceFields = new ArrayList<Field>(32);
504    for (Class<?> c = clazz; c != null; c = c.getSuperclass()) {
505      final Field[] fields = c.getDeclaredFields();
506      for (final Field f : fields) {
507        if (!Modifier.isStatic(f.getModifiers())) {
508          shallowInstanceSize = adjustForField(shallowInstanceSize, f);
509
510          if (!f.getType().isPrimitive()) {
511            f.setAccessible(true);
512            referenceFields.add(f);
513          }
514        }
515      }
516    }
517
518    cachedInfo = new ClassCache(
519        alignObjectSize(shallowInstanceSize), 
520        referenceFields.toArray(new Field[referenceFields.size()]));
521    return cachedInfo;
522  }

524  /**
525   * This method returns the maximum representation size of an object. <code>sizeSoFar</code>
526   * is the object's size measured so far. <code>f</code> is the field being probed.
527   * 
528   * <p>The returned offset will be the maximum of whatever was measured so far and 
529   * <code>f</code> field's offset and representation size (unaligned).
530   */
531  private static long adjustForField(long sizeSoFar, final Field f) {
532    final Class<?> type = f.getType();
533    final int fsize = type.isPrimitive() ? primitiveSizes.get(type) : NUM_BYTES_OBJECT_REF;
534    if (objectFieldOffsetMethod != null) {
535      try {
536        final long offsetPlusSize =
537          ((Number) objectFieldOffsetMethod.invoke(theUnsafe, f)).longValue() + fsize;
538        return Math.max(sizeSoFar, offsetPlusSize);
539      } catch (IllegalAccessException ex) {
540        throw new RuntimeException("Access problem with sun.misc.Unsafe", ex);
541      } catch (InvocationTargetException ite) {
542        final Throwable cause = ite.getCause();
543        if (cause instanceof RuntimeException)
544          throw (RuntimeException) cause;
545        if (cause instanceof Error)
546          throw (Error) cause;
547        // this should never happen (Unsafe does not declare
548        // checked Exceptions for this method), but who knows?
549        throw new RuntimeException("Call to Unsafe's objectFieldOffset() throwed "+
550          "checked Exception when accessing field " +
551          f.getDeclaringClass().getName() + "#" + f.getName(), cause);
552      }
553    } else {
554      // TODO: No alignments based on field type/ subclass fields alignments?
555      return sizeSoFar + fsize;
556    }
557  }

579  /**
580   * Returns <code>size</code> in human-readable units (GB, MB, KB or bytes). 
581   */
582  public static String humanReadableUnits(long bytes, DecimalFormat df) {
583    if (bytes / ONE_GB > 0) {
584      return df.format((float) bytes / ONE_GB) + " GB";
585    } else if (bytes / ONE_MB > 0) {
586      return df.format((float) bytes / ONE_MB) + " MB";
587    } else if (bytes / ONE_KB > 0) {
588      return df.format((float) bytes / ONE_KB) + " KB";
589    } else {
590      return bytes + " bytes";
591    }
592  }

603  /**
604   * An identity hash set implemented using open addressing. No null keys are allowed.
605   * 
606   * TODO: If this is useful outside this class, make it public - needs some work
607   */
608  static final class IdentityHashSet<KType> implements Iterable<KType> {
609    /**
610     * Default load factor.
611     */
612    public final static float DEFAULT_LOAD_FACTOR = 0.75f;
613
614    /**
615     * Minimum capacity for the set.
616     */
617    public final static int MIN_CAPACITY = 4;
618
619    /**
620     * All of set entries. Always of power of two length.
621     */
622    public Object[] keys;
623    
624    /**
625     * Cached number of assigned slots.
626     */
627    public int assigned;
628    
629    /**
630     * The load factor for this set (fraction of allocated or deleted slots before
631     * the buffers must be rehashed or reallocated).
632     */
633    public final float loadFactor;
634    
635    /**
636     * Cached capacity threshold at which we must resize the buffers.
637     */
638    private int resizeThreshold;
639    
640    /**
641     * Creates a hash set with the default capacity of 16.
642     * load factor of {@value #DEFAULT_LOAD_FACTOR}. `
643     */
644    public IdentityHashSet() {
645      this(16, DEFAULT_LOAD_FACTOR);
646    }
647    
648    /**
649     * Creates a hash set with the given capacity, load factor of
650     * {@value #DEFAULT_LOAD_FACTOR}.
651     */
652    public IdentityHashSet(int initialCapacity) {
653      this(initialCapacity, DEFAULT_LOAD_FACTOR);
654    }
655    
656    /**
657     * Creates a hash set with the given capacity and load factor.
658     */
659    public IdentityHashSet(int initialCapacity, float loadFactor) {
660      initialCapacity = Math.max(MIN_CAPACITY, initialCapacity);
661      
662      assert initialCapacity > 0 : "Initial capacity must be between (0, "
663          + Integer.MAX_VALUE + "].";
664      assert loadFactor > 0 && loadFactor < 1 : "Load factor must be between (0, 1).";
665      this.loadFactor = loadFactor;
666      allocateBuffers(roundCapacity(initialCapacity));
667    }
668    
669    /**
670     * Adds a reference to the set. Null keys are not allowed.
671     */
672    public boolean add(KType e) {
673      assert e != null : "Null keys not allowed.";
674      
675      if (assigned >= resizeThreshold) expandAndRehash();
676      
677      final int mask = keys.length - 1;
678      int slot = rehash(e) & mask;
679      Object existing;
680      while ((existing = keys[slot]) != null) {
681        if (e == existing) {
682          return false; // already found.
683        }
684        slot = (slot + 1) & mask;
685      }
686      assigned++;
687      keys[slot] = e;
688      return true;
689    }
690
691    /**
692     * Checks if the set contains a given ref.
693     */
694    public boolean contains(KType e) {
695      final int mask = keys.length - 1;
696      int slot = rehash(e) & mask;
697      Object existing;
698      while ((existing = keys[slot]) != null) {
699        if (e == existing) {
700          return true;
701        }
702        slot = (slot + 1) & mask;
703      }
704      return false;
705    }
706
707    /** Rehash via MurmurHash.
708     * 
709     * <p>The implementation is based on the
710     * finalization step from Austin Appleby's
711     * <code>MurmurHash3</code>.
712     * 
713     * @see "http://sites.google.com/site/murmurhash/"
714     */
715    private static int rehash(Object o) {
716      int k = System.identityHashCode(o);
717      k ^= k >>> 16;
718      k *= 0x85ebca6b;
719      k ^= k >>> 13;
720      k *= 0xc2b2ae35;
721      k ^= k >>> 16;
722      return k;
723    }
724    
725    /**
726     * Expand the internal storage buffers (capacity) or rehash current keys and
727     * values if there are a lot of deleted slots.
728     */
729    private void expandAndRehash() {
730      final Object[] oldKeys = this.keys;
731      
732      assert assigned >= resizeThreshold;
733      allocateBuffers(nextCapacity(keys.length));
734      
735      /*
736       * Rehash all assigned slots from the old hash table.
737       */
738      final int mask = keys.length - 1;
739      for (int i = 0; i < oldKeys.length; i++) {
740        final Object key = oldKeys[i];
741        if (key != null) {
742          int slot = rehash(key) & mask;
743          while (keys[slot] != null) {
744            slot = (slot + 1) & mask;
745          }
746          keys[slot] = key;
747        }
748      }
749      Arrays.fill(oldKeys, null);
750    }
751    
752    /**
753     * Allocate internal buffers for a given capacity.
754     * 
755     * @param capacity
756     *          New capacity (must be a power of two).
757     */
758    private void allocateBuffers(int capacity) {
759      this.keys = new Object[capacity];
760      this.resizeThreshold = (int) (capacity * DEFAULT_LOAD_FACTOR);
761    }
762    
763    /**
764     * Return the next possible capacity, counting from the current buffers' size.
765     */
766    protected int nextCapacity(int current) {
767      assert current > 0 && Long.bitCount(current) == 1 : "Capacity must be a power of two.";
768      assert ((current << 1) > 0) : "Maximum capacity exceeded ("
769          + (0x80000000 >>> 1) + ").";
770      
771      if (current < MIN_CAPACITY / 2) current = MIN_CAPACITY / 2;
772      return current << 1;
773    }
774    
775    /**
776     * Round the capacity to the next allowed value.
777     */
778    protected int roundCapacity(int requestedCapacity) {
779      // Maximum positive integer that is a power of two.
780      if (requestedCapacity > (0x80000000 >>> 1)) return (0x80000000 >>> 1);
781      
782      int capacity = MIN_CAPACITY;
783      while (capacity < requestedCapacity) {
784        capacity <<= 1;
785      }
786
787      return capacity;
788    }
789    
790    public void clear() {
791      assigned = 0;
792      Arrays.fill(keys, null);
793    }
794    
795    public int size() {
796      return assigned;
797    }
798    
799    public boolean isEmpty() {
800      return size() == 0;
801    }

803    //@Override
804    public Iterator<KType> iterator() {
805      return new Iterator<KType>() {
806        int pos = -1;
807        Object nextElement = fetchNext();
808
809        //@Override
810        public boolean hasNext() {
811          return nextElement != null;
812        }
813
814        @SuppressWarnings("unchecked")
815        //@Override
816        public KType next() {
817          Object r = this.nextElement;
818          if (r == null) {
819            throw new NoSuchElementException();
820          }
821          this.nextElement = fetchNext();
822          return (KType) r;
823        }
824
825        private Object fetchNext() {
826          pos++;
827          while (pos < keys.length && keys[pos] == null) {
828            pos++;
829          }
830
831          return (pos >= keys.length ? null : keys[pos]);
832        }
833
834        //@Override
835        public void remove() {
836          throw new UnsupportedOperationException();
837        }
838      };
839    }
