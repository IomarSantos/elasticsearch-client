020package org.elasticsearch.common.logging.jdk;
021
022import org.elasticsearch.common.logging.support.AbstractESLogger;
023
024import java.util.logging.Level;
025import java.util.logging.Logger;
026
027/**
028 *
029 */
030public class JdkESLogger extends AbstractESLogger {
031
032    private final Logger logger;
033
034    private final String name;
035
036    public JdkESLogger(String prefix, String name, Logger logger) {
037        super(prefix);
038        this.logger = logger;
039        this.name = name;
040    }
041
042    @Override
043    public void setLevel(String level) {
044        if ("error".equalsIgnoreCase(level)) {
045            logger.setLevel(Level.SEVERE);
046        } else if ("warn".equalsIgnoreCase(level)) {
047            logger.setLevel(Level.WARNING);
048        } else if ("info".equalsIgnoreCase(level)) {
049            logger.setLevel(Level.INFO);
050        } else if ("debug".equalsIgnoreCase(level)) {
051            logger.setLevel(Level.FINE);
052        } else if ("trace".equalsIgnoreCase(level)) {
053            logger.setLevel(Level.FINE);
054        }
055    }
056
057    @Override
058    public String getName() {
059        return logger.getName();
060    }
061
062    @Override
063    public boolean isTraceEnabled() {
064        return logger.isLoggable(Level.FINEST);
065    }
066
067    @Override
068    public boolean isDebugEnabled() {
069        return logger.isLoggable(Level.FINE);
070    }
071
072    @Override
073    public boolean isInfoEnabled() {
074        return logger.isLoggable(Level.INFO);
075    }
076
077    @Override
078    public boolean isWarnEnabled() {
079        return logger.isLoggable(Level.WARNING);
080    }
081
082    @Override
083    public boolean isErrorEnabled() {
084        return logger.isLoggable(Level.SEVERE);
085    }
086
087    @Override
088    protected void internalTrace(String msg) {
089        logger.logp(Level.FINEST, name, null, msg);
090    }
091
092    @Override
093    protected void internalTrace(String msg, Throwable cause) {
094        logger.logp(Level.FINEST, name, null, msg, cause);
095    }
096
097    @Override
098    protected void internalDebug(String msg) {
099        logger.logp(Level.FINE, name, null, msg);
100    }
101
102    @Override
103    protected void internalDebug(String msg, Throwable cause) {
104        logger.logp(Level.FINE, name, null, msg, cause);
105    }
106
107    @Override
108    protected void internalInfo(String msg) {
109        logger.logp(Level.INFO, name, null, msg);
110    }
111
112    @Override
113    protected void internalInfo(String msg, Throwable cause) {
114        logger.logp(Level.INFO, name, null, msg, cause);
115    }
116
117    @Override
118    protected void internalWarn(String msg) {
119        logger.logp(Level.WARNING, name, null, msg);
120    }
121
122    @Override
123    protected void internalWarn(String msg, Throwable cause) {
124        logger.logp(Level.WARNING, name, null, msg, cause);
125    }
126
127    @Override
128    protected void internalError(String msg) {
129        logger.logp(Level.SEVERE, name, null, msg);
130    }
131
132    @Override
133    protected void internalError(String msg, Throwable cause) {
134        logger.logp(Level.SEVERE, name, null, msg, cause);
135    }
136}
