020package org.elasticsearch.common.logging.log4j;
021
022import org.apache.log4j.Level;
023import org.apache.log4j.Logger;
024import org.elasticsearch.common.logging.support.AbstractESLogger;
025
026/**
027 *
028 */
029public class Log4jESLogger extends AbstractESLogger {
030
031    private final org.apache.log4j.Logger logger;
032
033    public Log4jESLogger(String prefix, Logger logger) {
034        super(prefix);
035        this.logger = logger;
036    }
037
038    public void setLevel(String level) {
039        if ("error".equalsIgnoreCase(level)) {
040            logger.setLevel(Level.ERROR);
041        } else if ("warn".equalsIgnoreCase(level)) {
042            logger.setLevel(Level.WARN);
043        } else if ("info".equalsIgnoreCase(level)) {
044            logger.setLevel(Level.INFO);
045        } else if ("debug".equalsIgnoreCase(level)) {
046            logger.setLevel(Level.DEBUG);
047        } else if ("trace".equalsIgnoreCase(level)) {
048            logger.setLevel(Level.TRACE);
049        }
050    }
051
052    @Override
053    public String getName() {
054        return logger.getName();
055    }
056
057    @Override
058    public boolean isTraceEnabled() {
059        return logger.isTraceEnabled();
060    }
061
062    @Override
063    public boolean isDebugEnabled() {
064        return logger.isDebugEnabled();
065    }
066
067    @Override
068    public boolean isInfoEnabled() {
069        return logger.isInfoEnabled();
070    }
071
072    @Override
073    public boolean isWarnEnabled() {
074        return logger.isEnabledFor(Level.WARN);
075    }
076
077    @Override
078    public boolean isErrorEnabled() {
079        return logger.isEnabledFor(Level.ERROR);
080    }
081
082    @Override
083    protected void internalTrace(String msg) {
084        logger.trace(msg);
085    }
086
087    @Override
088    protected void internalTrace(String msg, Throwable cause) {
089        logger.trace(msg, cause);
090    }
091
092    @Override
093    protected void internalDebug(String msg) {
094        logger.debug(msg);
095    }
096
097    @Override
098    protected void internalDebug(String msg, Throwable cause) {
099        logger.debug(msg, cause);
100    }
101
102    @Override
103    protected void internalInfo(String msg) {
104        logger.info(msg);
105    }
106
107    @Override
108    protected void internalInfo(String msg, Throwable cause) {
109        logger.info(msg, cause);
110    }
111
112    @Override
113    protected void internalWarn(String msg) {
114        logger.warn(msg);
115    }
116
117    @Override
118    protected void internalWarn(String msg, Throwable cause) {
119        logger.warn(msg, cause);
120    }
121
122    @Override
123    protected void internalError(String msg) {
124        logger.error(msg);
125    }
126
127    @Override
128    protected void internalError(String msg, Throwable cause) {
129        logger.error(msg, cause);
130    }
131}
