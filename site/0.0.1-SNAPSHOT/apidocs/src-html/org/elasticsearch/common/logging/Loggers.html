078    public static ESLogger getLogger(Class clazz, Settings settings, String... prefixes) {
079        return getLogger(buildClassLoggerName(clazz), settings, prefixes);
080    }

082    public static ESLogger getLogger(String loggerName, Settings settings, String... prefixes) {
083        List<String> prefixesList = newArrayList();
084        if (settings.getAsBoolean("logger.logHostAddress", false)) {
085            try {
086                prefixesList.add(InetAddress.getLocalHost().getHostAddress());
087            } catch (UnknownHostException e) {
088                // ignore
089            }
090        }
091        if (settings.getAsBoolean("logger.logHostName", false)) {
092            try {
093                prefixesList.add(InetAddress.getLocalHost().getHostName());
094            } catch (UnknownHostException e) {
095                // ignore
096            }
097        }
098        String name = settings.get("name");
099        if (name != null) {
100            prefixesList.add(name);
101        }
102        if (prefixes != null && prefixes.length > 0) {
103            prefixesList.addAll(asList(prefixes));
104        }
105        return getLogger(getLoggerName(loggerName), prefixesList.toArray(new String[prefixesList.size()]));
106    }

123
124    public static ESLogger getLogger(String name, String... prefixes) {
125        String prefix = null;
126        if (prefixes != null && prefixes.length > 0) {
127            StringBuilder sb = new StringBuilder();
128            for (String prefixX : prefixes) {
129                if (prefixX != null) {
130                    if (prefixX.equals(SPACE)) {
131                        sb.append(" ");
132                    } else {
133                        sb.append("[").append(prefixX).append("]");
134                    }
135                }
136            }
137            if (sb.length() > 0) {
138                sb.append(" ");
139                prefix = sb.toString();
140            }
141        }
142        return ESLoggerFactory.getLogger(prefix, getLoggerName(name));
143    }
144
145    private static String buildClassLoggerName(Class clazz) {
146        String name = clazz.getName();
147        if (name.startsWith("org.elasticsearch.")) {
148            name = Classes.getPackageName(clazz);
149        }
150        return name;
151    }
152
153    private static String getLoggerName(String name) {
154        if (name.startsWith("org.elasticsearch.")) {
155            name = name.substring("org.elasticsearch.".length());
156        }
157        return commonPrefix + name;
158    }
