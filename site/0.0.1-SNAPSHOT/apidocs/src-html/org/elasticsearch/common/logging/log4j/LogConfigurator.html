020package org.elasticsearch.common.logging.log4j;
021
022import com.google.common.collect.ImmutableMap;
023import org.apache.log4j.PropertyConfigurator;
024import org.elasticsearch.common.collect.MapBuilder;
025import org.elasticsearch.common.settings.ImmutableSettings;
026import org.elasticsearch.common.settings.Settings;
027import org.elasticsearch.env.ClientEnvironment;
028import org.elasticsearch.env.FailedToResolveConfigException;
029
030import java.util.Map;
031import java.util.Properties;
032
033import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
034
035/**
036 *
037 */
038public class LogConfigurator {
039
040    private static boolean loaded;
041
042    private static ImmutableMap<String, String> replacements = new MapBuilder<String, String>()
043            .put("console", "org.elasticsearch.common.logging.log4j.ConsoleAppender")
044            .put("async", "org.apache.log4j.AsyncAppender")
045            .put("dailyRollingFile", "org.apache.log4j.DailyRollingFileAppender")
046            .put("externallyRolledFile", "org.apache.log4j.ExternallyRolledFileAppender")
047            .put("file", "org.apache.log4j.FileAppender")
048            .put("jdbc", "org.apache.log4j.jdbc.JDBCAppender")
049            .put("jms", "org.apache.log4j.net.JMSAppender")
050            .put("lf5", "org.apache.log4j.lf5.LF5Appender")
051            .put("ntevent", "org.apache.log4j.nt.NTEventLogAppender")
052            .put("null", "org.apache.log4j.NullAppender")
053            .put("rollingFile", "org.apache.log4j.RollingFileAppender")
054            .put("smtp", "org.apache.log4j.net.SMTPAppender")
055            .put("socket", "org.apache.log4j.net.SocketAppender")
056            .put("socketHub", "org.apache.log4j.net.SocketHubAppender")
057            .put("syslog", "org.apache.log4j.net.SyslogAppender")
058            .put("telnet", "org.apache.log4j.net.TelnetAppender")
059                    // layouts
060            .put("simple", "org.apache.log4j.SimpleLayout")
061            .put("html", "org.apache.log4j.HTMLLayout")
062            .put("pattern", "org.apache.log4j.PatternLayout")
063            .put("consolePattern", "org.apache.log4j.PatternLayout")
064            .put("ttcc", "org.apache.log4j.TTCCLayout")
065            .put("xml", "org.apache.log4j.XMLLayout")
066            .immutableMap();
067
068    public static void configure(Settings settings) {
069        if (loaded) {
070            return;
071        }
072        loaded = true;
073        ClientEnvironment environment = new ClientEnvironment(settings);
074        ImmutableSettings.Builder settingsBuilder = settingsBuilder().put(settings);
075        try {
076            settingsBuilder.loadFromUrl(environment.resolveConfig("logging.yml"));
077        } catch (FailedToResolveConfigException e) {
078            // ignore
079        } catch (NoClassDefFoundError e) {
080            // ignore, no yaml
081        }
082        try {
083            settingsBuilder.loadFromUrl(environment.resolveConfig("logging.json"));
084        } catch (FailedToResolveConfigException e) {
085            // ignore
086        }
087        try {
088            settingsBuilder.loadFromUrl(environment.resolveConfig("logging.properties"));
089        } catch (FailedToResolveConfigException e) {
090            // ignore
091        }
092        settingsBuilder
093                .putProperties("elasticsearch.", System.getProperties())
094                .putProperties("es.", System.getProperties())
095                .replacePropertyPlaceholders();
096        Properties props = new Properties();
097        for (Map.Entry<String, String> entry : settingsBuilder.build().getAsMap().entrySet()) {
098            String key = "log4j." + entry.getKey();
099            String value = entry.getValue();
100            if (replacements.containsKey(value)) {
101                value = replacements.get(value);
102            }
103            if (key.endsWith(".value")) {
104                props.setProperty(key.substring(0, key.length() - ".value".length()), value);
105            } else if (key.endsWith(".type")) {
106                props.setProperty(key.substring(0, key.length() - ".type".length()), value);
107            } else {
108                props.setProperty(key, value);
109            }
110        }
111        PropertyConfigurator.configure(props);
112    }
113}
