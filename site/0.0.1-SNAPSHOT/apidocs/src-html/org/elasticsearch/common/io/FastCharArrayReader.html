020package org.elasticsearch.common.io;
021
022import java.io.IOException;
023import java.io.Reader;
024
025/**
026 *
027 */
028public class FastCharArrayReader extends Reader {
029
030    /**
031     * The character buffer.
032     */
033    protected char buf[];
034
035    /**
036     * The current buffer position.
037     */
038    protected int pos;
039
040    /**
041     * The position of mark in buffer.
042     */
043    protected int markedPos = 0;
044
045    /**
046     * The index of the end of this buffer.  There is not valid
047     * data at or beyond this index.
048     */
049    protected int count;
050
051    /**
052     * Creates a CharArrayReader from the specified array of chars.
053     *
054     * @param buf Input buffer (not copied)
055     */
056    public FastCharArrayReader(char buf[]) {
057        this.buf = buf;
058        this.pos = 0;
059        this.count = buf.length;
060    }
061
062    /**
063     * Creates a CharArrayReader from the specified array of chars.
064     * <p/>
065     * <p> The resulting reader will start reading at the given
066     * <tt>offset</tt>.  The total number of <tt>char</tt> values that can be
067     * read from this reader will be either <tt>length</tt> or
068     * <tt>buf.length-offset</tt>, whichever is smaller.
069     *
070     * @param buf    Input buffer (not copied)
071     * @param offset Offset of the first char to read
072     * @param length Number of chars to read
073     * @throws IllegalArgumentException If <tt>offset</tt> is negative or greater than
074     *                                  <tt>buf.length</tt>, or if <tt>length</tt> is negative, or if
075     *                                  the sum of these two values is negative.
076     */
077    public FastCharArrayReader(char buf[], int offset, int length) {
078        if ((offset < 0) || (offset > buf.length) || (length < 0) ||
079                ((offset + length) < 0)) {
080            throw new IllegalArgumentException();
081        }
082        this.buf = buf;
083        this.pos = offset;
084        this.count = Math.min(offset + length, buf.length);
085        this.markedPos = offset;
086    }
087
088    /**
089     * Checks to make sure that the stream has not been closed
090     */
091    private void ensureOpen() throws IOException {
092        if (buf == null)
093            throw new IOException("Stream closed");
094    }
095
096    /**
097     * Reads a single character.
098     *
099     * @throws IOException If an I/O error occurs
100     */
101    public int read() throws IOException {
102        ensureOpen();
103        if (pos >= count)
104            return -1;
105        else
106            return buf[pos++];
107    }
108
109    /**
110     * Reads characters into a portion of an array.
111     *
112     * @param b   Destination buffer
113     * @param off Offset at which to start storing characters
114     * @param len Maximum number of characters to read
115     * @return The actual number of characters read, or -1 if
116     *         the end of the stream has been reached
117     * @throws IOException If an I/O error occurs
118     */
119    public int read(char b[], int off, int len) throws IOException {
120        ensureOpen();
121        if ((off < 0) || (off > b.length) || (len < 0) ||
122                ((off + len) > b.length) || ((off + len) < 0)) {
123            throw new IndexOutOfBoundsException();
124        } else if (len == 0) {
125            return 0;
126        }
127
128        if (pos >= count) {
129            return -1;
130        }
131        if (pos + len > count) {
132            len = count - pos;
133        }
134        if (len <= 0) {
135            return 0;
136        }
137        System.arraycopy(buf, pos, b, off, len);
138        pos += len;
139        return len;
140    }
141
142    /**
143     * Skips characters.  Returns the number of characters that were skipped.
144     * <p/>
145     * <p>The <code>n</code> parameter may be negative, even though the
146     * <code>skip</code> method of the {@link Reader} superclass throws
147     * an exception in this case. If <code>n</code> is negative, then
148     * this method does nothing and returns <code>0</code>.
149     *
150     * @param n The number of characters to skip
151     * @return The number of characters actually skipped
152     * @throws IOException If the stream is closed, or an I/O error occurs
153     */
154    public long skip(long n) throws IOException {
155        ensureOpen();
156        if (pos + n > count) {
157            n = count - pos;
158        }
159        if (n < 0) {
160            return 0;
161        }
162        pos += n;
163        return n;
164    }
165
166    /**
167     * Tells whether this stream is ready to be read.  Character-array readers
168     * are always ready to be read.
169     *
170     * @throws IOException If an I/O error occurs
171     */
172    public boolean ready() throws IOException {
173        ensureOpen();
174        return (count - pos) > 0;
175    }
176
177    /**
178     * Tells whether this stream supports the mark() operation, which it does.
179     */
180    public boolean markSupported() {
181        return true;
182    }
183
184    /**
185     * Marks the present position in the stream.  Subsequent calls to reset()
186     * will reposition the stream to this point.
187     *
188     * @param readAheadLimit Limit on the number of characters that may be
189     *                       read while still preserving the mark.  Because
190     *                       the stream's input comes from a character array,
191     *                       there is no actual limit; hence this argument is
192     *                       ignored.
193     * @throws IOException If an I/O error occurs
194     */
195    public void mark(int readAheadLimit) throws IOException {
196        ensureOpen();
197        markedPos = pos;
198    }
199
200    /**
201     * Resets the stream to the most recent mark, or to the beginning if it has
202     * never been marked.
203     *
204     * @throws IOException If an I/O error occurs
205     */
206    public void reset() throws IOException {
207        ensureOpen();
208        pos = markedPos;
209    }
210
211    /**
212     * Closes the stream and releases any system resources associated with
213     * it.  Once the stream has been closed, further read(), ready(),
214     * mark(), reset(), or skip() invocations will throw an IOException.
215     * Closing a previously closed stream has no effect.
216     */
217    public void close() {
218        buf = null;
219    }
220}
