020package org.elasticsearch.common.io;
021
022import java.io.IOException;
023import java.io.InputStream;
024
025/**
026 *
027 */
028public class FastByteArrayInputStream extends InputStream {
029
030    /**
031     * An array of bytes that was provided
032     * by the creator of the stream. Elements <code>buf[0]</code>
033     * through <code>buf[count-1]</code> are the
034     * only bytes that can ever be read from the
035     * stream;  element <code>buf[pos]</code> is
036     * the next byte to be read.
037     */
038    protected byte buf[];
039
040    /**
041     * The index of the next character to read from the input stream buffer.
042     * This value should always be nonnegative
043     * and not larger than the value of <code>count</code>.
044     * The next byte to be read from the input stream buffer
045     * will be <code>buf[pos]</code>.
046     */
047    protected int pos;
048
049    /**
050     * The currently marked position in the stream.
051     * ByteArrayInputStream objects are marked at position zero by
052     * default when constructed.  They may be marked at another
053     * position within the buffer by the <code>mark()</code> method.
054     * The current buffer position is set to this point by the
055     * <code>reset()</code> method.
056     * <p/>
057     * If no mark has been set, then the value of mark is the offset
058     * passed to the constructor (or 0 if the offset was not supplied).
059     *
060     * @since JDK1.1
061     */
062    protected int mark = 0;
063
064    /**
065     * The index one greater than the last valid character in the input
066     * stream buffer.
067     * This value should always be nonnegative
068     * and not larger than the length of <code>buf</code>.
069     * It  is one greater than the position of
070     * the last byte within <code>buf</code> that
071     * can ever be read  from the input stream buffer.
072     */
073    protected int count;
074
075    /**
076     * Creates a <code>ByteArrayInputStream</code>
077     * so that it  uses <code>buf</code> as its
078     * buffer array.
079     * The buffer array is not copied.
080     * The initial value of <code>pos</code>
081     * is <code>0</code> and the initial value
082     * of  <code>count</code> is the length of
083     * <code>buf</code>.
084     *
085     * @param buf the input buffer.
086     */
087    public FastByteArrayInputStream(byte buf[]) {
088        this.buf = buf;
089        this.pos = 0;
090        this.count = buf.length;
091    }
092
093    /**
094     * Creates <code>ByteArrayInputStream</code>
095     * that uses <code>buf</code> as its
096     * buffer array. The initial value of <code>pos</code>
097     * is <code>offset</code> and the initial value
098     * of <code>count</code> is the minimum of <code>offset+length</code>
099     * and <code>buf.length</code>.
100     * The buffer array is not copied. The buffer's mark is
101     * set to the specified offset.
102     *
103     * @param buf    the input buffer.
104     * @param offset the offset in the buffer of the first byte to read.
105     * @param length the maximum number of bytes to read from the buffer.
106     */
107    public FastByteArrayInputStream(byte buf[], int offset, int length) {
108        this.buf = buf;
109        this.pos = offset;
110        this.count = Math.min(offset + length, buf.length);
111        this.mark = offset;
112    }
113
114    /**
115     * Reads the next byte of data from this input stream. The value
116     * byte is returned as an <code>int</code> in the range
117     * <code>0</code> to <code>255</code>. If no byte is available
118     * because the end of the stream has been reached, the value
119     * <code>-1</code> is returned.
120     * <p/>
121     * This <code>read</code> method
122     * cannot block.
123     *
124     * @return the next byte of data, or <code>-1</code> if the end of the
125     *         stream has been reached.
126     */
127    public int read() {
128        return (pos < count) ? (buf[pos++] & 0xff) : -1;
129    }
130
131    /**
132     * Reads up to <code>len</code> bytes of data into an array of bytes
133     * from this input stream.
134     * If <code>pos</code> equals <code>count</code>,
135     * then <code>-1</code> is returned to indicate
136     * end of file. Otherwise, the  number <code>k</code>
137     * of bytes read is equal to the smaller of
138     * <code>len</code> and <code>count-pos</code>.
139     * If <code>k</code> is positive, then bytes
140     * <code>buf[pos]</code> through <code>buf[pos+k-1]</code>
141     * are copied into <code>b[off]</code>  through
142     * <code>b[off+k-1]</code> in the manner performed
143     * by <code>System.arraycopy</code>. The
144     * value <code>k</code> is added into <code>pos</code>
145     * and <code>k</code> is returned.
146     * <p/>
147     * This <code>read</code> method cannot block.
148     *
149     * @param b   the buffer into which the data is read.
150     * @param off the start offset in the destination array <code>b</code>
151     * @param len the maximum number of bytes read.
152     * @return the total number of bytes read into the buffer, or
153     *         <code>-1</code> if there is no more data because the end of
154     *         the stream has been reached.
155     * @throws NullPointerException      If <code>b</code> is <code>null</code>.
156     * @throws IndexOutOfBoundsException If <code>off</code> is negative,
157     *                                   <code>len</code> is negative, or <code>len</code> is greater than
158     *                                   <code>b.length - off</code>
159     */
160    public int read(byte b[], int off, int len) {
161        if (b == null) {
162            throw new NullPointerException();
163        } else if (off < 0 || len < 0 || len > b.length - off) {
164            throw new IndexOutOfBoundsException();
165        }
166        if (pos >= count) {
167            return -1;
168        }
169        if (pos + len > count) {
170            len = count - pos;
171        }
172        if (len <= 0) {
173            return 0;
174        }
175        System.arraycopy(buf, pos, b, off, len);
176        pos += len;
177        return len;
178    }
179
180    /**
181     * Skips <code>n</code> bytes of input from this input stream. Fewer
182     * bytes might be skipped if the end of the input stream is reached.
183     * The actual number <code>k</code>
184     * of bytes to be skipped is equal to the smaller
185     * of <code>n</code> and  <code>count-pos</code>.
186     * The value <code>k</code> is added into <code>pos</code>
187     * and <code>k</code> is returned.
188     *
189     * @param n the number of bytes to be skipped.
190     * @return the actual number of bytes skipped.
191     */
192    public long skip(long n) {
193        if (pos + n > count) {
194            n = count - pos;
195        }
196        if (n < 0) {
197            return 0;
198        }
199        pos += n;
200        return n;
201    }
202
203    /**
204     * Returns the number of remaining bytes that can be read (or skipped over)
205     * from this input stream.
206     * <p/>
207     * The value returned is <code>count&nbsp;- pos</code>,
208     * which is the number of bytes remaining to be read from the input buffer.
209     *
210     * @return the number of remaining bytes that can be read (or skipped
211     *         over) from this input stream without blocking.
212     */
213    public int available() {
214        return count - pos;
215    }
216
217    public int position() {
218        return pos;
219    }
220
221    /**
222     * Tests if this <code>InputStream</code> supports mark/reset. The
223     * <code>markSupported</code> method of <code>ByteArrayInputStream</code>
224     * always returns <code>true</code>.
225     *
226     * @since JDK1.1
227     */
228    public boolean markSupported() {
229        return true;
230    }
231
232    /**
233     * Set the current marked position in the stream.
234     * ByteArrayInputStream objects are marked at position zero by
235     * default when constructed.  They may be marked at another
236     * position within the buffer by this method.
237     * <p/>
238     * If no mark has been set, then the value of the mark is the
239     * offset passed to the constructor (or 0 if the offset was not
240     * supplied).
241     * <p/>
242     * <p> Note: The <code>readAheadLimit</code> for this class
243     * has no meaning.
244     *
245     * @since JDK1.1
246     */
247    public void mark(int readAheadLimit) {
248        mark = pos;
249    }
250
251    /**
252     * Resets the buffer to the marked position.  The marked position
253     * is 0 unless another position was marked or an offset was specified
254     * in the constructor.
255     */
256    public void reset() {
257        pos = mark;
258    }
259
260    /**
261     * Closing a <tt>ByteArrayInputStream</tt> has no effect. The methods in
262     * this class can be called after the stream has been closed without
263     * generating an <tt>IOException</tt>.
264     * <p/>
265     */
266    public void close() throws IOException {
267    }
268}
