036    private static ESLogger logger = ESLoggerFactory.getLogger(FileSystemUtils.class.getName());
037
038    private static final long mkdirsStallTimeout = TimeValue.timeValueMinutes(5).millis();
039    private static final Object mkdirsMutex = new Object();
040    private static volatile Thread mkdirsThread;
041    private static volatile long mkdirsStartTime;
042
043    public static boolean mkdirs(File dir) {
044        synchronized (mkdirsMutex) {
045            try {
046                mkdirsThread = Thread.currentThread();
047                mkdirsStartTime = System.currentTimeMillis();
048                return dir.mkdirs();
049            } finally {
050                mkdirsThread = null;
051            }
052        }
053    }
054
055    public static void checkMkdirsStall(long currentTime) {
056        Thread mkdirsThread1 = mkdirsThread;
057        long stallTime = currentTime - mkdirsStartTime;
058        if (mkdirsThread1 != null && (stallTime > mkdirsStallTimeout)) {
059            logger.error("mkdirs stalled for {} on {}, trying to interrupt", new TimeValue(stallTime), mkdirsThread1.getName());
060            mkdirsThread1.interrupt(); // try and interrupt it...
061        }
062    }

064    public static int maxOpenFiles(File testDir) {
065        boolean dirCreated = false;
066        if (!testDir.exists()) {
067            dirCreated = true;
068            testDir.mkdirs();
069        }
070        List<RandomAccessFile> files = new ArrayList<RandomAccessFile>();
071        try {
072            while (true) {
073                files.add(new RandomAccessFile(new File(testDir, "tmp" + files.size()), "rw"));
074            }
075        } catch (IOException ioe) {
076            int i = 0;
077            for (RandomAccessFile raf : files) {
078                try {
079                    raf.close();
080                } catch (IOException e) {
081                    // ignore
082                }
083                new File(testDir, "tmp" + i++).delete();
084            }
085            if (dirCreated) {
086                deleteRecursively(testDir);
087            }
088        }
089        return files.size();
090    }

093    public static boolean hasExtensions(File root, String... extensions) {
094        if (root != null && root.exists()) {
095            if (root.isDirectory()) {
096                File[] children = root.listFiles();
097                if (children != null) {
098                    for (File child : children) {
099                        if (child.isDirectory()) {
100                            boolean has = hasExtensions(child, extensions);
101                            if (has) {
102                                return true;
103                            }
104                        } else {
105                            for (String extension : extensions) {
106                                if (child.getName().endsWith(extension)) {
107                                    return true;
108                                }
109                            }
110                        }
111                    }
112                }
113            }
114        }
115        return false;
116    }
117
118    public static boolean deleteRecursively(File[] roots) {
119        boolean deleted = true;
120        for (File root : roots) {
121            deleted &= deleteRecursively(root);
122        }
123        return deleted;
124    }
125
126    public static boolean deleteRecursively(File root) {
127        return deleteRecursively(root, true);
128    }
129
130    private static boolean innerDeleteRecursively(File root) {
131        return deleteRecursively(root, true);
132    }
133
134    /**
135     * Delete the supplied {@link java.io.File} - for directories,
136     * recursively delete any nested directories or files as well.
137     *
138     * @param root       the root <code>File</code> to delete
139     * @param deleteRoot whether or not to delete the root itself or just the content of the root.
140     * @return <code>true</code> if the <code>File</code> was deleted,
141     *         otherwise <code>false</code>
142     */
143    public static boolean deleteRecursively(File root, boolean deleteRoot) {
144        if (root != null && root.exists()) {
145            if (root.isDirectory()) {
146                File[] children = root.listFiles();
147                if (children != null) {
148                    for (File aChildren : children) {
149                        innerDeleteRecursively(aChildren);
150                    }
151                }
152            }
153
154            if (deleteRoot) {
155                return root.delete();
156            } else {
157                return true;
158            }
159        }
160        return false;
161    }

163    public static void syncFile(File fileToSync) throws IOException {
164        boolean success = false;
165        int retryCount = 0;
166        IOException exc = null;
167        while (!success && retryCount < 5) {
168            retryCount++;
169            RandomAccessFile file = null;
170            try {
171                try {
172                    file = new RandomAccessFile(fileToSync, "rw");
173                    file.getFD().sync();
174                    success = true;
175                } finally {
176                    if (file != null)
177                        file.close();
178                }
179            } catch (IOException ioe) {
180                if (exc == null)
181                    exc = ioe;
182                try {
183                    // Pause 5 msec
184                    Thread.sleep(5);
185                } catch (InterruptedException ie) {
186                    throw new InterruptedIOException(ie.getMessage());
187                }
188            }
189        }
190    }

192    public static void copyFile(File sourceFile, File destinationFile) throws IOException {
193        FileInputStream sourceIs = null;
194        FileChannel source = null;
195        FileOutputStream destinationOs = null;
196        FileChannel destination = null;
197        try {
198            sourceIs = new FileInputStream(sourceFile);
199            source = sourceIs.getChannel();
200            destinationOs = new FileOutputStream(destinationFile);
201            destination = destinationOs.getChannel();
202            destination.transferFrom(source, 0, source.size());
203        } finally {
204            if (source != null) {
205                source.close();
206            }
207            if (sourceIs != null) {
208                sourceIs.close();
209            }
210            if (destination != null) {
211                destination.close();
212            }
213            if (destinationOs != null) {
214                destinationOs.close();
215            }
216        }
217    }
