001package org.elasticsearch.common.io.stream;
002
003import org.elasticsearch.Version;
004import org.elasticsearch.common.bytes.BytesReference;
005import org.elasticsearch.common.text.Text;
006
007import java.io.IOException;
008
009/**
010 */
011public abstract class AdapterStreamInput extends StreamInput {
012
013    protected StreamInput in;
014
015    protected AdapterStreamInput() {
016    }
017
018    public AdapterStreamInput(StreamInput in) {
019        this.in = in;
020        super.setVersion(in.getVersion());
021    }
022
023    @Override
024    public StreamInput setVersion(Version version) {
025        in.setVersion(version);
026        return super.setVersion(version);
027    }
028
029    public void reset(StreamInput in) {
030        this.in = in;
031    }
032
033    @Override
034    public byte readByte() throws IOException {
035        return in.readByte();
036    }
037
038    @Override
039    public void readBytes(byte[] b, int offset, int len) throws IOException {
040        in.readBytes(b, offset, len);
041    }
042
043    @Override
044    public BytesReference readBytesReference() throws IOException {
045        return in.readBytesReference();
046    }
047
048    @Override
049    public BytesReference readBytesReference(int length) throws IOException {
050        return in.readBytesReference(length);
051    }
052
053    @Override
054    public void reset() throws IOException {
055        in.reset();
056    }
057
058    @Override
059    public void close() throws IOException {
060        in.close();
061    }
062
063    @Override
064    public int read() throws IOException {
065        return in.read();
066    }
067
068    // override ones to direct them
069
070
071    @Override
072    public void readFully(byte[] b) throws IOException {
073        in.readFully(b);
074    }
075
076    @Override
077    public short readShort() throws IOException {
078        return in.readShort();
079    }
080
081    @Override
082    public int readInt() throws IOException {
083        return in.readInt();
084    }
085
086    @Override
087    public int readVInt() throws IOException {
088        return in.readVInt();
089    }
090
091    @Override
092    public long readLong() throws IOException {
093        return in.readLong();
094    }
095
096    @Override
097    public long readVLong() throws IOException {
098        return in.readVLong();
099    }
100
101    @Override
102    public String readUTF() throws IOException {
103        return in.readUTF();
104    }
105
106    @Override
107    public String readString() throws IOException {
108        return in.readString();
109    }
110
111    @Override
112    public Text readText() throws IOException {
113        return in.readText();
114    }
115
116    @Override
117    public int read(byte[] b) throws IOException {
118        return in.read(b);
119    }
120
121    @Override
122    public int read(byte[] b, int off, int len) throws IOException {
123        return in.read(b, off, len);
124    }
125
126    @Override
127    public long skip(long n) throws IOException {
128        return in.skip(n);
129    }
130
131    @Override
132    public int available() throws IOException {
133        return in.available();
134    }
135
136    @Override
137    public void mark(int readlimit) {
138        in.mark(readlimit);
139    }
140
141    @Override
142    public boolean markSupported() {
143        return in.markSupported();
144    }
145
146    @Override
147    public String toString() {
148        return in.toString();
149    }
150}
