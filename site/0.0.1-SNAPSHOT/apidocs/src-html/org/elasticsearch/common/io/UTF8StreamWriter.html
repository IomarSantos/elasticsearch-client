020package org.elasticsearch.common.io;
021
022import java.io.CharConversionException;
023import java.io.IOException;
024import java.io.OutputStream;
025import java.io.Writer;
026
027
028/**
029 */
030public final class UTF8StreamWriter extends Writer {
031
032    /**
033     * Holds the current output stream or <code>null</code> if closed.
034     */
035    private OutputStream _outputStream;
036
037    /**
038     * Holds the bytes' buffer.
039     */
040    private final byte[] _bytes;
041
042    /**
043     * Holds the bytes buffer index.
044     */
045    private int _index;
046
047    /**
048     * Creates a UTF-8 writer having a byte buffer of moderate capacity (2048).
049     */
050    public UTF8StreamWriter() {
051        _bytes = new byte[2048];
052    }
053
054    /**
055     * Creates a UTF-8 writer having a byte buffer of specified capacity.
056     *
057     * @param capacity the capacity of the byte buffer.
058     */
059    public UTF8StreamWriter(int capacity) {
060        _bytes = new byte[capacity];
061    }
062
063    /**
064     * Sets the output stream to use for writing until this writer is closed.
065     * For example:[code]
066     * Writer writer = new UTF8StreamWriter().setOutputStream(out);
067     * [/code] is equivalent but writes faster than [code]
068     * Writer writer = new java.io.OutputStreamWriter(out, "UTF-8");
069     * [/code]
070     *
071     * @param out the output stream.
072     * @return this UTF-8 writer.
073     * @throws IllegalStateException if this writer is being reused and
074     *                               it has not been {@link #close closed} or {@link #reset reset}.
075     */
076    public UTF8StreamWriter setOutput(OutputStream out) {
077        if (_outputStream != null)
078            throw new IllegalStateException("Writer not closed or reset");
079        _outputStream = out;
080        return this;
081    }
082
083    /**
084     * Writes a single character. This method supports 16-bits
085     * character surrogates.
086     *
087     * @param c <code>char</code> the character to be written (possibly
088     *          a surrogate).
089     * @throws IOException if an I/O error occurs.
090     */
091    public void write(char c) throws IOException {
092        if ((c < 0xd800) || (c > 0xdfff)) {
093            write((int) c);
094        } else if (c < 0xdc00) { // High surrogate.
095            _highSurrogate = c;
096        } else { // Low surrogate.
097            int code = ((_highSurrogate - 0xd800) << 10) + (c - 0xdc00)
098                    + 0x10000;
099            write(code);
100        }
101    }
102
103    private char _highSurrogate;
104
105    /**
106     * Writes a character given its 31-bits Unicode.
107     *
108     * @param code the 31 bits Unicode of the character to be written.
109     * @throws IOException if an I/O error occurs.
110     */
111    public void write(int code) throws IOException {
112        if ((code & 0xffffff80) == 0) {
113            _bytes[_index] = (byte) code;
114            if (++_index >= _bytes.length) {
115                flushBuffer();
116            }
117        } else { // Writes more than one byte.
118            write2(code);
119        }
120    }
121
122    private void write2(int c) throws IOException {
123        if ((c & 0xfffff800) == 0) { // 2 bytes.
124            _bytes[_index] = (byte) (0xc0 | (c >> 6));
125            if (++_index >= _bytes.length) {
126                flushBuffer();
127            }
128            _bytes[_index] = (byte) (0x80 | (c & 0x3f));
129            if (++_index >= _bytes.length) {
130                flushBuffer();
131            }
132        } else if ((c & 0xffff0000) == 0) { // 3 bytes.
133            _bytes[_index] = (byte) (0xe0 | (c >> 12));
134            if (++_index >= _bytes.length) {
135                flushBuffer();
136            }
137            _bytes[_index] = (byte) (0x80 | ((c >> 6) & 0x3f));
138            if (++_index >= _bytes.length) {
139                flushBuffer();
140            }
141            _bytes[_index] = (byte) (0x80 | (c & 0x3f));
142            if (++_index >= _bytes.length) {
143                flushBuffer();
144            }
145        } else if ((c & 0xff200000) == 0) { // 4 bytes.
146            _bytes[_index] = (byte) (0xf0 | (c >> 18));
147            if (++_index >= _bytes.length) {
148                flushBuffer();
149            }
150            _bytes[_index] = (byte) (0x80 | ((c >> 12) & 0x3f));
151            if (++_index >= _bytes.length) {
152                flushBuffer();
153            }
154            _bytes[_index] = (byte) (0x80 | ((c >> 6) & 0x3f));
155            if (++_index >= _bytes.length) {
156                flushBuffer();
157            }
158            _bytes[_index] = (byte) (0x80 | (c & 0x3f));
159            if (++_index >= _bytes.length) {
160                flushBuffer();
161            }
162        } else if ((c & 0xf4000000) == 0) { // 5 bytes.
163            _bytes[_index] = (byte) (0xf8 | (c >> 24));
164            if (++_index >= _bytes.length) {
165                flushBuffer();
166            }
167            _bytes[_index] = (byte) (0x80 | ((c >> 18) & 0x3f));
168            if (++_index >= _bytes.length) {
169                flushBuffer();
170            }
171            _bytes[_index] = (byte) (0x80 | ((c >> 12) & 0x3f));
172            if (++_index >= _bytes.length) {
173                flushBuffer();
174            }
175            _bytes[_index] = (byte) (0x80 | ((c >> 6) & 0x3f));
176            if (++_index >= _bytes.length) {
177                flushBuffer();
178            }
179            _bytes[_index] = (byte) (0x80 | (c & 0x3f));
180            if (++_index >= _bytes.length) {
181                flushBuffer();
182            }
183        } else if ((c & 0x80000000) == 0) { // 6 bytes.
184            _bytes[_index] = (byte) (0xfc | (c >> 30));
185            if (++_index >= _bytes.length) {
186                flushBuffer();
187            }
188            _bytes[_index] = (byte) (0x80 | ((c >> 24) & 0x3f));
189            if (++_index >= _bytes.length) {
190                flushBuffer();
191            }
192            _bytes[_index] = (byte) (0x80 | ((c >> 18) & 0x3f));
193            if (++_index >= _bytes.length) {
194                flushBuffer();
195            }
196            _bytes[_index] = (byte) (0x80 | ((c >> 12) & 0x3F));
197            if (++_index >= _bytes.length) {
198                flushBuffer();
199            }
200            _bytes[_index] = (byte) (0x80 | ((c >> 6) & 0x3F));
201            if (++_index >= _bytes.length) {
202                flushBuffer();
203            }
204            _bytes[_index] = (byte) (0x80 | (c & 0x3F));
205            if (++_index >= _bytes.length) {
206                flushBuffer();
207            }
208        } else {
209            throw new CharConversionException("Illegal character U+"
210                    + Integer.toHexString(c));
211        }
212    }
213
214    /**
215     * Writes a portion of an array of characters.
216     *
217     * @param cbuf the array of characters.
218     * @param off  the offset from which to start writing characters.
219     * @param len  the number of characters to write.
220     * @throws IOException if an I/O error occurs.
221     */
222    public void write(char cbuf[], int off, int len) throws IOException {
223        final int off_plus_len = off + len;
224        for (int i = off; i < off_plus_len; ) {
225            char c = cbuf[i++];
226            if (c < 0x80) {
227                _bytes[_index] = (byte) c;
228                if (++_index >= _bytes.length) {
229                    flushBuffer();
230                }
231            } else {
232                write(c);
233            }
234        }
235    }
236
237    /**
238     * Writes a portion of a string.
239     *
240     * @param str a String.
241     * @param off the offset from which to start writing characters.
242     * @param len the number of characters to write.
243     * @throws IOException if an I/O error occurs
244     */
245    public void write(String str, int off, int len) throws IOException {
246        final int off_plus_len = off + len;
247        for (int i = off; i < off_plus_len; ) {
248            char c = str.charAt(i++);
249            if (c < 0x80) {
250                _bytes[_index] = (byte) c;
251                if (++_index >= _bytes.length) {
252                    flushBuffer();
253                }
254            } else {
255                write(c);
256            }
257        }
258    }
259
260    /**
261     * Writes the specified character sequence.
262     *
263     * @param csq the character sequence.
264     * @throws IOException if an I/O error occurs
265     */
266    public void write(CharSequence csq) throws IOException {
267        final int length = csq.length();
268        for (int i = 0; i < length; ) {
269            char c = csq.charAt(i++);
270            if (c < 0x80) {
271                _bytes[_index] = (byte) c;
272                if (++_index >= _bytes.length) {
273                    flushBuffer();
274                }
275            } else {
276                write(c);
277            }
278        }
279    }
280
281    /**
282     * Flushes the stream.  If the stream has saved any characters from the
283     * various write() methods in a buffer, write them immediately to their
284     * intended destination.  Then, if that destination is another character or
285     * byte stream, flush it.  Thus one flush() invocation will flush all the
286     * buffers in a chain of Writers and OutputStreams.
287     *
288     * @throws IOException if an I/O error occurs.
289     */
290    public void flush() throws IOException {
291        flushBuffer();
292        _outputStream.flush();
293    }
294
295    /**
296     * Closes and {@link #reset resets} this writer for reuse.
297     *
298     * @throws IOException if an I/O error occurs
299     */
300    public void close() throws IOException {
301        if (_outputStream != null) {
302            flushBuffer();
303            _outputStream.close();
304            reset();
305        }
306    }
307
308    /**
309     * Flushes the internal bytes buffer.
310     *
311     * @throws IOException if an I/O error occurs
312     */
313    private void flushBuffer() throws IOException {
314        if (_outputStream == null)
315            throw new IOException("Stream closed");
316        _outputStream.write(_bytes, 0, _index);
317        _index = 0;
318    }
319
320    // Implements Reusable.
321    public void reset() {
322        _highSurrogate = 0;
323        _index = 0;
324        _outputStream = null;
325    }
326
327    /**
328     * @deprecated Replaced by {@link #setOutput(OutputStream)}
329     */
330    public UTF8StreamWriter setOutputStream(OutputStream out) {
331        return this.setOutput(out);
332    }
