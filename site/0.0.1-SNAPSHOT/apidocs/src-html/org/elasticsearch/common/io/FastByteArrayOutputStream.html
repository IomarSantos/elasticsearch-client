020package org.elasticsearch.common.io;
021
022import org.elasticsearch.common.Bytes;
023import org.elasticsearch.common.bytes.BytesArray;
024import org.elasticsearch.common.bytes.BytesReference;
025
026import java.io.IOException;
027import java.io.OutputStream;
028import java.io.UnsupportedEncodingException;
029import java.util.Arrays;
030
031/**
032 * Similar to {@link java.io.ByteArrayOutputStream} just not synced.
033 */
034public class FastByteArrayOutputStream extends OutputStream implements BytesStream {
035
036    /**
037     * The buffer where data is stored.
038     */
039    protected byte buf[];
040
041    /**
042     * The number of valid bytes in the buffer.
043     */
044    protected int count;
045
046    /**
047     * Creates a new byte array output stream. The buffer capacity is
048     * initially 1024 bytes, though its size increases if necessary.
049     * <p/>
050     * ES: We use 1024 bytes since we mainly use this to build json/smile
051     * content in memory, and rarely does the 32 byte default in ByteArrayOutputStream fits...
052     */
053    public FastByteArrayOutputStream() {
054        this(1024);
055    }
056
057    /**
058     * Creates a new byte array output stream, with a buffer capacity of
059     * the specified size, in bytes.
060     *
061     * @param size the initial size.
062     * @throws IllegalArgumentException if size is negative.
063     */
064    public FastByteArrayOutputStream(int size) {
065        if (size < 0) {
066            throw new IllegalArgumentException("Negative initial size: "
067                    + size);
068        }
069        buf = new byte[size];
070    }
071
072    /**
073     * Writes the specified byte to this byte array output stream.
074     *
075     * @param b the byte to be written.
076     */
077    public void write(int b) {
078        int newcount = count + 1;
079        if (newcount > buf.length) {
080            buf = Arrays.copyOf(buf, Bytes.oversize(newcount, 1));
081        }
082        buf[count] = (byte) b;
083        count = newcount;
084    }
085
086    /**
087     * Writes <code>len</code> bytes from the specified byte array
088     * starting at offset <code>off</code> to this byte array output stream.
089     * <p/>
090     * <b>NO checks for bounds, parameters must be ok!</b>
091     *
092     * @param b   the data.
093     * @param off the start offset in the data.
094     * @param len the number of bytes to write.
095     */
096    public void write(byte b[], int off, int len) {
097        if (len == 0) {
098            return;
099        }
100        int newcount = count + len;
101        if (newcount > buf.length) {
102            buf = Arrays.copyOf(buf, Bytes.oversize(newcount, 1));
103        }
104        System.arraycopy(b, off, buf, count, len);
105        count = newcount;
106    }
107
108    /**
109     * Writes the complete contents of this byte array output stream to
110     * the specified output stream argument, as if by calling the output
111     * stream's write method using <code>out.write(buf, 0, count)</code>.
112     *
113     * @param out the output stream to which to write the data.
114     * @throws IOException if an I/O error occurs.
115     */
116    public void writeTo(OutputStream out) throws IOException {
117        out.write(buf, 0, count);
118    }
119
120    /**
121     * Resets the <code>count</code> field of this byte array output
122     * stream to zero, so that all currently accumulated output in the
123     * output stream is discarded. The output stream can be used again,
124     * reusing the already allocated buffer space.
125     *
126     * @see java.io.ByteArrayInputStream#count
127     */
128    public void reset() {
129        count = 0;
130    }
131
132    /**
133     * Returns the underlying byte array. Note, use {@link #size()} in order to know
134     * the length of it.
135     */
136    @Override
137    public BytesReference bytes() {
138        return new BytesArray(buf, 0, count);
139    }
140
141    /**
142     * Returns the current size of the buffer.
143     *
144     * @return the value of the <code>count</code> field, which is the number
145     *         of valid bytes in this output stream.
146     * @see java.io.ByteArrayOutputStream#count
147     */
148    public int size() {
149        return count;
150    }
151
152    /**
153     * Seeks back to the given position. Size will become the seeked location.
154     */
155    public void seek(int position) {
156        this.count = position;
157    }
158
159    /**
160     * Converts the buffer's contents into a string decoding bytes using the
161     * platform's default character set. The length of the new <tt>String</tt>
162     * is a function of the character set, and hence may not be equal to the
163     * size of the buffer.
164     * <p/>
165     * <p> This method always replaces malformed-input and unmappable-character
166     * sequences with the default replacement string for the platform's
167     * default character set. The {@linkplain java.nio.charset.CharsetDecoder}
168     * class should be used when more control over the decoding process is
169     * required.
170     *
171     * @return String decoded from the buffer's contents.
172     * @since JDK1.1
173     */
174    public String toString() {
175        return new String(buf, 0, count);
176    }
177
178    /**
179     * Converts the buffer's contents into a string by decoding the bytes using
180     * the specified {@link java.nio.charset.Charset charsetName}. The length of
181     * the new <tt>String</tt> is a function of the charset, and hence may not be
182     * equal to the length of the byte array.
183     * <p/>
184     * <p> This method always replaces malformed-input and unmappable-character
185     * sequences with this charset's default replacement string. The {@link
186     * java.nio.charset.CharsetDecoder} class should be used when more control
187     * over the decoding process is required.
188     *
189     * @param charsetName the name of a supported
190     *                    {@linkplain java.nio.charset.Charset </code>charset<code>}
191     * @return String decoded from the buffer's contents.
192     * @throws java.io.UnsupportedEncodingException
193     *          If the named charset is not supported
194     * @since JDK1.1
195     */
196    public String toString(String charsetName)
197            throws UnsupportedEncodingException {
198        return new String(buf, 0, count, charsetName);
199    }
200
201    /**
202     * Closing a <tt>ByteArrayOutputStream</tt> has no effect. The methods in
203     * this class can be called after the stream has been closed without
204     * generating an <tt>IOException</tt>.
205     * <p/>
206     */
207    public void close() throws IOException {
208    }
209}
