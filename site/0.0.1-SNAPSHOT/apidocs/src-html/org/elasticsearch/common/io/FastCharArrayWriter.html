020package org.elasticsearch.common.io;
021
022import java.io.IOException;
023import java.io.Writer;
024import java.util.Arrays;
025
026/**
027 * A similar class to {@link java.io.CharArrayWriter} allowing to get the underlying <tt>char[]</tt> buffer.
028 */
029public class FastCharArrayWriter extends Writer {
030
031    /**
032     * The buffer where data is stored.
033     */
034    protected char buf[];
035
036    /**
037     * The number of chars in the buffer.
038     */
039    protected int count;
040
041    /**
042     * Creates a new CharArrayWriter.
043     */
044    public FastCharArrayWriter() {
045        this(32);
046    }
047
048    /**
049     * Creates a new CharArrayWriter with the specified initial size.
050     *
051     * @param initialSize an int specifying the initial buffer size.
052     * @throws IllegalArgumentException if initialSize is negative
053     */
054    public FastCharArrayWriter(int initialSize) {
055        if (initialSize < 0) {
056            throw new IllegalArgumentException("Negative initial size: "
057                    + initialSize);
058        }
059        buf = new char[initialSize];
060    }
061
062    /**
063     * Writes a character to the buffer.
064     */
065    public void write(int c) {
066        int newcount = count + 1;
067        if (newcount > buf.length) {
068            buf = Arrays.copyOf(buf, Math.max(buf.length << 1, newcount));
069        }
070        buf[count] = (char) c;
071        count = newcount;
072    }
073
074    /**
075     * Writes characters to the buffer.
076     *
077     * @param c   the data to be written
078     * @param off the start offset in the data
079     * @param len the number of chars that are written
080     */
081    public void write(char c[], int off, int len) {
082        if ((off < 0) || (off > c.length) || (len < 0) ||
083                ((off + len) > c.length) || ((off + len) < 0)) {
084            throw new IndexOutOfBoundsException();
085        } else if (len == 0) {
086            return;
087        }
088        int newcount = count + len;
089        if (newcount > buf.length) {
090            buf = Arrays.copyOf(buf, Math.max(buf.length << 1, newcount));
091        }
092        System.arraycopy(c, off, buf, count, len);
093        count = newcount;
094    }
095
096    /**
097     * Write a portion of a string to the buffer.
098     *
099     * @param str String to be written from
100     * @param off Offset from which to start reading characters
101     * @param len Number of characters to be written
102     */
103    public void write(String str, int off, int len) {
104        int newcount = count + len;
105        if (newcount > buf.length) {
106            buf = Arrays.copyOf(buf, Math.max(buf.length << 1, newcount));
107        }
108        str.getChars(off, off + len, buf, count);
109        count = newcount;
110    }
111
112    /**
113     * Writes the contents of the buffer to another character stream.
114     *
115     * @param out the output stream to write to
116     * @throws java.io.IOException If an I/O error occurs.
117     */
118    public void writeTo(Writer out) throws IOException {
119        out.write(buf, 0, count);
120    }
121
122    /**
123     * Appends the specified character sequence to this writer.
124     * <p/>
125     * <p> An invocation of this method of the form <tt>out.append(csq)</tt>
126     * behaves in exactly the same way as the invocation
127     * <p/>
128     * <pre>
129     *     out.write(csq.toString()) </pre>
130     *
131     * <p> Depending on the specification of <tt>toString</tt> for the
132     * character sequence <tt>csq</tt>, the entire sequence may not be
133     * appended. For instance, invoking the <tt>toString</tt> method of a
134     * character buffer will return a subsequence whose content depends upon
135     * the buffer's position and limit.
136     *
137     * @param csq The character sequence to append.  If <tt>csq</tt> is
138     *            <tt>null</tt>, then the four characters <tt>"null"</tt> are
139     *            appended to this writer.
140     * @return This writer
141     * @since 1.5
142     */
143    public FastCharArrayWriter append(CharSequence csq) {
144        String s = (csq == null ? "null" : csq.toString());
145        write(s, 0, s.length());
146        return this;
147    }
148
149    /**
150     * Appends a subsequence of the specified character sequence to this writer.
151     * <p/>
152     * <p> An invocation of this method of the form <tt>out.append(csq, start,
153     * end)</tt> when <tt>csq</tt> is not <tt>null</tt>, behaves in
154     * exactly the same way as the invocation
155     * <p/>
156     * <pre>
157     *     out.write(csq.subSequence(start, end).toString()) </pre>
158     *
159     * @param csq   The character sequence from which a subsequence will be
160     *              appended.  If <tt>csq</tt> is <tt>null</tt>, then characters
161     *              will be appended as if <tt>csq</tt> contained the four
162     *              characters <tt>"null"</tt>.
163     * @param start The index of the first character in the subsequence
164     * @param end   The index of the character following the last character in the
165     *              subsequence
166     * @return This writer
167     * @throws IndexOutOfBoundsException If <tt>start</tt> or <tt>end</tt> are negative, <tt>start</tt>
168     *                                   is greater than <tt>end</tt>, or <tt>end</tt> is greater than
169     *                                   <tt>csq.length()</tt>
170     * @since 1.5
171     */
172    public FastCharArrayWriter append(CharSequence csq, int start, int end) {
173        String s = (csq == null ? "null" : csq).subSequence(start, end).toString();
174        write(s, 0, s.length());
175        return this;
176    }
177
178    /**
179     * Appends the specified character to this writer.
180     * <p/>
181     * <p> An invocation of this method of the form <tt>out.append(c)</tt>
182     * behaves in exactly the same way as the invocation
183     * <p/>
184     * <pre>
185     *     out.write(c) </pre>
186     *
187     * @param c The 16-bit character to append
188     * @return This writer
189     * @since 1.5
190     */
191    public FastCharArrayWriter append(char c) {
192        write(c);
193        return this;
194    }
195
196    /**
197     * Resets the buffer so that you can use it again without
198     * throwing away the already allocated buffer.
199     */
200    public void reset() {
201        count = 0;
202    }
203
204    /**
205     * Returns a copy of the input data.
206     *
207     * @return an array of chars copied from the input data.
208     */
209    public char toCharArray()[] {
210        return Arrays.copyOf(buf, count);
211    }
212
213    /**
214     * Returns the underlying char array. Note, use {@link #size()} in order to know the size of
215     * of the actual content within the array.
216     */
217    public char[] unsafeCharArray() {
218        return buf;
219    }
220
221    /**
222     * Returns the current size of the buffer.
223     *
224     * @return an int representing the current size of the buffer.
225     */
226    public int size() {
227        return count;
228    }
229
230    /**
231     * Converts input data to a string.
232     *
233     * @return the string.
234     */
235    public String toString() {
236        return new String(buf, 0, count);
237    }
238
239    /**
240     * Converts the input data to a string with trimmed whitespaces.
241     */
242    public String toStringTrim() {
243        int st = 0;
244        int len = count;
245        char[] val = buf;    /* avoid getfield opcode */
246
247        while ((st < len) && (val[st] <= ' ')) {
248            st++;
249            len--;
250        }
251        while ((st < len) && (val[len - 1] <= ' ')) {
252            len--;
253        }
254        return new String(buf, st, len);
255    }
256
257    /**
258     * Flush the stream.
259     */
260    public void flush() {
261    }
262
263    /**
264     * Close the stream.  This method does not release the buffer, since its
265     * contents might still be required. Note: Invoking this method in this class
266     * will have no effect.
267     */
268    public void close() {
269    }
270
271}
