195    @Override
196    public String toString() {
197        if (duration < 0) {
198            return Long.toString(duration);
199        }
200        long nanos = nanos();
201        if (nanos == 0) {
202            return "0s";
203        }
204        double value = nanos;
205        String suffix = "nanos";
206        if (nanos >= C6) {
207            value = daysFrac();
208            suffix = "d";
209        } else if (nanos >= C5) {
210            value = hoursFrac();
211            suffix = "h";
212        } else if (nanos >= C4) {
213            value = minutesFrac();
214            suffix = "m";
215        } else if (nanos >= C3) {
216            value = secondsFrac();
217            suffix = "s";
218        } else if (nanos >= C2) {
219            value = millisFrac();
220            suffix = "ms";
221        } else if (nanos >= C1) {
222            value = microsFrac();
223            suffix = "micros";
224        }
225        return Strings.format1Decimals(value, suffix);
226    }

228    public static TimeValue parseTimeValue(String sValue, TimeValue defaultValue) {
229        if (sValue == null) {
230            return defaultValue;
231        }
232        try {
233            long millis;
234            if (sValue.endsWith("S")) {
235                millis = Long.parseLong(sValue.substring(0, sValue.length() - 1));
236            } else if (sValue.endsWith("ms")) {
237                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - "ms".length())));
238            } else if (sValue.endsWith("s")) {
239                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - 1)) * 1000);
240            } else if (sValue.endsWith("m")) {
241                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - 1)) * 60 * 1000);
242            } else if (sValue.endsWith("H") || sValue.endsWith("h")) {
243                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - 1)) * 60 * 60 * 1000);
244            } else if (sValue.endsWith("d")) {
245                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - 1)) * 24 * 60 * 60 * 1000);
246            } else if (sValue.endsWith("w")) {
247                millis = (long) (Double.parseDouble(sValue.substring(0, sValue.length() - 1)) * 7 * 24 * 60 * 60 * 1000);
248            } else {
249                millis = Long.parseLong(sValue);
250            }
251            return new TimeValue(millis, TimeUnit.MILLISECONDS);
252        } catch (NumberFormatException e) {
253            throw new ElasticSearchParseException("Failed to parse [" + sValue + "]", e);
254        }
255    }

257    static final long C0 = 1L;
258    static final long C1 = C0 * 1000L;
259    static final long C2 = C1 * 1000L;
260    static final long C3 = C2 * 1000L;
261    static final long C4 = C3 * 60L;
262    static final long C5 = C4 * 60L;
263    static final long C6 = C5 * 24L;
264
265    public static TimeValue readTimeValue(StreamInput in) throws IOException {
266        TimeValue timeValue = new TimeValue();
267        timeValue.readFrom(in);
268        return timeValue;
269    }
270
271    @Override
272    public void readFrom(StreamInput in) throws IOException {
273        duration = in.readLong();
274        timeUnit = TimeUnit.NANOSECONDS;
275    }
276
277    @Override
278    public void writeTo(StreamOutput out) throws IOException {
279        out.writeLong(nanos());
280    }

282    @Override
283    public boolean equals(Object o) {
284        if (this == o) return true;
285        if (o == null || getClass() != o.getClass()) return false;
286
287        TimeValue timeValue = (TimeValue) o;
288
289        if (duration != timeValue.duration) return false;
290        if (timeUnit != timeValue.timeUnit) return false;
291
292        return true;
293    }
294
295    @Override
296    public int hashCode() {
297        int result = (int) (duration ^ (duration >>> 32));
298        result = 31 * result + (timeUnit != null ? timeUnit.hashCode() : 0);
299        return result;
300    }
