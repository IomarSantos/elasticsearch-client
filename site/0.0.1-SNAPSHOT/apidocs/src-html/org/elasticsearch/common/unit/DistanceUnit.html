020package org.elasticsearch.common.unit;
021
022import org.elasticsearch.ElasticSearchIllegalArgumentException;
023import org.elasticsearch.common.io.stream.StreamInput;
024import org.elasticsearch.common.io.stream.StreamOutput;
025
026import java.io.IOException;
027
028/**
029 *
030 */
031public enum DistanceUnit {
032    MILES(3959, 24902) {
033        @Override
034        public String toString() {
035            return "miles";
036        }
037
038        @Override
039        public double toMiles(double distance) {
040            return distance;
041        }
042
043        @Override
044        public double toKilometers(double distance) {
045            return distance * MILES_KILOMETRES_RATIO;
046        }
047
048        @Override
049        public String toString(double distance) {
050            return distance + "mi";
051        }
052    },
053    KILOMETERS(6371, 40076) {
054        @Override
055        public String toString() {
056            return "km";
057        }
058
059        @Override
060        public double toMiles(double distance) {
061            return distance / MILES_KILOMETRES_RATIO;
062        }
063
064        @Override
065        public double toKilometers(double distance) {
066            return distance;
067        }
068
069        @Override
070        public String toString(double distance) {
071            return distance + "km";
072        }
073    };
074
075    static final double MILES_KILOMETRES_RATIO = 1.609344;
076
077    /**
078     * Converts the given distance from the given DistanceUnit, to the given DistanceUnit
079     *
080     * @param distance Distance to convert
081     * @param from     Unit to convert the distance from
082     * @param to       Unit of distance to convert to
083     * @return Given distance converted to the distance in the given uni
084     */
085    public static double convert(double distance, DistanceUnit from, DistanceUnit to) {
086        if (from == to) {
087            return distance;
088        }
089        return (to == MILES) ? distance / MILES_KILOMETRES_RATIO : distance * MILES_KILOMETRES_RATIO;
090    }
091
092    public static double parse(String distance, DistanceUnit defaultUnit, DistanceUnit to) {
093        if (distance.endsWith("mi")) {
094            return convert(Double.parseDouble(distance.substring(0, distance.length() - "mi".length())), MILES, to);
095        } else if (distance.endsWith("miles")) {
096            return convert(Double.parseDouble(distance.substring(0, distance.length() - "miles".length())), MILES, to);
097        } else if (distance.endsWith("km")) {
098            return convert(Double.parseDouble(distance.substring(0, distance.length() - "km".length())), KILOMETERS, to);
099        } else {
100            return convert(Double.parseDouble(distance), defaultUnit, to);
101        }
102    }
103
104    public static DistanceUnit parseUnit(String distance, DistanceUnit defaultUnit) {
105        if (distance.endsWith("mi")) {
106            return MILES;
107        } else if (distance.endsWith("miles")) {
108            return MILES;
109        } else if (distance.endsWith("km")) {
110            return KILOMETERS;
111        } else {
112            return defaultUnit;
113        }
114    }
115
116    protected final double earthCircumference;
117    protected final double earthRadius;
118    protected final double distancePerDegree;
119
120    DistanceUnit(double earthRadius, double earthCircumference) {
121        this.earthCircumference = earthCircumference;
122        this.earthRadius = earthRadius;
123        this.distancePerDegree = earthCircumference / 360;
124    }
125
126    public double getEarthCircumference() {
127        return earthCircumference;
128    }
129
130    public double getEarthRadius() {
131        return earthRadius;
132    }
133
134    public double getDistancePerDegree() {
135        return distancePerDegree;
136    }
137
138    public abstract double toMiles(double distance);
139
140    public abstract double toKilometers(double distance);
141
142    public abstract String toString(double distance);
143
144    public static DistanceUnit fromString(String unit) {
145        if ("km".equals(unit)) {
146            return KILOMETERS;
147        } else if ("mi".equals(unit)) {
148            return MILES;
149        } else if ("miles".equals(unit)) {
150            return MILES;
151        }
152        throw new ElasticSearchIllegalArgumentException("No distance unit match [" + unit + "]");
153    }
154
155    public static void writeDistanceUnit(StreamOutput out, DistanceUnit unit) throws IOException {
156        if (unit == MILES) {
157            out.writeByte((byte) 0);
158        } else if (unit == KILOMETERS) {
159            out.writeByte((byte) 1);
160        }
161    }
162
163    public static DistanceUnit readDistanceUnit(StreamInput in) throws IOException {
164        byte b = in.readByte();
165        if (b == 0) {
166            return MILES;
167        } else if (b == 1) {
168            return KILOMETERS;
169        } else {
170            throw new ElasticSearchIllegalArgumentException("No type for distance unit matching [" + b + "]");
171        }
172    }
173}
