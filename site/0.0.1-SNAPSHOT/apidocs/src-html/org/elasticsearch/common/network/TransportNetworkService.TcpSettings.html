065    /**
066     * A custom name resolver can support custom lookup keys (my_net_key:ipv4) and also change
067     * the default inet address used in case no settings is provided.
068     */
069    public static interface CustomNameResolver {
070        /**
071         * Resolves the default value if possible. If not, return <tt>null</tt>.
072         */
073        InetAddress resolveDefault();
074
075        /**
076         * Resolves a custom value handling, return <tt>null</tt> if can't handle it.
077         */
078        InetAddress resolveIfPossible(String value);
079    }
080
081    private final List<CustomNameResolver> customNameResolvers = new CopyOnWriteArrayList<CustomNameResolver>();
082
083    public TransportNetworkService(Settings settings) {
084        this.settings = settings;
085        InetSocketTransportAddress.setResolveAddress(settings.getAsBoolean("network.address.serialization.resolve", false));
086    }
087
088    /**
089     * Add a custom name resolver.
090     */
091    public void addCustomNameResolver(CustomNameResolver customNameResolver) {
092        customNameResolvers.add(customNameResolver);
093    }

096    public InetAddress resolveBindHostAddress(String bindHost) throws IOException {
097        return resolveBindHostAddress(bindHost, null);
098    }
099
100    public InetAddress resolveBindHostAddress(String bindHost, String defaultValue2) throws IOException {
101        return resolveInetAddress(bindHost, settings.get(GLOBAL_NETWORK_BINDHOST_SETTING, settings.get(GLOBAL_NETWORK_HOST_SETTING)), defaultValue2);
102    }
103
104    public InetAddress resolvePublishHostAddress(String publishHost) throws IOException {
105        InetAddress address = resolvePublishHostAddress(publishHost, null);
106        // verify that its not a local address
107        if (address == null || address.isAnyLocalAddress()) {
108            address = NetworkUtils.getFirstNonLoopbackAddress(NetworkUtils.StackType.IPv4);
109            if (address == null) {
110                address = NetworkUtils.getFirstNonLoopbackAddress(NetworkUtils.getIpStackType());
111                if (address == null) {
112                    address = NetworkUtils.getLocalAddress();
113                    if (address == null) {
114                        return NetworkUtils.getLocalhost(NetworkUtils.StackType.IPv4);
115                    }
116                }
117            }
118        }
119        return address;
120    }
121
122    public InetAddress resolvePublishHostAddress(String publishHost, String defaultValue2) throws IOException {
123        return resolveInetAddress(publishHost, settings.get(GLOBAL_NETWORK_PUBLISHHOST_SETTING, settings.get(GLOBAL_NETWORK_HOST_SETTING)), defaultValue2);
124    }

126    public InetAddress resolveInetAddress(String host, String defaultValue1, String defaultValue2) throws UnknownHostException, IOException {
127        if (host == null) {
128            host = defaultValue1;
129        }
130        if (host == null) {
131            host = defaultValue2;
132        }
133        if (host == null) {
134            for (CustomNameResolver customNameResolver : customNameResolvers) {
135                InetAddress inetAddress = customNameResolver.resolveDefault();
136                if (inetAddress != null) {
137                    return inetAddress;
138                }
139            }
140            return null;
141        }
142        String origHost = host;
143        if ((host.startsWith("#") && host.endsWith("#")) || (host.startsWith("_") && host.endsWith("_"))) {
144            host = host.substring(1, host.length() - 1);
145
146            for (CustomNameResolver customNameResolver : customNameResolvers) {
147                InetAddress inetAddress = customNameResolver.resolveIfPossible(host);
148                if (inetAddress != null) {
149                    return inetAddress;
150                }
151            }
152
153            if (host.equals("local")) {
154                return NetworkUtils.getLocalAddress();
155            } else if (host.startsWith("non_loopback")) {
156                if (host.toLowerCase().endsWith(":ipv4")) {
157                    return NetworkUtils.getFirstNonLoopbackAddress(NetworkUtils.StackType.IPv4);
158                } else if (host.toLowerCase().endsWith(":ipv6")) {
159                    return NetworkUtils.getFirstNonLoopbackAddress(NetworkUtils.StackType.IPv6);
160                } else {
161                    return NetworkUtils.getFirstNonLoopbackAddress(NetworkUtils.getIpStackType());
162                }
163            } else {
164                NetworkUtils.StackType stackType = NetworkUtils.getIpStackType();
165                if (host.toLowerCase().endsWith(":ipv4")) {
166                    stackType = NetworkUtils.StackType.IPv4;
167                    host = host.substring(0, host.length() - 5);
168                } else if (host.toLowerCase().endsWith(":ipv6")) {
169                    stackType = NetworkUtils.StackType.IPv6;
170                    host = host.substring(0, host.length() - 5);
171                }
172                Collection<NetworkInterface> allInterfs = NetworkUtils.getAllAvailableInterfaces();
173                for (NetworkInterface ni : allInterfs) {
174                    if (!ni.isUp() || ni.isLoopback()) {
175                        continue;
176                    }
177                    if (host.equals(ni.getName()) || host.equals(ni.getDisplayName())) {
178                        return NetworkUtils.getFirstNonLoopbackAddress(ni, stackType);
179                    }
180                }
181            }
182            throw new IOException("Failed to find network interface for [" + origHost + "]");
183        }
184        return InetAddress.getByName(host);
185    }
