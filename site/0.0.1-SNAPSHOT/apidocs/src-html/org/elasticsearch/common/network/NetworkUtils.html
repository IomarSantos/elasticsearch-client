092    /**
093     * Returns the first non-loopback address on any interface on the current host.
094     *
095     * @param ip_version Constraint on IP version of address to be returned, 4 or 6
096     */
097    public static InetAddress getFirstNonLoopbackAddress(StackType ip_version) throws SocketException {
098        InetAddress address = null;
099
100        Enumeration intfs = NetworkInterface.getNetworkInterfaces();
101
102        List<NetworkInterface> intfsList = Lists.newArrayList();
103        while (intfs.hasMoreElements()) {
104            intfsList.add((NetworkInterface) intfs.nextElement());
105        }
106
107        // order by index, assuming first ones are more interesting
108        try {
109            final Method getIndexMethod = NetworkInterface.class.getDeclaredMethod("getIndex");
110            getIndexMethod.setAccessible(true);
111
112            Collections.sort(intfsList, new Comparator<NetworkInterface>() {
113                @Override
114                public int compare(NetworkInterface o1, NetworkInterface o2) {
115                    try {
116                        return ((Integer) getIndexMethod.invoke(o1)).intValue() - ((Integer) getIndexMethod.invoke(o2)).intValue();
117                    } catch (Exception e) {
118                        throw new ElasticSearchIllegalStateException("failed to fetch index of network interface");
119                    }
120                }
121            });
122        } catch (Exception e) {
123            // ignore
124        }
125
126        for (NetworkInterface intf : intfsList) {
127            try {
128                if (!intf.isUp() || intf.isLoopback())
129                    continue;
130            } catch (Exception e) {
131                // might happen when calling on a network interface that does not exists
132                continue;
133            }
134            address = getFirstNonLoopbackAddress(intf, ip_version);
135            if (address != null) {
136                return address;
137            }
138        }
139
140        return null;
141    }

144    /**
145     * Returns the first non-loopback address on the given interface on the current host.
146     *
147     * @param intf      the interface to be checked
148     * @param ipVersion Constraint on IP version of address to be returned, 4 or 6
149     */
150    public static InetAddress getFirstNonLoopbackAddress(NetworkInterface intf, StackType ipVersion) throws SocketException {
151        if (intf == null)
152            throw new IllegalArgumentException("Network interface pointer is null");
153
154        for (Enumeration addresses = intf.getInetAddresses(); addresses.hasMoreElements(); ) {
155            InetAddress address = (InetAddress) addresses.nextElement();
156            if (!address.isLoopbackAddress()) {
157                if ((address instanceof Inet4Address && ipVersion == StackType.IPv4) ||
158                        (address instanceof Inet6Address && ipVersion == StackType.IPv6))
159                    return address;
160            }
161        }
162        return null;
163    }

165    /**
166     * A function to check if an interface supports an IP version (i.e has addresses
167     * defined for that IP version).
168     *
169     * @param intf
170     * @return
171     */
172    public static boolean interfaceHasIPAddresses(NetworkInterface intf, StackType ipVersion) throws SocketException, UnknownHostException {
173        boolean supportsVersion = false;
174        if (intf != null) {
175            // get all the InetAddresses defined on the interface
176            Enumeration addresses = intf.getInetAddresses();
177            while (addresses != null && addresses.hasMoreElements()) {
178                // get the next InetAddress for the current interface
179                InetAddress address = (InetAddress) addresses.nextElement();
180
181                // check if we find an address of correct version
182                if ((address instanceof Inet4Address && (ipVersion == StackType.IPv4)) ||
183                        (address instanceof Inet6Address && (ipVersion == StackType.IPv6))) {
184                    supportsVersion = true;
185                    break;
186                }
187            }
188        } else {
189            throw new UnknownHostException("network interface not found");
190        }
191        return supportsVersion;
192    }

194    /**
195     * Tries to determine the type of IP stack from the available interfaces and their addresses and from the
196     * system properties (java.net.preferIPv4Stack and java.net.preferIPv6Addresses)
197     *
198     * @return StackType.IPv4 for an IPv4 only stack, StackYTypeIPv6 for an IPv6 only stack, and StackType.Unknown
199     *         if the type cannot be detected
200     */
201    public static StackType getIpStackType() {
202        boolean isIPv4StackAvailable = isStackAvailable(true);
203        boolean isIPv6StackAvailable = isStackAvailable(false);
204
205        // if only IPv4 stack available
206        if (isIPv4StackAvailable && !isIPv6StackAvailable) {
207            return StackType.IPv4;
208        }
209        // if only IPv6 stack available
210        else if (isIPv6StackAvailable && !isIPv4StackAvailable) {
211            return StackType.IPv6;
212        }
213        // if dual stack
214        else if (isIPv4StackAvailable && isIPv6StackAvailable) {
215            // get the System property which records user preference for a stack on a dual stack machine
216            if (Boolean.getBoolean(IPv4_SETTING)) // has preference over java.net.preferIPv6Addresses
217                return StackType.IPv4;
218            if (Boolean.getBoolean(IPv6_SETTING))
219                return StackType.IPv6;
220            return StackType.IPv6;
221        }
222        return StackType.Unknown;
223    }

226    public static boolean isStackAvailable(boolean ipv4) {
227        Collection<InetAddress> allAddrs = getAllAvailableAddresses();
228        for (InetAddress addr : allAddrs)
229            if (ipv4 && addr instanceof Inet4Address || (!ipv4 && addr instanceof Inet6Address))
230                return true;
231        return false;
232    }

235    /**
236     * Returns all the available interfaces, including first level sub interfaces.
237     */
238    public static List<NetworkInterface> getAllAvailableInterfaces() throws SocketException {
239        List<NetworkInterface> allInterfaces = new ArrayList<NetworkInterface>();
240        for (Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces(); interfaces.hasMoreElements(); ) {
241            NetworkInterface intf = interfaces.nextElement();
242            allInterfaces.add(intf);
243
244            Enumeration<NetworkInterface> subInterfaces = intf.getSubInterfaces();
245            if (subInterfaces != null && subInterfaces.hasMoreElements()) {
246                while (subInterfaces.hasMoreElements()) {
247                    allInterfaces.add(subInterfaces.nextElement());
248                }
249            }
250        }
251        return allInterfaces;
252    }
253
254    public static Collection<InetAddress> getAllAvailableAddresses() {
255        Set<InetAddress> retval = new HashSet<InetAddress>();
256        Enumeration en;
257
258        try {
259            en = NetworkInterface.getNetworkInterfaces();
260            if (en == null)
261                return retval;
262            while (en.hasMoreElements()) {
263                NetworkInterface intf = (NetworkInterface) en.nextElement();
264                Enumeration<InetAddress> addrs = intf.getInetAddresses();
265                while (addrs.hasMoreElements())
266                    retval.add(addrs.nextElement());
267            }
268        } catch (SocketException e) {
269            logger.warn("Failed to derive all available interfaces", e);
270        }
271
272        return retval;
273    }
