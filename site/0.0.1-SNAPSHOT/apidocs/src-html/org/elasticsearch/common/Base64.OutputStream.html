005/**
006 * <p>Encodes and decodes to and from Base64 notation.</p>
007 * <p>Homepage: <a href="http://iharder.net/base64">http://iharder.net/base64</a>.</p>
008 * <p/>
009 * <p>Example:</p>
010 * <p/>
011 * <code>String encoded = Base64.encode( myByteArray );</code>
012 * <br />
013 * <code>byte[] myByteArray = Base64.decode( encoded );</code>
014 * <p/>
015 * <p>The <tt>options</tt> parameter, which appears in a few places, is used to pass
016 * several pieces of information to the encoder. In the "higher level" methods such as
017 * encodeBytes( bytes, options ) the options parameter can be used to indicate such
018 * things as first gzipping the bytes before encoding them, not inserting linefeeds,
019 * and encoding using the URL-safe and Ordered dialects.</p>
020 * <p/>
021 * <p>Note, according to <a href="http://www.faqs.org/rfcs/rfc3548.html">RFC3548</a>,
022 * Section 2.1, implementations should not add line feeds unless explicitly told
023 * to do so. I've got Base64 set to this behavior now, although earlier versions
024 * broke lines by default.</p>
025 * <p/>
026 * <p>The constants defined in Base64 can be OR-ed together to combine options, so you
027 * might make a call like this:</p>
028 * <p/>
029 * <code>String encoded = Base64.encodeBytes( mybytes, Base64.GZIP | Base64.DO_BREAK_LINES );</code>
030 * <p>to compress the data before encoding it and then making the output have newline characters.</p>
031 * <p>Also...</p>
032 * <code>String encoded = Base64.encodeBytes( crazyString.getBytes() );</code>
033 * <p/>
034 * <p/>
035 * <p/>
036 * <p>

045 * something that has bad characters in it.</li>

047 * string ended in the last column; the buffer was not properly shrunk and
048 * contained an extra (null) byte that made it into the string.</li>

050 * was wrong for files of size 31, 34, and 37 bytes.</li>

052 * the Base64.OutputStream closed the Base64 encoding (by padding with equals
053 * signs) too soon. Also added an option to suppress the automatic decoding
054 * of gzipped streams. Also added experimental support for specifying a
055 * class loader when using the

494    /**
495     * <p>Encodes up to three bytes of the array <var>source</var>
496     * and writes the resulting four Base64 bytes to <var>destination</var>.
497     * The source and destination arrays can be manipulated
498     * anywhere along their length by specifying
499     * <var>srcOffset</var> and <var>destOffset</var>.
500     * This method does not check to make sure your arrays
501     * are large enough to accomodate <var>srcOffset</var> + 3 for
502     * the <var>source</var> array or <var>destOffset</var> + 4 for
503     * the <var>destination</var> array.
504     * The actual number of significant bytes in your array is
505     * given by <var>numSigBytes</var>.</p>
506     * <p>This is the lowest level of the encoding methods with
507     * all possible parameters.</p>
508     *
509     * @param source      the array to convert
510     * @param srcOffset   the index where conversion begins
511     * @param numSigBytes the number of significant bytes in your array
512     * @param destination the array to hold the conversion
513     * @param destOffset  the index where output will be put
514     * @return the <var>destination</var> array
515     * @since 1.3
516     */
517    private static byte[] encode3to4(
518            byte[] source, int srcOffset, int numSigBytes,
519            byte[] destination, int destOffset, int options) {
520
521        byte[] ALPHABET = getAlphabet(options);
522
523        //           1         2         3
524        // 01234567890123456789012345678901 Bit position
525        // --------000000001111111122222222 Array position from threeBytes
526        // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
527        //          >>18  >>12  >> 6  >> 0  Right shift necessary
528        //                0x3f  0x3f  0x3f  Additional AND
529
530        // Create buffer with zero-padding if there are only one or two
531        // significant bytes passed in the array.
532        // We have to shift left 24 in order to flush out the 1's that appear
533        // when Java treats a value as negative that is cast from a byte to an int.
534        int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0)
535                | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
536                | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
537
538        switch (numSigBytes) {
539            case 3:
540                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
541                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
542                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
543                destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
544                return destination;
545
546            case 2:
547                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
548                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
549                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
550                destination[destOffset + 3] = EQUALS_SIGN;
551                return destination;
552
553            case 1:
554                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
555                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
556                destination[destOffset + 2] = EQUALS_SIGN;
557                destination[destOffset + 3] = EQUALS_SIGN;
558                return destination;
559
560            default:
561                return destination;
562        }   // end switch
563    }   // end encode3to4

645     * cannot be serialized or there is another error,
646     * the method will throw an java.io.IOException. <b>This is new to v2.3!</b>
647     * In earlier versions, it just returned a null value, but
648     * in retrospect that's a pretty poor way to handle it.</p>
649     * <p/>
650     * The object is not GZip-compressed before being encoded.
651     * <p/>
652     * Example options:<pre>
653     *   GZIP: gzip-compresses object before encoding it.
654     *   DO_BREAK_LINES: break lines at 76 characters
655     * </pre>
656     * <p/>
657     * Example: <code>encodeObject( myObj, Base64.GZIP )</code> or
658     * <p/>
659     * Example: <code>encodeObject( myObj, Base64.GZIP | Base64.DO_BREAK_LINES )</code>
660     *
661     * @param serializableObject The object to encode
662     * @param options            Specified options
663     * @return The Base64-encoded object
664     * @throws java.io.IOException if there is an error
665     * @see Base64#GZIP
666     * @see Base64#DO_BREAK_LINES
667     * @since 2.0
668     */
669    public static String encodeObject(java.io.Serializable serializableObject, int options)
670            throws java.io.IOException {
671
672        if (serializableObject == null) {
673            throw new NullPointerException("Cannot serialize a null object.");
674        }   // end if: null
675
676        // Streams
677        java.io.ByteArrayOutputStream baos = null;
678        java.io.OutputStream b64os = null;
679        java.util.zip.GZIPOutputStream gzos = null;
680        java.io.ObjectOutputStream oos = null;
681
682
683        try {
684            // ObjectOutputStream -> (GZIP) -> Base64 -> ByteArrayOutputStream
685            baos = new java.io.ByteArrayOutputStream();
686            b64os = new Base64.OutputStream(baos, ENCODE | options);
687            if ((options & GZIP) != 0) {
688                // Gzip
689                gzos = new java.util.zip.GZIPOutputStream(b64os);
690                oos = new java.io.ObjectOutputStream(gzos);
691            } else {
692                // Not gzipped
693                oos = new java.io.ObjectOutputStream(b64os);
694            }
695            oos.writeObject(serializableObject);
696        }   // end try
697        catch (java.io.IOException e) {
698            // Catch it and then throw it immediately so that
699            // the finally{} block is called for cleanup.
700            throw e;
701        }   // end catch
702        finally {
703            try {
704                oos.close();
705            } catch (Exception e) {
706            }
707            try {
708                gzos.close();
709            } catch (Exception e) {
710            }
711            try {
712                b64os.close();
713            } catch (Exception e) {
714            }
715            try {
716                baos.close();
717            } catch (Exception e) {
718            }
719        }   // end finally
720
721        // Return value according to relevant encoding.
722        return new String(baos.toByteArray(), PREFERRED_ENCODING);
723
724    }   // end encode

877    /**
878     * Similar to {@link #encodeBytes(byte[], int, int, int)} but returns
879     * a byte array instead of instantiating a String. This is more efficient
880     * if you're working with I/O streams and have large data sets to encode.
881     *
882     * @param source  The data to convert
883     * @param off     Offset in array where conversion should begin
884     * @param len     Length of data to convert
885     * @param options Specified options
886     * @return The Base64-encoded data as a String
887     * @throws java.io.IOException      if there is an error
888     * @throws NullPointerException     if source array is null
889     * @throws IllegalArgumentException if source array, offset, or length are invalid
890     * @see Base64#GZIP
891     * @see Base64#DO_BREAK_LINES
892     * @since 2.3.1
893     */
894    public static byte[] encodeBytesToBytes(byte[] source, int off, int len, int options) throws java.io.IOException {
895
896        if (source == null) {
897            throw new NullPointerException("Cannot serialize a null array.");
898        }   // end if: null
899
900        if (off < 0) {
901            throw new IllegalArgumentException("Cannot have negative offset: " + off);
902        }   // end if: off < 0
903
904        if (len < 0) {
905            throw new IllegalArgumentException("Cannot have length offset: " + len);
906        }   // end if: len < 0
907
908        if (off + len > source.length) {
909            throw new IllegalArgumentException(
910                    String.format("Cannot have offset of %d and length of %d with array of length %d", off, len, source.length));
911        }   // end if: off < 0
912
913
914        // Compress?
915        if ((options & GZIP) != 0) {
916            java.io.ByteArrayOutputStream baos = null;
917            java.util.zip.GZIPOutputStream gzos = null;
918            Base64.OutputStream b64os = null;
919
920            try {
921                // GZip -> Base64 -> ByteArray
922                baos = new java.io.ByteArrayOutputStream();
923                b64os = new Base64.OutputStream(baos, ENCODE | options);
924                gzos = new java.util.zip.GZIPOutputStream(b64os);
925
926                gzos.write(source, off, len);
927                gzos.close();
928            }   // end try
929            catch (java.io.IOException e) {
930                // Catch it and then throw it immediately so that
931                // the finally{} block is called for cleanup.
932                throw e;
933            }   // end catch
934            finally {
935                try {
936                    gzos.close();
937                } catch (Exception e) {
938                }
939                try {
940                    b64os.close();
941                } catch (Exception e) {
942                }
943                try {
944                    baos.close();
945                } catch (Exception e) {
946                }
947            }   // end finally
948
949            return baos.toByteArray();
950        }   // end if: compress
951
952        // Else, don't compress. Better not to use streams at all then.
953        else {
954            boolean breakLines = (options & DO_BREAK_LINES) != 0;
955
956            //int    len43   = len * 4 / 3;
957            //byte[] outBuff = new byte[   ( len43 )                      // Main 4:3
958            //                           + ( (len % 3) > 0 ? 4 : 0 )      // Account for padding
959            //                           + (breakLines ? ( len43 / MAX_LINE_LENGTH ) : 0) ]; // New lines
960            // Try to determine more precisely how big the array needs to be.
961            // If we get it right, we don't have to do an array copy, and
962            // we save a bunch of memory.
963            int encLen = (len / 3) * 4 + (len % 3 > 0 ? 4 : 0); // Bytes needed for actual encoding
964            if (breakLines) {
965                encLen += encLen / MAX_LINE_LENGTH; // Plus extra newline characters
966            }
967            byte[] outBuff = new byte[encLen];
968
969
970            int d = 0;
971            int e = 0;
972            int len2 = len - 2;
973            int lineLength = 0;
974            for (; d < len2; d += 3, e += 4) {
975                encode3to4(source, d + off, 3, outBuff, e, options);
976
977                lineLength += 4;
978                if (breakLines && lineLength >= MAX_LINE_LENGTH) {
979                    outBuff[e + 4] = NEW_LINE;
980                    e++;
981                    lineLength = 0;
982                }   // end if: end of line
983            }   // en dfor: each piece of array

1156            throws java.io.IOException {
1157
1158        // Lots of error checking and exception throwing
1159        if (source == null) {
1160            throw new NullPointerException("Cannot decode null source array.");
1161        }   // end if
1162        if (off < 0 || off + len > source.length) {
1163            throw new IllegalArgumentException(String.format(
1164                    "Source array with length %d cannot have offset of %d and process %d bytes.", source.length, off, len));
1165        }   // end if
1166
1167        if (len == 0) {
1168            return new byte[0];
1169        } else if (len < 4) {
1170            throw new IllegalArgumentException(
1171                    "Base64-encoded string must have at least four characters, but length specified was " + len);
1172        }   // end if

1174        byte[] DECODABET = getDecodabet(options);
1175
1176        int len34 = len * 3 / 4;       // Estimate on array size
1177        byte[] outBuff = new byte[len34]; // Upper limit on size of output
1178        int outBuffPosn = 0;             // Keep track of where we're writing
1179
1180        byte[] b4 = new byte[4];     // Four byte buffer from source, eliminating white space
1181        int b4Posn = 0;               // Keep track of four byte input buffer
1182        int i = 0;               // Source array counter

1190            // Note the values such as -5 and -9 in the

1653    /**
1654     * A {@link Base64.InputStream} will read data from another
1655     * <tt>java.io.InputStream</tt>, given in the constructor,
1656     * and encode/decode to/from Base64 notation on the fly.
1657     *
1658     * @see Base64
1659     * @since 1.3
1660     */
1661    public static class InputStream extends java.io.FilterInputStream {
1662
1663        private boolean encode;         // Encoding or decoding
1664        private int position;       // Current position in the buffer
1665        private byte[] buffer;         // Small buffer holding converted data
1666        private int bufferLength;   // Length of buffer (3 or 4)
1667        private int numSigBytes;    // Number of meaningful bytes in the buffer
1668        private int lineLength;
1669        private boolean breakLines;     // Break lines at less than 80 characters
1670        private int options;        // Record options used to create the stream.
1671        private byte[] decodabet;      // Local copies to avoid extra method calls
1672
1673
1674        /**
1675         * Constructs a {@link Base64.InputStream} in DECODE mode.
1676         *
1677         * @param in the <tt>java.io.InputStream</tt> from which to read data.
1678         * @since 1.3
1679         */
1680        public InputStream(java.io.InputStream in) {
1681            this(in, DECODE);
1682        }   // end constructor
1683
1684
1685        /**
1686         * Constructs a {@link Base64.InputStream} in
1687         * either ENCODE or DECODE mode.
1688         * <p/>
1689         * Valid options:<pre>
1690         *   ENCODE or DECODE: Encode or Decode as data is read.
1691         *   DO_BREAK_LINES: break lines at 76 characters
1692         *     (only meaningful when encoding)</i>
1693         * </pre>
1694         * <p/>
1695         * Example: <code>new Base64.InputStream( in, Base64.DECODE )</code>
1696         *
1697         * @param in      the <tt>java.io.InputStream</tt> from which to read data.
1698         * @param options Specified options
1699         * @see Base64#ENCODE
1700         * @see Base64#DECODE
1701         * @see Base64#DO_BREAK_LINES
1702         * @since 2.0
1703         */
1704        public InputStream(java.io.InputStream in, int options) {
1705
1706            super(in);
1707            this.options = options; // Record for later
1708            this.breakLines = (options & DO_BREAK_LINES) > 0;
1709            this.encode = (options & ENCODE) > 0;
1710            this.bufferLength = encode ? 4 : 3;
1711            this.buffer = new byte[bufferLength];
1712            this.position = -1;
1713            this.lineLength = 0;
1714            this.decodabet = getDecodabet(options);
1715        }   // end constructor
1716
1717        /**
1718         * Reads enough of the input stream to convert
1719         * to/from Base64 and returns the next byte.
1720         *
1721         * @return next byte
1722         * @since 1.3
1723         */
1724        @Override
1725        public int read() throws java.io.IOException {
1726
1727            // Do we need to get data?
1728            if (position < 0) {
1729                if (encode) {
1730                    byte[] b3 = new byte[3];
1731                    int numBinaryBytes = 0;
1732                    for (int i = 0; i < 3; i++) {
1733                        int b = in.read();
1734
1735                        // If end of stream, b is -1.
1736                        if (b >= 0) {
1737                            b3[i] = (byte) b;
1738                            numBinaryBytes++;
1739                        } else {
1740                            break; // out of for loop
1741                        }   // end else: end of stream
1742
1743                    }   // end for: each needed input byte
1744
1745                    if (numBinaryBytes > 0) {
1746                        encode3to4(b3, 0, numBinaryBytes, buffer, 0, options);
1747                        position = 0;
1748                        numSigBytes = 4;
1749                    }   // end if: got data
1750                    else {
1751                        return -1;  // Must be end of stream
1752                    }   // end else
1753                }   // end if: encoding
1754
1755                // Else decoding
1756                else {
1757                    byte[] b4 = new byte[4];
1758                    int i = 0;
1759                    for (i = 0; i < 4; i++) {
1760                        // Read four "meaningful" bytes:
1761                        int b = 0;
1762                        do {
1763                            b = in.read();
1764                        }
1765                        while (b >= 0 && decodabet[b & 0x7f] <= WHITE_SPACE_ENC);
1766
1767                        if (b < 0) {
1768                            break; // Reads a -1 if end of stream
1769                        }   // end if: end of stream
1770
1771                        b4[i] = (byte) b;
1772                    }   // end for: each needed input byte
1773
1774                    if (i == 4) {
1775                        numSigBytes = decode4to3(b4, 0, buffer, 0, options);
1776                        position = 0;
1777                    }   // end if: got four characters
1778                    else if (i == 0) {
1779                        return -1;
1780                    }   // end else if: also padded correctly
1781                    else {
1782                        // Must have broken out from above.
1783                        throw new java.io.IOException("Improperly padded Base64 input.");
1784                    }   // end
1785
1786                }   // end else: decode
1787            }   // end else: get data
1788
1789            // Got data?
1790            if (position >= 0) {
1791                // End of relevant data?
1792                if ( /*!encode &&*/ position >= numSigBytes) {
1793                    return -1;
1794                }   // end if: got data
1795
1796                if (encode && breakLines && lineLength >= MAX_LINE_LENGTH) {
1797                    lineLength = 0;
1798                    return '\n';
1799                }   // end if
1800                else {
1801                    lineLength++;   // This isn't important when decoding
1802                    // but throwing an extra "if" seems
1803                    // just as wasteful.
1804
1805                    int b = buffer[position++];
1806
1807                    if (position >= bufferLength) {
1808                        position = -1;
1809                    }   // end if: end
1810
1811                    return b & 0xFF; // This is how you "cast" a byte that's
1812                    // intended to be unsigned.
1813                }   // end else
1814            }   // end if: position >= 0
1815
1816            // Else error
1817            else {
1818                throw new java.io.IOException("Error in Base64 code reading stream.");
1819            }   // end else
1820        }   // end read
1821
1822
1823        /**
1824         * Calls {@link #read()} repeatedly until the end of stream
1825         * is reached or <var>len</var> bytes are read.
1826         * Returns number of bytes read into array or -1 if
1827         * end of stream is encountered.
1828         *
1829         * @param dest array to hold values
1830         * @param off  offset for array
1831         * @param len  max number of bytes to read into array
1832         * @return bytes read into array or -1 if end of stream is encountered.
1833         * @since 1.3
1834         */
1835        @Override
1836        public int read(byte[] dest, int off, int len)
1837                throws java.io.IOException {
1838            int i;
1839            int b;
1840            for (i = 0; i < len; i++) {
1841                b = read();
1842
1843                if (b >= 0) {
1844                    dest[off + i] = (byte) b;
1845                } else if (i == 0) {
1846                    return -1;
1847                } else {
1848                    break; // Out of 'for' loop
1849                } // Out of 'for' loop
1850            }   // end for: each byte read
1851            return i;
1852        }   // end read
1853
1854    }   // end inner class InputStream

1895         * <p/>
1896         * Valid options:<pre>
1897         *   ENCODE or DECODE: Encode or Decode as data is read.
1898         *   DO_BREAK_LINES: don't break lines at 76 characters
1899         *     (only meaningful when encoding)</i>
1900         * </pre>
1901         * <p/>
1902         * Example: <code>new Base64.OutputStream( out, Base64.ENCODE )</code>
1903         *

1906         * @see Base64#ENCODE
1907         * @see Base64#DECODE
1908         * @see Base64#DO_BREAK_LINES
1909         * @since 1.3
1910         */

1912            super(out);
1913            this.breakLines = (options & DO_BREAK_LINES) != 0;
1914            this.encode = (options & ENCODE) != 0;
1915            this.bufferLength = encode ? 3 : 4;
1916            this.buffer = new byte[bufferLength];
1917            this.position = 0;
1918            this.lineLength = 0;
1919            this.suspendEncoding = false;
1920            this.b4 = new byte[4];
1921            this.options = options;
1922            this.decodabet = getDecodabet(options);

1926        /**
1927         * Writes the byte to the output stream after
1928         * converting to/from Base64 notation.
1929         * When encoding, bytes are buffered three
1930         * at a time before the output stream actually
1931         * gets a write() call.
1932         * When decoding, bytes are buffered four
1933         * at a time.
1934         *
1935         * @param theByte the byte to write
1936         * @since 1.3
1937         */
1938        @Override
1939        public void write(int theByte)
1940                throws java.io.IOException {
1941            // Encoding suspended?
1942            if (suspendEncoding) {
1943                this.out.write(theByte);
1944                return;
1945            }   // end if: supsended
1946
1947            // Encode?
1948            if (encode) {
1949                buffer[position++] = (byte) theByte;
1950                if (position >= bufferLength) { // Enough to encode.
1951
1952                    this.out.write(encode3to4(b4, buffer, bufferLength, options));
1953
1954                    lineLength += 4;
1955                    if (breakLines && lineLength >= MAX_LINE_LENGTH) {
1956                        this.out.write(NEW_LINE);
1957                        lineLength = 0;
1958                    }   // end if: end of line
1959
1960                    position = 0;
1961                }   // end if: enough to output
1962            }   // end if: encoding
1963
1964            // Else, Decoding
1965            else {
1966                // Meaningful Base64 character?
1967                if (decodabet[theByte & 0x7f] > WHITE_SPACE_ENC) {
1968                    buffer[position++] = (byte) theByte;
1969                    if (position >= bufferLength) { // Enough to output.
1970
1971                        int len = Base64.decode4to3(buffer, 0, b4, 0, options);
1972                        out.write(b4, 0, len);
1973                        position = 0;
1974                    }   // end if: enough to output
1975                }   // end if: meaningful base64 character
1976                else if (decodabet[theByte & 0x7f] != WHITE_SPACE_ENC) {
1977                    throw new java.io.IOException("Invalid character in Base64 data.");
1978                }   // end else: not white space either
1979            }   // end else: decoding
1980        }   // end write

1984         * Calls {@link #write(int)} repeatedly until <var>len</var>
1985         * bytes are written.
1986         *
1987         * @param theBytes array from which to read bytes
1988         * @param off      offset for array
1989         * @param len      max number of bytes to read into array
1990         * @since 1.3
1991         */
1992        @Override
1993        public void write(byte[] theBytes, int off, int len)

1995            // Encoding suspended?
1996            if (suspendEncoding) {
1997                this.out.write(theBytes, off, len);
1998                return;
1999            }   // end if: supsended
2000
2001            for (int i = 0; i < len; i++) {
2002                write(theBytes[off + i]);
