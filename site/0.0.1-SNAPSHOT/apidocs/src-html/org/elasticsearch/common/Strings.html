047    public static void tabify(int tabs, String from, StringBuilder to) throws Exception {
048        BufferedReader reader = new BufferedReader(new FastStringReader(from));
049        String line;
050        while ((line = reader.readLine()) != null) {
051            for (int i = 0; i < tabs; i++) {
052                to.append('\t');
053            }
054            to.append(line).append('\n');
055        }
056    }

058    public static void spaceify(int spaces, String from, StringBuilder to) throws Exception {
059        BufferedReader reader = new BufferedReader(new FastStringReader(from));
060        String line;
061        while ((line = reader.readLine()) != null) {
062            for (int i = 0; i < spaces; i++) {
063                to.append(' ');
064            }
065            to.append(line).append('\n');
066        }

069    /**
070     * Splits a backslash escaped string on the separator.
071     * <p/>
072     * Current backslash escaping supported:
073     * <br> \n \t \r \b \f are escaped the same as a Java String
074     * <br> Other characters following a backslash are produced verbatim (\c => c)
075     *
076     * @param s         the string to split
077     * @param separator the separator to split on
078     * @param decode    decode backslash escaping
079     */
080    public static List<String> splitSmart(String s, String separator, boolean decode) {
081        ArrayList<String> lst = new ArrayList<String>(2);
082        StringBuilder sb = new StringBuilder();
083        int pos = 0, end = s.length();
084        while (pos < end) {
085            if (s.startsWith(separator, pos)) {
086                if (sb.length() > 0) {
087                    lst.add(sb.toString());
088                    sb = new StringBuilder();
089                }
090                pos += separator.length();
091                continue;
092            }
093
094            char ch = s.charAt(pos++);
095            if (ch == '\\') {
096                if (!decode) sb.append(ch);
097                if (pos >= end) break;  // ERROR, or let it go?
098                ch = s.charAt(pos++);
099                if (decode) {
100                    switch (ch) {
101                        case 'n':
102                            ch = '\n';
103                            break;
104                        case 't':
105                            ch = '\t';
106                            break;
107                        case 'r':
108                            ch = '\r';
109                            break;
110                        case 'b':
111                            ch = '\b';
112                            break;
113                        case 'f':
114                            ch = '\f';
115                            break;
116                    }
117                }
118            }
119
120            sb.append(ch);
121        }
122
123        if (sb.length() > 0) {
124            lst.add(sb.toString());
125        }
126
127        return lst;
128    }

131    public static List<String> splitWS(String s, boolean decode) {
132        ArrayList<String> lst = new ArrayList<String>(2);
133        StringBuilder sb = new StringBuilder();
134        int pos = 0, end = s.length();
135        while (pos < end) {
136            char ch = s.charAt(pos++);
137            if (Character.isWhitespace(ch)) {
138                if (sb.length() > 0) {
139                    lst.add(sb.toString());
140                    sb = new StringBuilder();
141                }
142                continue;
143            }
144
145            if (ch == '\\') {
146                if (!decode) sb.append(ch);
147                if (pos >= end) break;  // ERROR, or let it go?
148                ch = s.charAt(pos++);
149                if (decode) {
150                    switch (ch) {
151                        case 'n':
152                            ch = '\n';
153                            break;
154                        case 't':
155                            ch = '\t';
156                            break;
157                        case 'r':
158                            ch = '\r';
159                            break;
160                        case 'b':
161                            ch = '\b';
162                            break;
163                        case 'f':
164                            ch = '\f';
165                            break;
166                    }
167                }
168            }
169
170            sb.append(ch);
171        }
172
173        if (sb.length() > 0) {
174            lst.add(sb.toString());
175        }
176
177        return lst;
178    }

215     * Check whether the given CharSequence has actual text.
216     * More specifically, returns <code>true</code> if the string not <code>null</code>,
217     * its length is greater than 0, and it contains at least one non-whitespace character.
218     * <p><pre>
219     * StringUtils.hasText(null) = false
220     * StringUtils.hasText("") = false
221     * StringUtils.hasText(" ") = false
222     * StringUtils.hasText("12345") = true
223     * StringUtils.hasText(" 12345 ") = true
224     * </pre>
225     *
226     * @param str the CharSequence to check (may be <code>null</code>)
227     * @return <code>true</code> if the CharSequence is not <code>null</code>,
228     *         its length is greater than 0, and it does not contain whitespace only
229     * @see java.lang.Character#isWhitespace
230     */
231    public static boolean hasText(CharSequence str) {
232        if (!hasLength(str)) {
233            return false;
234        }
235        int strLen = str.length();
236        for (int i = 0; i < strLen; i++) {
237            if (!Character.isWhitespace(str.charAt(i))) {
238                return true;
239            }
240        }
241        return false;
242    }

299        if (!hasLength(str)) {

301        }

303        while (sb.length() > 0 && Character.isWhitespace(sb.charAt(0))) {

305        }

308        }
309        return sb.toString();

412     *
413     * @param str    the String to check
414     * @param prefix the prefix to look for
415     * @see java.lang.String#startsWith

417    public static boolean startsWithIgnoreCase(String str, String prefix) {
418        if (str == null || prefix == null) {
419            return false;

422            return true;

474    /**
475     * Count the occurrences of the substring in string s.
476     *
477     * @param str string to search in. Return 0 if this is null.
478     * @param sub string to search for. Return 0 if this is null.
479     */
480    public static int countOccurrencesOf(String str, String sub) {
481        if (str == null || sub == null || str.length() == 0 || sub.length() == 0) {
482            return 0;
483        }
484        int count = 0;
485        int pos = 0;
486        int idx;
487        while ((idx = str.indexOf(sub, pos)) != -1) {
488            ++count;
489            pos = idx + sub.length();
490        }
491        return count;
492    }

494    /**
495     * Replace all occurences of a substring within a string with
496     * another string.
497     *
498     * @param inString   String to examine
499     * @param oldPattern String to replace
500     * @param newPattern String to insert
501     * @return a String with the replacements
502     */
503    public static String replace(String inString, String oldPattern, String newPattern) {
504        if (!hasLength(inString) || !hasLength(oldPattern) || newPattern == null) {
505            return inString;
506        }
507        StringBuilder sb = new StringBuilder();
508        int pos = 0; // our position in the old string
509        int index = inString.indexOf(oldPattern);
510        // the index of an occurrence we've found, or -1
511        int patLen = oldPattern.length();
512        while (index >= 0) {
513            sb.append(inString.substring(pos, index));
514            sb.append(newPattern);
515            pos = index + patLen;
516            index = inString.indexOf(oldPattern, pos);
517        }
518        sb.append(inString.substring(pos));
519        // remember to append any characters to the right of a match
520        return sb.toString();
521    }

550                sb.append(c);
551            }
552        }
553        return sb.toString();

629    private static String changeFirstCharacterCase(String str, boolean capitalize) {
630        if (str == null || str.length() == 0) {
631            return str;
632        }
633        StringBuilder sb = new StringBuilder(str.length());
634        if (capitalize) {
635            sb.append(Character.toUpperCase(str.charAt(0)));
636        } else {

638        }
639        sb.append(str.substring(1));
640        return sb.toString();
641    }

711    /**
712     * Apply the given relative path to the given path,
713     * assuming standard Java folder separation (i.e. "/" separators);
714     *
715     * @param path         the path to start from (usually a full file path)
716     * @param relativePath the relative path to apply
717     *                     (relative to the full file path above)
718     * @return the full file path that results from applying the relative path
719     */
720    public static String applyRelativePath(String path, String relativePath) {
721        int separatorIndex = path.lastIndexOf(FOLDER_SEPARATOR);
722        if (separatorIndex != -1) {
723            String newPath = path.substring(0, separatorIndex);
724            if (!relativePath.startsWith(FOLDER_SEPARATOR)) {
725                newPath += FOLDER_SEPARATOR;
726            }
727            return newPath + relativePath;
728        } else {
729            return relativePath;

733    /**
734     * Normalize the path by suppressing sequences like "path/.." and
735     * inner simple dots.
736     * <p>The result is convenient for path comparison. For other uses,
737     * notice that Windows separators ("\") are replaced by simple slashes.
738     *
739     * @param path the original path
740     * @return the normalized path
741     */
742    public static String cleanPath(String path) {
743        if (path == null) {
744            return null;
745        }
746        String pathToUse = replace(path, WINDOWS_FOLDER_SEPARATOR, FOLDER_SEPARATOR);
747
748        // Strip prefix from path to analyze, to not treat it as part of the
749        // first path element. This is necessary to correctly parse paths like
750        // "file:core/../core/io/Resource.class", where the ".." should just
751        // strip the first "core" directory while keeping the "file:" prefix.
752        int prefixIndex = pathToUse.indexOf(":");
753        String prefix = "";
754        if (prefixIndex != -1) {
755            prefix = pathToUse.substring(0, prefixIndex + 1);
756            pathToUse = pathToUse.substring(prefixIndex + 1);
757        }
758        if (pathToUse.startsWith(FOLDER_SEPARATOR)) {
759            prefix = prefix + FOLDER_SEPARATOR;
760            pathToUse = pathToUse.substring(1);
761        }
762
763        String[] pathArray = delimitedListToStringArray(pathToUse, FOLDER_SEPARATOR);
764        List<String> pathElements = new LinkedList<String>();
765        int tops = 0;
766
767        for (int i = pathArray.length - 1; i >= 0; i--) {
768            String element = pathArray[i];
769            if (CURRENT_PATH.equals(element)) {
770                // Points to current directory - drop it.
771            } else if (TOP_PATH.equals(element)) {
772                // Registering top path found.
773                tops++;
774            } else {
775                if (tops > 0) {
776                    // Merging path element with element corresponding to top path.
777                    tops--;
778                } else {
779                    // Normal path element found.
780                    pathElements.add(0, element);
781                }
782            }
783        }
784
785        // Remaining top paths need to be retained.
786        for (int i = 0; i < tops; i++) {
787            pathElements.add(0, TOP_PATH);
788        }
789
790        return prefix + collectionToDelimitedString(pathElements, FOLDER_SEPARATOR);
791    }

798     * @return whether the two paths are equivalent after normalization

804    /**
805     * Parse the given <code>localeString</code> into a {@link Locale}.
806     * <p>This is the inverse operation of {@link Locale#toString Locale's toString}.
807     *
808     * @param localeString the locale string, following <code>Locale's</code>
809     *                     <code>toString()</code> format ("en", "en_UK", etc);
810     *                     also accepts spaces as separators, as an alternative to underscores
811     * @return a corresponding <code>Locale</code> instance
812     */
813    public static Locale parseLocaleString(String localeString) {
814        String[] parts = tokenizeToStringArray(localeString, "_ ", false, false);
815        String language = (parts.length != 0 ? parts[0] : "");
816        String country = (parts.length > 1 ? parts[1] : "");
817        String variant = "";
818        if (parts.length >= 2) {
819            // There is definitely a variant, and it is everything after the country
820            // code sans the separator between the country code and the variant.
821            int endIndexOfCountryCode = localeString.indexOf(country) + country.length();
822            // Strip off any leading '_' and whitespace, what's left is the variant.
823            variant = trimLeadingWhitespace(localeString.substring(endIndexOfCountryCode));
824            if (variant.startsWith("_")) {
825                variant = trimLeadingCharacter(variant, '_');
826            }
827        }
828        return (language.length() > 0 ? new Locale(language, country, variant) : null);
829    }

1003    public static Set<String> splitStringToSet(final String s, final char c) {
1004        final char[] chars = s.toCharArray();
1005        int count = 1;
1006        for (final char x : chars) {
1007            if (x == c) {
1008                count++;
1009            }
1010        }
1011        final THashSet<String> result = new THashSet<String>(count);
1012        final int len = chars.length;
1013        int start = 0;  // starting index in chars of the current substring.
1014        int pos = 0;    // current index in chars.
1015        int i = 0;      // number of the current substring.
1016        for (; pos < len; pos++) {
1017            if (chars[pos] == c) {
1018                int size = pos - start;
1019                if (size > 0) { // only add non empty strings
1020                    result.add(new String(chars, start, size));
1021                }
1022                start = pos + 1;
1023            }
1024        }
1025        int size = pos - start;
1026        if (size > 0) {
1027            result.add(new String(chars, start, size));
1028        }
1029        return result;
1030    }

1033        if (s.length() == 0) {
1034            return Strings.EMPTY_ARRAY;
1035        }
1036        final char[] chars = s.toCharArray();
1037        int count = 1;
1038        for (final char x : chars) {
1039            if (x == c) {
1040                count++;
1041            }
1042        }
1043        final String[] result = new String[count];
1044        final int len = chars.length;
1045        int start = 0;  // starting index in chars of the current substring.
1046        int pos = 0;    // current index in chars.
1047        int i = 0;      // number of the current substring.
1048        for (; pos < len; pos++) {
1049            if (chars[pos] == c) {
1050                int size = pos - start;
1051                if (size > 0) {
1052                    result[i++] = new String(chars, start, size);
1053                }
1054                start = pos + 1;
1055            }
1056        }
1057        int size = pos - start;
1058        if (size > 0) {
1059            result[i++] = new String(chars, start, size);
1060        }
1061        if (i != count) {
1062            // we have empty strings, copy over to a new array
1063            String[] result1 = new String[i];
1064            System.arraycopy(result, 0, result1, 0, i);
1065            return result1;
1066        }
1067        return result;
1068    }

1072     * Does not include the delimiter in the result.
1073     *

1089        String afterDelimiter = toSplit.substring(offset + delimiter.length());
1090        return new String[]{beforeDelimiter, afterDelimiter};

1393    public static String toCamelCase(String value, StringBuilder sb) {
1394        boolean changed = false;
1395        for (int i = 0; i < value.length(); i++) {
1396            char c = value.charAt(i);
1397            if (c == '_') {
1398                if (!changed) {
1399                    if (sb != null) {
1400                        sb.setLength(0);
1401                    } else {
1402                        sb = new StringBuilder();
1403                    }
1404                    // copy it over here
1405                    for (int j = 0; j < i; j++) {
1406                        sb.append(value.charAt(j));
1407                    }
1408                    changed = true;
1409                }
1410                sb.append(Character.toUpperCase(value.charAt(++i)));
1411            } else {
1412                if (changed) {
1413                    sb.append(c);
1414                }
1415            }
1416        }
1417        if (!changed) {
1418            return value;
1419        }
1420        return sb.toString();
1421    }

1428        boolean changed = false;

1437                    }
1438                    // copy it over here

1440                        sb.append(value.charAt(j));
1441                    }
1442                    changed = true;
1443                    if (i == 0) {
1444                        sb.append(Character.toLowerCase(c));

1456                }
