030    /**
031     * Returns an array size >= minTargetSize, generally
032     * over-allocating exponentially to achieve amortized
033     * linear-time cost as the array grows.
034     * <p/>
035     * NOTE: this was originally borrowed from Python 2.4.2
036     * listobject.c sources (attribution in LICENSE.txt), but
037     * has now been substantially changed based on
038     * discussions from java-dev thread with subject "Dynamic
039     * array reallocation algorithms", started on Jan 12
040     * 2010.
041     *
042     * @param minTargetSize   Minimum required value to be returned.
043     * @param bytesPerElement Bytes used by each element of
044     *                        the array.  See constants in {@link RamUsageEstimator}.
045     * @lucene.internal
046     */
047
048    public static int oversize(int minTargetSize, int bytesPerElement) {
049
050        if (minTargetSize < 0) {
051            // catch usage that accidentally overflows int
052            throw new IllegalArgumentException("invalid array size " + minTargetSize);
053        }
054
055        if (minTargetSize == 0) {
056            // wait until at least one element is requested
057            return 0;
058        }
059
060        // asymptotic exponential growth by 1/8th, favors
061        // spending a bit more CPU to not tie up too much wasted
062        // RAM:
063        int extra = minTargetSize >> 3;
064
065        if (extra < 3) {
066            // for very small arrays, where constant overhead of
067            // realloc is presumably relatively high, we grow
068            // faster
069            extra = 3;
070        }
071
072        int newSize = minTargetSize + extra;
073
074        // add 7 to allow for worst case byte alignment addition below:
075        if (newSize + 7 < 0) {
076            // int overflowed -- return max allowed array size
077            return Integer.MAX_VALUE;
078        }
079
080        if (JvmUtils.JRE_IS_64BIT) {
081            // round up to 8 byte alignment in 64bit env
082            switch (bytesPerElement) {
083                case 4:
084                    // round up to multiple of 2
085                    return (newSize + 1) & 0x7ffffffe;
086                case 2:
087                    // round up to multiple of 4
088                    return (newSize + 3) & 0x7ffffffc;
089                case 1:
090                    // round up to multiple of 8
091                    return (newSize + 7) & 0x7ffffff8;
092                case 8:
093                    // no rounding
094                default:
095                    // odd (invalid?) size
096                    return newSize;
097            }
098        } else {
099            // round up to 4 byte alignment in 64bit env
100            switch (bytesPerElement) {
101                case 2:
102                    // round up to multiple of 2
103                    return (newSize + 1) & 0x7ffffffe;
104                case 1:
105                    // round up to multiple of 4
106                    return (newSize + 3) & 0x7ffffffc;
107                case 4:
108                case 8:
109                    // no rounding
110                default:
111                    // odd (invalid?) size
112                    return newSize;
113            }
114        }
115    }

182    static void getChars(int i, int index, byte[] buf) {
183        int q, r;
184        int charPos = index;
185        byte sign = 0;
186
187        if (i < 0) {
188            sign = '-';
189            i = -i;
190        }
191
192        // Generate two digits per iteration
193        while (i >= 65536) {
194            q = i / 100;
195            // really: r = i - (q * 100);
196            r = i - ((q << 6) + (q << 5) + (q << 2));
197            i = q;
198            buf[--charPos] = DigitOnes[r];
199            buf[--charPos] = DigitTens[r];
200        }
201
202        // Fall thru to fast mode for smaller numbers
203        // assert(i <= 65536, i);
204        for (; ; ) {
205            q = (i * 52429) >>> (16 + 3);
206            r = i - ((q << 3) + (q << 1));  // r = i-(q*10) ...
207            buf[--charPos] = digits[r];
208            i = q;
209            if (i == 0) break;
210        }
211        if (sign != 0) {
212            buf[--charPos] = sign;
213        }
214    }

216    public static int atoi(byte[] s)
217            throws NumberFormatException {
218        int result = 0;
219        boolean negative = false;
220        int i = 0, len = s.length;
221        int limit = -Integer.MAX_VALUE;
222        int multmin;
223        int digit;
224
225        if (len > 0) {
226            byte firstChar = s[0];
227            if (firstChar < '0') { // Possible leading "-"
228                if (firstChar == '-') {
229                    negative = true;
230                    limit = Integer.MIN_VALUE;
231                } else
232                    throw new NumberFormatException();
233
234                if (len == 1) // Cannot have lone "-"
235                    throw new NumberFormatException();
236                i++;
237            }
238            multmin = limit / 10;
239            while (i < len) {
240                // Accumulating negatively avoids surprises near MAX_VALUE
241                digit = Character.digit(s[i++], 10);
242                if (digit < 0) {
243                    throw new NumberFormatException();
244                }
245                if (result < multmin) {
246                    throw new NumberFormatException();
247                }
248                result *= 10;
249                if (result < limit + digit) {
250                    throw new NumberFormatException();
251                }
252                result -= digit;
253            }
254        } else {
255            throw new NumberFormatException();
256        }
257        return negative ? result : -result;
258    }

260    public static byte[] ltoa(long i) {
261        if (i == Long.MIN_VALUE)
262            return LONG_MIN_VALUE_BYTES;
263        int size = (i < 0) ? stringSize(-i) + 1 : stringSize(i);
264        byte[] buf = new byte[size];
265        getChars(i, size, buf);
266        return buf;
267    }

269    /**
270     * Places characters representing the integer i into the
271     * character array buf. The characters are placed into
272     * the buffer backwards starting with the least significant
273     * digit at the specified index (exclusive), and working
274     * backwards from there.
275     * <p/>
276     * Will fail if i == Long.MIN_VALUE
277     */
278    static void getChars(long i, int index, byte[] buf) {
279        long q;
280        int r;
281        int charPos = index;
282        byte sign = 0;
283
284        if (i < 0) {
285            sign = '-';
286            i = -i;
287        }
288
289        // Get 2 digits/iteration using longs until quotient fits into an int
290        while (i > Integer.MAX_VALUE) {
291            q = i / 100;
292            // really: r = i - (q * 100);
293            r = (int) (i - ((q << 6) + (q << 5) + (q << 2)));
294            i = q;
295            buf[--charPos] = DigitOnes[r];
296            buf[--charPos] = DigitTens[r];
297        }
298
299        // Get 2 digits/iteration using ints
300        int q2;
301        int i2 = (int) i;
302        while (i2 >= 65536) {
303            q2 = i2 / 100;
304            // really: r = i2 - (q * 100);
305            r = i2 - ((q2 << 6) + (q2 << 5) + (q2 << 2));
306            i2 = q2;
307            buf[--charPos] = DigitOnes[r];
308            buf[--charPos] = DigitTens[r];
309        }
310
311        // Fall thru to fast mode for smaller numbers
312        // assert(i2 <= 65536, i2);
313        for (; ; ) {
314            q2 = (i2 * 52429) >>> (16 + 3);
315            r = i2 - ((q2 << 3) + (q2 << 1));  // r = i2-(q2*10) ...
316            buf[--charPos] = digits[r];
317            i2 = q2;
318            if (i2 == 0) break;
319        }
320        if (sign != 0) {
321            buf[--charPos] = sign;
322        }
323    }

325    // Requires positive x
326
327    static int stringSize(long x) {
328        long p = 10;
329        for (int i = 1; i < 19; i++) {
330            if (x < p)
331                return i;
332            p = 10 * p;
333        }
334        return 19;
335    }
