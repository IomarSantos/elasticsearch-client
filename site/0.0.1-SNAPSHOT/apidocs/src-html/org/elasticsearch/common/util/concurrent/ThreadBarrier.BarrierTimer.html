020package org.elasticsearch.common.util.concurrent;
021
022import java.util.concurrent.BrokenBarrierException;
023import java.util.concurrent.CyclicBarrier;
024import java.util.concurrent.TimeUnit;
025import java.util.concurrent.TimeoutException;
026
027
028/**
029 * A synchronization aid that allows a set of threads to all wait for each other
030 * to reach a common barrier point. Barriers are useful in programs involving a
031 * fixed sized party of threads that must occasionally wait for each other.
032 * <code>ThreadBarrier</code> adds a <i>cause</i> to
033 * {@link BrokenBarrierException} thrown by a {@link #reset()} operation defined
034 * by {@link CyclicBarrier}.
035 * <p/>
036 * <p/>
037 * <b>Sample usage:</b> <br>
038 * <li>Barrier as a synchronization and Exception handling aid </li>
039 * <li>Barrier as a trigger for elapsed notification events </li>
040 * <p/>
041 * <pre>
042 *    class MyTestClass implements RemoteEventListener
043 *    {
044 *      final ThreadBarrier     barrier;
045 *
046 *      class Worker implements Runnable
047 *        {
048 *              public void run()
049 *            {
050 *                      barrier.await();        //wait for all threads to reach run
051 *                      try
052 *                {
053 *                              prepare();
054 *                              barrier.await();        //wait for all threads to prepare
055 *                              process();
056 *                              barrier.await();        //wait for all threads to process
057 *                }
058 *                      catch(Throwable t){
059 *                              log(&quot;Worker thread caught exception&quot;, t);
060 *                              barrier.reset(t);
061 *                }
062 *            }
063 *        }
064 *
065 *      public void testThreads() {
066 *              barrier = new ThreadBarrier(N_THREADS + 1);
067 *              for (int i = 0; i &lt; N; ++i)
068 *           new Thread(new Worker()).start();
069 *
070 *              try{
071 *                      barrier.await();        //wait for all threads to reach run
072 *                      barrier.await();        //wait for all threads to prepare
073 *                      barrier.await();        //wait for all threads to process
074 *            }
075 *              catch(BrokenBarrierException bbe) {
076 *                      Assert.fail(bbe);
077 *            }
078 *       }
079 *
080 *      int actualNotificationCount = 0;
081 *      public synchronized void notify (RemoteEvent event) {
082 *              try{
083 *                      actualNotificationCount++;
084 *                      if (actualNotificationCount == EXPECTED_COUNT)
085 *                              barrier.await();        //signal when all notifications arrive
086 *
087 *                       // too many notifications?
088 *                       Assert.assertFalse(&quot;Exceeded notification count&quot;,
089 *                                                                              actualNotificationCount > EXPECTED_COUNT);
090 *            }
091 *              catch(Throwable t) {
092 *                      log(&quot;Worker thread caught exception&quot;, t);
093 *                      barrier.reset(t);
094 *            }
095 *        }
096 *
097 *      public void testNotify() {
098 *              barrier = new ThreadBarrier(N_LISTENERS + 1);
099 *              registerNotification();
100 *              triggerNotifications();
101 *
102 *              //wait until either all notifications arrive, or
103 *              //until a MAX_TIMEOUT is reached.
104 *              barrier.await(MAX_TIMEOUT);
105 *
106 *              //check if all notifications were accounted for or timed-out
107 *              Assert.assertEquals(&quot;Notification count&quot;,
108 *                                                                      EXPECTED_COUNT, actualNotificationCount);
109 *
110 *              //inspect that the barrier isn't broken
111 *              barrier.inspect(); //throws BrokenBarrierException if broken
112 *        }
113 *    }
114 * </pre>
115 *
116 *
117 */
118public class ThreadBarrier extends CyclicBarrier {
119    /**
120     * The cause of a {@link BrokenBarrierException} and {@link TimeoutException}
121     * thrown from an await() when {@link #reset(Throwable)} was invoked.
122     */
123    private Throwable cause;
124
125    public ThreadBarrier(int parties) {
126        super(parties);
127    }
128
129    public ThreadBarrier(int parties, Runnable barrierAction) {
130        super(parties, barrierAction);
131    }
132
133    @Override
134    public int await() throws InterruptedException, BrokenBarrierException {
135        try {
136            breakIfBroken();
137            return super.await();
138        } catch (BrokenBarrierException bbe) {
139            initCause(bbe);
140            throw bbe;
141        }
142    }
143
144    @Override
145    public int await(long timeout, TimeUnit unit) throws InterruptedException, BrokenBarrierException, TimeoutException {
146        try {
147            breakIfBroken();
148            return super.await(timeout, unit);
149        } catch (BrokenBarrierException bbe) {
150            initCause(bbe);
151            throw bbe;
152        } catch (TimeoutException te) {
153            initCause(te);
154            throw te;
155        }
156    }
157
158    /**
159     * Resets the barrier to its initial state.  If any parties are
160     * currently waiting at the barrier, they will return with a
161     * {@link BrokenBarrierException}. Note that resets <em>after</em>
162     * a breakage has occurred for other reasons can be complicated to
163     * carry out; threads need to re-synchronize in some other way,
164     * and choose one to perform the reset.  It may be preferable to
165     * instead create a new barrier for subsequent use.
166     *
167     * @param cause The cause of the BrokenBarrierException
168     */
169    public synchronized void reset(Throwable cause) {
170        if (!isBroken()) {
171            super.reset();
172        }
173
174        if (this.cause == null) {
175            this.cause = cause;
176        }
177    }
178
179    /**
180     * Queries if this barrier is in a broken state. Note that if
181     * {@link #reset(Throwable)} is invoked the barrier will remain broken, while
182     * {@link #reset()} will reset the barrier to its initial state and
183     * {@link #isBroken()} will return false.
184     *
185     * @return {@code true} if one or more parties broke out of this barrier due
186     *         to interruption or timeout since construction or the last reset,
187     *         or a barrier action failed due to an exception; {@code false}
188     *         otherwise.
189     * @see #inspect()
190     */
191    @Override
192    public synchronized boolean isBroken() {
193        return this.cause != null || super.isBroken();
194    }
195
196    /**
197     * Inspects if the barrier is broken. If for any reason, the barrier
198     * was broken, a {@link BrokenBarrierException} will be thrown. Otherwise,
199     * would return gracefully.
200     *
201     * @throws BrokenBarrierException With a nested broken cause.
202     */
203    public synchronized void inspect() throws BrokenBarrierException {
204        try {
205            breakIfBroken();
206        } catch (BrokenBarrierException bbe) {
207            initCause(bbe);
208            throw bbe;
209        }
210    }
211
212    /**
213     * breaks this barrier if it has been reset or broken for any other reason.
214     * <p/>
215     * Note: This call is not atomic in respect to await/reset calls. A
216     * breakIfBroken() may be context switched to invoke a reset() prior to
217     * await(). This resets the barrier to its initial state - parties not
218     * currently waiting at the barrier will not be accounted for! An await that
219     * wasn't time limited, will block indefinitely.
220     *
221     * @throws BrokenBarrierException an empty BrokenBarrierException.
222     */
223    private synchronized void breakIfBroken()
224            throws BrokenBarrierException {
225        if (isBroken()) {
226            throw new BrokenBarrierException();
227        }
228    }
229
230    /**
231     * Initializes the cause of this throwable to the specified value. The cause
232     * is the throwable that was initialized by {@link #reset(Throwable)}.
233     *
234     * @param t throwable.
235     */
236    private synchronized void initCause(Throwable t) {
237        t.initCause(this.cause);
238    }
239
240    /**
241     * A Barrier action to be used in conjunction with {@link ThreadBarrier} to
242     * measure performance between barrier awaits. This runnable will execute
243     * when the barrier is tripped. Make sure to reset() the timer before next
244     * Measurement.
245     *
246     * @see ThreadBarrier#ThreadBarrier(int, Runnable)
247     *      <p/>
248     *      <B>Usage example:</B><br>
249     *      <pre><code>
250     *                                                                                             BarrierTimer timer = new BarrierTimer();
251     *                                                                                             ThreadBarrier barrier = new ThreadBarrier( nTHREADS + 1, timer );
252     *                                                                                             ..
253     *                                                                                             barrier.await(); // starts timer when all threads trip on await
254     *                                                                                             barrier.await(); // stops  timer when all threads trip on await
255     *                                                                                             ..
256     *                                                                                             long time = timer.getTimeInNanos();
257     *                                                                                             long tpi = time / ((long)nREPEATS * nTHREADS); //throughput per thread iteration
258     *                                                                                             long secs = timer.getTimeInSeconds();        //total runtime in seconds
259     *                                                                                             ..
260     *                                                                                             timer.reset();  // reuse timer
261     *                                                                                           </code></pre>
262     */
263    public static class BarrierTimer implements Runnable {
264        volatile boolean started;
265        volatile long startTime;
266        volatile long endTime;
267
268        public void run() {
269            long t = System.nanoTime();
270            if (!started) {
271                started = true;
272                startTime = t;
273            } else
274                endTime = t;
275        }
276
277        /**
278         * resets (clears) this timer before next execution.
279         */
280        public void reset() {
281            started = false;
282        }
283
284        /**
285         * Returns the elapsed time between two successive barrier executions.
286         *
287         * @return elapsed time in nanoseconds.
288         */
289        public long getTimeInNanos() {
290            return endTime - startTime;
291        }
292
293        /**
294         * Returns the elapsed time between two successive barrier executions.
295         *
296         * @return elapsed time in seconds.
297         */
298        public double getTimeInSeconds() {
299            long time = endTime - startTime;
300            return (time) / 1000000000.0;
301        }
302    }
303}
