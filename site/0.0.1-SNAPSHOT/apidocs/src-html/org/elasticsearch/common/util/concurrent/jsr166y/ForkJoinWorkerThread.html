007package org.elasticsearch.common.util.concurrent.jsr166y;
008
009/**
010 * A thread managed by a {@link ForkJoinPool}, which executes
011 * {@link ForkJoinTask}s.
012 * This class is subclassable solely for the sake of adding
013 * functionality -- there are no overridable methods dealing with
014 * scheduling or execution.  However, you can override initialization
015 * and termination methods surrounding the main task processing loop.
016 * If you do create such a subclass, you will also need to supply a
017 * custom {@link ForkJoinPool.ForkJoinWorkerThreadFactory} to use it
018 * in a {@code ForkJoinPool}.
019 *
020 * @since 1.7
021 * @author Doug Lea
022 */
023public class ForkJoinWorkerThread extends Thread {
024    /*
025     * ForkJoinWorkerThreads are managed by ForkJoinPools and perform
026     * ForkJoinTasks. For explanation, see the internal documentation
027     * of class ForkJoinPool.
028     */
029
030    final ForkJoinPool.WorkQueue workQueue; // Work-stealing mechanics
031    final ForkJoinPool pool;                // the pool this thread works in
032
033    /**
034     * Creates a ForkJoinWorkerThread operating in the given pool.
035     *
036     * @param pool the pool this thread works in
037     * @throws NullPointerException if pool is null
038     */
039    protected ForkJoinWorkerThread(ForkJoinPool pool) {
040        super(pool.nextWorkerName());
041        setDaemon(true);
042        Thread.UncaughtExceptionHandler ueh = pool.ueh;
043        if (ueh != null)
044            setUncaughtExceptionHandler(ueh);
045        this.pool = pool;
046        pool.registerWorker(this.workQueue = new ForkJoinPool.WorkQueue
047                            (pool, this, pool.localMode));
048    }
049
050    /**
051     * Returns the pool hosting this thread.
052     *
053     * @return the pool
054     */
055    public ForkJoinPool getPool() {
056        return pool;
057    }
058
059    /**
060     * Returns the index number of this thread in its pool.  The
061     * returned value ranges from zero to the maximum number of
062     * threads (minus one) that have ever been created in the pool.
063     * This method may be useful for applications that track status or
064     * collect results per-worker rather than per-task.
065     *
066     * @return the index number
067     */
068    public int getPoolIndex() {
069        return workQueue.poolIndex;
070    }
071
072    /**
073     * Initializes internal state after construction but before
074     * processing any tasks. If you override this method, you must
075     * invoke {@code super.onStart()} at the beginning of the method.
076     * Initialization requires care: Most fields must have legal
077     * default values, to ensure that attempted accesses from other
078     * threads work correctly even before this thread starts
079     * processing tasks.
080     */
081    protected void onStart() {
082    }
083
084    /**
085     * Performs cleanup associated with termination of this worker
086     * thread.  If you override this method, you must invoke
087     * {@code super.onTermination} at the end of the overridden method.
088     *
089     * @param exception the exception causing this thread to abort due
090     * to an unrecoverable error, or {@code null} if completed normally
091     */
092    protected void onTermination(Throwable exception) {
093    }
094
095    /**
096     * This method is required to be public, but should never be
097     * called explicitly. It performs the main run loop to execute
098     * {@link ForkJoinTask}s.
099     */
100    public void run() {
101        Throwable exception = null;
102        try {
103            onStart();
104            pool.runWorker(workQueue);
105        } catch (Throwable ex) {
106            exception = ex;
107        } finally {
108            try {
109                onTermination(exception);
110            } catch (Throwable ex) {
111                if (exception == null)
112                    exception = ex;
113            } finally {
114                pool.deregisterWorker(this, exception);
115            }
116        }
117    }
118}
