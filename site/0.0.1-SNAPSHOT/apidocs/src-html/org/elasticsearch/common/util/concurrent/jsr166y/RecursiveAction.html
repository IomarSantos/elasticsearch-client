007package org.elasticsearch.common.util.concurrent.jsr166y;

009/**
010 * A recursive resultless {@link ForkJoinTask}.  This class
011 * establishes conventions to parameterize resultless actions as
012 * {@code Void} {@code ForkJoinTask}s. Because {@code null} is the
013 * only valid value of type {@code Void}, methods such as {@code join}
014 * always return {@code null} upon completion.
015 *
016 * <p><b>Sample Usages.</b> Here is a simple but complete ForkJoin
017 * sort that sorts a given {@code long[]} array:
018 *
019 *  <pre> {@code
020 * static class SortTask extends RecursiveAction {
021 *   final long[] array; final int lo, hi;
022 *   SortTask(long[] array, int lo, int hi) {
023 *     this.array = array; this.lo = lo; this.hi = hi;
024 *   }
025 *   SortTask(long[] array) { this(array, 0, array.length); }
026 *   protected void compute() {
027 *     if (hi - lo < THRESHOLD)
028 *       sortSequentially(lo, hi);
029 *     else {
030 *       int mid = (lo + hi) >>> 1;
031 *       invokeAll(new SortTask(array, lo, mid),
032 *                 new SortTask(array, mid, hi));
033 *       merge(lo, mid, hi);
034 *     }
035 *   }
036 *   // implementation details follow:
037 *   final static int THRESHOLD = 1000;
038 *   void sortSequentially(int lo, int hi) {
039 *     Arrays.sort(array, lo, hi);
040 *   }
041 *   void merge(int lo, int mid, int hi) {
042 *     long[] buf = Arrays.copyOfRange(array, lo, mid);
043 *     for (int i = 0, j = lo, k = mid; i < buf.length; j++)
044 *       array[j] = (k == hi || buf[i] < array[k]) ?
045 *         buf[i++] : array[k++];
046 *   }
047 * }}</pre>
048 *
049 * You could then sort {@code anArray} by creating {@code new
050 * SortTask(anArray)} and invoking it in a ForkJoinPool.  As a more
051 * concrete simple example, the following task increments each element
052 * of an array:
053 *  <pre> {@code
054 * class IncrementTask extends RecursiveAction {
055 *   final long[] array; final int lo, hi;
056 *   IncrementTask(long[] array, int lo, int hi) {
057 *     this.array = array; this.lo = lo; this.hi = hi;
058 *   }
059 *   protected void compute() {
060 *     if (hi - lo < THRESHOLD) {
061 *       for (int i = lo; i < hi; ++i)
062 *         array[i]++;
063 *     }
064 *     else {
065 *       int mid = (lo + hi) >>> 1;
066 *       invokeAll(new IncrementTask(array, lo, mid),
067 *                 new IncrementTask(array, mid, hi));
068 *     }
069 *   }
070 * }}</pre>
071 *
072 * <p>The following example illustrates some refinements and idioms
073 * that may lead to better performance: RecursiveActions need not be
074 * fully recursive, so long as they maintain the basic
075 * divide-and-conquer approach. Here is a class that sums the squares
076 * of each element of a double array, by subdividing out only the
077 * right-hand-sides of repeated divisions by two, and keeping track of
078 * them with a chain of {@code next} references. It uses a dynamic
079 * threshold based on method {@code getSurplusQueuedTaskCount}, but
080 * counterbalances potential excess partitioning by directly
081 * performing leaf actions on unstolen tasks rather than further
082 * subdividing.
083 *
084 *  <pre> {@code
085 * double sumOfSquares(ForkJoinPool pool, double[] array) {
086 *   int n = array.length;
087 *   Applyer a = new Applyer(array, 0, n, null);
088 *   pool.invoke(a);
089 *   return a.result;
090 * }
091 *
092 * class Applyer extends RecursiveAction {
093 *   final double[] array;
094 *   final int lo, hi;
095 *   double result;
096 *   Applyer next; // keeps track of right-hand-side tasks
097 *   Applyer(double[] array, int lo, int hi, Applyer next) {
098 *     this.array = array; this.lo = lo; this.hi = hi;
099 *     this.next = next;
100 *   }
101 *
102 *   double atLeaf(int l, int h) {
103 *     double sum = 0;
104 *     for (int i = l; i < h; ++i) // perform leftmost base step
105 *       sum += array[i] * array[i];
106 *     return sum;
107 *   }
108 *
109 *   protected void compute() {
110 *     int l = lo;
111 *     int h = hi;
112 *     Applyer right = null;
113 *     while (h - l > 1 && getSurplusQueuedTaskCount() <= 3) {
114 *        int mid = (l + h) >>> 1;
115 *        right = new Applyer(array, mid, h, right);
116 *        right.fork();
117 *        h = mid;
118 *     }
119 *     double sum = atLeaf(l, h);
120 *     while (right != null) {
121 *        if (right.tryUnfork()) // directly calculate if not stolen
122 *          sum += right.atLeaf(right.lo, right.hi);
123 *       else {
124 *          right.join();
125 *          sum += right.result;
126 *        }
127 *        right = right.next;
128 *      }
129 *     result = sum;
130 *   }
131 * }}</pre>
132 *
133 * @since 1.7
134 * @author Doug Lea
135 */
136public abstract class RecursiveAction extends ForkJoinTask<Void> {
137    private static final long serialVersionUID = 5232453952276485070L;
138
139    /**
140     * The main computation performed by this task.
141     */
142    protected abstract void compute();
143
144    /**
145     * Always returns {@code null}.
146     *
147     * @return {@code null} always
148     */
149    public final Void getRawResult() { return null; }
150
151    /**
152     * Requires null completion value.
153     */
154    protected final void setRawResult(Void mustBeNull) { }
155
156    /**
157     * Implements execution conventions for RecursiveActions.
158     */
159    protected final boolean exec() {
160        compute();
161        return true;
162    }
163
164}
