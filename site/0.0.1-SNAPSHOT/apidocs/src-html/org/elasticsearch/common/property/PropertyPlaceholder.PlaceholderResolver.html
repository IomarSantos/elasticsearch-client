020package org.elasticsearch.common.property;
021
022import org.elasticsearch.common.Preconditions;
023import org.elasticsearch.common.Strings;
024
025import java.util.HashSet;
026import java.util.Properties;
027import java.util.Set;
028
029/**
030 * Utility class for working with Strings that have placeholder values in them. A placeholder takes the form
031 * <tt>${name}</tt>. Using <tt>PropertyPlaceholder</tt> these placeholders can be substituted for
032 * user-supplied values.
033 * <p/>
034 * <p> Values for substitution can be supplied using a {@link Properties} instance or using a
035 * {@link PlaceholderResolver}.
036 *
037 *
038 */
039public class PropertyPlaceholder {
040
041    private final String placeholderPrefix;
042
043    private final String placeholderSuffix;
044
045    private final boolean ignoreUnresolvablePlaceholders;
046
047    /**
048     * Creates a new <code>PropertyPlaceholderHelper</code> that uses the supplied prefix and suffix. Unresolvable
049     * placeholders are ignored.
050     *
051     * @param placeholderPrefix the prefix that denotes the start of a placeholder.
052     * @param placeholderSuffix the suffix that denotes the end of a placeholder.
053     */
054    public PropertyPlaceholder(String placeholderPrefix, String placeholderSuffix) {
055        this(placeholderPrefix, placeholderSuffix, true);
056    }
057
058    /**
059     * Creates a new <code>PropertyPlaceholderHelper</code> that uses the supplied prefix and suffix.
060     *
061     * @param placeholderPrefix              the prefix that denotes the start of a placeholder.
062     * @param placeholderSuffix              the suffix that denotes the end of a placeholder.
063     * @param ignoreUnresolvablePlaceholders indicates whether unresolvable placeholders should be ignored
064     *                                       (<code>true</code>) or cause an exception (<code>false</code>).
065     */
066    public PropertyPlaceholder(String placeholderPrefix, String placeholderSuffix,
067                               boolean ignoreUnresolvablePlaceholders) {
068        Preconditions.checkNotNull(placeholderPrefix, "Argument 'placeholderPrefix' must not be null.");
069        Preconditions.checkNotNull(placeholderSuffix, "Argument 'placeholderSuffix' must not be null.");
070        this.placeholderPrefix = placeholderPrefix;
071        this.placeholderSuffix = placeholderSuffix;
072        this.ignoreUnresolvablePlaceholders = ignoreUnresolvablePlaceholders;
073    }
074
075    /**
076     * Replaces all placeholders of format <code>${name}</code> with the corresponding property from the supplied {@link
077     * Properties}.
078     *
079     * @param value      the value containing the placeholders to be replaced.
080     * @param properties the <code>Properties</code> to use for replacement.
081     * @return the supplied value with placeholders replaced inline.
082     */
083    public String replacePlaceholders(String value, final Properties properties) {
084        Preconditions.checkNotNull(properties, "Argument 'properties' must not be null.");
085        return replacePlaceholders(value, new PlaceholderResolver() {
086
087            public String resolvePlaceholder(String placeholderName) {
088                return properties.getProperty(placeholderName);
089            }
090        });
091    }
092
093    /**
094     * Replaces all placeholders of format <code>${name}</code> with the value returned from the supplied {@link
095     * PlaceholderResolver}.
096     *
097     * @param value               the value containing the placeholders to be replaced.
098     * @param placeholderResolver the <code>PlaceholderResolver</code> to use for replacement.
099     * @return the supplied value with placeholders replaced inline.
100     */
101    public String replacePlaceholders(String value, PlaceholderResolver placeholderResolver) {
102        Preconditions.checkNotNull(value, "Argument 'value' must not be null.");
103        return parseStringValue(value, placeholderResolver, new HashSet<String>());
104    }
105
106    protected String parseStringValue(String strVal, PlaceholderResolver placeholderResolver,
107                                      Set<String> visitedPlaceholders) {
108        StringBuilder buf = new StringBuilder(strVal);
109
110        int startIndex = strVal.indexOf(this.placeholderPrefix);
111        while (startIndex != -1) {
112            int endIndex = findPlaceholderEndIndex(buf, startIndex);
113            if (endIndex != -1) {
114                String placeholder = buf.substring(startIndex + this.placeholderPrefix.length(), endIndex);
115                if (!visitedPlaceholders.add(placeholder)) {
116                    throw new IllegalArgumentException(
117                            "Circular placeholder reference '" + placeholder + "' in property definitions");
118                }
119                // Recursive invocation, parsing placeholders contained in the placeholder key.
120                placeholder = parseStringValue(placeholder, placeholderResolver, visitedPlaceholders);
121
122                // Now obtain the value for the fully resolved key...
123                int defaultValueIdx = placeholder.indexOf(':');
124                String defaultValue = null;
125                if (defaultValueIdx != -1) {
126                    defaultValue = placeholder.substring(defaultValueIdx + 1);
127                    placeholder = placeholder.substring(0, defaultValueIdx);
128                }
129                String propVal = placeholderResolver.resolvePlaceholder(placeholder);
130                if (propVal == null) {
131                    propVal = defaultValue;
132                }
133                if (propVal != null) {
134                    // Recursive invocation, parsing placeholders contained in the
135                    // previously resolved placeholder value.
136                    propVal = parseStringValue(propVal, placeholderResolver, visitedPlaceholders);
137                    buf.replace(startIndex, endIndex + this.placeholderSuffix.length(), propVal);
138                    startIndex = buf.indexOf(this.placeholderPrefix, startIndex + propVal.length());
139                } else if (this.ignoreUnresolvablePlaceholders) {
140                    // Proceed with unprocessed value.
141                    startIndex = buf.indexOf(this.placeholderPrefix, endIndex + this.placeholderSuffix.length());
142                } else {
143                    throw new IllegalArgumentException("Could not resolve placeholder '" + placeholder + "'");
144                }
145
146                visitedPlaceholders.remove(placeholder);
147            } else {
148                startIndex = -1;
149            }
150        }
151
152        return buf.toString();
153    }
154
155    private int findPlaceholderEndIndex(CharSequence buf, int startIndex) {
156        int index = startIndex + this.placeholderPrefix.length();
157        int withinNestedPlaceholder = 0;
158        while (index < buf.length()) {
159            if (Strings.substringMatch(buf, index, this.placeholderSuffix)) {
160                if (withinNestedPlaceholder > 0) {
161                    withinNestedPlaceholder--;
162                    index = index + this.placeholderPrefix.length() - 1;
163                } else {
164                    return index;
165                }
166            } else if (Strings.substringMatch(buf, index, this.placeholderPrefix)) {
167                withinNestedPlaceholder++;
168                index = index + this.placeholderPrefix.length();
169            } else {
170                index++;
171            }
172        }
173        return -1;
174    }
175
176    /**
177     * Strategy interface used to resolve replacement values for placeholders contained in Strings.
178     *
179     * @see PropertyPlaceholder
180     */
181    public static interface PlaceholderResolver {
182
183        /**
184         * Resolves the supplied placeholder name into the replacement value.
185         *
186         * @param placeholderName the name of the placeholder to resolve.
187         * @return the replacement value or <code>null</code> if no replacement is to be made.
188         */
189        String resolvePlaceholder(String placeholderName);
190    }
191}
