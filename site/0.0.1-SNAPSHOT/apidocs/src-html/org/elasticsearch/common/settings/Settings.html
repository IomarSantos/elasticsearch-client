020package org.elasticsearch.common.settings;
021
022import com.google.common.collect.ImmutableMap;
023import org.elasticsearch.Version;
024import org.elasticsearch.common.Nullable;
025import org.elasticsearch.common.unit.ByteSizeValue;
026import org.elasticsearch.common.unit.SizeValue;
027import org.elasticsearch.common.unit.TimeValue;
028
029import java.util.Map;
030
031/**
032 * Immutable settings allowing to control the configuration.
033 * <p/>
034 * <p>Using {@link ImmutableSettings#settingsBuilder()} in order to create a builder
035 * which in turn can create an immutable implementation of settings.
036 *
037 * @see ImmutableSettings
038 */
039public interface Settings {
040
041    /**
042     * Component settings for a specific component. Returns all the settings for the given class, where the
043     * FQN of the class is used, without the <tt>org.elasticsearch<tt> prefix. If there is no <tt>org.elasticsearch</tt>
044     * prefix, then the prefix used is the first part of the package name (<tt>org</tt> / <tt>com</tt> / ...)
045     */
046    Settings getComponentSettings(Class component);
047
048    /**
049     * Component settings for a specific component. Returns all the settings for the given class, where the
050     * FQN of the class is used, without provided prefix.
051     */
052    Settings getComponentSettings(String prefix, Class component);
053
054    /**
055     * A settings that are filtered (and key is removed) with the specified prefix.
056     */
057    Settings getByPrefix(String prefix);
058
059    /**
060     * The class loader associated with this settings, or {@link org.elasticsearch.common.Classes#getDefaultClassLoader()}
061     * if not set.
062     */
063    ClassLoader getClassLoader();
064
065    /**
066     * The class loader associated with this settings, but only if explicitly set, otherwise <tt>null</tt>.
067     */
068    @Nullable
069    ClassLoader getClassLoaderIfSet();
070
071    /**
072     * The settings as a {@link java.util.Map}.
073     */
074    ImmutableMap<String, String> getAsMap();
075
076    /**
077     * Returns the setting value associated with the setting key.
078     *
079     * @param setting The setting key
080     * @return The setting value, <tt>null</tt> if it does not exists.
081     */
082    String get(String setting);
083
084    /**
085     * Returns the setting value associated with the setting key. If it does not exists,
086     * returns the default value provided.
087     *
088     * @param setting      The setting key
089     * @param defaultValue The value to return if no value is associated with the setting
090     * @return The setting value, or the default value if no value exists
091     */
092    String get(String setting, String defaultValue);
093
094    /**
095     * Returns group settings for the given setting prefix.
096     */
097    Map<String, Settings> getGroups(String settingPrefix) throws SettingsException;
098
099    /**
100     * Returns the setting value (as float) associated with the setting key. If it does not exists,
101     * returns the default value provided.
102     *
103     * @param setting      The setting key
104     * @param defaultValue The value to return if no value is associated with the setting
105     * @return The (float) value, or the default value if no value exists.
106     * @throws SettingsException Failure to parse the setting
107     */
108    Float getAsFloat(String setting, Float defaultValue) throws SettingsException;
109
110    /**
111     * Returns the setting value (as double) associated with the setting key. If it does not exists,
112     * returns the default value provided.
113     *
114     * @param setting      The setting key
115     * @param defaultValue The value to return if no value is associated with the setting
116     * @return The (double) value, or the default value if no value exists.
117     * @throws SettingsException Failure to parse the setting
118     */
119    Double getAsDouble(String setting, Double defaultValue) throws SettingsException;
120
121    /**
122     * Returns the setting value (as int) associated with the setting key. If it does not exists,
123     * returns the default value provided.
124     *
125     * @param setting      The setting key
126     * @param defaultValue The value to return if no value is associated with the setting
127     * @return The (int) value, or the default value if no value exists.
128     * @throws SettingsException Failure to parse the setting
129     */
130    Integer getAsInt(String setting, Integer defaultValue) throws SettingsException;
131
132    /**
133     * Returns the setting value (as long) associated with the setting key. If it does not exists,
134     * returns the default value provided.
135     *
136     * @param setting      The setting key
137     * @param defaultValue The value to return if no value is associated with the setting
138     * @return The (long) value, or the default value if no value exists.
139     * @throws SettingsException Failure to parse the setting
140     */
141    Long getAsLong(String setting, Long defaultValue) throws SettingsException;
142
143    /**
144     * Returns the setting value (as boolean) associated with the setting key. If it does not exists,
145     * returns the default value provided.
146     *
147     * @param setting      The setting key
148     * @param defaultValue The value to return if no value is associated with the setting
149     * @return The (boolean) value, or the default value if no value exists.
150     * @throws SettingsException Failure to parse the setting
151     */
152    Boolean getAsBoolean(String setting, Boolean defaultValue) throws SettingsException;
153
154    /**
155     * Returns the setting value (as time) associated with the setting key. If it does not exists,
156     * returns the default value provided.
157     *
158     * @param setting      The setting key
159     * @param defaultValue The value to return if no value is associated with the setting
160     * @return The (time) value, or the default value if no value exists.
161     * @throws SettingsException Failure to parse the setting
162     * @see TimeValue#parseTimeValue(String, org.elasticsearch.common.unit.TimeValue)
163     */
164    TimeValue getAsTime(String setting, TimeValue defaultValue) throws SettingsException;
165
166    /**
167     * Returns the setting value (as size) associated with the setting key. If it does not exists,
168     * returns the default value provided.
169     *
170     * @param setting      The setting key
171     * @param defaultValue The value to return if no value is associated with the setting
172     * @return The (size) value, or the default value if no value exists.
173     * @throws SettingsException Failure to parse the setting
174     * @see org.elasticsearch.common.unit.ByteSizeValue#parseBytesSizeValue(String, org.elasticsearch.common.unit.ByteSizeValue)
175     */
176    ByteSizeValue getAsBytesSize(String setting, ByteSizeValue defaultValue) throws SettingsException;
177
178    /**
179     * Returns the setting value (as size) associated with the setting key. If it does not exists,
180     * returns the default value provided.
181     *
182     * @param setting      The setting key
183     * @param defaultValue The value to return if no value is associated with the setting
184     * @return The (size) value, or the default value if no value exists.
185     * @throws SettingsException Failure to parse the setting
186     * @see org.elasticsearch.common.unit.ByteSizeValue#parseBytesSizeValue(String, org.elasticsearch.common.unit.ByteSizeValue)
187     */
188    SizeValue getAsSize(String setting, SizeValue defaultValue) throws SettingsException;
189
190    /**
191     * Returns the setting value (as a class) associated with the setting key. If it does not exists,
192     * returns the default class provided.
193     *
194     * @param setting      The setting key
195     * @param defaultClazz The class to return if no value is associated with the setting
196     * @param <T>          The type of the class
197     * @return The class setting value, or the default class provided is no value exists
198     * @throws NoClassSettingsException Failure to load a class
199     */
200    <T> Class<? extends T> getAsClass(String setting, Class<? extends T> defaultClazz) throws NoClassSettingsException;
201
202    /**
203     * Returns the setting value (as a class) associated with the setting key. If the value itself fails to
204     * represent a loadable class, the value will be appended to the <tt>prefixPackage</tt> and suffixed with the
205     * <tt>suffixClassName</tt> and it will try to be loaded with it.
206     *
207     * @param setting         The setting key
208     * @param defaultClazz    The class to return if no value is associated with the setting
209     * @param prefixPackage   The prefix package to prefix the value with if failing to load the class as is
210     * @param suffixClassName The suffix class name to prefix the value with if failing to load the class as is
211     * @param <T>             The type of the class
212     * @return The class represented by the setting value, or the default class provided if no value exists
213     * @throws NoClassSettingsException Failure to load the class
214     */
215    <T> Class<? extends T> getAsClass(String setting, Class<? extends T> defaultClazz, String prefixPackage, String suffixClassName) throws NoClassSettingsException;
216
217    /**
218     * The values associated with a setting prefix as an array. The settings array is in the format of:
219     * <tt>settingPrefix.[index]</tt>.
220     * <p/>
221     * <p>It will also automatically load a comma separated list under the settingPrefix and merge with
222     * the numbered format.
223     *
224     * @param settingPrefix The setting prefix to load the array by
225     * @return The setting array values
226     * @throws SettingsException
227     */
228    String[] getAsArray(String settingPrefix, String[] defaultArray) throws SettingsException;
229
230    /**
231     * The values associated with a setting prefix as an array. The settings array is in the format of:
232     * <tt>settingPrefix.[index]</tt>.
233     * <p/>
234     * <p>It will also automatically load a comma separated list under the settingPrefix and merge with
235     * the numbered format.
236     *
237     * @param settingPrefix The setting prefix to load the array by
238     * @return The setting array values
239     * @throws SettingsException
240     */
241    String[] getAsArray(String settingPrefix) throws SettingsException;
242
243    /**
244     * Retruns a parsed version.
245     */
246    Version getAsVersion(String setting, Version defaultVersion) throws SettingsException;
247
248    /**
249     * A settings builder interface.
250     */
251    interface Builder {
252
253        /**
254         * Builds the settings.
255         */
256        Settings build();
257    }
258}
