020package org.elasticsearch.common.settings.loader;
021
022import org.elasticsearch.common.Nullable;
023
024import java.io.IOException;
025import java.util.List;
026import java.util.Map;
027
028import static com.google.common.collect.Lists.newArrayList;
029import static com.google.common.collect.Maps.newHashMap;
030
031/**
032 * Provides the ability to load settings (in the form of a simple Map) from
033 * the actual source content that represents them.
034 */
035public interface SettingsLoader {
036
037    static class Helper {
038
039        public static Map<String, String> loadNestedFromMap(@Nullable Map map) {
040            Map<String, String> settings = newHashMap();
041            if (map == null) {
042                return settings;
043            }
044            StringBuilder sb = new StringBuilder();
045            List<String> path = newArrayList();
046            serializeMap(settings, sb, path, map);
047            return settings;
048        }
049
050        private static void serializeMap(Map<String, String> settings, StringBuilder sb, List<String> path, Map<Object, Object> map) {
051            for (Map.Entry<Object, Object> entry : map.entrySet()) {
052                if (entry.getValue() instanceof Map) {
053                    path.add((String) entry.getKey());
054                    serializeMap(settings, sb, path, (Map<Object, Object>) entry.getValue());
055                    path.remove(path.size() - 1);
056                } else if (entry.getValue() instanceof List) {
057                    path.add((String) entry.getKey());
058                    serializeList(settings, sb, path, (List) entry.getValue());
059                    path.remove(path.size() - 1);
060                } else {
061                    serializeValue(settings, sb, path, (String) entry.getKey(), entry.getValue());
062                }
063            }
064        }
065
066        private static void serializeList(Map<String, String> settings, StringBuilder sb, List<String> path, List list) {
067            int counter = 0;
068            for (Object listEle : list) {
069                if (listEle instanceof Map) {
070                    path.add(Integer.toString(counter));
071                    serializeMap(settings, sb, path, (Map<Object, Object>) listEle);
072                    path.remove(path.size() - 1);
073                } else if (listEle instanceof List) {
074                    path.add(Integer.toString(counter));
075                    serializeList(settings, sb, path, (List) listEle);
076                    path.remove(path.size() - 1);
077                } else {
078                    serializeValue(settings, sb, path, Integer.toString(counter), listEle);
079                }
080                counter++;
081            }
082        }
083
084        private static void serializeValue(Map<String, String> settings, StringBuilder sb, List<String> path, String name, Object value) {
085            if (value == null) {
086                return;
087            }
088            sb.setLength(0);
089            for (String pathEle : path) {
090                sb.append(pathEle).append('.');
091            }
092            sb.append(name);
093            settings.put(sb.toString(), value.toString());
094        }
095    }
096
097
098    /**
099     * Loads (parses) the settings from a source string.
100     */
101    Map<String, String> load(String source) throws IOException;
102
103    /**
104     * Loads (parses) the settings from a source bytes.
105     */
106    Map<String, String> load(byte[] source) throws IOException;
107}
