020package org.elasticsearch.common.transport;
021
022import gnu.trove.list.array.TIntArrayList;
023
024import java.util.StringTokenizer;
025
026/**
027 *
028 */
029public class PortsRange {
030
031    private final String portRange;
032
033    public PortsRange(String portRange) {
034        this.portRange = portRange;
035    }
036
037    public int[] ports() throws NumberFormatException {
038        final TIntArrayList ports = new TIntArrayList();
039        iterate(new PortCallback() {
040            @Override
041            public boolean onPortNumber(int portNumber) {
042                ports.add(portNumber);
043                return false;
044            }
045        });
046        return ports.toArray(new int[ports.size()]);
047    }
048
049    public boolean iterate(PortCallback callback) throws NumberFormatException {
050        StringTokenizer st = new StringTokenizer(portRange, ",");
051        boolean success = false;
052        while (st.hasMoreTokens() && !success) {
053            String portToken = st.nextToken().trim();
054            int index = portToken.indexOf('-');
055            if (index == -1) {
056                int portNumber = Integer.parseInt(portToken.trim());
057                success = callback.onPortNumber(portNumber);
058                if (success) {
059                    break;
060                }
061            } else {
062                int startPort = Integer.parseInt(portToken.substring(0, index).trim());
063                int endPort = Integer.parseInt(portToken.substring(index + 1).trim());
064                if (endPort < startPort) {
065                    throw new IllegalArgumentException("Start port [" + startPort + "] must be greater than end port [" + endPort + "]");
066                }
067                for (int i = startPort; i <= endPort; i++) {
068                    success = callback.onPortNumber(i);
069                    if (success) {
070                        break;
071                    }
072                }
073            }
074        }
075        return success;
076    }
077
078    public static interface PortCallback {
079        boolean onPortNumber(int portNumber);
080    }
081}
