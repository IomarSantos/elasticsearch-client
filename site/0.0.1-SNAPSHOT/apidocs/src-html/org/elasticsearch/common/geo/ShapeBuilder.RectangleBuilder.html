020package org.elasticsearch.common.geo;
021
022import com.spatial4j.core.shape.Point;
023import com.spatial4j.core.shape.Rectangle;
024import com.spatial4j.core.shape.Shape;
025import com.spatial4j.core.shape.impl.PointImpl;
026import com.spatial4j.core.shape.impl.RectangleImpl;
027import com.spatial4j.core.shape.jts.JtsGeometry;
028import com.spatial4j.core.shape.jts.JtsPoint;
029import com.vividsolutions.jts.geom.*;
030
031import java.util.ArrayList;
032import java.util.List;
033
034/**
035 * Utility class for building {@link Shape} instances like {@link Point},
036 * {@link Rectangle} and Polygons.
037 */
038public class ShapeBuilder {
039
040    private static final GeometryFactory GEOMETRY_FACTORY = new GeometryFactory();
041
042    private ShapeBuilder() {
043    }
044
045    /**
046     * Creates a new {@link Point}
047     *
048     * @param lon Longitude of point
049     * @param lat Latitude of point
050     * @return Point with the latitude and longitude
051     */
052    public static Point newPoint(double lon, double lat) {
053        return new PointImpl(lon, lat, GeoShapeConstants.SPATIAL_CONTEXT);
054    }
055
056    /**
057     * Creates a new {@link RectangleBuilder} to build a {@link Rectangle}
058     *
059     * @return RectangleBuilder instance
060     */
061    public static RectangleBuilder newRectangle() {
062        return new RectangleBuilder();
063    }
064
065    /**
066     * Creates a new {@link PolygonBuilder} to build a Polygon
067     *
068     * @return PolygonBuilder instance
069     */
070    public static PolygonBuilder newPolygon() {
071        return new PolygonBuilder();
072    }
073
074    /**
075     * Converts the given Shape into the JTS {@link Geometry} representation.
076     * If the Shape already uses a Geometry, that is returned.
077     *
078     * @param shape Shape to convert
079     * @return Geometry representation of the Shape
080     */
081    public static Geometry toJTSGeometry(Shape shape) {
082        if (shape instanceof JtsGeometry) {
083            return ((JtsGeometry) shape).getGeom();
084        } else if (shape instanceof JtsPoint) {
085            return ((JtsPoint) shape).getGeom();
086        } else if (shape instanceof Rectangle) {
087            Rectangle rectangle = (Rectangle) shape;
088
089            if (rectangle.getCrossesDateLine()) {
090                throw new IllegalArgumentException("Cannot convert Rectangles that cross the dateline into JTS Geometrys");
091            }
092
093            return newPolygon().point(rectangle.getMinX(), rectangle.getMaxY())
094                    .point(rectangle.getMaxX(), rectangle.getMaxY())
095                    .point(rectangle.getMaxX(), rectangle.getMinY())
096                    .point(rectangle.getMinX(), rectangle.getMinY())
097                    .point(rectangle.getMinX(), rectangle.getMaxY()).toPolygon();
098        } else if (shape instanceof Point) {
099            Point point = (Point) shape;
100            return GEOMETRY_FACTORY.createPoint(new Coordinate(point.getX(), point.getY()));
101        }
102
103        throw new IllegalArgumentException("Shape type [" + shape.getClass().getSimpleName() + "] not supported");
104    }
105
106    /**
107     * Builder for creating a {@link Rectangle} instance
108     */
109    public static class RectangleBuilder {
110
111        private Point topLeft;
112        private Point bottomRight;
113
114        /**
115         * Sets the top left point of the Rectangle
116         *
117         * @param lon Longitude of the top left point
118         * @param lat Latitude of the top left point
119         * @return this
120         */
121        public RectangleBuilder topLeft(double lon, double lat) {
122            this.topLeft = new PointImpl(lon, lat, GeoShapeConstants.SPATIAL_CONTEXT);
123            return this;
124        }
125
126        /**
127         * Sets the bottom right point of the Rectangle
128         *
129         * @param lon Longitude of the bottom right point
130         * @param lat Latitude of the bottom right point
131         * @return this
132         */
133        public RectangleBuilder bottomRight(double lon, double lat) {
134            this.bottomRight = new PointImpl(lon, lat, GeoShapeConstants.SPATIAL_CONTEXT);
135            return this;
136        }
137
138        /**
139         * Builds the {@link Rectangle} instance
140         *
141         * @return Built Rectangle
142         */
143        public Rectangle build() {
144            return new RectangleImpl(topLeft.getX(), bottomRight.getX(), bottomRight.getY(), topLeft.getY(), GeoShapeConstants.SPATIAL_CONTEXT);
145        }
146    }
147
148    /**
149     * Builder for creating a {@link Shape} instance of a Polygon
150     */
151    public static class PolygonBuilder {
152
153        private final List<Point> points = new ArrayList<Point>();
154
155        /**
156         * Adds a point to the Polygon
157         *
158         * @param lon Longitude of the point
159         * @param lat Latitude of the point
160         * @return this
161         */
162        public PolygonBuilder point(double lon, double lat) {
163            points.add(new PointImpl(lon, lat, GeoShapeConstants.SPATIAL_CONTEXT));
164            return this;
165        }
166
167        /**
168         * Builds a {@link Shape} instance representing the polygon
169         *
170         * @return Built polygon
171         */
172        public Shape build() {
173            return new JtsGeometry(toPolygon(), GeoShapeConstants.SPATIAL_CONTEXT, true);
174        }
175
176        /**
177         * Creates the raw {@link Polygon}
178         *
179         * @return Built polygon
180         */
181        public Polygon toPolygon() {
182            Coordinate[] coordinates = new Coordinate[points.size()];
183            for (int i = 0; i < points.size(); i++) {
184                coordinates[i] = new Coordinate(points.get(i).getX(), points.get(i).getY());
185            }
186
187            LinearRing ring = GEOMETRY_FACTORY.createLinearRing(coordinates);
188            return GEOMETRY_FACTORY.createPolygon(ring, null);
189        }
190    }
191}
