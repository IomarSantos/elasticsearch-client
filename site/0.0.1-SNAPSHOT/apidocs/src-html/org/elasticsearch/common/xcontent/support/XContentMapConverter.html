020package org.elasticsearch.common.xcontent.support;
021
022import org.elasticsearch.common.bytes.BytesArray;
023import org.elasticsearch.common.bytes.BytesReference;
024import org.elasticsearch.common.text.Text;
025import org.elasticsearch.common.xcontent.XContentBuilder;
026import org.elasticsearch.common.xcontent.XContentGenerator;
027import org.elasticsearch.common.xcontent.XContentParser;
028
029import java.io.IOException;
030import java.util.*;
031
032/**
033 *
034 */
035public class XContentMapConverter {
036
037    public static interface MapFactory {
038        Map<String, Object> newMap();
039    }
040
041    public static final MapFactory SIMPLE_MAP_FACTORY = new MapFactory() {
042        @Override
043        public Map<String, Object> newMap() {
044            return new HashMap<String, Object>();
045        }
046    };
047
048    public static final MapFactory ORDERED_MAP_FACTORY = new MapFactory() {
049        @Override
050        public Map<String, Object> newMap() {
051            return new LinkedHashMap<String, Object>();
052        }
053    };
054
055    public static Map<String, Object> readMap(XContentParser parser) throws IOException {
056        return readMap(parser, SIMPLE_MAP_FACTORY);
057    }
058
059    public static Map<String, Object> readOrderedMap(XContentParser parser) throws IOException {
060        return readMap(parser, ORDERED_MAP_FACTORY);
061    }
062
063    public static Map<String, Object> readMap(XContentParser parser, MapFactory mapFactory) throws IOException {
064        Map<String, Object> map = mapFactory.newMap();
065        XContentParser.Token t = parser.currentToken();
066        if (t == null) {
067            t = parser.nextToken();
068        }
069        if (t == XContentParser.Token.START_OBJECT) {
070            t = parser.nextToken();
071        }
072        for (; t == XContentParser.Token.FIELD_NAME; t = parser.nextToken()) {
073            // Must point to field name
074            String fieldName = parser.currentName();
075            // And then the value...
076            t = parser.nextToken();
077            Object value = readValue(parser, mapFactory, t);
078            map.put(fieldName, value);
079        }
080        return map;
081    }
082
083    private static List<Object> readList(XContentParser parser, MapFactory mapFactory, XContentParser.Token t) throws IOException {
084        ArrayList<Object> list = new ArrayList<Object>();
085        while ((t = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
086            list.add(readValue(parser, mapFactory, t));
087        }
088        return list;
089    }
090
091    private static Object readValue(XContentParser parser, MapFactory mapFactory, XContentParser.Token t) throws IOException {
092        if (t == XContentParser.Token.VALUE_NULL) {
093            return null;
094        } else if (t == XContentParser.Token.VALUE_STRING) {
095            return parser.text();
096        } else if (t == XContentParser.Token.VALUE_NUMBER) {
097            XContentParser.NumberType numberType = parser.numberType();
098            if (numberType == XContentParser.NumberType.INT) {
099                return parser.intValue();
100            } else if (numberType == XContentParser.NumberType.LONG) {
101                return parser.longValue();
102            } else if (numberType == XContentParser.NumberType.FLOAT) {
103                return parser.floatValue();
104            } else if (numberType == XContentParser.NumberType.DOUBLE) {
105                return parser.doubleValue();
106            }
107        } else if (t == XContentParser.Token.VALUE_BOOLEAN) {
108            return parser.booleanValue();
109        } else if (t == XContentParser.Token.START_OBJECT) {
110            return readMap(parser, mapFactory);
111        } else if (t == XContentParser.Token.START_ARRAY) {
112            return readList(parser, mapFactory, t);
113        } else if (t == XContentParser.Token.VALUE_EMBEDDED_OBJECT) {
114            return parser.binaryValue();
115        }
116        return null;
117    }
118
119    public static void writeMap(XContentGenerator gen, Map<String, Object> map) throws IOException {
120        gen.writeStartObject();
121
122        for (Map.Entry<String, Object> entry : map.entrySet()) {
123            gen.writeFieldName(entry.getKey());
124            Object value = entry.getValue();
125            if (value == null) {
126                gen.writeNull();
127            } else {
128                writeValue(gen, value);
129            }
130        }
131
132        gen.writeEndObject();
133    }
134
135    private static void writeIterable(XContentGenerator gen, Iterable iterable) throws IOException {
136        gen.writeStartArray();
137        for (Object value : iterable) {
138            writeValue(gen, value);
139        }
140        gen.writeEndArray();
141    }
142
143    private static void writeObjectArray(XContentGenerator gen, Object[] array) throws IOException {
144        gen.writeStartArray();
145        for (Object value : array) {
146            writeValue(gen, value);
147        }
148        gen.writeEndArray();
149    }
150
151    private static void writeValue(XContentGenerator gen, Object value) throws IOException {
152        if (value == null) {
153            gen.writeNull();
154            return;
155        }
156        Class type = value.getClass();
157        if (type == String.class) {
158            gen.writeString((String) value);
159        } else if (type == Integer.class) {
160            gen.writeNumber(((Integer) value).intValue());
161        } else if (type == Long.class) {
162            gen.writeNumber(((Long) value).longValue());
163        } else if (type == Float.class) {
164            gen.writeNumber(((Float) value).floatValue());
165        } else if (type == Double.class) {
166            gen.writeNumber(((Double) value).doubleValue());
167        } else if (type == Short.class) {
168            gen.writeNumber(((Short) value).shortValue());
169        } else if (type == Boolean.class) {
170            gen.writeBoolean(((Boolean) value).booleanValue());
171        } else if (value instanceof Map) {
172            writeMap(gen, (Map) value);
173        } else if (value instanceof Iterable) {
174            writeIterable(gen, (Iterable) value);
175        } else if (value instanceof Object[]) {
176            writeObjectArray(gen, (Object[]) value);
177        } else if (type == byte[].class) {
178            gen.writeBinary((byte[]) value);
179        } else if (value instanceof Date) {
180            gen.writeString(XContentBuilder.defaultDatePrinter.print(((Date) value).getTime()));
181        } else if (value instanceof BytesReference) {
182            BytesReference bytes = (BytesReference) value;
183            if (!bytes.hasArray()) {
184                bytes = bytes.toBytesArray();
185            }
186            gen.writeBinary(bytes.array(), bytes.arrayOffset(), bytes.length());
187        } else if (value instanceof Text) {
188            Text text = (Text) value;
189            if (text.hasBytes() && text.bytes().hasArray()) {
190                gen.writeUTF8String(text.bytes().array(), text.bytes().arrayOffset(), text.bytes().length());
191            } else if (text.hasString()) {
192                gen.writeString(text.string());
193            } else {
194                BytesArray bytesArray = text.bytes().toBytesArray();
195                gen.writeUTF8String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length());
196            }
197        } else {
198            gen.writeString(value.toString());
199        }
200    }
201}
