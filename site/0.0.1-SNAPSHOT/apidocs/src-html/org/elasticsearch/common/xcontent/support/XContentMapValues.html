037    /**
038     * Extracts raw values (string, int, and so on) based on the path provided returning all of them
039     * as a single list.
040     */
041    public static List<Object> extractRawValues(String path, Map<String, Object> map) {
042        List<Object> values = Lists.newArrayList();
043        String[] pathElements = Strings.splitStringToArray(path, '.');
044        if (pathElements.length == 0) {
045            return values;
046        }
047        extractRawValues(values, map, pathElements, 0);
048        return values;
049    }
050
051    @SuppressWarnings({"unchecked"})
052    private static void extractRawValues(List values, Map<String, Object> part, String[] pathElements, int index) {
053        if (index == pathElements.length) {
054            return;
055        }
056
057        String key = pathElements[index];
058        Object currentValue = part.get(key);
059        int nextIndex = index + 1;
060        while (currentValue == null && nextIndex != pathElements.length) {
061            key += "." + pathElements[nextIndex];
062            currentValue = part.get(key);
063            nextIndex++;
064        }
065
066        if (currentValue == null) {
067            return;
068        }
069
070        if (currentValue instanceof Map) {
071            extractRawValues(values, (Map<String, Object>) currentValue, pathElements, nextIndex);
072        } else if (currentValue instanceof List) {
073            extractRawValues(values, (List) currentValue, pathElements, nextIndex);
074        } else {
075            values.add(currentValue);
076        }
077    }
078
079    @SuppressWarnings({"unchecked"})
080    private static void extractRawValues(List values, List<Object> part, String[] pathElements, int index) {
081        for (Object value : part) {
082            if (value == null) {
083                continue;
084            }
085            if (value instanceof Map) {
086                extractRawValues(values, (Map<String, Object>) value, pathElements, index);
087            } else if (value instanceof List) {
088                extractRawValues(values, (List) value, pathElements, index);
089            } else {
090                values.add(value);
091            }
092        }
093    }

095    public static Object extractValue(String path, Map<String, Object> map) {
096        String[] pathElements = Strings.splitStringToArray(path, '.');
097        if (pathElements.length == 0) {
098            return null;
099        }
100        return extractValue(pathElements, 0, map);
101    }
102
103    @SuppressWarnings({"unchecked"})
104    private static Object extractValue(String[] pathElements, int index, Object currentValue) {
105        if (index == pathElements.length) {
106            return currentValue;
107        }
108        if (currentValue == null) {
109            return null;
110        }
111        if (currentValue instanceof Map) {
112            Map map = (Map) currentValue;
113            String key = pathElements[index];
114            Object mapValue = map.get(key);
115            int nextIndex = index + 1;
116            while (mapValue == null && nextIndex != pathElements.length) {
117                key += "." + pathElements[nextIndex];
118                mapValue = map.get(key);
119                nextIndex++;
120            }
121            return extractValue(pathElements, nextIndex, mapValue);
122        }
123        if (currentValue instanceof List) {
124            List valueList = (List) currentValue;
125            List newList = new ArrayList(valueList.size());
126            for (Object o : valueList) {
127                Object listValue = extractValue(pathElements, index, o);
128                if (listValue != null) {
129                    newList.add(listValue);
130                }
131            }
132            return newList;
133        }
134        return null;
135    }

137    public static Map<String, Object> filter(Map<String, Object> map, String[] includes, String[] excludes) {
138        Map<String, Object> result = Maps.newHashMap();
139        filter(map, result, includes, excludes, new StringBuilder());
140        return result;
141    }
142
143    private static void filter(Map<String, Object> map, Map<String, Object> into, String[] includes, String[] excludes, StringBuilder sb) {
144        for (Map.Entry<String, Object> entry : map.entrySet()) {
145            String key = entry.getKey();
146            int mark = sb.length();
147            if (sb.length() > 0) {
148                sb.append('.');
149            }
150            sb.append(key);
151            String path = sb.toString();
152            boolean excluded = false;
153            for (String exclude : excludes) {
154                if (Regex.simpleMatch(exclude, path)) {
155                    excluded = true;
156                    break;
157                }
158            }
159            if (excluded) {
160                sb.setLength(mark);
161                continue;
162            }
163            if (includes.length > 0) {
164                boolean atLeastOnOneIncludeMatched = false;
165                for (String include : includes) {
166                    // check for prefix as well, something like: obj1.arr1.*
167                    // note, this does not work well with middle matches, like obj1.*.obj3
168                    if (include.startsWith(path) || Regex.simpleMatch(include, path)) {
169                        atLeastOnOneIncludeMatched = true;
170                        break;
171                    }
172                }
173                if (!atLeastOnOneIncludeMatched) {
174                    sb.setLength(mark);
175                    continue;
176                }
177            }
178
179
180            if (entry.getValue() instanceof Map) {
181                Map<String, Object> innerInto = Maps.newHashMap();
182                filter((Map<String, Object>) entry.getValue(), innerInto, includes, excludes, sb);
183                if (!innerInto.isEmpty()) {
184                    into.put(entry.getKey(), innerInto);
185                }
186            } else if (entry.getValue() instanceof List) {
187                List<Object> list = (List<Object>) entry.getValue();
188                List<Object> innerInto = new ArrayList<Object>(list.size());
189                filter(list, innerInto, includes, excludes, sb);
190                into.put(entry.getKey(), innerInto);
191            } else {
192                into.put(entry.getKey(), entry.getValue());
193            }
194            sb.setLength(mark);
195        }
196    }
197
198    private static void filter(List<Object> from, List<Object> to, String[] includes, String[] excludes, StringBuilder sb) {
199        for (Object o : from) {
200            if (o instanceof Map) {
201                Map<String, Object> innerInto = Maps.newHashMap();
202                filter((Map<String, Object>) o, innerInto, includes, excludes, sb);
203                if (!innerInto.isEmpty()) {
204                    to.add(innerInto);
205                }
206            } else if (o instanceof List) {
207                List<Object> innerInto = new ArrayList<Object>();
208                filter((List<Object>) o, innerInto, includes, excludes, sb);
209            } else {
210                to.add(o);
211            }
212        }
213    }

324    public static boolean nodeBooleanValue(Object node) {
325        if (node instanceof Boolean) {
326            return (Boolean) node;
327        }
328        if (node instanceof Number) {
329            return ((Number) node).intValue() != 0;
330        }
331        String value = node.toString();
332        return !(value.equals("false") || value.equals("0") || value.equals("off"));
333    }

343        if (node instanceof Number) {

345        }
346        return TimeValue.parseTimeValue(node.toString(), null);
