044    public static XContentParser createParser(BytesReference bytes) throws IOException {
045        if (bytes.hasArray()) {
046            return createParser(bytes.array(), bytes.arrayOffset(), bytes.length());
047        }
048        BasicCompressor compressor = BasicCompressorFactory.compressor(bytes);
049        if (compressor != null) {
050            CompressedStreamInput compressedInput = compressor.streamInput(bytes.streamInput());
051            XContentType contentType = XContentFactory.xContentType(compressedInput);
052            compressedInput.resetToBufferStart();
053            return XContentFactory.xContent(contentType).createParser(compressedInput);
054        } else {
055            return XContentFactory.xContent(bytes).createParser(bytes.streamInput());
056        }
057    }

060    public static XContentParser createParser(byte[] data, int offset, int length) throws IOException {
061        BasicCompressor compressor = BasicCompressorFactory.compressor(data, offset, length);
062        if (compressor != null) {
063            CompressedStreamInput compressedInput = compressor.streamInput(new BytesStreamInput(data, offset, length, false));
064            XContentType contentType = XContentFactory.xContentType(compressedInput);
065            compressedInput.resetToBufferStart();
066            return XContentFactory.xContent(contentType).createParser(compressedInput);
067        } else {
068            return XContentFactory.xContent(data, offset, length).createParser(data, offset, length);
069        }
070    }

072    public static Tuple<XContentType, Map<String, Object>> convertToMap(BytesReference bytes, boolean ordered) throws ElasticSearchParseException {
073        if (bytes.hasArray()) {
074            return convertToMap(bytes.array(), bytes.arrayOffset(), bytes.length(), ordered);
075        }
076        try {
077            XContentParser parser;
078            XContentType contentType;
079            BasicCompressor compressor = BasicCompressorFactory.compressor(bytes);
080            if (compressor != null) {
081                CompressedStreamInput compressedStreamInput = compressor.streamInput(bytes.streamInput());
082                contentType = XContentFactory.xContentType(compressedStreamInput);
083                compressedStreamInput.resetToBufferStart();
084                parser = XContentFactory.xContent(contentType).createParser(compressedStreamInput);
085            } else {
086                contentType = XContentFactory.xContentType(bytes);
087                parser = XContentFactory.xContent(contentType).createParser(bytes.streamInput());
088            }
089            if (ordered) {
090                return Tuple.tuple(contentType, parser.mapOrderedAndClose());
091            } else {
092                return Tuple.tuple(contentType, parser.mapAndClose());
093            }
094        } catch (IOException e) {
095            throw new ElasticSearchParseException("Failed to parse content to map", e);
096        }
097    }

131    public static String convertToJson(BytesReference bytes, boolean reformatJson, boolean prettyPrint) throws IOException {
132        if (bytes.hasArray()) {
133            return convertToJson(bytes.array(), bytes.arrayOffset(), bytes.length(), reformatJson, prettyPrint);
134        }
135        XContentType xContentType = XContentFactory.xContentType(bytes);
136        if (xContentType == XContentType.JSON && !reformatJson) {
137            BytesArray bytesArray = bytes.toBytesArray();
138            return new String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length(), Charsets.UTF_8);
139        }
140        XContentParser parser = null;
141        try {
142            parser = XContentFactory.xContent(xContentType).createParser(bytes.streamInput());
143            parser.nextToken();
144            XContentBuilder builder = XContentFactory.jsonBuilder();
145            if (prettyPrint) {
146                builder.prettyPrint();
147            }
148            builder.copyCurrentStructure(parser);
149            return builder.string();
150        } finally {
151            if (parser != null) {
152                parser.close();
153            }
154        }
155    }

183    /**
184     * Updates the provided changes into the source. If the key exists in the changes, it overrides the one in source
185     * unless both are Maps, in which case it recuersively updated it.
186     */
187    public static void update(Map<String, Object> source, Map<String, Object> changes) {
188        for (Map.Entry<String, Object> changesEntry : changes.entrySet()) {
189            if (!source.containsKey(changesEntry.getKey())) {
190                // safe to copy, change does not exist in source
191                source.put(changesEntry.getKey(), changesEntry.getValue());
192            } else {
193                if (source.get(changesEntry.getKey()) instanceof Map && changesEntry.getValue() instanceof Map) {
194                    // recursive merge maps
195                    update((Map<String, Object>) source.get(changesEntry.getKey()), (Map<String, Object>) changesEntry.getValue());
196                } else {
197                    // update the field
198                    source.put(changesEntry.getKey(), changesEntry.getValue());
199                }
200            }
201        }
202    }

204    /**
205     * Merges the defaults provided as the second parameter into the content of the first. Only does recursive merge
206     * for inner maps.
207     */
208    @SuppressWarnings({"unchecked"})
209    public static void mergeDefaults(Map<String, Object> content, Map<String, Object> defaults) {
210        for (Map.Entry<String, Object> defaultEntry : defaults.entrySet()) {
211            if (!content.containsKey(defaultEntry.getKey())) {
212                // copy it over, it does not exists in the content
213                content.put(defaultEntry.getKey(), defaultEntry.getValue());
214            } else {
215                // in the content and in the default, only merge compound ones (maps)
216                if (content.get(defaultEntry.getKey()) instanceof Map && defaultEntry.getValue() instanceof Map) {
217                    mergeDefaults((Map<String, Object>) content.get(defaultEntry.getKey()), (Map<String, Object>) defaultEntry.getValue());
218                } else if (content.get(defaultEntry.getKey()) instanceof List && defaultEntry.getValue() instanceof List) {
219                    List defaultList = (List) defaultEntry.getValue();
220                    List contentList = (List) content.get(defaultEntry.getKey());
221
222                    List mergedList = new ArrayList();
223                    if (allListValuesAreMapsOfOne(defaultList) && allListValuesAreMapsOfOne(contentList)) {
224                        // all are in the form of [ {"key1" : {}}, {"key2" : {}} ], merge based on keys
225                        Map<String, Map<String, Object>> processed = Maps.newLinkedHashMap();
226                        for (Object o : contentList) {
227                            Map<String, Object> map = (Map<String, Object>) o;
228                            Map.Entry<String, Object> entry = map.entrySet().iterator().next();
229                            processed.put(entry.getKey(), map);
230                        }
231                        for (Object o : defaultList) {
232                            Map<String, Object> map = (Map<String, Object>) o;
233                            Map.Entry<String, Object> entry = map.entrySet().iterator().next();
234                            if (processed.containsKey(entry.getKey())) {
235                                mergeDefaults(processed.get(entry.getKey()), map);
236                            }
237                        }
238                        for (Map<String, Object> map : processed.values()) {
239                            mergedList.add(map);
240                        }
241                    } else {
242                        // if both are lists, simply combine them, first the defaults, then the content
243                        // just make sure not to add the same value twice
244                        mergedList.addAll(defaultList);
245                        for (Object o : contentList) {
246                            if (!mergedList.contains(o)) {
247                                mergedList.add(o);
248                            }
249                        }
250                    }
251                    content.put(defaultEntry.getKey(), mergedList);
252                }
253            }
254        }
255    }
256
257    private static boolean allListValuesAreMapsOfOne(List list) {
258        for (Object o : list) {
259            if (!(o instanceof Map)) {
260                return false;
261            }
262            if (((Map) o).size() != 1) {
263                return false;
264            }
265        }
266        return true;
267    }

269    public static void copyCurrentStructure(XContentGenerator generator, XContentParser parser) throws IOException {
270        XContentParser.Token t = parser.currentToken();
271
272        // Let's handle field-name separately first
273        if (t == XContentParser.Token.FIELD_NAME) {
274            generator.writeFieldName(parser.currentName());
275            t = parser.nextToken();
276            // fall-through to copy the associated value
277        }
278
279        switch (t) {
280            case START_ARRAY:
281                generator.writeStartArray();
282                while (parser.nextToken() != XContentParser.Token.END_ARRAY) {
283                    copyCurrentStructure(generator, parser);
284                }
285                generator.writeEndArray();
286                break;
287            case START_OBJECT:
288                generator.writeStartObject();
289                while (parser.nextToken() != XContentParser.Token.END_OBJECT) {
290                    copyCurrentStructure(generator, parser);
291                }
292                generator.writeEndObject();
293                break;
294            default: // others are simple:
295                copyCurrentEvent(generator, parser);
296        }
297    }

299    public static void copyCurrentEvent(XContentGenerator generator, XContentParser parser) throws IOException {
300        switch (parser.currentToken()) {
301            case START_OBJECT:
302                generator.writeStartObject();
303                break;
304            case END_OBJECT:
305                generator.writeEndObject();
306                break;
307            case START_ARRAY:
308                generator.writeStartArray();
309                break;
310            case END_ARRAY:
311                generator.writeEndArray();
312                break;
313            case FIELD_NAME:
314                generator.writeFieldName(parser.currentName());
315                break;
316            case VALUE_STRING:
317                if (parser.hasTextCharacters()) {
318                    generator.writeString(parser.textCharacters(), parser.textOffset(), parser.textLength());
319                } else {
320                    generator.writeString(parser.text());
321                }
322                break;
323            case VALUE_NUMBER:
324                switch (parser.numberType()) {
325                    case INT:
326                        generator.writeNumber(parser.intValue());
327                        break;
328                    case LONG:
329                        generator.writeNumber(parser.longValue());
330                        break;
331                    case FLOAT:
332                        generator.writeNumber(parser.floatValue());
333                        break;
334                    case DOUBLE:
335                        generator.writeNumber(parser.doubleValue());
336                        break;
337                }
338                break;
339            case VALUE_BOOLEAN:
340                generator.writeBoolean(parser.booleanValue());
341                break;
342            case VALUE_NULL:
343                generator.writeNull();
344                break;
345            case VALUE_EMBEDDED_OBJECT:
346                generator.writeBinary(parser.binaryValue());
347        }
348    }
