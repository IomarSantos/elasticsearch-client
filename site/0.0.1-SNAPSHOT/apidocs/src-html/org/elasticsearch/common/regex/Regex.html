020package org.elasticsearch.common.regex;
021
022import org.elasticsearch.ElasticSearchIllegalArgumentException;
023import org.elasticsearch.common.Strings;
024
025import java.util.regex.Pattern;
026
027/**
028 *
029 */
030public class Regex {
031
032    /**
033     * Is the str a simple match pattern.
034     */
035    public static boolean isSimpleMatchPattern(String str) {
036        return str.indexOf('*') != -1;
037    }
038
039    /**
040     * Match a String against the given pattern, supporting the following simple
041     * pattern styles: "xxx*", "*xxx", "*xxx*" and "xxx*yyy" matches (with an
042     * arbitrary number of pattern parts), as well as direct equality.
043     *
044     * @param pattern the pattern to match against
045     * @param str     the String to match
046     * @return whether the String matches the given pattern
047     */
048    public static boolean simpleMatch(String pattern, String str) {
049        if (pattern == null || str == null) {
050            return false;
051        }
052        int firstIndex = pattern.indexOf('*');
053        if (firstIndex == -1) {
054            return pattern.equals(str);
055        }
056        if (firstIndex == 0) {
057            if (pattern.length() == 1) {
058                return true;
059            }
060            int nextIndex = pattern.indexOf('*', firstIndex + 1);
061            if (nextIndex == -1) {
062                return str.endsWith(pattern.substring(1));
063            }
064            String part = pattern.substring(1, nextIndex);
065            int partIndex = str.indexOf(part);
066            while (partIndex != -1) {
067                if (simpleMatch(pattern.substring(nextIndex), str.substring(partIndex + part.length()))) {
068                    return true;
069                }
070                partIndex = str.indexOf(part, partIndex + 1);
071            }
072            return false;
073        }
074        return (str.length() >= firstIndex &&
075                pattern.substring(0, firstIndex).equals(str.substring(0, firstIndex)) &&
076                simpleMatch(pattern.substring(firstIndex), str.substring(firstIndex)));
077    }
078
079    /**
080     * Match a String against the given patterns, supporting the following simple
081     * pattern styles: "xxx*", "*xxx", "*xxx*" and "xxx*yyy" matches (with an
082     * arbitrary number of pattern parts), as well as direct equality.
083     *
084     * @param patterns the patterns to match against
085     * @param str      the String to match
086     * @return whether the String matches any of the given patterns
087     */
088    public static boolean simpleMatch(String[] patterns, String str) {
089        if (patterns != null) {
090            for (String pattern : patterns) {
091                if (simpleMatch(pattern, str)) {
092                    return true;
093                }
094            }
095        }
096        return false;
097    }
098
099    public static Pattern compile(String regex, String flags) {
100        int pFlags = flags == null ? 0 : flagsFromString(flags);
101        return Pattern.compile(regex, pFlags);
102    }
103
104    public static int flagsFromString(String flags) {
105        int pFlags = 0;
106        for (String s : Strings.delimitedListToStringArray(flags, "|")) {
107            if (s.isEmpty()) {
108                continue;
109            }
110            if ("CASE_INSENSITIVE".equalsIgnoreCase(s)) {
111                pFlags |= Pattern.CASE_INSENSITIVE;
112            } else if ("MULTILINE".equalsIgnoreCase(s)) {
113                pFlags |= Pattern.MULTILINE;
114            } else if ("DOTALL".equalsIgnoreCase(s)) {
115                pFlags |= Pattern.DOTALL;
116            } else if ("UNICODE_CASE".equalsIgnoreCase(s)) {
117                pFlags |= Pattern.UNICODE_CASE;
118            } else if ("CANON_EQ".equalsIgnoreCase(s)) {
119                pFlags |= Pattern.CANON_EQ;
120            } else if ("UNIX_LINES".equalsIgnoreCase(s)) {
121                pFlags |= Pattern.UNIX_LINES;
122            } else if ("LITERAL".equalsIgnoreCase(s)) {
123                pFlags |= Pattern.LITERAL;
124            } else if ("COMMENTS".equalsIgnoreCase(s)) {
125                pFlags |= Pattern.COMMENTS;
126            } else {
127                throw new ElasticSearchIllegalArgumentException("Unknown regex flag [" + s + "]");
128            }
129        }
130        return pFlags;
131    }
132
133    public static String flagsToString(int flags) {
134        StringBuilder sb = new StringBuilder();
135        if ((flags & Pattern.CASE_INSENSITIVE) != 0) {
136            sb.append("CASE_INSENSITIVE|");
137        }
138        if ((flags & Pattern.MULTILINE) != 0) {
139            sb.append("MULTILINE|");
140        }
141        if ((flags & Pattern.DOTALL) != 0) {
142            sb.append("DOTALL|");
143        }
144        if ((flags & Pattern.UNICODE_CASE) != 0) {
145            sb.append("UNICODE_CASE|");
146        }
147        if ((flags & Pattern.CANON_EQ) != 0) {
148            sb.append("CANON_EQ|");
149        }
150        if ((flags & Pattern.UNIX_LINES) != 0) {
151            sb.append("UNIX_LINES|");
152        }
153        if ((flags & Pattern.LITERAL) != 0) {
154            sb.append("LITERAL|");
155        }
156        if ((flags & Pattern.COMMENTS) != 0) {
157            sb.append("COMMENTS|");
158        }
159        return sb.toString();
160    }
161}
