034    /**
035     * Parses a joda based pattern, including some named ones (similar to the built in Joda ISO ones).
036     */
037    public static FormatDateTimeFormatter forPattern(String input) {
038        DateTimeFormatter formatter;
039        if ("basicDate".equals(input) || "basic_date".equals(input)) {
040            formatter = ISODateTimeFormat.basicDate();
041        } else if ("basicDateTime".equals(input) || "basic_date_time".equals(input)) {
042            formatter = ISODateTimeFormat.basicDateTime();
043        } else if ("basicDateTimeNoMillis".equals(input) || "basic_date_time_no_millis".equals(input)) {
044            formatter = ISODateTimeFormat.basicDateTimeNoMillis();
045        } else if ("basicOrdinalDate".equals(input) || "basic_ordinal_date".equals(input)) {
046            formatter = ISODateTimeFormat.basicOrdinalDate();
047        } else if ("basicOrdinalDateTime".equals(input) || "basic_ordinal_date_time".equals(input)) {
048            formatter = ISODateTimeFormat.basicOrdinalDateTime();
049        } else if ("basicOrdinalDateTimeNoMillis".equals(input) || "basic_ordinal_date_time_no_millis".equals(input)) {
050            formatter = ISODateTimeFormat.basicOrdinalDateTimeNoMillis();
051        } else if ("basicTime".equals(input) || "basic_time".equals(input)) {
052            formatter = ISODateTimeFormat.basicTime();
053        } else if ("basicTimeNoMillis".equals(input) || "basic_time_no_millis".equals(input)) {
054            formatter = ISODateTimeFormat.basicTimeNoMillis();
055        } else if ("basicTTime".equals(input) || "basic_t_Time".equals(input)) {
056            formatter = ISODateTimeFormat.basicTTime();
057        } else if ("basicTTimeNoMillis".equals(input) || "basic_t_time_no_millis".equals(input)) {
058            formatter = ISODateTimeFormat.basicTTimeNoMillis();
059        } else if ("basicWeekDate".equals(input) || "basic_week_date".equals(input)) {
060            formatter = ISODateTimeFormat.basicWeekDate();
061        } else if ("basicWeekDateTime".equals(input) || "basic_week_date_time".equals(input)) {
062            formatter = ISODateTimeFormat.basicWeekDateTime();
063        } else if ("basicWeekDateTimeNoMillis".equals(input) || "basic_week_date_time_no_millis".equals(input)) {
064            formatter = ISODateTimeFormat.basicWeekDateTimeNoMillis();
065        } else if ("date".equals(input)) {
066            formatter = ISODateTimeFormat.date();
067        } else if ("dateHour".equals(input) || "date_hour".equals(input)) {
068            formatter = ISODateTimeFormat.dateHour();
069        } else if ("dateHourMinute".equals(input) || "date_hour_minute".equals(input)) {
070            formatter = ISODateTimeFormat.dateHourMinute();
071        } else if ("dateHourMinuteSecond".equals(input) || "date_hour_minute_second".equals(input)) {
072            formatter = ISODateTimeFormat.dateHourMinuteSecond();
073        } else if ("dateHourMinuteSecondFraction".equals(input) || "date_hour_minute_second_fraction".equals(input)) {
074            formatter = ISODateTimeFormat.dateHourMinuteSecondFraction();
075        } else if ("dateHourMinuteSecondMillis".equals(input) || "date_hour_minute_second_millis".equals(input)) {
076            formatter = ISODateTimeFormat.dateHourMinuteSecondMillis();
077        } else if ("dateOptionalTime".equals(input) || "date_optional_time".equals(input)) {
078            // in this case, we have a separate parser and printer since the dataOptionalTimeParser can't print
079            return new FormatDateTimeFormatter(input,
080                    ISODateTimeFormat.dateOptionalTimeParser().withZone(DateTimeZone.UTC),
081                    ISODateTimeFormat.dateTime().withZone(DateTimeZone.UTC));
082        } else if ("dateTime".equals(input) || "date_time".equals(input)) {
083            formatter = ISODateTimeFormat.dateTime();
084        } else if ("dateTimeNoMillis".equals(input) || "date_time_no_millis".equals(input)) {
085            formatter = ISODateTimeFormat.dateTimeNoMillis();
086        } else if ("hour".equals(input)) {
087            formatter = ISODateTimeFormat.hour();
088        } else if ("hourMinute".equals(input) || "hour_minute".equals(input)) {
089            formatter = ISODateTimeFormat.hourMinute();
090        } else if ("hourMinuteSecond".equals(input) || "hour_minute_second".equals(input)) {
091            formatter = ISODateTimeFormat.hourMinuteSecond();
092        } else if ("hourMinuteSecondFraction".equals(input) || "hour_minute_second_fraction".equals(input)) {
093            formatter = ISODateTimeFormat.hourMinuteSecondFraction();
094        } else if ("hourMinuteSecondMillis".equals(input) || "hour_minute_second_millis".equals(input)) {
095            formatter = ISODateTimeFormat.hourMinuteSecondMillis();
096        } else if ("ordinalDate".equals(input) || "ordinal_date".equals(input)) {
097            formatter = ISODateTimeFormat.ordinalDate();
098        } else if ("ordinalDateTime".equals(input) || "ordinal_date_time".equals(input)) {
099            formatter = ISODateTimeFormat.ordinalDateTime();
100        } else if ("ordinalDateTimeNoMillis".equals(input) || "ordinal_date_time_no_millis".equals(input)) {
101            formatter = ISODateTimeFormat.ordinalDateTimeNoMillis();
102        } else if ("time".equals(input)) {
103            formatter = ISODateTimeFormat.time();
104        } else if ("tTime".equals(input) || "t_time".equals(input)) {
105            formatter = ISODateTimeFormat.tTime();
106        } else if ("tTimeNoMillis".equals(input) || "t_time_no_millis".equals(input)) {
107            formatter = ISODateTimeFormat.tTimeNoMillis();
108        } else if ("weekDate".equals(input) || "week_date".equals(input)) {
109            formatter = ISODateTimeFormat.weekDate();
110        } else if ("weekDateTime".equals(input) || "week_date_time".equals(input)) {
111            formatter = ISODateTimeFormat.weekDateTime();
112        } else if ("weekyear".equals(input) || "week_year".equals(input)) {
113            formatter = ISODateTimeFormat.weekyear();
114        } else if ("weekyearWeek".equals(input)) {
115            formatter = ISODateTimeFormat.weekyearWeek();
116        } else if ("year".equals(input)) {
117            formatter = ISODateTimeFormat.year();
118        } else if ("yearMonth".equals(input) || "year_month".equals(input)) {
119            formatter = ISODateTimeFormat.yearMonth();
120        } else if ("yearMonthDay".equals(input) || "year_month_day".equals(input)) {
121            formatter = ISODateTimeFormat.yearMonthDay();
122        } else {
123            String[] formats = Strings.delimitedListToStringArray(input, "||");
124            if (formats == null || formats.length == 1) {
125                formatter = DateTimeFormat.forPattern(input);
126            } else {
127                DateTimeParser[] parsers = new DateTimeParser[formats.length];
128                for (int i = 0; i < formats.length; i++) {
129                    parsers[i] = DateTimeFormat.forPattern(formats[i]).withZone(DateTimeZone.UTC).getParser();
130                }
131                DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder()
132                        .append(DateTimeFormat.forPattern(formats[0]).withZone(DateTimeZone.UTC).getPrinter(), parsers);
133                formatter = builder.toFormatter();
134            }
135        }
136        return new FormatDateTimeFormatter(input, formatter.withZone(DateTimeZone.UTC));
137    }

140    public static final DurationFieldType Quarters = new DurationFieldType("quarters") {
141        private static final long serialVersionUID = -8167713675442491871L;
142
143        public DurationField getField(Chronology chronology) {
144            return new ScaledDurationField(chronology.months(), Quarters, 3);
145        }
146    };
147
148    public static final DateTimeFieldType QuarterOfYear = new DateTimeFieldType("quarterOfYear") {
149        private static final long serialVersionUID = -5677872459807379123L;
150
151        public DurationFieldType getDurationType() {
152            return Quarters;
153        }
154
155        public DurationFieldType getRangeDurationType() {
156            return DurationFieldType.years();
157        }
158
159        public DateTimeField getField(Chronology chronology) {
160            return new OffsetDateTimeField(new DividedDateTimeField(new OffsetDateTimeField(chronology.monthOfYear(), -1), QuarterOfYear, 3), 1);
161        }
162    };
