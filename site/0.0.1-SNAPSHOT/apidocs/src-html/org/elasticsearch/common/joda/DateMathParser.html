001package org.elasticsearch.common.joda;
002
003import org.elasticsearch.ElasticSearchParseException;
004import org.joda.time.DateTimeZone;
005import org.joda.time.MutableDateTime;
006
007import java.util.concurrent.TimeUnit;
008
009/**
010 */
011public class DateMathParser {
012
013    private final FormatDateTimeFormatter dateTimeFormatter;
014
015    private final TimeUnit timeUnit;
016
017    public DateMathParser(FormatDateTimeFormatter dateTimeFormatter, TimeUnit timeUnit) {
018        this.dateTimeFormatter = dateTimeFormatter;
019        this.timeUnit = timeUnit;
020    }
021
022    public long parse(String text, long now) {
023        return parse(text, now, false, false);
024    }
025
026    public long parseUpperInclusive(String text, long now) {
027        return parse(text, now, true, true);
028    }
029
030    public long parse(String text, long now, boolean roundUp, boolean upperInclusive) {
031        long time;
032        String mathString;
033        if (text.startsWith("now")) {
034            time = now;
035            mathString = text.substring("now".length());
036        } else {
037            int index = text.indexOf("||");
038            String parseString;
039            if (index == -1) {
040                parseString = text;
041                mathString = ""; // nothing else
042            } else {
043                parseString = text.substring(0, index);
044                mathString = text.substring(index + 2);
045            }
046            if (upperInclusive) {
047                time = parseUpperInclusiveStringValue(parseString);
048            } else {
049                time = parseStringValue(parseString);
050            }
051        }
052
053        if (mathString.isEmpty()) {
054            return time;
055        }
056
057        return parseMath(mathString, time, roundUp);
058    }
059
060    private long parseMath(String mathString, long time, boolean roundUp) throws ElasticSearchParseException {
061        MutableDateTime dateTime = new MutableDateTime(time, DateTimeZone.UTC);
062        try {
063            for (int i = 0; i < mathString.length(); ) {
064                char c = mathString.charAt(i++);
065                int type;
066                if (c == '/') {
067                    type = 0;
068                } else if (c == '+') {
069                    type = 1;
070                } else if (c == '-') {
071                    type = 2;
072                } else {
073                    throw new ElasticSearchParseException("operator not supported for date math [" + mathString + "]");
074                }
075
076                int num;
077                if (!Character.isDigit(mathString.charAt(i))) {
078                    num = 1;
079                } else {
080                    int numFrom = i;
081                    while (Character.isDigit(mathString.charAt(i))) {
082                        i++;
083                    }
084                    num = Integer.parseInt(mathString.substring(numFrom, i));
085                }
086                if (type == 0) {
087                    // rounding is only allowed on whole numbers
088                    if (num != 1) {
089                        throw new ElasticSearchParseException("rounding `/` can only be used on single unit types [" + mathString + "]");
090                    }
091                }
092                char unit = mathString.charAt(i++);
093                switch (unit) {
094                    case 'M':
095                        if (type == 0) {
096                            if (roundUp) {
097                                dateTime.monthOfYear().roundCeiling();
098                            } else {
099                                dateTime.monthOfYear().roundFloor();
100                            }
101                        } else if (type == 1) {
102                            dateTime.addMonths(num);
103                        } else if (type == 2) {
104                            dateTime.addMonths(-num);
105                        }
106                        break;
107                    case 'w':
108                        if (type == 0) {
109                            if (roundUp) {
110                                dateTime.weekOfWeekyear().roundCeiling();
111                            } else {
112                                dateTime.weekOfWeekyear().roundFloor();
113                            }
114                        } else if (type == 1) {
115                            dateTime.addWeeks(num);
116                        } else if (type == 2) {
117                            dateTime.addWeeks(-num);
118                        }
119                        break;
120                    case 'd':
121                        if (type == 0) {
122                            if (roundUp) {
123                                dateTime.dayOfMonth().roundCeiling();
124                            } else {
125                                dateTime.dayOfMonth().roundFloor();
126                            }
127                        } else if (type == 1) {
128                            dateTime.addDays(num);
129                        } else if (type == 2) {
130                            dateTime.addDays(-num);
131                        }
132                        break;
133                    case 'h':
134                    case 'H':
135                        if (type == 0) {
136                            if (roundUp) {
137                                dateTime.hourOfDay().roundCeiling();
138                            } else {
139                                dateTime.hourOfDay().roundFloor();
140                            }
141                        } else if (type == 1) {
142                            dateTime.addHours(num);
143                        } else if (type == 2) {
144                            dateTime.addHours(-num);
145                        }
146                        break;
147                    case 'm':
148                        if (type == 0) {
149                            if (roundUp) {
150                                dateTime.minuteOfHour().roundCeiling();
151                            } else {
152                                dateTime.minuteOfHour().roundFloor();
153                            }
154                        } else if (type == 1) {
155                            dateTime.addMinutes(num);
156                        } else if (type == 2) {
157                            dateTime.addMinutes(-num);
158                        }
159                        break;
160                    case 's':
161                        if (type == 0) {
162                            if (roundUp) {
163                                dateTime.secondOfMinute().roundCeiling();
164                            } else {
165                                dateTime.secondOfMinute().roundFloor();
166                            }
167                        } else if (type == 1) {
168                            dateTime.addSeconds(num);
169                        } else if (type == 2) {
170                            dateTime.addSeconds(-num);
171                        }
172                        break;
173                    default:
174                        throw new ElasticSearchParseException("unit [" + unit + "] not supported for date math [" + mathString + "]");
175                }
176            }
177        } catch (Exception e) {
178            if (e instanceof ElasticSearchParseException) {
179                throw (ElasticSearchParseException) e;
180            }
181            throw new ElasticSearchParseException("failed to parse date math [" + mathString + "]");
182        }
183        return dateTime.getMillis();
184    }
185
186    private long parseStringValue(String value) {
187        try {
188            return dateTimeFormatter.parser().parseMillis(value);
189        } catch (RuntimeException e) {
190            try {
191                long time = Long.parseLong(value);
192                return timeUnit.toMillis(time);
193            } catch (NumberFormatException e1) {
194                throw new ElasticSearchParseException("failed to parse date field [" + value + "], tried both date format [" + dateTimeFormatter.format() + "], and timestamp number", e);
195            }
196        }
197    }
198
199    private long parseUpperInclusiveStringValue(String value) {
200        try {
201            MutableDateTime dateTime = new MutableDateTime(3000, 12, 31, 23, 59, 59, 999, DateTimeZone.UTC);
202            int location = dateTimeFormatter.parser().parseInto(dateTime, value, 0);
203            // if we parsed all the string value, we are good
204            if (location == value.length()) {
205                return dateTime.getMillis();
206            }
207            // if we did not manage to parse, or the year is really high year which is unreasonable
208            // see if its a number
209            if (location <= 0 || dateTime.getYear() > 5000) {
210                try {
211                    long time = Long.parseLong(value);
212                    return timeUnit.toMillis(time);
213                } catch (NumberFormatException e1) {
214                    throw new ElasticSearchParseException("failed to parse date field [" + value + "], tried both date format [" + dateTimeFormatter.format() + "], and timestamp number");
215                }
216            }
217            return dateTime.getMillis();
218        } catch (RuntimeException e) {
219            try {
220                long time = Long.parseLong(value);
221                return timeUnit.toMillis(time);
222            } catch (NumberFormatException e1) {
223                throw new ElasticSearchParseException("failed to parse date field [" + value + "], tried both date format [" + dateTimeFormatter.format() + "], and timestamp number", e);
224            }
225        }
226    }
227}
