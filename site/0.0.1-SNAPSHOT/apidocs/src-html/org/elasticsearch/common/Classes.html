020package org.elasticsearch.common;
021
022import java.lang.reflect.Modifier;
023
024/**
025 *
026 */
027public class Classes {
028
029    /**
030     * The package separator character '.'
031     */
032    private static final char PACKAGE_SEPARATOR = '.';
033
034    /**
035     * Return the default ClassLoader to use: typically the thread context
036     * ClassLoader, if available; the ClassLoader that loaded the ClassUtils
037     * class will be used as fallback.
038     * <p/>
039     * <p>Call this method if you intend to use the thread context ClassLoader
040     * in a scenario where you absolutely need a non-null ClassLoader reference:
041     * for example, for class path resource loading (but not necessarily for
042     * <code>Class.forName</code>, which accepts a <code>null</code> ClassLoader
043     * reference as well).
044     *
045     * @return the default ClassLoader (never <code>null</code>)
046     * @see java.lang.Thread#getContextClassLoader()
047     */
048    public static ClassLoader getDefaultClassLoader() {
049        ClassLoader cl = null;
050        try {
051            cl = Thread.currentThread().getContextClassLoader();
052        } catch (Throwable ex) {
053            // Cannot access thread context ClassLoader - falling back to system class loader...
054        }
055        if (cl == null) {
056            // No thread context class loader -> use class loader of this class.
057            cl = Classes.class.getClassLoader();
058        }
059        return cl;
060    }
061
062    /**
063     * Determine the name of the package of the given class:
064     * e.g. "java.lang" for the <code>java.lang.String</code> class.
065     *
066     * @param clazz the class
067     * @return the package name, or the empty String if the class
068     *         is defined in the default package
069     */
070    public static String getPackageName(Class clazz) {
071        String className = clazz.getName();
072        int lastDotIndex = className.lastIndexOf(PACKAGE_SEPARATOR);
073        return (lastDotIndex != -1 ? className.substring(0, lastDotIndex) : "");
074    }
075
076    public static String getPackageNameNoDomain(Class clazz) {
077        String fullPackage = getPackageName(clazz);
078        if (fullPackage.startsWith("org.") || fullPackage.startsWith("com.") || fullPackage.startsWith("net.")) {
079            return fullPackage.substring(4);
080        }
081        return fullPackage;
082    }
083
084    public static boolean isInnerClass(Class<?> clazz) {
085        return !Modifier.isStatic(clazz.getModifiers())
086                && clazz.getEnclosingClass() != null;
087    }
088
089    public static boolean isConcrete(Class<?> clazz) {
090        int modifiers = clazz.getModifiers();
091        return !clazz.isInterface() && !Modifier.isAbstract(modifiers);
092    }
093
094    private Classes() {
095
096    }
097}
