020package org.elasticsearch.common.compress;
021
022import org.elasticsearch.common.lucene.util.UnicodeUtil;
023import org.elasticsearch.common.Unicode;
024import org.elasticsearch.common.bytes.BytesArray;
025import org.elasticsearch.common.bytes.BytesReference;
026import org.elasticsearch.common.io.stream.StreamInput;
027import org.elasticsearch.common.io.stream.StreamOutput;
028import org.elasticsearch.common.io.stream.Streamable;
029
030import java.io.IOException;
031import java.util.Arrays;
032
033/**
034 *
035 */
036public class CompressedString implements Streamable {
037
038    private byte[] bytes;
039
040    CompressedString() {
041    }
042
043    /**
044     * Constructor assuming the data provided is compressed (UTF8). It uses the provided
045     * array without copying it.
046     */
047    public CompressedString(byte[] compressed) {
048        this.bytes = compressed;
049    }
050
051    public CompressedString(BytesReference data) throws IOException {
052        BasicCompressor compressor = BasicCompressorFactory.compressor(data);
053        if (compressor != null) {
054            // already compressed...
055            this.bytes = data.toBytes();
056        } else {
057            BytesArray bytesArray = data.toBytesArray();
058            this.bytes = BasicCompressorFactory.defaultCompressor().compress(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length());
059        }
060    }
061
062    /**
063     * Constructs a new compressed string, assuming the bytes are UTF8, by copying it over.
064     *
065     * @param data   The byte array
066     * @param offset Offset into the byte array
067     * @param length The length of the data
068     * @throws IOException
069     */
070    public CompressedString(byte[] data, int offset, int length) throws IOException {
071        BasicCompressor compressor = BasicCompressorFactory.compressor(data, offset, length);
072        if (compressor != null) {
073            // already compressed...
074            this.bytes = Arrays.copyOfRange(data, offset, offset + length);
075        } else {
076            // default to LZF
077            this.bytes = BasicCompressorFactory.defaultCompressor().compress(data, offset, length);
078        }
079    }
080
081    public CompressedString(String str) throws IOException {
082        UnicodeUtil.UTF8Result result = Unicode.unsafeFromStringAsUtf8(str);
083        this.bytes = BasicCompressorFactory.defaultCompressor().compress(result.result, 0, result.length);
084    }
085
086    public byte[] compressed() {
087        return this.bytes;
088    }
089
090    public byte[] uncompressed() throws IOException {
091        BasicCompressor compressor = BasicCompressorFactory.compressor(bytes);
092        return compressor.uncompress(bytes, 0, bytes.length);
093    }
094
095    public String string() throws IOException {
096        return Unicode.fromBytes(uncompressed());
097    }
098
099    public static CompressedString readCompressedString(StreamInput in) throws IOException {
100        CompressedString compressedString = new CompressedString();
101        compressedString.readFrom(in);
102        return compressedString;
103    }
104
105    @Override
106    public void readFrom(StreamInput in) throws IOException {
107        bytes = new byte[in.readVInt()];
108        in.readBytes(bytes, 0, bytes.length);
109    }
110
111    @Override
112    public void writeTo(StreamOutput out) throws IOException {
113        out.writeVInt(bytes.length);
114        out.writeBytes(bytes);
115    }
116
117    @Override
118    public boolean equals(Object o) {
119        if (this == o) return true;
120        if (o == null || getClass() != o.getClass()) return false;
121
122        CompressedString that = (CompressedString) o;
123
124        if (!Arrays.equals(bytes, that.bytes)) return false;
125
126        return true;
127    }
128
129    @Override
130    public int hashCode() {
131        return bytes != null ? Arrays.hashCode(bytes) : 0;
132    }
133
134    @Override
135    public String toString() {
136        try {
137            return string();
138        } catch (IOException e) {
139            return "_na_";
140        }
141    }
142}
