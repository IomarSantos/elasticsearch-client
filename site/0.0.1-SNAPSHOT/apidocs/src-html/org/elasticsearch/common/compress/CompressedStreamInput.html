020package org.elasticsearch.common.compress;
021
022import org.elasticsearch.Version;
023import org.elasticsearch.common.io.stream.StreamInput;
024
025import java.io.EOFException;
026import java.io.IOException;
027
028/**
029 */
030public abstract class CompressedStreamInput<T extends CompressorContext> extends StreamInput {
031
032    private final StreamInput in;
033    protected final CompressorContext context;
034
035    private boolean closed;
036
037    protected byte[] uncompressed;
038    private int position = 0;
039    private int valid = 0;
040
041    public CompressedStreamInput(StreamInput in, T context) throws IOException {
042        this.in = in;
043        this.context = context;
044        super.setVersion(in.getVersion());
045        readHeader(in);
046    }
047
048    @Override
049    public StreamInput setVersion(Version version) {
050        in.setVersion(version);
051        return super.setVersion(version);
052    }
053
054    /**
055     * Expert!, resets to buffer start, without the need to decompress it again.
056     */
057    public void resetToBufferStart() {
058        this.position = 0;
059    }
060
061    /**
062     * Method is overridden to report number of bytes that can now be read
063     * from decoded data buffer, without reading bytes from the underlying
064     * stream.
065     * Never throws an exception; returns number of bytes available without
066     * further reads from underlying source; -1 if stream has been closed, or
067     * 0 if an actual read (and possible blocking) is needed to find out.
068     */
069    @Override
070    public int available() throws IOException {
071        // if closed, return -1;
072        if (closed) {
073            return -1;
074        }
075        int left = (valid - position);
076        return (left <= 0) ? 0 : left;
077    }
078
079    @Override
080    public int read() throws IOException {
081        if (!readyBuffer()) {
082            return -1;
083        }
084        return uncompressed[position++] & 255;
085    }
086
087    @Override
088    public byte readByte() throws IOException {
089        if (!readyBuffer()) {
090            throw new EOFException();
091        }
092        return uncompressed[position++];
093    }
094
095    @Override
096    public int read(byte[] buffer, int offset, int length) throws IOException {
097        return read(buffer, offset, length, false);
098    }
099
100    public int read(byte[] buffer, int offset, int length, boolean fullRead) throws IOException {
101        if (length < 1) {
102            return 0;
103        }
104        if (!readyBuffer()) {
105            return -1;
106        }
107        // First let's read however much data we happen to have...
108        int chunkLength = Math.min(valid - position, length);
109        System.arraycopy(uncompressed, position, buffer, offset, chunkLength);
110        position += chunkLength;
111
112        if (chunkLength == length || !fullRead) {
113            return chunkLength;
114        }
115        // Need more data, then
116        int totalRead = chunkLength;
117        do {
118            offset += chunkLength;
119            if (!readyBuffer()) {
120                break;
121            }
122            chunkLength = Math.min(valid - position, (length - totalRead));
123            System.arraycopy(uncompressed, position, buffer, offset, chunkLength);
124            position += chunkLength;
125            totalRead += chunkLength;
126        } while (totalRead < length);
127
128        return totalRead;
129    }
130
131    @Override
132    public void readBytes(byte[] b, int offset, int len) throws IOException {
133        int result = read(b, offset, len, true /* we want to have full reads, thats the contract... */);
134        if (result < len) {
135            throw new EOFException();
136        }
137    }
138
139    @Override
140    public void reset() throws IOException {
141        this.position = 0;
142        this.valid = 0;
143        in.reset();
144    }
145
146    @Override
147    public void close() throws IOException {
148        position = valid = 0;
149        if (!closed) {
150            closed = true;
151            doClose();
152            in.close();
153        }
154    }
155
156    protected abstract void doClose() throws IOException;
157
158    /**
159     * Fill the uncompressed bytes buffer by reading the underlying inputStream.
160     */
161    protected boolean readyBuffer() throws IOException {
162        if (position < valid) {
163            return true;
164        }
165        if (closed) {
166            return false;
167        }
168        valid = uncompress(in, uncompressed);
169        if (valid < 0) {
170            return false;
171        }
172        position = 0;
173        return (position < valid);
174    }
175
176    protected abstract void readHeader(StreamInput in) throws IOException;
177
178    /**
179     * Uncompress the data into the out array, returning the size uncompressed
180     */
181    protected abstract int uncompress(StreamInput in, byte[] out) throws IOException;
182
183}
