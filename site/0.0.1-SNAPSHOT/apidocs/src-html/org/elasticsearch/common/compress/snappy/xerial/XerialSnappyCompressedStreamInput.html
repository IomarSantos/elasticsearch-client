020package org.elasticsearch.common.compress.snappy.xerial;
021
022import org.elasticsearch.common.compress.snappy.ClientSnappyCompressedStreamInput;
023import org.elasticsearch.common.compress.snappy.SnappyCompressorContext;
024import org.elasticsearch.common.io.stream.StreamInput;
025import org.xerial.snappy.Snappy;
026
027import java.io.IOException;
028
029/**
030 */
031public class XerialSnappyCompressedStreamInput extends ClientSnappyCompressedStreamInput {
032
033    public XerialSnappyCompressedStreamInput(StreamInput in, SnappyCompressorContext context) throws IOException {
034        super(in, context);
035    }
036
037    @Override
038    protected int uncompress(StreamInput in, byte[] out) throws IOException {
039        int compressedByte = in.read();
040        // if we are on the "tip", just return 0 uncompressed data...
041        if (compressedByte == -1) {
042            return 0;
043        }
044        boolean compressed = compressedByte == 1;
045        int length = in.readVInt();
046        if (!compressed) {
047            in.readBytes(out, 0, length);
048            return length;
049        } else {
050            in.readBytes(inputBuffer, 0, length);
051            return Snappy.rawUncompress(inputBuffer, 0, length, out, 0);
052        }
053    }
054}
