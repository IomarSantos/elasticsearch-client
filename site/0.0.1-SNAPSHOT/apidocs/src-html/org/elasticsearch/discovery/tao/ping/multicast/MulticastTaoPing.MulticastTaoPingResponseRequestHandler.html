076    public MulticastTaoPing(Settings settings,
077            TransportThreadPool threadPool, 
078            ClientTransportService transportService,
079            ClusterName clusterName, DiscoveryNode localNode) {
080        this.settings = settings;
081        this.logger = Loggers.getLogger(getClass(), settings);
082        this.threadPool = threadPool;
083        this.transportService = transportService;
084        this.localNode = localNode;
085        this.clusterName = clusterName;
086        this.networkService = new TransportNetworkService(settings);
087
088        this.address = settings.get("address");
089        this.port = settings.getAsInt("port", 54328);
090        this.group = settings.get("group", "224.2.2.4");
091        this.bufferSize = settings.getAsInt("buffer_size", 2048);
092        this.ttl = settings.getAsInt("ttl", 3);
093
094        this.pingEnabled = settings.getAsBoolean("ping.enabled", true);
095
096        logger.debug("using group [{}], with port [{}], ttl [{}], and address [{}]", group, port, ttl, address);
097
098        this.transportService.registerHandler(MulticastTaoPingResponseRequestHandler.ACTION, new MulticastTaoPingResponseRequestHandler());
099    }

101    public void start() {
102        try {
103            this.datagramPacketReceive = new DatagramPacket(new byte[bufferSize], bufferSize);
104            this.datagramPacketSend = new DatagramPacket(new byte[bufferSize], bufferSize, InetAddress.getByName(group), port);
105        } catch (Exception e) {
106            logger.warn("disabled, failed to setup multicast (datagram) discovery : {}", e.getMessage());
107            if (logger.isDebugEnabled()) {
108                logger.debug("disabled, failed to setup multicast (datagram) discovery", e);
109            }
110            return;
111        }
112
113        InetAddress multicastInterface = null;
114        try {
115            MulticastSocket multicastSocket;
116            multicastSocket = new MulticastSocket(port);
117
118            multicastSocket.setTimeToLive(ttl);
119
120            // set the send interface
121            multicastInterface = networkService.resolvePublishHostAddress(address);
122            multicastSocket.setInterface(multicastInterface);
123            multicastSocket.joinGroup(InetAddress.getByName(group));
124
125            multicastSocket.setReceiveBufferSize(bufferSize);
126            multicastSocket.setSendBufferSize(bufferSize);
127            multicastSocket.setSoTimeout(60000);
128
129            this.multicastSocket = multicastSocket;
130
131            this.receiver = new Receiver();
132            this.receiverThread = daemonThreadFactory(settings, "discovery#multicast#receiver").newThread(receiver);
133            this.receiverThread.start();
134        } catch (Exception e) {
135            datagramPacketReceive = null;
136            datagramPacketSend = null;
137            if (multicastSocket != null) {
138                multicastSocket.close();
139                multicastSocket = null;
140            }
141            logger.warn("disabled, failed to setup multicast discovery on port [{}], [{}]: {}", port, multicastInterface, e.getMessage());
142            if (logger.isDebugEnabled()) {
143                logger.debug("disabled, failed to setup multicast discovery on {}", e, multicastInterface);
144            }
145        }
146    }
147
148    public void stop() throws ElasticSearchException {
149        if (receiver != null) {
150            receiver.stop();
151        }
152        if (receiverThread != null) {
153            receiverThread.interrupt();
154        }
155        if (multicastSocket != null) {
156            multicastSocket.close();
157            multicastSocket = null;
158        }
159    }

161    public PingResponse[] pingAndWait(TimeValue timeout) {
162        final AtomicReference<PingResponse[]> response = new AtomicReference<PingResponse[]>();
163        final CountDownLatch latch = new CountDownLatch(1);
164        ping(new PingListener() {
165            @Override
166            public void onPing(PingResponse[] pings) {
167                response.set(pings);
168                latch.countDown();
169            }
170        }, timeout);
171        try {
172            latch.await();
173            return response.get();
174        } catch (InterruptedException e) {
175            return null;
176        }
177    }

179    @Override
180    public void ping(final PingListener listener, final TimeValue timeout) {
181        if (!pingEnabled) {
182            threadPool.generic().execute(new Runnable() {
183                @Override
184                public void run() {
185                    listener.onPing(new PingResponse[0]);
186                }
187            });
188            return;
189        }
190        final int id = pingIdGenerator.incrementAndGet();
191        receivedResponses.put(id, ConcurrentCollections.<DiscoveryNode, PingResponse>newConcurrentMap());
192        sendPingRequest(id);
193        // try and send another ping request halfway through (just in case someone woke up during it...)
194        // this can be a good trade-off to nailing the initial lookup or un-delivered messages
195        threadPool.schedule(TimeValue.timeValueMillis(timeout.millis() / 2), TransportThreadPool.Names.GENERIC, new Runnable() {
196            @Override
197            public void run() {
198                try {
199                    sendPingRequest(id);
200                } catch (Exception e) {
201                    logger.warn("[{}] failed to send second ping request", e, id);
202                }
203            }
204        });
205        threadPool.schedule(timeout, TransportThreadPool.Names.GENERIC, new Runnable() {
206            @Override
207            public void run() {
208                ConcurrentMap<DiscoveryNode, PingResponse> responses = receivedResponses.remove(id);
209                listener.onPing(responses.values().toArray(new PingResponse[responses.size()]));
210            }
211        });
212    }
213
214    private void sendPingRequest(int id) {
215        if (multicastSocket == null) {
216            return;
217        }
218        synchronized (sendMutex) {
219            BasicCachedStreamOutput.Entry cachedEntry = BasicCachedStreamOutput.popEntry();
220            try {
221                StreamOutput out = cachedEntry.handles();
222                out.writeBytes(INTERNAL_HEADER);
223                Version.writeVersion(Version.CURRENT, out);
224                out.writeInt(id);
225                clusterName.writeTo(out);
226                localNode.writeTo(out);
227                out.close();
228                datagramPacketSend.setData(cachedEntry.bytes().bytes().copyBytesArray().toBytes());
229                multicastSocket.send(datagramPacketSend);
230                if (logger.isTraceEnabled()) {
231                    logger.trace("[{}] sending ping request", id);
232                }
233            } catch (Exception e) {
234                if (logger.isDebugEnabled()) {
235                    logger.debug("failed to send multicast ping request", e);
236                } else {
237                    logger.warn("failed to send multicast ping request: {}", ExceptionsHelper.detailedMessage(e));
238                }
239            } finally {
240                BasicCachedStreamOutput.pushEntry(cachedEntry);
241            }
242        }
243    }

245    public class MulticastTaoPingResponseRequestHandler extends BaseTransportRequestHandler<MulticastPingResponse> {
246
247        public static final String ACTION = "discovery/tao/multicast";
248
249        @Override
250        public MulticastPingResponse newInstance() {
251            return new MulticastPingResponse();
252        }
253
254        @Override
255        public void messageReceived(MulticastPingResponse request, TransportChannel channel) throws Exception {
256            if (logger.isTraceEnabled()) {
257                logger.trace("[{}] received {}", request.id, request.pingResponse);
258            }
259            ConcurrentMap<DiscoveryNode, PingResponse> responses = receivedResponses.get(request.id);
260            if (responses == null) {
261                logger.warn("received ping response {} with no matching id [{}]", request.pingResponse, request.id);
262            } else {
263                responses.put(request.pingResponse.target(), request.pingResponse);
264            }
265            channel.sendResponse(TransportResponse.Empty.INSTANCE);
266        }
267
268        @Override
269        public String executor() {
270            return TransportThreadPool.Names.SAME;
271        }
272    }

295    private class Receiver implements Runnable {
296
297        private volatile boolean running = true;
298
299        public void stop() {
300            running = false;
301        }
302
303        @Override
304        public void run() {
305            while (running) {
306                try {
307                    int id = -1;
308                    DiscoveryNode requestingNodeX = null;
309                    ClusterName clusterName = null;
310
311                    Map<String, Object> externalPingData = null;
312                    XContentType xContentType = null;
313
314                    synchronized (receiveMutex) {
315                        try {
316                            multicastSocket.receive(datagramPacketReceive);
317                        } catch (SocketTimeoutException ignore) {
318                            continue;
319                        } catch (Exception e) {
320                            if (running) {
321                                logger.warn("failed to receive packet", e);
322                            }
323                            continue;
324                        }
325                        try {
326                            boolean internal = false;
327                            if (datagramPacketReceive.getLength() > 4) {
328                                int counter = 0;
329                                for (; counter < INTERNAL_HEADER.length; counter++) {
330                                    if (datagramPacketReceive.getData()[datagramPacketReceive.getOffset() + counter] != INTERNAL_HEADER[counter]) {
331                                        break;
332                                    }
333                                }
334                                if (counter == INTERNAL_HEADER.length) {
335                                    internal = true;
336                                }
337                            }
338                            if (internal) {
339                                StreamInput input = BasicCachedStreamInput.cachedHandles(new BytesStreamInput(datagramPacketReceive.getData(), datagramPacketReceive.getOffset() + INTERNAL_HEADER.length, datagramPacketReceive.getLength(), true));
340                                Version version = Version.readVersion(input);
341                                id = input.readInt();
342                                clusterName = ClusterName.readClusterName(input);
343                                requestingNodeX = readNode(input);
344                            } else {
345                                xContentType = XContentFactory.xContentType(datagramPacketReceive.getData(), datagramPacketReceive.getOffset(), datagramPacketReceive.getLength());
346                                if (xContentType != null) {
347                                    // an external ping
348                                    externalPingData = XContentFactory.xContent(xContentType)
349                                            .createParser(datagramPacketReceive.getData(), datagramPacketReceive.getOffset(), datagramPacketReceive.getLength())
350                                            .mapAndClose();
351                                } else {
352                                    throw new ElasticSearchIllegalStateException("failed multicast message, probably message from previous version");
353                                }
354                            }
355                        } catch (Exception e) {
356                            logger.warn("failed to read requesting data from {}", e, datagramPacketReceive.getSocketAddress());
357                            continue;
358                        }
359                    }
360                    if (externalPingData != null) {
361                        handleExternalPingRequest(externalPingData, xContentType, datagramPacketReceive.getSocketAddress());
362                    } else {
363                        handleNodePingRequest(id, requestingNodeX, clusterName);
364                    }
365                } catch (Exception e) {
366                    logger.warn("unexpected exception in multicast receiver", e);
367                }
368            }
369        }
370
371        @SuppressWarnings("unchecked")
372        private void handleExternalPingRequest(Map<String, Object> externalPingData, XContentType contentType, SocketAddress remoteAddress) {
373            if (externalPingData.containsKey("response")) {
374                // ignoring responses sent over the multicast channel
375                logger.trace("got an external ping response (ignoring) from {}, content {}", remoteAddress, externalPingData);
376                return;
377            }
378
379            if (multicastSocket == null) {
380                logger.debug("can't send ping response, no socket, from {}, content {}", remoteAddress, externalPingData);
381                return;
382            }
383
384            Map<String, Object> request = (Map<String, Object>) externalPingData.get("request");
385            if (request == null) {
386                logger.warn("malformed external ping request, no 'request' element from {}, content {}", remoteAddress, externalPingData);
387                return;
388            }
389
390            String clusterName = request.containsKey("cluster_name") ? request.get("cluster_name").toString() : request.containsKey("clusterName") ? request.get("clusterName").toString() : null;
391            if (clusterName == null) {
392                logger.warn("malformed external ping request, missing 'cluster_name' element within request, from {}, content {}", remoteAddress, externalPingData);
393                return;
394            }
395
396            if (!clusterName.equals(MulticastTaoPing.this.clusterName.value())) {
397                logger.trace("got request for cluster_name {}, but our cluster_name is {}, from {}, content {}", clusterName, MulticastTaoPing.this.clusterName.value(), remoteAddress, externalPingData);
398                return;
399            }
400            if (logger.isTraceEnabled()) {
401                logger.trace("got external ping request from {}, content {}", remoteAddress, externalPingData);
402            }
403
404            try {
405
406                XContentBuilder builder = XContentFactory.contentBuilder(contentType);
407                builder.startObject().startObject("response");
408                builder.field("cluster_name", MulticastTaoPing.this.clusterName.value());
409                builder.startObject("version").field("number", Version.CURRENT.number()).field("snapshot_build", Version.CURRENT.snapshot).endObject();
410                builder.field("transport_address", localNode.address().toString());
411
412                builder.startObject("attributes");
413                // no attributes
414                builder.endObject();
415
416                builder.endObject().endObject();
417                synchronized (sendMutex) {
418                    BytesReference bytes = builder.bytes();
419                    datagramPacketSend.setData(bytes.array(), bytes.arrayOffset(), bytes.length());
420                    multicastSocket.send(datagramPacketSend);
421                    if (logger.isTraceEnabled()) {
422                        logger.trace("sending external ping response {}", builder.string());
423                    }
424                }
425            } catch (Exception e) {
426                logger.warn("failed to send external multicast response", e);
427            }
428        }

430        private void handleNodePingRequest(int id, DiscoveryNode requestingNodeX, ClusterName clusterName) {
431            if (!pingEnabled) {
432                return;
433            }
434            //DiscoveryNodes discoveryNodes = nodesProvider.nodes();
435            final DiscoveryNode requestingNode = requestingNodeX;
436            if (requestingNode.id().equals(localNode.id())) {
437                // that's me, ignore
438                return;
439            }
440            if (!clusterName.equals(MulticastTaoPing.this.clusterName)) {
441                if (logger.isTraceEnabled()) {
442                    logger.trace("[{}] received ping_request from [{}], but wrong cluster_name [{}], expected [{}], ignoring", id, requestingNode, clusterName, MulticastTaoPing.this.clusterName);
443                }
444                return;
445            }
446            // don't connect between two client nodes, no need for that...
447            if (!localNode.shouldConnectTo(requestingNode)) {
448                if (logger.isTraceEnabled()) {
449                    logger.trace("[{}] received ping_request from [{}], both are client nodes, ignoring", id, requestingNode, clusterName);
450                }
451                return;
452            }
453            final MulticastPingResponse multicastPingResponse = new MulticastPingResponse();
454            multicastPingResponse.id = id;
455            multicastPingResponse.pingResponse = new PingResponse(localNode, clusterName);
456
457            if (logger.isTraceEnabled()) {
458                logger.trace("[{}] received ping_request from [{}], sending {}", id, requestingNode, multicastPingResponse.pingResponse);
459            }
460
461            if (!transportService.nodeConnected(requestingNode)) {
462                // do the connect and send on a thread pool
463                threadPool.generic().execute(new Runnable() {
464                    @Override
465                    public void run() {
466                        // connect to the node if possible
467                        try {
468                            transportService.connectToNode(requestingNode);
469                            transportService.sendRequest(requestingNode, MulticastTaoPingResponseRequestHandler.ACTION, multicastPingResponse, new EmptyTransportResponseHandler(TransportThreadPool.Names.SAME) {
470                                @Override
471                                public void handleException(TransportException exp) {
472                                    logger.warn("failed to receive confirmation on sent ping response to [{}]", exp, requestingNode);
473                                }
474                            });
475                        } catch (Exception e) {
476                            logger.warn("failed to connect to requesting node {}", e, requestingNode);
477                        }
478                    }
479                });
480            } else {
481                transportService.sendRequest(requestingNode, MulticastTaoPingResponseRequestHandler.ACTION, multicastPingResponse, new EmptyTransportResponseHandler(TransportThreadPool.Names.SAME) {
482                    @Override
483                    public void handleException(TransportException exp) {
484                        logger.warn("failed to receive confirmation on sent ping response to [{}]", exp, requestingNode);
485                    }
486                });
487            }
488        }
489    }
