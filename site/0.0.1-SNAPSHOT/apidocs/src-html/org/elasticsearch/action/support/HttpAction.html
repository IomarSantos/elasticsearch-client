020package org.elasticsearch.action.support;
021
022import com.ning.http.client.AsyncHandler;
023import com.ning.http.client.AsyncHandler.STATE;
024import com.ning.http.client.AsyncHttpClient;
025import com.ning.http.client.HttpResponseBodyPart;
026import com.ning.http.client.HttpResponseHeaders;
027import com.ning.http.client.HttpResponseStatus;
028
029import org.elasticsearch.ElasticSearchException;
030import org.elasticsearch.action.ActionFuture;
031import org.elasticsearch.action.ActionListener;
032import org.elasticsearch.action.ActionRequest;
033import org.elasticsearch.action.ActionRequestValidationException;
034import org.elasticsearch.action.ActionResponse;
035import org.elasticsearch.common.io.stream.BytesStreamOutput;
036import org.elasticsearch.common.logging.ESLogger;
037import org.elasticsearch.common.logging.ESLoggerFactory;
038import org.elasticsearch.common.settings.Settings;
039
040import static org.elasticsearch.action.support.PlainActionFuture.newFuture;
041
042import java.io.IOException;
043
044public abstract class HttpAction<Request extends ActionRequest, Response extends ActionResponse> {
045
046    protected ESLogger logger = ESLoggerFactory.getLogger(HttpAction.class.getName());
047
048    public ActionFuture<Response> execute(HttpClient client, Request request) throws ElasticSearchException {
049        PlainActionFuture<Response> future = newFuture();
050        request.listenerThreaded(false);
051        execute(client, request, future);
052        return future;
053    }
054
055    public void execute(HttpClient client, Request request, ActionListener<Response> listener) {
056        ActionRequestValidationException validationException = request.validate();
057        if (validationException != null) {
058            listener.onFailure(validationException);
059            return;
060        }
061        try {
062            doExecute(client, request, listener);
063        } catch (Exception e) {
064            logger.error(e.getMessage(), e);
065            listener.onFailure(e);
066        }
067    }
068
069    protected abstract void doExecute(HttpClient client, Request request, ActionListener<Response> listener);
070
071    protected abstract Response toResponse(HttpResponse response) throws IOException;
072
073    protected void submit(HttpClient client, HttpRequest request, ActionListener<Response> listener) {        
074        AsyncHttpClient.BoundRequestBuilder builder = client.prepareRequest(request.buildRequest(client.settings()));
075        if (logger.isDebugEnabled()) {
076            logger.debug("submitting request = {}, body = {}", builder.build().toString(), builder.build().getStringData());
077        }
078        AsyncHandler<HttpResponse> handler = new WrapHandler(client.settings(), listener);
079        try {
080            builder.execute(handler);
081        } catch (Exception e) {
082            logger.error(e.getMessage(), e);
083            listener.onFailure(e);
084        }
085    }
086
087    private class WrapHandler implements AsyncHandler<HttpResponse> {
088
089        private final ActionListener<Response> listener;
090        private final BytesStreamOutput body;
091        private HttpResponseStatus statuscode;
092        private HttpResponseHeaders headers;
093        private String contentType;
094
095        WrapHandler(Settings settings, ActionListener<Response> listener) {
096            this.listener = listener;
097            this.body = new BytesStreamOutput();
098        }
099
100        @Override
101        public STATE onStatusReceived(HttpResponseStatus hrs) throws Exception {
102            if (logger.isDebugEnabled()) {
103                logger.debug("onStatusReceived {}", hrs);
104            }
105            this.statuscode = hrs;
106            return STATE.CONTINUE;
107        }
108
109        @Override
110        public STATE onHeadersReceived(HttpResponseHeaders hrh) throws Exception {
111            if (logger.isDebugEnabled()) {
112                logger.debug("onHeadersReceived {}", hrh);
113            }
114            this.headers = hrh;
115            this.contentType = hrh.getHeaders().getFirstValue("Content-type");
116            return STATE.CONTINUE;
117        }
118
119        @Override
120        public STATE onBodyPartReceived(HttpResponseBodyPart hrbp) throws Exception {
121            if (logger.isDebugEnabled()) {
122                logger.debug("onBodyPartReceived {}", new String(hrbp.getBodyPartBytes()));
123            }
124            body.writeBytes(hrbp.getBodyPartBytes());
125            return STATE.CONTINUE;
126        }
127
128        @Override
129        public void onThrowable(Throwable t) {
130            logger.error(t.getMessage(), t);
131            listener.onFailure(t);
132        }
133
134        @Override
135        public HttpResponse onCompleted() {
136            HttpResponse response = new HttpResponse(
137                    statuscode != null ? statuscode.getStatusCode() : -1,
138                    contentType,
139                    headers != null ? headers.getHeaders() : null,
140                    body.bytes());
141            if (logger.isDebugEnabled()) {
142                logger.debug("onCompleted {}", response);
143            }
144            try {
145                if (response.getStatusCode() >= 200 && response.getStatusCode() < 300) {
146                    listener.onResponse(toResponse(response));
147                } else {
148                    throw new IOException("HTTP error " + response.getStatusCode() + " message: " + response.getBody().toUtf8());
149                }
150            } catch (IOException e) {
151                logger.error(e.getMessage(), e);
152                listener.onFailure(e);
153            }
154            return response;
155        }
156    }
157}
