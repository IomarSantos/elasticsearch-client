020package org.elasticsearch.action.mlt;
021
022import org.elasticsearch.ElasticSearchIllegalArgumentException;
023import org.elasticsearch.action.ActionListener;
024import org.elasticsearch.action.ActionRequestBuilder;
025import org.elasticsearch.action.search.SearchResponse;
026import org.elasticsearch.action.search.SearchType;
027import org.elasticsearch.client.SearchClient;
028import org.elasticsearch.common.xcontent.XContentBuilder;
029import org.elasticsearch.search.Scroll;
030import org.elasticsearch.search.builder.SearchSourceBuilder;
031
032import java.util.Map;
033
034/**
035 */
036public class MoreLikeThisRequestBuilder extends ActionRequestBuilder<MoreLikeThisRequest, SearchResponse, MoreLikeThisRequestBuilder> {
037
038    public MoreLikeThisRequestBuilder(SearchClient client) {
039        super(client, new MoreLikeThisRequest());
040    }
041
042    public MoreLikeThisRequestBuilder(SearchClient client, String index, String type, String id) {
043        super(client, new MoreLikeThisRequest(index).type(type).id(id));
044    }
045
046    /**
047     * The fields of the document to use in order to find documents "like" this one. Defaults to run
048     * against all the document fields.
049     */
050    public MoreLikeThisRequestBuilder setField(String... fields) {
051        request.fields(fields);
052        return this;
053    }
054
055    /**
056     * The percent of the terms to match for each field. Defaults to <tt>0.3f</tt>.
057     */
058    public MoreLikeThisRequestBuilder setPercentTermsToMatch(float percentTermsToMatch) {
059        request.percentTermsToMatch(percentTermsToMatch);
060        return this;
061    }
062
063    /**
064     * The frequency below which terms will be ignored in the source doc. Defaults to <tt>2</tt>.
065     */
066    public MoreLikeThisRequestBuilder setMinTermFreq(int minTermFreq) {
067        request.minTermFreq(minTermFreq);
068        return this;
069    }
070
071    /**
072     * The maximum number of query terms that will be included in any generated query. Defaults to <tt>25</tt>.
073     */
074    public MoreLikeThisRequestBuilder maxQueryTerms(int maxQueryTerms) {
075        request.maxQueryTerms(maxQueryTerms);
076        return this;
077    }
078
079    /**
080     * Any word in this set is considered "uninteresting" and ignored.
081     * <p/>
082     * <p>Even if your Analyzer allows stopwords, you might want to tell the MoreLikeThis code to ignore them, as
083     * for the purposes of document similarity it seems reasonable to assume that "a stop word is never interesting".
084     * <p/>
085     * <p>Defaults to no stop words.
086     */
087    public MoreLikeThisRequestBuilder setStopWords(String... stopWords) {
088        request.stopWords(stopWords);
089        return this;
090    }
091
092    /**
093     * The frequency at which words will be ignored which do not occur in at least this
094     * many docs. Defaults to <tt>5</tt>.
095     */
096    public MoreLikeThisRequestBuilder setMinDocFreq(int minDocFreq) {
097        request.minDocFreq(minDocFreq);
098        return this;
099    }
100
101    /**
102     * The maximum frequency in which words may still appear. Words that appear
103     * in more than this many docs will be ignored. Defaults to unbounded.
104     */
105    public MoreLikeThisRequestBuilder setMaxDocFreq(int maxDocFreq) {
106        request.maxDocFreq(maxDocFreq);
107        return this;
108    }
109
110    /**
111     * The minimum word length below which words will be ignored. Defaults to <tt>0</tt>.
112     */
113    public MoreLikeThisRequestBuilder setMinWordLen(int minWordLen) {
114        request.minWordLen(minWordLen);
115        return this;
116    }
117
118    /**
119     * The maximum word length above which words will be ignored. Defaults to unbounded.
120     */
121    public MoreLikeThisRequestBuilder setMaxWordLen(int maxWordLen) {
122        request().maxWordLen(maxWordLen);
123        return this;
124    }
125
126    /**
127     * The boost factor to use when boosting terms. Defaults to <tt>1</tt>.
128     */
129    public MoreLikeThisRequestBuilder setBoostTerms(float boostTerms) {
130        request.boostTerms(boostTerms);
131        return this;
132    }
133
134    /**
135     * An optional search source request allowing to control the search request for the
136     * more like this documents.
137     */
138    public MoreLikeThisRequestBuilder setSearchSource(SearchSourceBuilder sourceBuilder) {
139        request.searchSource(sourceBuilder);
140        return this;
141    }
142
143    /**
144     * An optional search source request allowing to control the search request for the
145     * more like this documents.
146     */
147    public MoreLikeThisRequestBuilder setSearchSource(String searchSource) {
148        request.searchSource(searchSource);
149        return this;
150    }
151
152    /**
153     * An optional search source request allowing to control the search request for the
154     * more like this documents.
155     */
156    public MoreLikeThisRequestBuilder setSearchSource(Map searchSource) {
157        request.searchSource(searchSource);
158        return this;
159    }
160
161    /**
162     * An optional search source request allowing to control the search request for the
163     * more like this documents.
164     */
165    public MoreLikeThisRequestBuilder setSearchSource(XContentBuilder builder) {
166        request.searchSource(builder);
167        return this;
168    }
169
170    /**
171     * An optional search source request allowing to control the search request for the
172     * more like this documents.
173     */
174    public MoreLikeThisRequestBuilder setSearchSource(byte[] searchSource) {
175        request.searchSource(searchSource);
176        return this;
177    }
178
179    /**
180     * The search type of the mlt search query.
181     */
182    public MoreLikeThisRequestBuilder setSearchType(SearchType searchType) {
183        request.searchType(searchType);
184        return this;
185    }
186
187    /**
188     * The search type of the mlt search query.
189     */
190    public MoreLikeThisRequestBuilder setSearchType(String searchType) throws ElasticSearchIllegalArgumentException {
191        request.searchType(searchType);
192        return this;
193    }
194
195    /**
196     * The indices the resulting mlt query will run against. If not set, will run
197     * against the index the document was fetched from.
198     */
199    public MoreLikeThisRequestBuilder setSearchIndices(String... searchIndices) {
200        request.searchIndices(searchIndices);
201        return this;
202    }
203
204    /**
205     * The types the resulting mlt query will run against. If not set, will run
206     * against the type of the document fetched.
207     */
208    public MoreLikeThisRequestBuilder setSearchTypes(String... searchTypes) {
209        request.searchTypes(searchTypes);
210        return this;
211    }
212
213    /**
214     * An optional search scroll request to be able to continue and scroll the search
215     * operation.
216     */
217    public MoreLikeThisRequestBuilder setSearchScroll(Scroll searchScroll) {
218        request.searchScroll(searchScroll);
219        return this;
220    }
221
222    /**
223     * The number of documents to return, defaults to 10.
224     */
225    public MoreLikeThisRequestBuilder setSearchSize(int size) {
226        request.searchSize(size);
227        return this;
228    }
229
230    /**
231     * From which search result set to return.
232     */
233    public MoreLikeThisRequestBuilder setSearchFrom(int from) {
234        request.searchFrom(from);
235        return this;
236    }
237
238
239    @Override
240    protected void doExecute(ActionListener<SearchResponse> listener) {
241        ((SearchClient) client).moreLikeThis(request, listener);
242    }
243}
