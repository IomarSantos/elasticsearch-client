020package org.elasticsearch.action.delete;
021
022import org.elasticsearch.action.ActionRequestValidationException;
023import org.elasticsearch.action.support.replication.ShardReplicationOperationRequest;
024import org.elasticsearch.common.Nullable;
025import org.elasticsearch.common.Required;
026import org.elasticsearch.common.io.stream.StreamInput;
027import org.elasticsearch.common.io.stream.StreamOutput;
028import org.elasticsearch.index.VersionType;
029
030import java.io.IOException;
031
032import static org.elasticsearch.action.ValidateActions.addValidationError;
033
034/**
035 * A request to delete a document from an index based on its type and id. Best created using
036 * {@link org.elasticsearch.client.Requests#deleteRequest(String)}.
037 * <p/>
038 * <p>The operation requires the {@link #index()}, {@link #type(String)} and {@link #id(String)} to
039 * be set.
040 *
041 * @see DeleteResponse
042 * @see org.elasticsearch.client.Client#delete(DeleteRequest)
043 * @see org.elasticsearch.client.Requests#deleteRequest(String)
044 */
045public class DeleteRequest extends ShardReplicationOperationRequest<DeleteRequest> {
046
047    private String type;
048    private String id;
049    @Nullable
050    private String routing;
051    @Nullable
052    private String parent;
053    private boolean refresh;
054    private long version;
055    private VersionType versionType = VersionType.INTERNAL;
056
057    /**
058     * Constructs a new delete request against the specified index. The {@link #type(String)} and {@link #id(String)}
059     * must be set.
060     */
061    public DeleteRequest(String index) {
062        this.index = index;
063    }
064
065    /**
066     * Constructs a new delete request against the specified index with the type and id.
067     *
068     * @param index The index to get the document from
069     * @param type  The type of the document
070     * @param id    The id of the document
071     */
072    public DeleteRequest(String index, String type, String id) {
073        this.index = index;
074        this.type = type;
075        this.id = id;
076    }
077
078    public DeleteRequest(DeleteRequest request) {
079        super(request);
080        this.type = request.type();
081        this.id = request.id();
082        this.routing = request.routing();
083        this.refresh = request.refresh();
084        this.version = request.version();
085        this.versionType = request.versionType();
086    }
087
088    public DeleteRequest() {
089    }
090
091    @Override
092    public ActionRequestValidationException validate() {
093        ActionRequestValidationException validationException = super.validate();
094        if (type == null) {
095            validationException = addValidationError("type is missing", validationException);
096        }
097        if (id == null) {
098            validationException = addValidationError("id is missing", validationException);
099        }
100        return validationException;
101    }
102
103    /**
104     * The type of the document to delete.
105     */
106    public String type() {
107        return type;
108    }
109
110    /**
111     * Sets the type of the document to delete.
112     */
113    @Required
114    public DeleteRequest type(String type) {
115        this.type = type;
116        return this;
117    }
118
119    /**
120     * The id of the document to delete.
121     */
122    public String id() {
123        return id;
124    }
125
126    /**
127     * Sets the id of the document to delete.
128     */
129    @Required
130    public DeleteRequest id(String id) {
131        this.id = id;
132        return this;
133    }
134
135    /**
136     * Sets the parent id of this document. Will simply set the routing to this value, as it is only
137     * used for routing with delete requests.
138     */
139    public DeleteRequest parent(String parent) {
140        this.parent = parent;
141        if (routing == null) {
142            routing = parent;
143        }
144        return this;
145    }
146    
147    public String parent() {
148        return parent;
149    }
150
151    /**
152     * Controls the shard routing of the request. Using this value to hash the shard
153     * and not the id.
154     */
155    public DeleteRequest routing(String routing) {
156        if (routing != null && routing.length() == 0) {
157            this.routing = null;
158        } else {
159            this.routing = routing;
160        }
161        return this;
162    }
163
164    /**
165     * Controls the shard routing of the delete request. Using this value to hash the shard
166     * and not the id.
167     */
168    public String routing() {
169        return this.routing;
170    }
171
172    /**
173     * Should a refresh be executed post this index operation causing the operation to
174     * be searchable. Note, heavy indexing should not set this to <tt>true</tt>. Defaults
175     * to <tt>false</tt>.
176     */
177    public DeleteRequest refresh(boolean refresh) {
178        this.refresh = refresh;
179        return this;
180    }
181
182    public boolean refresh() {
183        return this.refresh;
184    }
185
186    /**
187     * Sets the version, which will cause the delete operation to only be performed if a matching
188     * version exists and no changes happened on the doc since then.
189     */
190    public DeleteRequest version(long version) {
191        this.version = version;
192        return this;
193    }
194
195    public long version() {
196        return this.version;
197    }
198
199    public DeleteRequest versionType(VersionType versionType) {
200        this.versionType = versionType;
201        return this;
202    }
203
204    public VersionType versionType() {
205        return this.versionType;
206    }
207
208    @Override
209    public void readFrom(StreamInput in) throws IOException {
210        super.readFrom(in);
211        type = in.readString();
212        id = in.readString();
213        routing = in.readOptionalString();
214        refresh = in.readBoolean();
215        version = in.readLong();
216        versionType = VersionType.fromValue(in.readByte());
217    }
218
219    @Override
220    public void writeTo(StreamOutput out) throws IOException {
221        super.writeTo(out);
222        out.writeString(type);
223        out.writeString(id);
224        out.writeOptionalString(routing());
225        out.writeBoolean(refresh);
226        out.writeLong(version);
227        out.writeByte(versionType.getValue());
228    }
229
230    @Override
231    public String toString() {
232        return "delete {[" + index + "][" + type + "][" + id + "]}";
233    }
234}
