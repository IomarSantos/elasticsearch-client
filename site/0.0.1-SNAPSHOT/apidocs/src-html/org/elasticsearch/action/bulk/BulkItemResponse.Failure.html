020package org.elasticsearch.action.bulk;
021
022import org.elasticsearch.action.ActionResponse;
023import org.elasticsearch.action.delete.DeleteResponse;
024import org.elasticsearch.action.index.IndexResponse;
025import org.elasticsearch.common.io.stream.StreamInput;
026import org.elasticsearch.common.io.stream.StreamOutput;
027import org.elasticsearch.common.io.stream.Streamable;
028
029import java.io.IOException;
030
031/**
032 * Represents a single item response for an action executed as part of the bulk API. Holds the index/type/id
033 * of the relevant action, and if it has failed or not (with the failure message incase it failed).
034 *
035 *
036 */
037public class BulkItemResponse implements Streamable {
038
039    /**
040     * Represents a failure.
041     */
042    public static class Failure {
043        private final String index;
044        private final String type;
045        private final String id;
046        private final String message;
047
048        public Failure(String index, String type, String id, String message) {
049            this.index = index;
050            this.type = type;
051            this.id = id;
052            this.message = message;
053        }
054
055        /**
056         * The index name of the action.
057         */
058        public String index() {
059            return this.index;
060        }
061
062        /**
063         * The index name of the action.
064         */
065        public String getIndex() {
066            return index();
067        }
068
069        /**
070         * The type of the action.
071         */
072        public String type() {
073            return type;
074        }
075
076        /**
077         * The type of the action.
078         */
079        public String getType() {
080            return type();
081        }
082
083        /**
084         * The id of the action.
085         */
086        public String id() {
087            return id;
088        }
089
090        /**
091         * The id of the action.
092         */
093        public String getId() {
094            return this.id;
095        }
096
097        /**
098         * The failure message.
099         */
100        public String message() {
101            return this.message;
102        }
103
104        /**
105         * The failure message.
106         */
107        public String getMessage() {
108            return message();
109        }
110    }
111
112    private int id;
113
114    private String opType;
115
116    private ActionResponse response;
117
118    private Failure failure;
119
120    BulkItemResponse() {
121
122    }
123
124    public BulkItemResponse(int id, String opType, ActionResponse response) {
125        this.id = id;
126        this.opType = opType;
127        this.response = response;
128    }
129
130    public BulkItemResponse(int id, String opType, Failure failure) {
131        this.id = id;
132        this.opType = opType;
133        this.failure = failure;
134    }
135
136    /**
137     * The numeric order of the item matching the same request order in the bulk request.
138     */
139    public int itemId() {
140        return id;
141    }
142
143    /**
144     * The operation type ("index", "create" or "delete").
145     */
146    public String opType() {
147        return this.opType;
148    }
149
150    /**
151     * The index name of the action.
152     */
153    public String index() {
154        if (failure != null) {
155            return failure.index();
156        }
157        if (response instanceof IndexResponse) {
158            return ((IndexResponse) response).index();
159        } else if (response instanceof DeleteResponse) {
160            return ((DeleteResponse) response).index();
161        }
162        return null;
163    }
164
165    /**
166     * The index name of the action.
167     */
168    public String getIndex() {
169        return index();
170    }
171
172    /**
173     * The type of the action.
174     */
175    public String type() {
176        if (failure != null) {
177            return failure.type();
178        }
179        if (response instanceof IndexResponse) {
180            return ((IndexResponse) response).type();
181        } else if (response instanceof DeleteResponse) {
182            return ((DeleteResponse) response).type();
183        }
184        return null;
185    }
186
187    /**
188     * The type of the action.
189     */
190    public String getType() {
191        return this.type();
192    }
193
194    /**
195     * The id of the action.
196     */
197    public String id() {
198        if (failure != null) {
199            return failure.id();
200        }
201        if (response instanceof IndexResponse) {
202            return ((IndexResponse) response).id();
203        } else if (response instanceof DeleteResponse) {
204            return ((DeleteResponse) response).id();
205        }
206        return null;
207    }
208
209    /**
210     * The id of the action.
211     */
212    public String getId() {
213        return id();
214    }
215
216    /**
217     * The version of the action.
218     */
219    public long version() {
220        if (failure != null) {
221            return -1;
222        }
223        if (response instanceof IndexResponse) {
224            return ((IndexResponse) response).version();
225        } else if (response instanceof DeleteResponse) {
226            return ((DeleteResponse) response).version();
227        }
228        return -1;
229    }
230
231    /**
232     * The actual response ({@link IndexResponse} or {@link DeleteResponse}). <tt>null</tt> in
233     * case of failure.
234     */
235    public <T extends ActionResponse> T response() {
236        return (T) response;
237    }
238
239    /**
240     * Is this a failed execution of an operation.
241     */
242    public boolean failed() {
243        return failure != null;
244    }
245
246    /**
247     * Is this a failed execution of an operation.
248     */
249    public boolean isFailed() {
250        return failed();
251    }
252
253    /**
254     * The failure message, <tt>null</tt> if it did not fail.
255     */
256    public String failureMessage() {
257        if (failure != null) {
258            return failure.message();
259        }
260        return null;
261    }
262
263    /**
264     * The failure message, <tt>null</tt> if it did not fail.
265     */
266    public String getFailureMessage() {
267        return failureMessage();
268    }
269
270    /**
271     * The actual failure object if there was a failure.
272     */
273    public Failure failure() {
274        return this.failure;
275    }
276
277    /**
278     * The actual failure object if there was a failure.
279     */
280    public Failure getFailure() {
281        return failure();
282    }
283
284    public static BulkItemResponse readBulkItem(StreamInput in) throws IOException {
285        BulkItemResponse response = new BulkItemResponse();
286        response.readFrom(in);
287        return response;
288    }
289
290    @Override
291    public void readFrom(StreamInput in) throws IOException {
292        id = in.readVInt();
293        opType = in.readUTF();
294
295        byte type = in.readByte();
296        if (type == 0) {
297            response = new IndexResponse();
298            response.readFrom(in);
299        } else if (type == 1) {
300            response = new DeleteResponse();
301            response.readFrom(in);
302        }
303
304        if (in.readBoolean()) {
305            failure = new Failure(in.readUTF(), in.readUTF(), in.readUTF(), in.readUTF());
306        }
307    }
308
309    @Override
310    public void writeTo(StreamOutput out) throws IOException {
311        out.writeVInt(id);
312        out.writeUTF(opType);
313        if (response == null) {
314            out.writeByte((byte) 2);
315        } else {
316            if (response instanceof IndexResponse) {
317                out.writeByte((byte) 0);
318            } else if (response instanceof DeleteResponse) {
319                out.writeByte((byte) 1);
320            }
321            response.writeTo(out);
322        }
323        if (failure == null) {
324            out.writeBoolean(false);
325        } else {
326            out.writeBoolean(true);
327            out.writeUTF(failure.index());
328            out.writeUTF(failure.type());
329            out.writeUTF(failure.id());
330            out.writeUTF(failure.message());
331        }
332    }
333}
