046/**
047 * A bulk request holds an ordered {@link IndexRequest}s and {@link DeleteRequest}s and allows to executes
048 * it in a single batch.
049 *
050 * @see org.elasticsearch.client.Client#bulk(BulkRequest)
051 */
052public class BulkRequest extends ActionRequest<BulkRequest> {
053
054    private static final int REQUEST_OVERHEAD = 50;
055
056    final List<ActionRequest> requests = Lists.newArrayList();
057
058    private ReplicationType replicationType = ReplicationType.DEFAULT;
059    private WriteConsistencyLevel consistencyLevel = WriteConsistencyLevel.DEFAULT;
060    private boolean refresh = false;
061
062    private long sizeInBytes = 0;
063
064    /**
065     * Adds a list of requests to be executed. Either index or delete requests.
066     */
067    public BulkRequest add(ActionRequest... requests) {
068        for (ActionRequest request : requests) {
069            if (request instanceof IndexRequest) {
070                add((IndexRequest) request);
071            } else if (request instanceof DeleteRequest) {
072                add((DeleteRequest) request);
073            } else {
074                throw new ElasticSearchIllegalArgumentException("No support for request [" + request + "]");
075            }
076        }
077        return this;
078    }
079
080    /**
081     * Adds a list of requests to be executed. Either index or delete requests.
082     */
083    public BulkRequest add(Iterable<ActionRequest> requests) {
084        for (ActionRequest request : requests) {
085            if (request instanceof IndexRequest) {
086                add((IndexRequest) request);
087            } else if (request instanceof DeleteRequest) {
088                add((DeleteRequest) request);
089            } else {
090                throw new ElasticSearchIllegalArgumentException("No support for request [" + request + "]");
091            }
092        }
093        return this;
094    }
095
096    /**
097     * Adds an {@link IndexRequest} to the list of actions to execute. Follows the same behavior of {@link IndexRequest}
098     * (for example, if no id is provided, one will be generated, or usage of the create flag).
099     */
100    public BulkRequest add(IndexRequest request) {
101        request.beforeLocalFork();
102        return internalAdd(request);
103    }
104
105    BulkRequest internalAdd(IndexRequest request) {
106        requests.add(request);
107        sizeInBytes += request.source().length() + REQUEST_OVERHEAD;
108        return this;
109    }
110
111    /**
112     * Adds an {@link DeleteRequest} to the list of actions to execute.
113     */
114    public BulkRequest add(DeleteRequest request) {
115        requests.add(request);
116        sizeInBytes += REQUEST_OVERHEAD;
117        return this;
118    }

120    public List<ActionRequest> requests() {
121        return this.requests;

125        return requests.size();

139    /**
140     * Adds a framed data in binary format
141     */
142    public BulkRequest add(byte[] data, int from, int length, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
143        return add(new BytesArray(data, from, length), contentUnsafe, defaultIndex, defaultType);
144    }

146    /**
147     * Adds a framed data in binary format
148     */
149    public BulkRequest add(BytesReference data, boolean contentUnsafe, @Nullable String defaultIndex, @Nullable String defaultType) throws Exception {
150        XContent xContent = XContentFactory.xContent(data);
151        int from = 0;
152        int length = data.length();
153        byte marker = xContent.streamSeparator();
154        while (true) {
155            int nextMarker = findNextMarker(marker, from, data, length);
156            if (nextMarker == -1) {
157                break;
158            }
159            // now parse the action
160            XContentParser parser = xContent.createParser(data.slice(from, nextMarker - from));
161
162            try {
163                // move pointers
164                from = nextMarker + 1;
165
166                // Move to START_OBJECT
167                XContentParser.Token token = parser.nextToken();
168                if (token == null) {
169                    continue;
170                }
171                assert token == XContentParser.Token.START_OBJECT;
172                // Move to FIELD_NAME, that's the action
173                token = parser.nextToken();
174                assert token == XContentParser.Token.FIELD_NAME;
175                String action = parser.currentName();
176
177                String index = defaultIndex;
178                String type = defaultType;
179                String id = null;
180                String routing = null;
181                String parent = null;
182                String timestamp = null;
183                Long ttl = null;
184                String opType = null;
185                long version = 0;
186                VersionType versionType = VersionType.INTERNAL;
187                String percolate = null;
188
189                // at this stage, next token can either be END_OBJECT (and use default index and type, with auto generated id)
190                // or START_OBJECT which will have another set of parameters
191
192                String currentFieldName = null;
193                while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
194                    if (token == XContentParser.Token.FIELD_NAME) {
195                        currentFieldName = parser.currentName();
196                    } else if (token.isValue()) {
197                        if ("_index".equals(currentFieldName)) {
198                            index = parser.text();
199                        } else if ("_type".equals(currentFieldName)) {
200                            type = parser.text();
201                        } else if ("_id".equals(currentFieldName)) {
202                            id = parser.text();
203                        } else if ("_routing".equals(currentFieldName) || "routing".equals(currentFieldName)) {
204                            routing = parser.text();
205                        } else if ("_parent".equals(currentFieldName) || "parent".equals(currentFieldName)) {
206                            parent = parser.text();
207                        } else if ("_timestamp".equals(currentFieldName) || "timestamp".equals(currentFieldName)) {
208                            timestamp = parser.text();
209                        } else if ("_ttl".equals(currentFieldName) || "ttl".equals(currentFieldName)) {
210                            if (parser.currentToken() == XContentParser.Token.VALUE_STRING) {
211                                ttl = TimeValue.parseTimeValue(parser.text(), null).millis();
212                            } else {
213                                ttl = parser.longValue();
214                            }
215                        } else if ("op_type".equals(currentFieldName) || "opType".equals(currentFieldName)) {
216                            opType = parser.text();
217                        } else if ("_version".equals(currentFieldName) || "version".equals(currentFieldName)) {
218                            version = parser.longValue();
219                        } else if ("_version_type".equals(currentFieldName) || "_versionType".equals(currentFieldName) || "version_type".equals(currentFieldName) || "versionType".equals(currentFieldName)) {
220                            versionType = VersionType.fromString(parser.text());
221                        } else if ("percolate".equals(currentFieldName) || "_percolate".equals(currentFieldName)) {
222                            percolate = parser.textOrNull();
223                        }
224                    }
225                }
226
227                if ("delete".equals(action)) {
228                    add(new DeleteRequest(index, type, id).parent(parent).version(version).versionType(versionType).routing(routing));
229                } else {
230                    nextMarker = findNextMarker(marker, from, data, length);
231                    if (nextMarker == -1) {
232                        break;
233                    }
234                    // order is important, we set parent after routing, so routing will be set to parent if not set explicitly
235                    // we use internalAdd so we don't fork here, this allows us not to copy over the big byte array to small chunks
236                    // of index request. All index requests are still unsafe if applicable.
237                    if ("index".equals(action)) {
238                        if (opType == null) {
239                            internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
240                                    .source(data.slice(from, nextMarker - from), contentUnsafe)
241                                    .percolate(percolate));
242                        } else {
243                            internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
244                                    .create("create".equals(opType))
245                                    .source(data.slice(from, nextMarker - from), contentUnsafe)
246                                    .percolate(percolate));
247                        }
248                    } else if ("create".equals(action)) {
249                        internalAdd(new IndexRequest(index, type, id).routing(routing).parent(parent).timestamp(timestamp).ttl(ttl).version(version).versionType(versionType)
250                                .create(true)
251                                .source(data.slice(from, nextMarker - from), contentUnsafe)
252                                .percolate(percolate));
253                    }
254                    // move pointers
255                    from = nextMarker + 1;
256                }
257            } finally {
258                parser.close();
259            }
260        }
261        return this;
262    }

301
302    private int findNextMarker(byte marker, int from, BytesReference data, int length) {
303        for (int i = from; i < length; i++) {
304            if (data.get(i) == marker) {
305                return i;
306            }
307        }
308        return -1;
309    }
310
311    @Override
312    public ActionRequestValidationException validate() {
313        ActionRequestValidationException validationException = null;
314        if (requests.isEmpty()) {
315            validationException = addValidationError("no requests added", validationException);
316        }
317        for (int i = 0; i < requests.size(); i++) {
318            ActionRequestValidationException ex = requests.get(i).validate();
319            if (ex != null) {
320                if (validationException == null) {
321                    validationException = new ActionRequestValidationException();
322                }
323                validationException.addValidationErrors(ex.validationErrors());
324            }
325        }
326
327        return validationException;
328    }
329
330    @Override
331    public void readFrom(StreamInput in) throws IOException {
332        super.readFrom(in);
333        replicationType = ReplicationType.fromId(in.readByte());
334        consistencyLevel = WriteConsistencyLevel.fromId(in.readByte());
335        int size = in.readVInt();
336        for (int i = 0; i < size; i++) {
337            byte type = in.readByte();
338            if (type == 0) {
339                IndexRequest request = new IndexRequest();
340                request.readFrom(in);
341                requests.add(request);
342            } else if (type == 1) {
343                DeleteRequest request = new DeleteRequest();
344                request.readFrom(in);
345                requests.add(request);
346            }
347        }
348        refresh = in.readBoolean();
349    }
350
351    @Override
352    public void writeTo(StreamOutput out) throws IOException {
353        super.writeTo(out);
354        out.writeByte(replicationType.id());
355        out.writeByte(consistencyLevel.id());
356        out.writeVInt(requests.size());
357        for (ActionRequest request : requests) {
358            if (request instanceof IndexRequest) {
359                out.writeByte((byte) 0);
360            } else if (request instanceof DeleteRequest) {
361                out.writeByte((byte) 1);
362            }
363            request.writeTo(out);
364        }
365        out.writeBoolean(refresh);
366    }
