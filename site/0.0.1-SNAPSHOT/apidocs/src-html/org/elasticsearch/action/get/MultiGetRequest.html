041    /**
042     * A single get item.
043     */
044    public static class Item implements Streamable {
045        private String index;
046        private String type;
047        private String id;
048        private String routing;
049        private String[] fields;
050
051        Item() {
052
053        }
054
055        /**
056         * Constructs a single get item.
057         *
058         * @param index The index name
059         * @param type  The type (can be null)
060         * @param id    The id
061         */
062        public Item(String index, @Nullable String type, String id) {
063            this.index = index;
064            this.type = type;
065            this.id = id;
066        }
067
068        public String index() {
069            return this.index;
070        }
071
072        public Item index(String index) {
073            this.index = index;
074            return this;
075        }
076
077        public String type() {
078            return this.type;
079        }
080
081        public String id() {
082            return this.id;
083        }
084
085        /**
086         * The routing associated with this document.
087         */
088        public Item routing(String routing) {
089            this.routing = routing;
090            return this;
091        }
092
093        public String routing() {
094            return this.routing;
095        }
096
097        public Item fields(String... fields) {
098            this.fields = fields;
099            return this;
100        }
101
102        public String[] fields() {
103            return this.fields;
104        }
105
106        public static Item readItem(StreamInput in) throws IOException {
107            Item item = new Item();
108            item.readFrom(in);
109            return item;
110        }
111
112        @Override
113        public void readFrom(StreamInput in) throws IOException {
114            index = in.readUTF();
115            if (in.readBoolean()) {
116                type = in.readUTF();
117            }
118            id = in.readUTF();
119            if (in.readBoolean()) {
120                routing = in.readUTF();
121            }
122            int size = in.readVInt();
123            if (size > 0) {
124                fields = new String[size];
125                for (int i = 0; i < size; i++) {
126                    fields[i] = in.readUTF();
127                }
128            }
129        }
130
131        @Override
132        public void writeTo(StreamOutput out) throws IOException {
133            out.writeUTF(index);
134            if (type == null) {
135                out.writeBoolean(false);
136            } else {
137                out.writeBoolean(true);
138                out.writeUTF(type);
139            }
140            out.writeUTF(id);
141            if (routing == null) {
142                out.writeBoolean(false);
143            } else {
144                out.writeBoolean(true);
145                out.writeUTF(routing);
146            }
147            if (fields == null) {
148                out.writeVInt(0);
149            } else {
150                out.writeVInt(fields.length);
151                for (String field : fields) {
152                    out.writeUTF(field);
153                }
154            }
155        }
156    }

180    @Override
181    public ActionRequestValidationException validate() {
182        ActionRequestValidationException validationException = null;
183        if (items.isEmpty()) {
184            validationException = ValidateActions.addValidationError("no documents to get", validationException);
185        } else {
186            for (int i = 0; i < items.size(); i++) {
187                Item item = items.get(i);
188                if (item.index() == null) {
189                    validationException = ValidateActions.addValidationError("index is missing for doc " + i, validationException);
190                }
191                if (item.id() == null) {
192                    validationException = ValidateActions.addValidationError("id is missing for doc " + i, validationException);
193                }
194            }
195        }
196        return validationException;
197    }

231    public void add(@Nullable String defaultIndex, @Nullable String defaultType, @Nullable String[] defaultFields, byte[] data, int from, int length) throws Exception {
232        add(defaultIndex, defaultType, defaultFields, new BytesArray(data, from, length));

235    public void add(@Nullable String defaultIndex, @Nullable String defaultType, @Nullable String[] defaultFields, BytesReference data) throws Exception {
236        XContentParser parser = XContentFactory.xContent(data).createParser(data);
237        try {
238            XContentParser.Token token;
239            String currentFieldName = null;
240            while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
241                if (token == XContentParser.Token.FIELD_NAME) {
242                    currentFieldName = parser.currentName();
243                } else if (token == XContentParser.Token.START_ARRAY) {
244                    if ("docs".equals(currentFieldName)) {
245                        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
246                            if (token != XContentParser.Token.START_OBJECT) {
247                                throw new ElasticSearchIllegalArgumentException("docs array element should include an object");
248                            }
249                            String index = defaultIndex;
250                            String type = defaultType;
251                            String id = null;
252                            String routing = null;
253                            List<String> fields = null;
254                            while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
255                                if (token == XContentParser.Token.FIELD_NAME) {
256                                    currentFieldName = parser.currentName();
257                                } else if (token.isValue()) {
258                                    if ("_index".equals(currentFieldName)) {
259                                        index = parser.text();
260                                    } else if ("_type".equals(currentFieldName)) {
261                                        type = parser.text();
262                                    } else if ("_id".equals(currentFieldName)) {
263                                        id = parser.text();
264                                    } else if ("_routing".equals(currentFieldName) || "routing".equals(currentFieldName)) {
265                                        routing = parser.text();
266                                    }
267                                } else if (token == XContentParser.Token.START_ARRAY) {
268                                    if ("fields".equals(currentFieldName)) {
269                                        fields = new ArrayList<String>();
270                                        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
271                                            fields.add(parser.text());
272                                        }
273                                    }
274                                }
275                            }
276                            String[] aFields;
277                            if (fields != null) {
278                                aFields = fields.toArray(new String[fields.size()]);
279                            } else {
280                                aFields = defaultFields;
281                            }
282                            add(new Item(index, type, id).routing(routing).fields(aFields));
283                        }
284                    } else if ("ids".equals(currentFieldName)) {
285                        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
286                            if (!token.isValue()) {
287                                throw new ElasticSearchIllegalArgumentException("ids array element should only contain ids");
288                            }
289                            add(new Item(defaultIndex, defaultType, parser.text()).fields(defaultFields));
290                        }
291                    }
292                }
293            }
294        } finally {
295            parser.close();
296        }
297    }

299    @Override
300    public void readFrom(StreamInput in) throws IOException {
301        super.readFrom(in);
302        preference = in.readOptionalString();
303        refresh = in.readBoolean();
304        byte realtime = in.readByte();
305        if (realtime == 0) {
306            this.realtime = false;
307        } else if (realtime == 1) {
308            this.realtime = true;
309        }
310
311        int size = in.readVInt();
312        items = new ArrayList<Item>(size);
313        for (int i = 0; i < size; i++) {
314            items.add(Item.readItem(in));
315        }
316    }
317
318    @Override
319    public void writeTo(StreamOutput out) throws IOException {
320        super.writeTo(out);
321        out.writeOptionalString(preference);
322        out.writeBoolean(refresh);
323        if (realtime == null) {
324            out.writeByte((byte) -1);
325        } else if (realtime == false) {
326            out.writeByte((byte) 0);
327        } else {
328            out.writeByte((byte) 1);
329        }
330
331        out.writeVInt(items.size());
332        for (Item item : items) {
333            item.writeTo(out);
334        }
335    }
