020package org.elasticsearch.action.search;
021
022import org.elasticsearch.action.ActionListener;
023import org.elasticsearch.action.ActionRequestBuilder;
024import org.elasticsearch.action.support.IgnoreIndices;
025import org.elasticsearch.client.SearchClient;
026
027/**
028 * A request builder for multiple search requests.
029 */
030public class MultiSearchRequestBuilder extends ActionRequestBuilder<MultiSearchRequest, MultiSearchResponse, MultiSearchRequestBuilder> {
031
032    public MultiSearchRequestBuilder(SearchClient client) {
033        super(client, new MultiSearchRequest());
034    }
035
036    /**
037     * Add a search request to execute. Note, the order is important, the search response will be returned in the
038     * same order as the search requests.
039     * <p/>
040     * If ignoreIndices has been set on the search request, then the ignoreIndices of the multi search request
041     * will not be used (if set).
042     */
043    public MultiSearchRequestBuilder add(SearchRequest request) {
044        if (request.ignoreIndices() == IgnoreIndices.DEFAULT && request().ignoreIndices() != IgnoreIndices.DEFAULT) {
045            request.ignoreIndices(request().ignoreIndices());
046        }
047
048        super.request.add(request);
049        return this;
050    }
051
052    /**
053     * Add a search request to execute. Note, the order is important, the search response will be returned in the
054     * same order as the search requests.
055     */
056    public MultiSearchRequestBuilder add(SearchRequestBuilder request) {
057        if (request.request().ignoreIndices() == IgnoreIndices.DEFAULT && request().ignoreIndices() != IgnoreIndices.DEFAULT) {
058            request.request().ignoreIndices(request().ignoreIndices());
059        }
060
061        super.request.add(request);
062        return this;
063    }
064
065    /**
066     * Specifies what type of requested indices to ignore. For example indices that don't exist.
067     * Invoke this method before invoking {@link #add(SearchRequestBuilder)}.
068     */
069    public MultiSearchRequestBuilder setIgnoreIndices(IgnoreIndices ignoreIndices) {
070        request().ignoreIndices(ignoreIndices);
071        return this;
072    }
073
074    @Override
075    protected void doExecute(ActionListener<MultiSearchResponse> listener) {
076        ((SearchClient) client).multiSearch(request, listener);
077    }
078}
