001package org.elasticsearch.action.search;
002
003import com.google.common.collect.Iterators;
004import org.elasticsearch.action.ActionResponse;
005import org.elasticsearch.common.Nullable;
006import org.elasticsearch.common.io.stream.StreamInput;
007import org.elasticsearch.common.io.stream.StreamOutput;
008import org.elasticsearch.common.io.stream.Streamable;
009import org.elasticsearch.common.xcontent.ToXContent;
010import org.elasticsearch.common.xcontent.XContentBuilder;
011import org.elasticsearch.common.xcontent.XContentBuilderString;
012
013import java.io.IOException;
014import java.util.Iterator;
015
016/**
017 * A multi search response.
018 */
019public class MultiSearchResponse extends ActionResponse implements Iterable<MultiSearchResponse.Item>, ToXContent {
020
021    /**
022     * A search response item, holding the actual search response, or an error message if it failed.
023     */
024    public static class Item implements Streamable {
025        private SearchResponse response;
026        private String failureMessage;
027
028        Item() {
029
030        }
031
032        public Item(SearchResponse response, String failureMessage) {
033            this.response = response;
034            this.failureMessage = failureMessage;
035        }
036
037        /**
038         * Is it a failed search?
039         */
040        public boolean isFailure() {
041            return failureMessage != null;
042        }
043
044        /**
045         * The actual failure message, null if its not a failure.
046         */
047        @Nullable
048        public String failureMessage() {
049            return failureMessage;
050        }
051
052        /**
053         * The actual failure message, null if its not a failure.
054         */
055        @Nullable
056        public String getFailureMessage() {
057            return failureMessage;
058        }
059
060        /**
061         * The actual search response, null if its a failure.
062         */
063        @Nullable
064        public SearchResponse response() {
065            return this.response;
066        }
067
068        /**
069         * The actual search response, null if its a failure.
070         */
071        @Nullable
072        public SearchResponse getResponse() {
073            return this.response;
074        }
075
076        public static Item readItem(StreamInput in) throws IOException {
077            Item item = new Item();
078            item.readFrom(in);
079            return item;
080        }
081
082        @Override
083        public void readFrom(StreamInput in) throws IOException {
084            if (in.readBoolean()) {
085                this.response = new SearchResponse();
086                response.readFrom(in);
087            } else {
088                failureMessage = in.readString();
089            }
090        }
091
092        @Override
093        public void writeTo(StreamOutput out) throws IOException {
094            if (response != null) {
095                out.writeBoolean(true);
096                response.writeTo(out);
097            } else {
098                out.writeString(failureMessage);
099            }
100        }
101    }
102
103    private Item[] items;
104
105    MultiSearchResponse() {
106    }
107
108    public MultiSearchResponse(Item[] items) {
109        this.items = items;
110    }
111
112    @Override
113    public Iterator<Item> iterator() {
114        return Iterators.forArray(items);
115    }
116
117    /**
118     * The list of responses, the order is the same as the one provided in the request.
119     */
120    public Item[] responses() {
121        return this.items;
122    }
123
124    /**
125     * The list of responses, the order is the same as the one provided in the request.
126     */
127    public Item[] getResponses() {
128        return this.items;
129    }
130
131    @Override
132    public void readFrom(StreamInput in) throws IOException {
133        super.readFrom(in);
134        items = new Item[in.readVInt()];
135        for (int i = 0; i < items.length; i++) {
136            items[i] = Item.readItem(in);
137        }
138    }
139
140    @Override
141    public void writeTo(StreamOutput out) throws IOException {
142        super.writeTo(out);
143        out.writeVInt(items.length);
144        for (Item item : items) {
145            item.writeTo(out);
146        }
147    }
148
149    @Override
150    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
151        builder.startArray(Fields.RESPONSES);
152        for (Item item : items) {
153            if (item.isFailure()) {
154                builder.startObject();
155                builder.field(Fields.ERROR, item.failureMessage());
156                builder.endObject();
157            } else {
158                builder.startObject();
159                item.response().toXContent(builder, params);
160                builder.endObject();
161            }
162        }
163        builder.endArray();
164        return builder;
165    }
166
167    static final class Fields {
168        static final XContentBuilderString RESPONSES = new XContentBuilderString("responses");
169        static final XContentBuilderString ERROR = new XContentBuilderString("error");
170    }
171}
