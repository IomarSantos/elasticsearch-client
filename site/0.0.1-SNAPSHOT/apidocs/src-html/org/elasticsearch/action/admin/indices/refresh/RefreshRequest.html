020package org.elasticsearch.action.admin.indices.refresh;
021
022import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
023import org.elasticsearch.action.support.broadcast.BroadcastOperationThreading;
024import org.elasticsearch.common.io.stream.StreamInput;
025import org.elasticsearch.common.io.stream.StreamOutput;
026
027import java.io.IOException;
028
029/**
030 * A refresh request making all operations performed since the last refresh available for search. The (near) real-time
031 * capabilities depends on the index engine used. For example, the robin one requires refresh to be called, but by
032 * default a refresh is scheduled periodically.
033 *
034 * @see org.elasticsearch.client.Requests#refreshRequest(String...)
035 * @see org.elasticsearch.client.IndicesAdminClient#refresh(RefreshRequest)
036 * @see RefreshResponse
037 */
038public class RefreshRequest extends BroadcastOperationRequest<RefreshRequest> {
039
040    private boolean waitForOperations = true;
041
042    RefreshRequest() {
043    }
044
045    public RefreshRequest(String... indices) {
046        super(indices);
047        // we want to do the refresh in parallel on local shards...
048        operationThreading(BroadcastOperationThreading.THREAD_PER_SHARD);
049    }
050
051    public boolean waitForOperations() {
052        return waitForOperations;
053    }
054
055    public RefreshRequest waitForOperations(boolean waitForOperations) {
056        this.waitForOperations = waitForOperations;
057        return this;
058    }
059
060    public void readFrom(StreamInput in) throws IOException {
061        super.readFrom(in);
062        waitForOperations = in.readBoolean();
063    }
064
065    public void writeTo(StreamOutput out) throws IOException {
066        super.writeTo(out);
067        out.writeBoolean(waitForOperations);
068    }
069}
