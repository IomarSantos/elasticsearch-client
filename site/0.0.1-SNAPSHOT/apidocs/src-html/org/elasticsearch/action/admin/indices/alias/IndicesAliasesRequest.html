020package org.elasticsearch.action.admin.indices.alias;
021
022import com.google.common.collect.Lists;
023import org.elasticsearch.ElasticSearchGenerationException;
024import org.elasticsearch.action.ActionRequestValidationException;
025import org.elasticsearch.action.support.master.MasterNodeOperationRequest;
026import org.elasticsearch.cluster.metadata.AliasAction;
027import org.elasticsearch.common.io.stream.StreamInput;
028import org.elasticsearch.common.io.stream.StreamOutput;
029import org.elasticsearch.common.unit.TimeValue;
030import org.elasticsearch.common.xcontent.ToXContent;
031import org.elasticsearch.common.xcontent.XContentBuilder;
032import org.elasticsearch.common.xcontent.XContentFactory;
033import org.elasticsearch.common.xcontent.XContentType;
034import org.elasticsearch.index.query.FilterBuilder;
035
036import java.io.IOException;
037import java.util.List;
038import java.util.Map;
039
040import static org.elasticsearch.action.ValidateActions.addValidationError;
041import static org.elasticsearch.cluster.metadata.AliasAction.readAliasAction;
042import static org.elasticsearch.common.unit.TimeValue.readTimeValue;
043
044/**
045 * A request to add/remove aliases for one or more indices.
046 */
047public class IndicesAliasesRequest extends MasterNodeOperationRequest<IndicesAliasesRequest> {
048
049    private List<AliasAction> aliasActions = Lists.newArrayList();
050
051    private TimeValue timeout = TimeValue.timeValueSeconds(10);
052
053    public IndicesAliasesRequest() {
054
055    }
056
057    /**
058     * Adds an alias to the index.
059     *
060     * @param index The index
061     * @param alias The alias
062     */
063    public IndicesAliasesRequest addAlias(String index, String alias) {
064        aliasActions.add(new AliasAction(AliasAction.Type.ADD, index, alias));
065        return this;
066    }
067
068    /**
069     * Adds an alias to the index.
070     *
071     * @param index  The index
072     * @param alias  The alias
073     * @param filter The filter
074     */
075    public IndicesAliasesRequest addAlias(String index, String alias, String filter) {
076        aliasActions.add(new AliasAction(AliasAction.Type.ADD, index, alias, filter));
077        return this;
078    }
079
080    /**
081     * Adds an alias to the index.
082     *
083     * @param index  The index
084     * @param alias  The alias
085     * @param filter The filter
086     */
087    public IndicesAliasesRequest addAlias(String index, String alias, Map<String, Object> filter) {
088        if (filter == null || filter.isEmpty()) {
089            aliasActions.add(new AliasAction(AliasAction.Type.ADD, index, alias));
090            return this;
091        }
092        try {
093            XContentBuilder builder = XContentFactory.contentBuilder(XContentType.JSON);
094            builder.map(filter);
095            aliasActions.add(new AliasAction(AliasAction.Type.ADD, index, alias, builder.string()));
096            return this;
097        } catch (IOException e) {
098            throw new ElasticSearchGenerationException("Failed to generate [" + filter + "]", e);
099        }
100    }
101
102    /**
103     * Adds an alias to the index.
104     *
105     * @param index         The index
106     * @param alias         The alias
107     * @param filterBuilder The filter
108     */
109    public IndicesAliasesRequest addAlias(String index, String alias, FilterBuilder filterBuilder) {
110        if (filterBuilder == null) {
111            aliasActions.add(new AliasAction(AliasAction.Type.ADD, index, alias));
112            return this;
113        }
114        try {
115            XContentBuilder builder = XContentFactory.jsonBuilder();
116            filterBuilder.toXContent(builder, ToXContent.EMPTY_PARAMS);
117            builder.close();
118            return addAlias(index, alias, builder.string());
119        } catch (IOException e) {
120            throw new ElasticSearchGenerationException("Failed to build json for alias request", e);
121        }
122    }
123
124    /**
125     * Removes an alias to the index.
126     *
127     * @param index The index
128     * @param alias The alias
129     */
130    public IndicesAliasesRequest removeAlias(String index, String alias) {
131        aliasActions.add(new AliasAction(AliasAction.Type.REMOVE, index, alias));
132        return this;
133    }
134
135    public IndicesAliasesRequest addAliasAction(AliasAction action) {
136        aliasActions.add(action);
137        return this;
138    }
139
140    public List<AliasAction> aliasActions() {
141        return this.aliasActions;
142    }
143
144    /**
145     * Timeout to wait till the put mapping gets acknowledged of all current cluster nodes. Defaults to
146     * <tt>10s</tt>.
147     */
148    TimeValue timeout() {
149        return timeout;
150    }
151
152    /**
153     * Timeout to wait till the alias operations get acknowledged of all current cluster nodes. Defaults to
154     * <tt>10s</tt>.
155     */
156    public IndicesAliasesRequest timeout(TimeValue timeout) {
157        this.timeout = timeout;
158        return this;
159    }
160
161    /**
162     * Timeout to wait till the alias operations get acknowledged of all current cluster nodes. Defaults to
163     * <tt>10s</tt>.
164     */
165    public IndicesAliasesRequest timeout(String timeout) {
166        return timeout(TimeValue.parseTimeValue(timeout, TimeValue.timeValueSeconds(10)));
167    }
168
169    @Override
170    public ActionRequestValidationException validate() {
171        ActionRequestValidationException validationException = null;
172        if (aliasActions.isEmpty()) {
173            validationException = addValidationError("Must specify at least one alias action", validationException);
174        }
175        return validationException;
176    }
177
178    @Override
179    public void readFrom(StreamInput in) throws IOException {
180        super.readFrom(in);
181        int size = in.readVInt();
182        for (int i = 0; i < size; i++) {
183            aliasActions.add(readAliasAction(in));
184        }
185        timeout = readTimeValue(in);
186    }
187
188    @Override
189    public void writeTo(StreamOutput out) throws IOException {
190        super.writeTo(out);
191        out.writeVInt(aliasActions.size());
192        for (AliasAction aliasAction : aliasActions) {
193            aliasAction.writeTo(out);
194        }
195        timeout.writeTo(out);
196    }
197}
