020package org.elasticsearch.action.admin.indices.template.delete;
021
022import org.elasticsearch.action.ActionRequestValidationException;
023import org.elasticsearch.action.support.master.MasterNodeOperationRequest;
024import org.elasticsearch.common.io.stream.StreamInput;
025import org.elasticsearch.common.io.stream.StreamOutput;
026import org.elasticsearch.common.unit.TimeValue;
027
028import java.io.IOException;
029
030import static org.elasticsearch.action.ValidateActions.addValidationError;
031import static org.elasticsearch.common.unit.TimeValue.readTimeValue;
032import static org.elasticsearch.common.unit.TimeValue.timeValueSeconds;
033
034/**
035 * A request to delete an index template.
036 */
037public class DeleteIndexTemplateRequest extends MasterNodeOperationRequest<DeleteIndexTemplateRequest> {
038
039    private String name;
040
041    private TimeValue timeout = timeValueSeconds(10);
042
043    DeleteIndexTemplateRequest() {
044    }
045
046    /**
047     * Constructs a new delete index request for the specified name.
048     */
049    public DeleteIndexTemplateRequest(String name) {
050        this.name = name;
051    }
052
053    @Override
054    public ActionRequestValidationException validate() {
055        ActionRequestValidationException validationException = null;
056        if (name == null) {
057            validationException = addValidationError("name is missing", validationException);
058        }
059        return validationException;
060    }
061
062    /**
063     * The index template name to delete.
064     */
065    String name() {
066        return name;
067    }
068
069    /**
070     * Timeout to wait for the index deletion to be acknowledged by current cluster nodes. Defaults
071     * to <tt>10s</tt>.
072     */
073    TimeValue timeout() {
074        return timeout;
075    }
076
077    /**
078     * Timeout to wait for the index template deletion to be acknowledged by current cluster nodes. Defaults
079     * to <tt>10s</tt>.
080     */
081    public DeleteIndexTemplateRequest timeout(TimeValue timeout) {
082        this.timeout = timeout;
083        return this;
084    }
085
086    /**
087     * Timeout to wait for the index template deletion to be acknowledged by current cluster nodes. Defaults
088     * to <tt>10s</tt>.
089     */
090    public DeleteIndexTemplateRequest timeout(String timeout) {
091        return timeout(TimeValue.parseTimeValue(timeout, null));
092    }
093
094    @Override
095    public void readFrom(StreamInput in) throws IOException {
096        super.readFrom(in);
097        name = in.readString();
098        timeout = readTimeValue(in);
099    }
100
101    @Override
102    public void writeTo(StreamOutput out) throws IOException {
103        super.writeTo(out);
104        out.writeString(name);
105        timeout.writeTo(out);
106    }
107}
