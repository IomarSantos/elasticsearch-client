020package org.elasticsearch.action.admin.cluster.node.stats;
021
022import org.elasticsearch.action.support.nodes.NodesOperationRequest;
023import org.elasticsearch.common.io.stream.StreamInput;
024import org.elasticsearch.common.io.stream.StreamOutput;
025
026import java.io.IOException;
027
028/**
029 * A request to get node (cluster) level stats.
030 */
031public class NodesStatsRequest extends NodesOperationRequest<NodesStatsRequest> {
032
033    private boolean indices = true;
034    private boolean os;
035    private boolean process;
036    private boolean jvm;
037    private boolean threadPool;
038    private boolean network;
039    private boolean fs;
040    private boolean transport;
041    private boolean http;
042
043    protected NodesStatsRequest() {
044    }
045
046    /**
047     * Get stats from nodes based on the nodes ids specified. If none are passed, stats
048     * for all nodes will be returned.
049     */
050    public NodesStatsRequest(String... nodesIds) {
051        super(nodesIds);
052    }
053
054    /**
055     * Sets all the request flags.
056     */
057    public NodesStatsRequest all() {
058        this.indices = true;
059        this.os = true;
060        this.process = true;
061        this.jvm = true;
062        this.threadPool = true;
063        this.network = true;
064        this.fs = true;
065        this.transport = true;
066        this.http = true;
067        return this;
068    }
069
070    /**
071     * Clears all the request flags.
072     */
073    public NodesStatsRequest clear() {
074        this.indices = false;
075        this.os = false;
076        this.process = false;
077        this.jvm = false;
078        this.threadPool = false;
079        this.network = false;
080        this.fs = false;
081        this.transport = false;
082        this.http = false;
083        return this;
084    }
085
086    /**
087     * Should indices stats be returned.
088     */
089    public boolean indices() {
090        return this.indices;
091    }
092
093    /**
094     * Should indices stats be returned.
095     */
096    public NodesStatsRequest indices(boolean indices) {
097        this.indices = indices;
098        return this;
099    }
100
101    /**
102     * Should the node OS be returned.
103     */
104    public boolean os() {
105        return this.os;
106    }
107
108    /**
109     * Should the node OS be returned.
110     */
111    public NodesStatsRequest os(boolean os) {
112        this.os = os;
113        return this;
114    }
115
116    /**
117     * Should the node Process be returned.
118     */
119    public boolean process() {
120        return this.process;
121    }
122
123    /**
124     * Should the node Process be returned.
125     */
126    public NodesStatsRequest process(boolean process) {
127        this.process = process;
128        return this;
129    }
130
131    /**
132     * Should the node JVM be returned.
133     */
134    public boolean jvm() {
135        return this.jvm;
136    }
137
138    /**
139     * Should the node JVM be returned.
140     */
141    public NodesStatsRequest jvm(boolean jvm) {
142        this.jvm = jvm;
143        return this;
144    }
145
146    /**
147     * Should the node Thread Pool be returned.
148     */
149    public boolean threadPool() {
150        return this.threadPool;
151    }
152
153    /**
154     * Should the node Thread Pool be returned.
155     */
156    public NodesStatsRequest threadPool(boolean threadPool) {
157        this.threadPool = threadPool;
158        return this;
159    }
160
161    /**
162     * Should the node Network be returned.
163     */
164    public boolean network() {
165        return this.network;
166    }
167
168    /**
169     * Should the node Network be returned.
170     */
171    public NodesStatsRequest network(boolean network) {
172        this.network = network;
173        return this;
174    }
175
176    /**
177     * Should the node file system stats be returned.
178     */
179    public boolean fs() {
180        return this.fs;
181    }
182
183    /**
184     * Should the node file system stats be returned.
185     */
186    public NodesStatsRequest fs(boolean fs) {
187        this.fs = fs;
188        return this;
189    }
190
191    /**
192     * Should the node Transport be returned.
193     */
194    public boolean transport() {
195        return this.transport;
196    }
197
198    /**
199     * Should the node Transport be returned.
200     */
201    public NodesStatsRequest transport(boolean transport) {
202        this.transport = transport;
203        return this;
204    }
205
206    /**
207     * Should the node HTTP be returned.
208     */
209    public boolean http() {
210        return this.http;
211    }
212
213    /**
214     * Should the node HTTP be returned.
215     */
216    public NodesStatsRequest http(boolean http) {
217        this.http = http;
218        return this;
219    }
220
221    @Override
222    public void readFrom(StreamInput in) throws IOException {
223        super.readFrom(in);
224        indices = in.readBoolean();
225        os = in.readBoolean();
226        process = in.readBoolean();
227        jvm = in.readBoolean();
228        threadPool = in.readBoolean();
229        network = in.readBoolean();
230        fs = in.readBoolean();
231        transport = in.readBoolean();
232        http = in.readBoolean();
233    }
234
235    @Override
236    public void writeTo(StreamOutput out) throws IOException {
237        super.writeTo(out);
238        out.writeBoolean(indices);
239        out.writeBoolean(os);
240        out.writeBoolean(process);
241        out.writeBoolean(jvm);
242        out.writeBoolean(threadPool);
243        out.writeBoolean(network);
244        out.writeBoolean(fs);
245        out.writeBoolean(transport);
246        out.writeBoolean(http);
247    }
248
249}
