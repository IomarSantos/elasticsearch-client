020package org.elasticsearch.action.admin.indices.segments;
021
022import com.google.common.collect.ImmutableList;
023import org.elasticsearch.action.support.broadcast.BroadcastShardOperationResponse;
024import org.elasticsearch.cluster.routing.ShardRouting;
025import org.elasticsearch.common.io.stream.StreamInput;
026import org.elasticsearch.common.io.stream.StreamOutput;
027import org.elasticsearch.index.engine.Segment;
028
029import java.io.IOException;
030import java.util.ArrayList;
031import java.util.Iterator;
032import java.util.List;
033
034import static org.elasticsearch.cluster.routing.ImmutableShardRouting.readShardRoutingEntry;
035
036public class ShardSegments extends BroadcastShardOperationResponse implements Iterable<Segment> {
037
038    private ShardRouting shardRouting;
039
040    private List<Segment> segments;
041
042    ShardSegments() {
043    }
044
045    public ShardSegments(ShardRouting shardRouting, List<Segment> segments) {
046        super(shardRouting.index(), shardRouting.id());
047        this.shardRouting = shardRouting;
048        this.segments = segments;
049    }
050
051    @Override
052    public Iterator<Segment> iterator() {
053        return segments.iterator();
054    }
055
056    public ShardRouting shardRouting() {
057        return this.shardRouting;
058    }
059
060    public ShardRouting getShardRouting() {
061        return this.shardRouting;
062    }
063
064    public List<Segment> segments() {
065        return this.segments;
066    }
067
068    public List<Segment> getSegments() {
069        return segments;
070    }
071
072    public int numberOfCommitted() {
073        int count = 0;
074        for (Segment segment : segments) {
075            if (segment.committed()) {
076                count++;
077            }
078        }
079        return count;
080    }
081
082    public int numberOfSearch() {
083        int count = 0;
084        for (Segment segment : segments) {
085            if (segment.search()) {
086                count++;
087            }
088        }
089        return count;
090    }
091
092    public static ShardSegments readShardSegments(StreamInput in) throws IOException {
093        ShardSegments shard = new ShardSegments();
094        shard.readFrom(in);
095        return shard;
096    }
097
098    @Override
099    public void readFrom(StreamInput in) throws IOException {
100        super.readFrom(in);
101        shardRouting = readShardRoutingEntry(in);
102        int size = in.readVInt();
103        if (size == 0) {
104            segments = ImmutableList.of();
105        } else {
106            segments = new ArrayList<Segment>(size);
107            for (int i = 0; i < size; i++) {
108                segments.add(Segment.readSegment(in));
109            }
110        }
111    }
112
113    @Override
114    public void writeTo(StreamOutput out) throws IOException {
115        super.writeTo(out);
116        shardRouting.writeTo(out);
117        out.writeVInt(segments.size());
118        for (Segment segment : segments) {
119            segment.writeTo(out);
120        }
121    }
122}
