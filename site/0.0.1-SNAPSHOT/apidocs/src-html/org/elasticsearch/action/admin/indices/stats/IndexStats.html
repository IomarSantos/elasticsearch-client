020package org.elasticsearch.action.admin.indices.stats;
021
022import com.google.common.collect.Lists;
023import com.google.common.collect.Maps;
024
025import java.util.Iterator;
026import java.util.List;
027import java.util.Map;
028
029/**
030 */
031public class IndexStats implements Iterable<IndexShardStats> {
032
033    private final String index;
034
035    private final ShardStats shards[];
036
037    public IndexStats(String index, ShardStats[] shards) {
038        this.index = index;
039        this.shards = shards;
040    }
041
042    public String index() {
043        return this.index;
044    }
045
046    public ShardStats[] shards() {
047        return this.shards;
048    }
049
050    private Map<Integer, IndexShardStats> indexShards;
051
052    public Map<Integer, IndexShardStats> indexShards() {
053        if (indexShards != null) {
054            return indexShards;
055        }
056        Map<Integer, List<ShardStats>> tmpIndexShards = Maps.newHashMap();
057        for (ShardStats shard : shards) {
058            List<ShardStats> lst = tmpIndexShards.get(shard.shardRouting().id());
059            if (lst == null) {
060                lst = Lists.newArrayList();
061                tmpIndexShards.put(shard.shardRouting().id(), lst);
062            }
063            lst.add(shard);
064        }
065        indexShards = Maps.newHashMap();
066        for (Map.Entry<Integer, List<ShardStats>> entry : tmpIndexShards.entrySet()) {
067            indexShards.put(entry.getKey(), new IndexShardStats(entry.getValue().get(0).shardRouting().shardId(), entry.getValue().toArray(new ShardStats[entry.getValue().size()])));
068        }
069        return indexShards;
070    }
071
072    @Override
073    public Iterator<IndexShardStats> iterator() {
074        return indexShards().values().iterator();
075    }
076
077    private CommonStats total = null;
078
079    public CommonStats getTotal() {
080        return total();
081    }
082
083    public CommonStats total() {
084        if (total != null) {
085            return total;
086        }
087        CommonStats stats = new CommonStats();
088        for (ShardStats shard : shards) {
089            stats.add(shard.stats());
090        }
091        total = stats;
092        return stats;
093    }
094
095    private CommonStats primary = null;
096
097    public CommonStats getPrimaries() {
098        return primaries();
099    }
100
101    public CommonStats primaries() {
102        if (primary != null) {
103            return primary;
104        }
105        CommonStats stats = new CommonStats();
106        for (ShardStats shard : shards) {
107            if (shard.shardRouting().primary()) {
108                stats.add(shard.stats());
109            }
110        }
111        primary = stats;
112        return stats;
113    }
114}
