020package org.elasticsearch.action.admin.indices.stats;
021
022import org.elasticsearch.action.ActionListener;
023import org.elasticsearch.action.support.broadcast.BroadcastOperationRequestBuilder;
024import org.elasticsearch.client.IndicesAdminClient;
025import org.elasticsearch.client.internal.InternalIndicesAdminClient;
026
027/**
028 * A request to get indices level stats. Allow to enable different stats to be returned.
029 * <p/>
030 * <p>By default, the {@link #setDocs(boolean)}, {@link #setStore(boolean)}, {@link #setIndexing(boolean)}
031 * are enabled. Other stats can be enabled as well.
032 * <p/>
033 * <p>All the stats to be returned can be cleared using {@link #clear()}, at which point, specific
034 * stats can be enabled.
035 */
036public class IndicesStatsRequestBuilder extends BroadcastOperationRequestBuilder<IndicesStatsRequest, IndicesStats, IndicesStatsRequestBuilder> {
037
038    public IndicesStatsRequestBuilder(IndicesAdminClient indicesClient) {
039        super((InternalIndicesAdminClient) indicesClient, new IndicesStatsRequest());
040    }
041
042    /**
043     * Sets all flags to return all stats.
044     */
045    public IndicesStatsRequestBuilder all() {
046        request.all();
047        return this;
048    }
049
050    /**
051     * Clears all stats.
052     */
053    public IndicesStatsRequestBuilder clear() {
054        request.clear();
055        return this;
056    }
057
058    /**
059     * Document types to return stats for. Mainly affects {@link #setIndexing(boolean)} when
060     * enabled, returning specific indexing stats for those types.
061     */
062    public IndicesStatsRequestBuilder setTypes(String... types) {
063        request.types(types);
064        return this;
065    }
066
067    public IndicesStatsRequestBuilder setGroups(String... groups) {
068        request.groups(groups);
069        return this;
070    }
071
072    public IndicesStatsRequestBuilder setDocs(boolean docs) {
073        request.docs(docs);
074        return this;
075    }
076
077    public IndicesStatsRequestBuilder setStore(boolean store) {
078        request.store(store);
079        return this;
080    }
081
082    public IndicesStatsRequestBuilder setIndexing(boolean indexing) {
083        request.indexing(indexing);
084        return this;
085    }
086
087    public IndicesStatsRequestBuilder setGet(boolean get) {
088        request.get(get);
089        return this;
090    }
091
092    public IndicesStatsRequestBuilder setSearch(boolean search) {
093        request.search(search);
094        return this;
095    }
096
097    public IndicesStatsRequestBuilder setMerge(boolean merge) {
098        request.merge(merge);
099        return this;
100    }
101
102    public IndicesStatsRequestBuilder setRefresh(boolean refresh) {
103        request.refresh(refresh);
104        return this;
105    }
106
107    public IndicesStatsRequestBuilder setFlush(boolean flush) {
108        request.flush(flush);
109        return this;
110    }
111
112    public IndicesStatsRequestBuilder setWarmer(boolean warmer) {
113        request.warmer(warmer);
114        return this;
115    }
116
117    @Override
118    protected void doExecute(ActionListener<IndicesStats> listener) {
119        ((IndicesAdminClient) client).stats(request, listener);
120    }
121}
