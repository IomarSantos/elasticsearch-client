020package org.elasticsearch.action.admin.indices.warmer.delete;
021
022import org.elasticsearch.action.ActionRequestValidationException;
023import org.elasticsearch.action.support.master.MasterNodeOperationRequest;
024import org.elasticsearch.common.Nullable;
025import org.elasticsearch.common.Strings;
026import org.elasticsearch.common.io.stream.StreamInput;
027import org.elasticsearch.common.io.stream.StreamOutput;
028
029import java.io.IOException;
030
031/**
032 * A request to delete an index warmer.
033 */
034public class DeleteWarmerRequest extends MasterNodeOperationRequest<DeleteWarmerRequest> {
035
036    private String name;
037
038    private String[] indices = Strings.EMPTY_ARRAY;
039
040    DeleteWarmerRequest() {
041    }
042
043    /**
044     * Constructs a new delete warmer request for the specified name.
045     *
046     * @param name: the name (or wildcard expression) of the warmer to match, null to delete all.
047     */
048    public DeleteWarmerRequest(String name) {
049        this.name = name;
050    }
051
052    @Override
053    public ActionRequestValidationException validate() {
054        ActionRequestValidationException validationException = null;
055        return validationException;
056    }
057
058    /**
059     * The name to delete.
060     */
061    @Nullable
062    String name() {
063        return name;
064    }
065
066    /**
067     * The name (or wildcard expression) of the index warmer to delete, or null
068     * to delete all warmers.
069     */
070    public DeleteWarmerRequest name(@Nullable String name) {
071        this.name = name;
072        return this;
073    }
074
075    /**
076     * Sets the indices this put mapping operation will execute on.
077     */
078    public DeleteWarmerRequest indices(String[] indices) {
079        this.indices = indices;
080        return this;
081    }
082
083    /**
084     * The indices the mappings will be put.
085     */
086    public String[] indices() {
087        return indices;
088    }
089
090    @Override
091    public void readFrom(StreamInput in) throws IOException {
092        super.readFrom(in);
093        name = in.readOptionalString();
094        indices = in.readStringArray();
095    }
096
097    @Override
098    public void writeTo(StreamOutput out) throws IOException {
099        super.writeTo(out);
100        out.writeOptionalString(name);
101        out.writeStringArrayNullable(indices);
102    }
103}
