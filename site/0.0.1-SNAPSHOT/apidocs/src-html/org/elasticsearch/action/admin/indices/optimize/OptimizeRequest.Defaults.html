020package org.elasticsearch.action.admin.indices.optimize;
021
022import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
023import org.elasticsearch.action.support.broadcast.BroadcastOperationThreading;
024import org.elasticsearch.common.io.stream.StreamInput;
025import org.elasticsearch.common.io.stream.StreamOutput;
026
027import java.io.IOException;
028
029/**
030 * A request to optimize one or more indices. In order to optimize on all the indices, pass an empty array or
031 * <tt>null</tt> for the indices.
032 * <p/>
033 * <p>{@link #waitForMerge(boolean)} allows to control if the call will block until the optimize completes and
034 * defaults to <tt>true</tt>.
035 * <p/>
036 * <p>{@link #maxNumSegments(int)} allows to control the number of segments to optimize down to. By default, will
037 * cause the optimize process to optimize down to half the configured number of segments.
038 *
039 * @see org.elasticsearch.client.Requests#optimizeRequest(String...)
040 * @see org.elasticsearch.client.IndicesAdminClient#optimize(OptimizeRequest)
041 * @see OptimizeResponse
042 */
043public class OptimizeRequest extends BroadcastOperationRequest<OptimizeRequest> {
044
045    public static final class Defaults {
046        public static final boolean WAIT_FOR_MERGE = true;
047        public static final int MAX_NUM_SEGMENTS = -1;
048        public static final boolean ONLY_EXPUNGE_DELETES = false;
049        public static final boolean FLUSH = true;
050        public static final boolean REFRESH = true;
051    }
052
053    private boolean waitForMerge = Defaults.WAIT_FOR_MERGE;
054
055    private int maxNumSegments = Defaults.MAX_NUM_SEGMENTS;
056
057    private boolean onlyExpungeDeletes = Defaults.ONLY_EXPUNGE_DELETES;
058
059    private boolean flush = Defaults.FLUSH;
060
061    private boolean refresh = Defaults.FLUSH;
062
063    /**
064     * Constructs an optimization request over one or more indices.
065     *
066     * @param indices The indices to optimize, no indices passed means all indices will be optimized.
067     */
068    public OptimizeRequest(String... indices) {
069        super(indices);
070        // we want to do the optimize in parallel on local shards...
071        operationThreading(BroadcastOperationThreading.THREAD_PER_SHARD);
072    }
073
074    public OptimizeRequest() {
075
076    }
077
078    /**
079     * Should the call block until the optimize completes. Defaults to <tt>true</tt>.
080     */
081    public boolean waitForMerge() {
082        return waitForMerge;
083    }
084
085    /**
086     * Should the call block until the optimize completes. Defaults to <tt>true</tt>.
087     */
088    public OptimizeRequest waitForMerge(boolean waitForMerge) {
089        this.waitForMerge = waitForMerge;
090        return this;
091    }
092
093    /**
094     * Will optimize the index down to <= maxNumSegments. By default, will cause the optimize
095     * process to optimize down to half the configured number of segments.
096     */
097    public int maxNumSegments() {
098        return maxNumSegments;
099    }
100
101    /**
102     * Will optimize the index down to <= maxNumSegments. By default, will cause the optimize
103     * process to optimize down to half the configured number of segments.
104     */
105    public OptimizeRequest maxNumSegments(int maxNumSegments) {
106        this.maxNumSegments = maxNumSegments;
107        return this;
108    }
109
110    /**
111     * Should the optimization only expunge deletes from the index, without full optimization.
112     * Defaults to full optimization (<tt>false</tt>).
113     */
114    public boolean onlyExpungeDeletes() {
115        return onlyExpungeDeletes;
116    }
117
118    /**
119     * Should the optimization only expunge deletes from the index, without full optimization.
120     * Defaults to full optimization (<tt>false</tt>).
121     */
122    public OptimizeRequest onlyExpungeDeletes(boolean onlyExpungeDeletes) {
123        this.onlyExpungeDeletes = onlyExpungeDeletes;
124        return this;
125    }
126
127    /**
128     * Should flush be performed after the optimization. Defaults to <tt>true</tt>.
129     */
130    public boolean flush() {
131        return flush;
132    }
133
134    /**
135     * Should flush be performed after the optimization. Defaults to <tt>true</tt>.
136     */
137    public OptimizeRequest flush(boolean flush) {
138        this.flush = flush;
139        return this;
140    }
141
142    /**
143     * Should refresh be performed after the optimization. Defaults to <tt>true</tt>.
144     */
145    public boolean refresh() {
146        return refresh;
147    }
148
149    /**
150     * Should refresh be performed after the optimization. Defaults to <tt>true</tt>.
151     */
152    public OptimizeRequest refresh(boolean refresh) {
153        this.refresh = refresh;
154        return this;
155    }
156
157    public void readFrom(StreamInput in) throws IOException {
158        super.readFrom(in);
159        waitForMerge = in.readBoolean();
160        maxNumSegments = in.readInt();
161        onlyExpungeDeletes = in.readBoolean();
162        flush = in.readBoolean();
163        refresh = in.readBoolean();
164    }
165
166    public void writeTo(StreamOutput out) throws IOException {
167        super.writeTo(out);
168        out.writeBoolean(waitForMerge);
169        out.writeInt(maxNumSegments);
170        out.writeBoolean(onlyExpungeDeletes);
171        out.writeBoolean(flush);
172        out.writeBoolean(refresh);
173    }
174}
