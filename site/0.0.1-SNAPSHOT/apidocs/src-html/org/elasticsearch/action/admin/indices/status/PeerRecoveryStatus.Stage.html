020package org.elasticsearch.action.admin.indices.status;
021
022import org.elasticsearch.ElasticSearchIllegalArgumentException;
023import org.elasticsearch.common.unit.ByteSizeValue;
024import org.elasticsearch.common.unit.TimeValue;
025
026/**
027 *
028 */
029public class PeerRecoveryStatus {
030
031    public enum Stage {
032        INIT((byte) 0),
033        INDEX((byte) 1),
034        TRANSLOG((byte) 2),
035        FINALIZE((byte) 3),
036        DONE((byte) 4);
037
038        private final byte value;
039
040        Stage(byte value) {
041            this.value = value;
042        }
043
044        public byte value() {
045            return value;
046        }
047
048        public static Stage fromValue(byte value) {
049            if (value == 0) {
050                return INIT;
051            } else if (value == 1) {
052                return INDEX;
053            } else if (value == 2) {
054                return TRANSLOG;
055            } else if (value == 3) {
056                return FINALIZE;
057            } else if (value == 4) {
058                return DONE;
059            }
060            throw new ElasticSearchIllegalArgumentException("No stage found for [" + value + ']');
061        }
062    }
063
064    final Stage stage;
065
066    final long startTime;
067
068    final long time;
069
070    final long indexSize;
071
072    final long reusedIndexSize;
073
074    final long recoveredIndexSize;
075
076    final long recoveredTranslogOperations;
077
078    public PeerRecoveryStatus(Stage stage, long startTime, long time, long indexSize, long reusedIndexSize,
079                              long recoveredIndexSize, long recoveredTranslogOperations) {
080        this.stage = stage;
081        this.startTime = startTime;
082        this.time = time;
083        this.indexSize = indexSize;
084        this.reusedIndexSize = reusedIndexSize;
085        this.recoveredIndexSize = recoveredIndexSize;
086        this.recoveredTranslogOperations = recoveredTranslogOperations;
087    }
088
089    public Stage stage() {
090        return this.stage;
091    }
092
093    public long startTime() {
094        return this.startTime;
095    }
096
097    public long getStartTime() {
098        return this.startTime;
099    }
100
101    public TimeValue time() {
102        return TimeValue.timeValueMillis(time);
103    }
104
105    public TimeValue getTime() {
106        return time();
107    }
108
109    public ByteSizeValue indexSize() {
110        return new ByteSizeValue(indexSize);
111    }
112
113    public ByteSizeValue getIndexSize() {
114        return indexSize();
115    }
116
117    public ByteSizeValue reusedIndexSize() {
118        return new ByteSizeValue(reusedIndexSize);
119    }
120
121    public ByteSizeValue getReusedIndexSize() {
122        return reusedIndexSize();
123    }
124
125    public ByteSizeValue expectedRecoveredIndexSize() {
126        return new ByteSizeValue(indexSize - reusedIndexSize);
127    }
128
129    public ByteSizeValue getExpectedRecoveredIndexSize() {
130        return expectedRecoveredIndexSize();
131    }
132
133    /**
134     * How much of the index has been recovered.
135     */
136    public ByteSizeValue recoveredIndexSize() {
137        return new ByteSizeValue(recoveredIndexSize);
138    }
139
140    /**
141     * How much of the index has been recovered.
142     */
143    public ByteSizeValue getRecoveredIndexSize() {
144        return recoveredIndexSize();
145    }
146
147    public int indexRecoveryProgress() {
148        if (recoveredIndexSize == 0) {
149            if (indexSize != 0 && indexSize == reusedIndexSize) {
150                return 100;
151            }
152            return 0;
153        }
154        return (int) (((double) recoveredIndexSize) / expectedRecoveredIndexSize().bytes() * 100);
155    }
156
157    public int getIndexRecoveryProgress() {
158        return indexRecoveryProgress();
159    }
160
161    public long recoveredTranslogOperations() {
162        return recoveredTranslogOperations;
163    }
164
165    public long getRecoveredTranslogOperations() {
166        return recoveredTranslogOperations();
167    }
168}
