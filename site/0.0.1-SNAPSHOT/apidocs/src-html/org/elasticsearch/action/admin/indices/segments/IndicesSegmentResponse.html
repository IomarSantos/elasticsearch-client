020package org.elasticsearch.action.admin.indices.segments;
021
022import com.google.common.collect.Lists;
023import com.google.common.collect.Maps;
024import com.google.common.collect.Sets;
025import org.elasticsearch.action.ShardOperationFailedException;
026import org.elasticsearch.action.support.broadcast.BroadcastOperationResponse;
027import org.elasticsearch.cluster.ClusterState;
028import org.elasticsearch.common.io.stream.StreamInput;
029import org.elasticsearch.common.io.stream.StreamOutput;
030import org.elasticsearch.common.xcontent.ToXContent;
031import org.elasticsearch.common.xcontent.XContentBuilder;
032import org.elasticsearch.common.xcontent.XContentBuilderString;
033import org.elasticsearch.index.engine.Segment;
034
035import java.io.IOException;
036import java.util.List;
037import java.util.Map;
038import java.util.Set;
039
040public class IndicesSegmentResponse extends BroadcastOperationResponse implements ToXContent {
041
042    private ShardSegments[] shards;
043
044    private Map<String, IndexSegments> indicesSegments;
045
046    IndicesSegmentResponse() {
047
048    }
049
050    IndicesSegmentResponse(ShardSegments[] shards, ClusterState clusterState, int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures) {
051        super(totalShards, successfulShards, failedShards, shardFailures);
052        this.shards = shards;
053    }
054
055    public Map<String, IndexSegments> getIndices() {
056        return this.indices();
057    }
058
059    public Map<String, IndexSegments> indices() {
060        if (indicesSegments != null) {
061            return indicesSegments;
062        }
063        Map<String, IndexSegments> indicesSegments = Maps.newHashMap();
064
065        Set<String> indices = Sets.newHashSet();
066        for (ShardSegments shard : shards) {
067            indices.add(shard.index());
068        }
069
070        for (String index : indices) {
071            List<ShardSegments> shards = Lists.newArrayList();
072            for (ShardSegments shard : this.shards) {
073                if (shard.shardRouting().index().equals(index)) {
074                    shards.add(shard);
075                }
076            }
077            indicesSegments.put(index, new IndexSegments(index, shards.toArray(new ShardSegments[shards.size()])));
078        }
079        this.indicesSegments = indicesSegments;
080        return indicesSegments;
081    }
082
083
084    @Override
085    public void readFrom(StreamInput in) throws IOException {
086        super.readFrom(in);
087        shards = new ShardSegments[in.readVInt()];
088        for (int i = 0; i < shards.length; i++) {
089            shards[i] = ShardSegments.readShardSegments(in);
090        }
091    }
092
093    @Override
094    public void writeTo(StreamOutput out) throws IOException {
095        super.writeTo(out);
096        out.writeVInt(shards.length);
097        for (ShardSegments shard : shards) {
098            shard.writeTo(out);
099        }
100    }
101
102    @Override
103    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
104        builder.startObject(Fields.INDICES);
105
106        for (IndexSegments indexSegments : indices().values()) {
107            builder.startObject(indexSegments.index(), XContentBuilder.FieldCaseConversion.NONE);
108
109            builder.startObject(Fields.SHARDS);
110            for (IndexShardSegments indexSegment : indexSegments) {
111                builder.startArray(Integer.toString(indexSegment.shardId().id()));
112                for (ShardSegments shardSegments : indexSegment) {
113                    builder.startObject();
114
115                    builder.startObject(Fields.ROUTING);
116                    builder.field(Fields.STATE, shardSegments.shardRouting().state());
117                    builder.field(Fields.PRIMARY, shardSegments.shardRouting().primary());
118                    builder.field(Fields.NODE, shardSegments.shardRouting().currentNodeId());
119                    if (shardSegments.shardRouting().relocatingNodeId() != null) {
120                        builder.field(Fields.RELOCATING_NODE, shardSegments.shardRouting().relocatingNodeId());
121                    }
122                    builder.endObject();
123
124                    builder.field(Fields.NUM_COMMITTED_SEGMENTS, shardSegments.numberOfCommitted());
125                    builder.field(Fields.NUM_SEARCH_SEGMENTS, shardSegments.numberOfSearch());
126
127                    builder.startObject(Fields.SEGMENTS);
128                    for (Segment segment : shardSegments) {
129                        builder.startObject(segment.name());
130                        builder.field(Fields.GENERATION, segment.generation());
131                        builder.field(Fields.NUM_DOCS, segment.numDocs());
132                        builder.field(Fields.DELETED_DOCS, segment.deletedDocs());
133                        builder.field(Fields.SIZE, segment.size().toString());
134                        builder.field(Fields.SIZE_IN_BYTES, segment.sizeInBytes());
135                        builder.field(Fields.COMMITTED, segment.committed());
136                        builder.field(Fields.SEARCH, segment.search());
137                        builder.endObject();
138                    }
139                    builder.endObject();
140
141                    builder.endObject();
142                }
143                builder.endArray();
144            }
145            builder.endObject();
146
147            builder.endObject();
148        }
149
150        builder.endObject();
151        return builder;
152    }
153
154    static final class Fields {
155        static final XContentBuilderString INDICES = new XContentBuilderString("indices");
156        static final XContentBuilderString SHARDS = new XContentBuilderString("shards");
157        static final XContentBuilderString ROUTING = new XContentBuilderString("routing");
158        static final XContentBuilderString STATE = new XContentBuilderString("state");
159        static final XContentBuilderString PRIMARY = new XContentBuilderString("primary");
160        static final XContentBuilderString NODE = new XContentBuilderString("node");
161        static final XContentBuilderString RELOCATING_NODE = new XContentBuilderString("relocating_node");
162
163        static final XContentBuilderString SEGMENTS = new XContentBuilderString("segments");
164        static final XContentBuilderString GENERATION = new XContentBuilderString("generation");
165        static final XContentBuilderString NUM_COMMITTED_SEGMENTS = new XContentBuilderString("num_committed_segments");
166        static final XContentBuilderString NUM_SEARCH_SEGMENTS = new XContentBuilderString("num_search_segments");
167        static final XContentBuilderString NUM_DOCS = new XContentBuilderString("num_docs");
168        static final XContentBuilderString DELETED_DOCS = new XContentBuilderString("deleted_docs");
169        static final XContentBuilderString SIZE = new XContentBuilderString("size");
170        static final XContentBuilderString SIZE_IN_BYTES = new XContentBuilderString("size_in_bytes");
171        static final XContentBuilderString COMMITTED = new XContentBuilderString("committed");
172        static final XContentBuilderString SEARCH = new XContentBuilderString("search");
173    }
174}
