020package org.elasticsearch.action.admin.indices.validate.query;
021
022import org.elasticsearch.ElasticSearchGenerationException;
023import org.elasticsearch.action.ActionRequestValidationException;
024import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
025import org.elasticsearch.client.AdminRequests;
026import org.elasticsearch.common.Required;
027import org.elasticsearch.common.Strings;
028import org.elasticsearch.common.bytes.BytesArray;
029import org.elasticsearch.common.bytes.BytesReference;
030import org.elasticsearch.common.io.stream.StreamInput;
031import org.elasticsearch.common.io.stream.StreamOutput;
032import org.elasticsearch.common.xcontent.XContentBuilder;
033import org.elasticsearch.common.xcontent.XContentFactory;
034import org.elasticsearch.common.xcontent.XContentHelper;
035import org.elasticsearch.common.xcontent.XContentType;
036import org.elasticsearch.index.query.QueryBuilder;
037
038import java.io.IOException;
039import java.util.Arrays;
040import java.util.Map;
041
042/**
043 * A request to validate a specific query.
044 * <p/>
045 * <p>The request requires the query source to be set either using {@link #query(org.elasticsearch.index.query.QueryBuilder)},
046 * or {@link #query(byte[])}.
047 */
048public class ValidateQueryRequest extends BroadcastOperationRequest<ValidateQueryRequest> {
049
050    private static final XContentType contentType = AdminRequests.CONTENT_TYPE;
051
052    private BytesReference querySource;
053    private boolean querySourceUnsafe;
054
055    private boolean explain;
056
057    private String[] types = Strings.EMPTY_ARRAY;
058
059    ValidateQueryRequest() {
060    }
061
062    /**
063     * Constructs a new validate request against the provided indices. No indices provided means it will
064     * run against all indices.
065     */
066    public ValidateQueryRequest(String... indices) {
067        super(indices);
068    }
069
070    @Override
071    public ActionRequestValidationException validate() {
072        ActionRequestValidationException validationException = super.validate();
073        return validationException;
074    }
075
076    @Override
077    protected void beforeStart() {
078        if (querySourceUnsafe) {
079            querySource = querySource.copyBytesArray();
080            querySourceUnsafe = false;
081        }
082    }
083
084    /**
085     * The query source to execute.
086     */
087    BytesReference querySource() {
088        return querySource;
089    }
090
091    /**
092     * The query source to execute.
093     *
094     * @see org.elasticsearch.index.query.QueryBuilders
095     */
096    @Required
097    public ValidateQueryRequest query(QueryBuilder queryBuilder) {
098        this.querySource = queryBuilder.buildAsBytes();
099        this.querySourceUnsafe = false;
100        return this;
101    }
102
103    /**
104     * The query source to execute in the form of a map.
105     */
106    @Required
107    public ValidateQueryRequest query(Map querySource) {
108        try {
109            XContentBuilder builder = XContentFactory.contentBuilder(contentType);
110            builder.map(querySource);
111            return query(builder);
112        } catch (IOException e) {
113            throw new ElasticSearchGenerationException("Failed to generate [" + querySource + "]", e);
114        }
115    }
116
117    @Required
118    public ValidateQueryRequest query(XContentBuilder builder) {
119        this.querySource = builder.bytes();
120        this.querySourceUnsafe = false;
121        return this;
122    }
123
124    /**
125     * The query source to validate. It is preferable to use either {@link #query(byte[])}
126     * or {@link #query(org.elasticsearch.index.query.QueryBuilder)}.
127     */
128    @Required
129    public ValidateQueryRequest query(String querySource) {
130        this.querySource = new BytesArray(querySource);
131        ;
132        this.querySourceUnsafe = false;
133        return this;
134    }
135
136    /**
137     * The query source to validate.
138     */
139    @Required
140    public ValidateQueryRequest query(byte[] querySource) {
141        return query(querySource, 0, querySource.length, false);
142    }
143
144    /**
145     * The query source to validate.
146     */
147    @Required
148    public ValidateQueryRequest query(byte[] querySource, int offset, int length, boolean unsafe) {
149        return query(new BytesArray(querySource, offset, length), unsafe);
150    }
151
152    /**
153     * The query source to validate.
154     */
155    @Required
156    public ValidateQueryRequest query(BytesReference querySource, boolean unsafe) {
157        this.querySource = querySource;
158        this.querySourceUnsafe = unsafe;
159        return this;
160    }
161
162    /**
163     * The types of documents the query will run against. Defaults to all types.
164     */
165    String[] types() {
166        return this.types;
167    }
168
169    /**
170     * The types of documents the query will run against. Defaults to all types.
171     */
172    public ValidateQueryRequest types(String... types) {
173        this.types = types;
174        return this;
175    }
176
177    /**
178     * Indicate if detailed information about query is requested
179     */
180    public void explain(boolean explain) {
181        this.explain = explain;
182    }
183
184    /**
185     * Indicates if detailed information about query is requested
186     */
187    public boolean explain() {
188        return explain;
189    }
190
191    @Override
192    public void readFrom(StreamInput in) throws IOException {
193        super.readFrom(in);
194
195        querySourceUnsafe = false;
196        querySource = in.readBytesReference();
197
198        int typesSize = in.readVInt();
199        if (typesSize > 0) {
200            types = new String[typesSize];
201            for (int i = 0; i < typesSize; i++) {
202                types[i] = in.readUTF();
203            }
204        }
205
206        explain = in.readBoolean();
207
208    }
209
210    @Override
211    public void writeTo(StreamOutput out) throws IOException {
212        super.writeTo(out);
213
214        out.writeBytesReference(querySource);
215
216        out.writeVInt(types.length);
217        for (String type : types) {
218            out.writeUTF(type);
219        }
220
221        out.writeBoolean(explain);
222    }
223
224    @Override
225    public String toString() {
226        String sSource = "_na_";
227        try {
228            sSource = XContentHelper.convertToJson(querySource, false);
229        } catch (Exception e) {
230            // ignore
231        }
232        return "[" + Arrays.toString(indices) + "]" + Arrays.toString(types) + ", querySource[" + sSource + "], explain:" + explain;
233    }
234}
