045/**
046 * Index request to index a typed JSON document into a specific index and make it searchable. Best
047 * created using {@link org.elasticsearch.client.Requests#indexRequest(String)}.
048 * <p/>
049 * <p>The index requires the {@link #index()}, {@link #type(String)}, {@link #id(String)} and
050 * {@link #source(byte[])} to be set.
051 * <p/>
052 * <p>The source (content to index) can be set in its bytes form using ({@link #source(byte[])}),
053 * its string form ({@link #source(String)}) or using a {@link org.elasticsearch.common.xcontent.XContentBuilder}
054 * ({@link #source(org.elasticsearch.common.xcontent.XContentBuilder)}).
055 * <p/>
056 * <p>If the {@link #id(String)} is not set, it will be automatically generated.
057 *
058 * @see IndexResponse
059 * @see org.elasticsearch.client.Requests#indexRequest(String)
060 * @see org.elasticsearch.client.Client#index(IndexRequest)
061 */
062public class IndexRequest extends ShardReplicationOperationRequest<IndexRequest> {
063
064    /**
065     * Operation type controls if the type of the index operation.
066     */
067    public static enum OpType {
068        /**
069         * Index the source. If there an existing document with the id, it will
070         * be replaced.
071         */
072        INDEX((byte) 0),
073        /**
074         * Creates the resource. Simply adds it to the index, if there is an existing
075         * document with the id, then it won't be removed.
076         */
077        CREATE((byte) 1);
078
079        private byte id;
080
081        OpType(byte id) {
082            this.id = id;
083        }
084
085        /**
086         * The internal representation of the operation type.
087         */
088        public byte id() {
089            return id;
090        }
091
092        /**
093         * Constructs the operation type from its internal representation.
094         */
095        public static OpType fromId(byte id) {
096            if (id == 0) {
097                return INDEX;
098            } else if (id == 1) {
099                return CREATE;
100            } else {
101                throw new ElasticSearchIllegalArgumentException("No type match for [" + id + "]");
102            }
103        }
104    }

152    @Override
153    public ActionRequestValidationException validate() {
154        ActionRequestValidationException validationException = super.validate();
155        if (type == null) {

162    }

262     * Sets the relative ttl value. It musts be > 0 as it makes little sense otherwise. Setting it
263     * to <tt>null</tt> will reset to have no ttl.
264     */
265    public IndexRequest ttl(Long ttl) throws ElasticSearchGenerationException {
266        if (ttl == null) {
267            this.ttl = -1;
268            return this;
269        }
270        if (ttl <= 0) {
271            throw new ElasticSearchIllegalArgumentException("TTL value must be > 0. Illegal value provided [" + ttl + "]");
272        }
273        this.ttl = ttl;
274        return this;
275    }

315    public IndexRequest source(Map source, XContentType contentType) throws ElasticSearchGenerationException {
316        try {
317            XContentBuilder builder = XContentFactory.contentBuilder(contentType);
318            builder.map(source);
319            return source(builder);
320        } catch (IOException e) {
321            throw new ElasticSearchGenerationException("Failed to generate [" + source + "]", e);
322        }
323    }

444    /**
445     * Sets a string representation of the {@link #opType(org.elasticsearch.action.index.IndexRequest.OpType)}. Can
446     * be either "index" or "create".
447     */
448    public IndexRequest opType(String opType) throws ElasticSearchIllegalArgumentException {
449        if ("create".equals(opType)) {
450            return opType(OpType.CREATE);
451        } else if ("index".equals(opType)) {
452            return opType(OpType.INDEX);
453        } else {
454            throw new ElasticSearchIllegalArgumentException("No index opType matching [" + opType + "]");
455        }
456    }

464        } else {
465            return opType(OpType.INDEX);

529    @Override
530    public void readFrom(StreamInput in) throws IOException {
531        super.readFrom(in);
532        type = in.readString();
533        id = in.readOptionalString();
534        routing = in.readOptionalString();
535        parent = in.readOptionalString();
536        timestamp = in.readOptionalString();
537        ttl = in.readLong();
538        source = in.readBytesReference();
539        sourceUnsafe = false;
540
541        opType = OpType.fromId(in.readByte());
542        refresh = in.readBoolean();
543        version = in.readLong();
544        percolate = in.readOptionalString();
545        versionType = VersionType.fromValue(in.readByte());
546    }
547
548    @Override
549    public void writeTo(StreamOutput out) throws IOException {
550        super.writeTo(out);
551        out.writeString(type);
552        out.writeOptionalString(id);
553        out.writeOptionalString(routing);
554        out.writeOptionalString(parent);
555        out.writeOptionalString(timestamp);
556        out.writeLong(ttl);
557        out.writeBytesReference(source);
558        out.writeByte(opType.id());
559        out.writeBoolean(refresh);
560        out.writeLong(version);
561        out.writeOptionalString(percolate);
562        out.writeByte(versionType.getValue());
563    }
564
565    @Override
566    public String toString() {
567        String sSource = "_na_";
568        try {
569            sSource = XContentHelper.convertToJson(source, false);
570        } catch (Exception e) {
571            // ignore
572        }
573        return "index {[" + index + "][" + type + "][" + id + "], source[" + sSource + "]}";
574    }
