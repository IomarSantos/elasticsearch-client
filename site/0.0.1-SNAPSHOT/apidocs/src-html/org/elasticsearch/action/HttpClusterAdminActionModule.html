067
068    /**
069     * Registers an action.
070     *
071     * @param action The action type.
072     * @param htppAction The HTTP action implementing the actual action.
073     * @param supportHttpActions Any support actions that are needed by the HTTP
074     * action.
075     * @param <Request> The request type.
076     * @param <Response> The response type.
077     */
078    private static <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>> 
079            void registerAction(ClusterAction<Request, Response, RequestBuilder> action, HttpAction<Request, Response> httpAction, Class... supportHttpActions) {
080        actions.put(action.name(), new ActionEntry(action, httpAction, supportHttpActions));
081    }
082
083    public <Request extends ActionRequest, Response extends ActionResponse> 
084            HttpAction<Request, Response> getAction(String action) {
085        return actions.get(action).httpAction;
086    }
087
088    static class ActionEntry<Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>> {
089
090        public final ClusterAction<Request, Response, RequestBuilder> action;
091        public final HttpAction<Request, Response> httpAction;
092        public final Class[] supportHttpActions;
093
094        ActionEntry(ClusterAction<Request, Response, RequestBuilder> action, HttpAction<Request, Response> httpAction, Class... supportHttpActions) {
095            this.action = action;
096            this.httpAction = httpAction;
097            this.supportHttpActions = supportHttpActions;
098        }
099    }
