020package org.elasticsearch.cluster.routing.allocation.command;
021
022import com.google.common.collect.Lists;
023import org.elasticsearch.ElasticSearchException;
024import org.elasticsearch.ElasticSearchIllegalArgumentException;
025import org.elasticsearch.ElasticSearchParseException;
026import org.elasticsearch.cluster.routing.allocation.RoutingAllocation;
027import org.elasticsearch.common.Nullable;
028import org.elasticsearch.common.io.stream.StreamInput;
029import org.elasticsearch.common.io.stream.StreamOutput;
030import org.elasticsearch.common.xcontent.ToXContent;
031import org.elasticsearch.common.xcontent.XContentBuilder;
032import org.elasticsearch.common.xcontent.XContentParser;
033
034import java.io.IOException;
035import java.util.Arrays;
036import java.util.HashMap;
037import java.util.Iterator;
038import java.util.List;
039import java.util.Map;
040import java.util.ServiceLoader;
041
042/**
043 */
044public class AllocationCommands {
045
046    private final static Map<String, AllocationCommandFactory> factories = new HashMap();
047
048    static {
049        ServiceLoader<AllocationCommand> loader = ServiceLoader.load(AllocationCommand.class);
050        Iterator<AllocationCommand> it = loader.iterator();
051        while (it.hasNext()) {
052            AllocationCommand command = it.next();
053            factories.put(command.name(), command.factory());
054        }
055    }
056
057    @SuppressWarnings("unchecked")
058    @Nullable
059    public static <T extends AllocationCommand> AllocationCommandFactory<T> lookupFactory(String name) {
060        return factories.get(name);
061    }
062
063    @SuppressWarnings("unchecked")
064    public static <T extends AllocationCommand> AllocationCommandFactory<T> lookupFactorySafe(String name) throws ElasticSearchIllegalArgumentException {
065        AllocationCommandFactory<T> factory = factories.get(name);
066        if (factory == null) {
067            throw new ElasticSearchIllegalArgumentException("No allocation command factory registered for name [" + name + "]");
068        }
069        return factory;
070    }
071
072    private final List<AllocationCommand> commands = Lists.newArrayList();
073
074    public AllocationCommands(AllocationCommand... commands) {
075        if (commands != null) {
076            this.commands.addAll(Arrays.asList(commands));
077        }
078    }
079
080    public AllocationCommands add(AllocationCommand... commands) {
081        if (commands != null) {
082            this.commands.addAll(Arrays.asList(commands));
083        }
084        return this;
085    }
086
087    public List<AllocationCommand> commands() {
088        return this.commands;
089    }
090
091    public void execute(RoutingAllocation allocation) throws ElasticSearchException {
092        for (AllocationCommand command : commands) {
093            command.execute(allocation);
094        }
095    }
096
097    public static AllocationCommands readFrom(StreamInput in) throws IOException {
098        AllocationCommands commands = new AllocationCommands();
099        int size = in.readVInt();
100        for (int i = 0; i < size; i++) {
101            String name = in.readString();
102            commands.add(lookupFactorySafe(name).readFrom(in));
103        }
104        return commands;
105    }
106
107    public static void writeTo(AllocationCommands commands, StreamOutput out) throws IOException {
108        out.writeVInt(commands.commands.size());
109        for (AllocationCommand command : commands.commands) {
110            out.writeString(command.name());
111            lookupFactorySafe(command.name()).writeTo(command, out);
112        }
113    }
114
115    /**
116     * <pre>
117     *     {
118     *         "commands" : [
119     *              {"allocate" : {"index" : "test", "shard" : 0, "node" : "test"}}
120     *         ]
121     *     }
122     * </pre>
123     */
124    public static AllocationCommands fromXContent(XContentParser parser) throws IOException {
125        AllocationCommands commands = new AllocationCommands();
126
127        XContentParser.Token token = parser.currentToken();
128        if (token == null) {
129            throw new ElasticSearchParseException("No commands");
130        }
131        if (token == XContentParser.Token.FIELD_NAME) {
132            if (!parser.currentName().equals("commands")) {
133                throw new ElasticSearchParseException("expected field name to be named `commands`, got " + parser.currentName());
134            }
135            if (!parser.currentName().equals("commands")) {
136                throw new ElasticSearchParseException("expected field name to be named `commands`, got " + parser.currentName());
137            }
138            token = parser.nextToken();
139            if (token != XContentParser.Token.START_ARRAY) {
140                throw new ElasticSearchParseException("commands should follow with an array element");
141            }
142        } else if (token == XContentParser.Token.START_ARRAY) {
143            // ok...
144        } else {
145            throw new ElasticSearchParseException("expected either field name commands, or start array, got " + token);
146        }
147        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
148            if (token == XContentParser.Token.START_OBJECT) {
149                // move to the command name
150                token = parser.nextToken();
151                String commandName = parser.currentName();
152                token = parser.nextToken();
153                commands.add(AllocationCommands.lookupFactorySafe(commandName).fromXContent(parser));
154                // move to the end object one
155                if (parser.nextToken() != XContentParser.Token.END_OBJECT) {
156                    throw new ElasticSearchParseException("allocation command is malformed, done parsing a command, but didn't get END_OBJECT, got " + token);
157                }
158            } else {
159                throw new ElasticSearchParseException("allocation command is malformed, got token " + token);
160            }
161        }
162        return commands;
163    }
164
165    public static void toXContent(AllocationCommands commands, XContentBuilder builder, ToXContent.Params params) throws IOException {
166        builder.startArray("commands");
167        for (AllocationCommand command : commands.commands) {
168            builder.startObject();
169            builder.field(command.name());
170            AllocationCommands.lookupFactorySafe(command.name()).toXContent(command, builder, params);
171            builder.endObject();
172        }
173        builder.endArray();
174    }
175}
