020package org.elasticsearch.cluster.routing.allocation.decider;
021
022import org.elasticsearch.cluster.routing.MutableShardRouting;
023import org.elasticsearch.cluster.routing.RoutingNode;
024import org.elasticsearch.cluster.routing.ShardRouting;
025import org.elasticsearch.cluster.routing.allocation.RoutingAllocation;
026import org.elasticsearch.common.settings.Settings;
027
028import java.util.List;
029
030import static org.elasticsearch.common.settings.ImmutableSettings.Builder.EMPTY_SETTINGS;
031
032public class ClusterRebalanceAllocationDecider extends SimpleAllocationDecider {
033
034    public static enum ClusterRebalanceType {
035        ALWAYS,
036        INDICES_PRIMARIES_ACTIVE,
037        INDICES_ALL_ACTIVE
038    }
039
040    private final ClusterRebalanceType type;
041
042    public ClusterRebalanceAllocationDecider() {
043        this(EMPTY_SETTINGS);
044    }
045
046    public ClusterRebalanceAllocationDecider(Settings settings) {
047        super(settings);
048        String allowRebalance = settings.get("cluster.routing.allocation.allow_rebalance", "indices_all_active");
049        if ("always".equalsIgnoreCase(allowRebalance)) {
050            type = ClusterRebalanceType.ALWAYS;
051        } else if ("indices_primaries_active".equalsIgnoreCase(allowRebalance) || "indicesPrimariesActive".equalsIgnoreCase(allowRebalance)) {
052            type = ClusterRebalanceType.INDICES_PRIMARIES_ACTIVE;
053        } else if ("indices_all_active".equalsIgnoreCase(allowRebalance) || "indicesAllActive".equalsIgnoreCase(allowRebalance)) {
054            type = ClusterRebalanceType.INDICES_ALL_ACTIVE;
055        } else {
056            //logger.warn("[cluster.routing.allocation.allow_rebalance] has a wrong value {}, defaulting to 'indices_all_active'", allowRebalance);
057            type = ClusterRebalanceType.INDICES_ALL_ACTIVE;
058        }
059        //logger.debug("using [cluster.routing.allocation.allow_rebalance] with [{}]", type.toString().toLowerCase());
060    }
061
062    @Override
063    public boolean canRebalance(ShardRouting shardRouting, RoutingAllocation allocation) {
064        if (type == ClusterRebalanceType.INDICES_PRIMARIES_ACTIVE) {
065            for (MutableShardRouting shard : allocation.routingNodes().unassigned()) {
066                if (shard.primary()) {
067                    return false;
068                }
069            }
070            for (RoutingNode node : allocation.routingNodes()) {
071                List<MutableShardRouting> shards = node.shards();
072                for (int i = 0; i < shards.size(); i++) {
073                    MutableShardRouting shard = shards.get(i);
074                    if (shard.primary() && !shard.active() && shard.relocatingNodeId() == null) {
075                        return false;
076                    }
077                }
078            }
079            return true;
080        }
081        if (type == ClusterRebalanceType.INDICES_ALL_ACTIVE) {
082            if (!allocation.routingNodes().unassigned().isEmpty()) {
083                return false;
084            }
085            for (RoutingNode node : allocation.routingNodes()) {
086                List<MutableShardRouting> shards = node.shards();
087                for (int i = 0; i < shards.size(); i++) {
088                    MutableShardRouting shard = shards.get(i);
089                    if (!shard.active() && shard.relocatingNodeId() == null) {
090                        return false;
091                    }
092                }
093            }
094        }
095        // type == Type.ALWAYS
096        return true;
097    }
098}
