020package org.elasticsearch.cluster.routing;

022import gnu.trove.map.hash.TObjectIntHashMap;
023import org.elasticsearch.cluster.ClusterState;
024import org.elasticsearch.cluster.block.ClusterBlocks;
025import org.elasticsearch.cluster.metadata.IndexMetaData;
026import org.elasticsearch.cluster.metadata.MetaData;
027import org.elasticsearch.cluster.node.DiscoveryNode;
028
029import java.util.*;
030
031import static com.google.common.collect.Lists.newArrayList;
032import static com.google.common.collect.Maps.newHashMap;
033
034/**
035 *
036 */
037public class RoutingNodes implements Iterable<RoutingNode> {
038
039    private final MetaData metaData;
040
041    private final ClusterBlocks blocks;
042
043    private final RoutingTable routingTable;
044
045    private final Map<String, RoutingNode> nodesToShards = newHashMap();
046
047    private final List<MutableShardRouting> unassigned = newArrayList();
048
049    private final List<MutableShardRouting> ignoredUnassigned = newArrayList();
050
051    private final Map<String, TObjectIntHashMap<String>> nodesPerAttributeNames = new HashMap<String, TObjectIntHashMap<String>>();
052
053    public RoutingNodes(ClusterState clusterState) {
054        this.metaData = clusterState.metaData();
055        this.blocks = clusterState.blocks();
056        this.routingTable = clusterState.routingTable();
057        Map<String, List<MutableShardRouting>> nodesToShards = newHashMap();
058        // fill in the nodeToShards with the "live" nodes
059        for (DiscoveryNode node : clusterState.nodes().dataNodes().values()) {
060            nodesToShards.put(node.id(), new ArrayList<MutableShardRouting>());
061        }
062
063        // fill in the inverse of node -> shards allocated
064        for (IndexRoutingTable indexRoutingTable : routingTable.indicesRouting().values()) {
065            for (IndexShardRoutingTable indexShard : indexRoutingTable) {
066                for (ShardRouting shard : indexShard) {
067                    if (shard.assignedToNode()) {
068                        List<MutableShardRouting> entries = nodesToShards.get(shard.currentNodeId());
069                        if (entries == null) {
070                            entries = newArrayList();
071                            nodesToShards.put(shard.currentNodeId(), entries);
072                        }
073                        entries.add(new MutableShardRouting(shard));
074                        if (shard.relocating()) {
075                            entries = nodesToShards.get(shard.relocatingNodeId());
076                            if (entries == null) {
077                                entries = newArrayList();
078                                nodesToShards.put(shard.relocatingNodeId(), entries);
079                            }
080                            // add the counterpart shard with relocatingNodeId reflecting the source from which
081                            // it's relocating from.
082                            entries.add(new MutableShardRouting(shard.index(), shard.id(), shard.relocatingNodeId(),
083                                    shard.currentNodeId(), shard.primary(), ShardRoutingState.INITIALIZING, shard.version()));
084                        }
085                    } else {
086                        unassigned.add(new MutableShardRouting(shard));
087                    }
088                }
089            }
090        }
091        for (Map.Entry<String, List<MutableShardRouting>> entry : nodesToShards.entrySet()) {
092            String nodeId = entry.getKey();
093            this.nodesToShards.put(nodeId, new RoutingNode(nodeId, clusterState.nodes().get(nodeId), entry.getValue()));
094        }
095    }
096
097    @Override
098    public Iterator<RoutingNode> iterator() {
099        return nodesToShards.values().iterator();
100    }

115        return metaData();
116    }

118    public ClusterBlocks blocks() {
119        return this.blocks;
120    }
121
122    public ClusterBlocks getBlocks() {
123        return this.blocks;
124    }
125
126    public int requiredAverageNumberOfShardsPerNode() {
127        int totalNumberOfShards = 0;
128        // we need to recompute to take closed shards into account
129        for (IndexMetaData indexMetaData : metaData.indices().values()) {
130            if (indexMetaData.state() == IndexMetaData.State.OPEN) {
131                totalNumberOfShards += indexMetaData.totalNumberOfShards();
132            }
133        }
134        return totalNumberOfShards / nodesToShards.size();
135    }
136
137    public boolean hasUnassigned() {
138        return !unassigned.isEmpty();
139    }
140
141    public List<MutableShardRouting> ignoredUnassigned() {
142        return this.ignoredUnassigned;
143    }
144
145    public List<MutableShardRouting> unassigned() {
146        return this.unassigned;
147    }
148
149    public List<MutableShardRouting> getUnassigned() {
150        return unassigned();
151    }
152
153    public Map<String, RoutingNode> nodesToShards() {
154        return nodesToShards;
155    }
156
157    public Map<String, RoutingNode> getNodesToShards() {
158        return nodesToShards();
159    }
160
161    public RoutingNode node(String nodeId) {
162        return nodesToShards.get(nodeId);
163    }
164
165    public TObjectIntHashMap<String> nodesPerAttributesCounts(String attributeName) {
166        TObjectIntHashMap<String> nodesPerAttributesCounts = nodesPerAttributeNames.get(attributeName);
167        if (nodesPerAttributesCounts != null) {
168            return nodesPerAttributesCounts;
169        }
170        nodesPerAttributesCounts = new TObjectIntHashMap<String>();
171        for (RoutingNode routingNode : this) {
172            String attrValue = routingNode.node().attributes().get(attributeName);
173            nodesPerAttributesCounts.adjustOrPutValue(attrValue, 1, 1);
174        }
175        nodesPerAttributeNames.put(attributeName, nodesPerAttributesCounts);
176        return nodesPerAttributesCounts;
177    }
178
179    public MutableShardRouting findPrimaryForReplica(ShardRouting shard) {
180        assert !shard.primary();
181        for (RoutingNode routingNode : nodesToShards.values()) {
182            List<MutableShardRouting> shards = routingNode.shards();
183            for (int i = 0; i < shards.size(); i++) {
184                MutableShardRouting shardRouting = shards.get(i);
185                if (shardRouting.shardId().equals(shard.shardId()) && shardRouting.primary()) {
186                    return shardRouting;
187                }
188            }
189        }
190        return null;
191    }
192
193    public List<MutableShardRouting> shardsRoutingFor(ShardRouting shardRouting) {
194        return shardsRoutingFor(shardRouting.index(), shardRouting.id());
195    }
196
197    public List<MutableShardRouting> shardsRoutingFor(String index, int shardId) {
198        List<MutableShardRouting> shards = newArrayList();
199        for (RoutingNode routingNode : this) {
200            List<MutableShardRouting> nShards = routingNode.shards();
201            for (int i = 0; i < nShards.size(); i++) {
202                MutableShardRouting shardRouting = nShards.get(i);
203                if (shardRouting.index().equals(index) && shardRouting.id() == shardId) {
204                    shards.add(shardRouting);
205                }
206            }
207        }
208        for (int i = 0; i < unassigned.size(); i++) {
209            MutableShardRouting shardRouting = unassigned.get(i);
210            if (shardRouting.index().equals(index) && shardRouting.id() == shardId) {
211                shards.add(shardRouting);
212            }
213        }
214        return shards;
215    }
216
217    public int numberOfShardsOfType(ShardRoutingState state) {
218        int count = 0;
219        for (RoutingNode routingNode : this) {
220            count += routingNode.numberOfShardsWithState(state);
221        }
222        return count;
223    }
224
225    public List<MutableShardRouting> shardsWithState(ShardRoutingState... state) {
226        List<MutableShardRouting> shards = newArrayList();
227        for (RoutingNode routingNode : this) {
228            shards.addAll(routingNode.shardsWithState(state));
229        }
230        return shards;
231    }
232
233    public List<MutableShardRouting> shardsWithState(String index, ShardRoutingState... state) {
234        List<MutableShardRouting> shards = newArrayList();
235        for (RoutingNode routingNode : this) {
236            shards.addAll(routingNode.shardsWithState(index, state));
237        }
238        return shards;
239    }
240
241    public String prettyPrint() {
242        StringBuilder sb = new StringBuilder("routing_nodes:\n");
243        for (RoutingNode routingNode : this) {
244            sb.append(routingNode.prettyPrint());
245        }
246        sb.append("---- unassigned\n");
247        for (MutableShardRouting shardEntry : unassigned) {
248            sb.append("--------").append(shardEntry.shortSummary()).append('\n');
249        }
250        return sb.toString();
251    }
