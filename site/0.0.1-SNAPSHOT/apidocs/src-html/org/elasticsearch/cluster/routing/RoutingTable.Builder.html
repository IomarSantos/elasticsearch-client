109    /**
110     * All the shards (replicas) for the provided indices.
111     *
112     * @param indices The indices to return all the shards (replicas), can be <tt>null</tt> or empty array to indicate all indices
113     * @return All the shards matching the specific index
114     * @throws IndexMissingException If an index passed does not exists
115     */
116    public List<ShardRouting> allShards(String... indices) throws IndexMissingException {
117        List<ShardRouting> shards = Lists.newArrayList();
118        if (indices == null || indices.length == 0) {
119            indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
120        }
121        for (String index : indices) {
122            IndexRoutingTable indexRoutingTable = index(index);
123            if (indexRoutingTable == null) {
124                throw new IndexMissingException(new Index(index));
125            }
126            for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
127                for (ShardRouting shardRouting : indexShardRoutingTable) {
128                    shards.add(shardRouting);
129                }
130            }
131        }
132        return shards;
133    }

135    /**
136     * All the shards (primary + replicas) for the provided indices grouped (each group is a single element, consisting
137     * of the shard). This is handy for components that expect to get group iterators, but still want in some
138     * cases to iterate over all the shards (and not just one shard in replication group).
139     *
140     * @param indices The indices to return all the shards (replicas), can be <tt>null</tt> or empty array to indicate all indices
141     * @return All the shards grouped into a single shard element group each
142     * @throws IndexMissingException If an index passed does not exists
143     * @see IndexRoutingTable#groupByAllIt()
144     */
145    public GroupShardsIterator allShardsGrouped(String... indices) throws IndexMissingException {
146        // use list here since we need to maintain identity across shards
147        ArrayList<ShardIterator> set = new ArrayList<ShardIterator>();
148        if (indices == null || indices.length == 0) {
149            indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
150        }
151        for (String index : indices) {
152            IndexRoutingTable indexRoutingTable = index(index);
153            if (indexRoutingTable == null) {
154                continue;
155                // we simply ignore indices that don't exists (make sense for operations that use it currently)
156//                throw new IndexMissingException(new Index(index));
157            }
158            for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
159                for (ShardRouting shardRouting : indexShardRoutingTable) {
160                    set.add(shardRouting.shardsIt());
161                }
162            }
163        }
164        return new GroupShardsIterator(set);
165    }

256    public static class Builder {
257
258        private long version;
259
260        private final Map<String, IndexRoutingTable> indicesRouting = newHashMap();
261
262        public Builder routingTable(RoutingTable routingTable) {
263            version = routingTable.version;
264            for (IndexRoutingTable indexRoutingTable : routingTable) {
265                indicesRouting.put(indexRoutingTable.index(), indexRoutingTable);
266            }
267            return this;
268        }
269
270        public Builder updateNodes(RoutingNodes routingNodes) {
271            // this is being called without pre initializing the routing table, so we must copy over the version as well
272            this.version = routingNodes.routingTable().version();
273
274            Map<String, IndexRoutingTable.Builder> indexRoutingTableBuilders = newHashMap();
275            for (RoutingNode routingNode : routingNodes) {
276                for (MutableShardRouting shardRoutingEntry : routingNode) {
277                    // every relocating shard has a double entry, ignore the target one.
278                    if (shardRoutingEntry.state() == ShardRoutingState.INITIALIZING && shardRoutingEntry.relocatingNodeId() != null)
279                        continue;
280
281                    String index = shardRoutingEntry.index();
282                    IndexRoutingTable.Builder indexBuilder = indexRoutingTableBuilders.get(index);
283                    if (indexBuilder == null) {
284                        indexBuilder = new IndexRoutingTable.Builder(index);
285                        indexRoutingTableBuilders.put(index, indexBuilder);
286                    }
287
288                    IndexShardRoutingTable refData = routingNodes.routingTable().index(shardRoutingEntry.index()).shard(shardRoutingEntry.id());
289                    indexBuilder.addShard(refData, shardRoutingEntry);
290                }
291            }
292            for (MutableShardRouting shardRoutingEntry : Iterables.concat(routingNodes.unassigned(), routingNodes.ignoredUnassigned())) {
293                String index = shardRoutingEntry.index();
294                IndexRoutingTable.Builder indexBuilder = indexRoutingTableBuilders.get(index);
295                if (indexBuilder == null) {
296                    indexBuilder = new IndexRoutingTable.Builder(index);
297                    indexRoutingTableBuilders.put(index, indexBuilder);
298                }
299                IndexShardRoutingTable refData = routingNodes.routingTable().index(shardRoutingEntry.index()).shard(shardRoutingEntry.id());
300                indexBuilder.addShard(refData, shardRoutingEntry);
301            }
302            for (IndexRoutingTable.Builder indexBuilder : indexRoutingTableBuilders.values()) {
303                add(indexBuilder);
304            }
305            return this;
306        }
307
308        public Builder updateNumberOfReplicas(int numberOfReplicas, String... indices) throws IndexMissingException {
309            if (indices == null || indices.length == 0) {
310                indices = indicesRouting.keySet().toArray(new String[indicesRouting.keySet().size()]);
311            }
312            for (String index : indices) {
313                IndexRoutingTable indexRoutingTable = indicesRouting.get(index);
314                if (indexRoutingTable == null) {
315                    // ignore index missing failure, its closed...
316                    continue;
317                }
318                int currentNumberOfReplicas = indexRoutingTable.shards().get(0).size() - 1; // remove the required primary
319                IndexRoutingTable.Builder builder = new IndexRoutingTable.Builder(index);
320                // re-add all the shards
321                for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
322                    builder.addIndexShard(indexShardRoutingTable);
323                }
324                if (currentNumberOfReplicas < numberOfReplicas) {
325                    // now, add "empty" ones
326                    for (int i = 0; i < (numberOfReplicas - currentNumberOfReplicas); i++) {
327                        builder.addReplica();
328                    }
329                } else if (currentNumberOfReplicas > numberOfReplicas) {
330                    int delta = currentNumberOfReplicas - numberOfReplicas;
331                    if (delta <= 0) {
332                        // ignore, can't remove below the current one...
333                    } else {
334                        for (int i = 0; i < delta; i++) {
335                            builder.removeReplica();
336                        }
337                    }
338                }
339                indicesRouting.put(index, builder.build());
340            }
341            return this;
342        }

344        public Builder addAsNew(IndexMetaData indexMetaData) {
345            if (indexMetaData.state() == IndexMetaData.State.OPEN) {
346                IndexRoutingTable.Builder indexRoutingBuilder = new IndexRoutingTable.Builder(indexMetaData.index())
347                        .initializeAsNew(indexMetaData);
348                add(indexRoutingBuilder);
349            }
350            return this;
351        }
352
353        public Builder addAsRecovery(IndexMetaData indexMetaData) {
354            if (indexMetaData.state() == IndexMetaData.State.OPEN) {
355                IndexRoutingTable.Builder indexRoutingBuilder = new IndexRoutingTable.Builder(indexMetaData.index())
356                        .initializeAsRecovery(indexMetaData);
357                add(indexRoutingBuilder);
358            }
359            return this;
360        }

362        public Builder add(IndexRoutingTable indexRoutingTable) {
363            indexRoutingTable.validate();
364            indicesRouting.put(indexRoutingTable.index(), indexRoutingTable);
365            return this;
366        }
367
368        public Builder add(IndexRoutingTable.Builder indexRoutingTableBuilder) {
369            add(indexRoutingTableBuilder.build());
370            return this;
371        }
372
373        public Builder remove(String index) {
374            indicesRouting.remove(index);

378        public Builder version(long version) {
379            this.version = version;
380            return this;
381        }
382
383        public RoutingTable build() {
384            // normalize the versions right before we build it...
385            for (IndexRoutingTable indexRoutingTable : indicesRouting.values()) {
386                indicesRouting.put(indexRoutingTable.index(), indexRoutingTable.normalizeVersions());
387            }
388            return new RoutingTable(version, indicesRouting);
389        }
390
391        public static RoutingTable readFrom(StreamInput in) throws IOException {
392            Builder builder = new Builder();
393            builder.version = in.readLong();
394            int size = in.readVInt();
395            for (int i = 0; i < size; i++) {
396                IndexRoutingTable index = IndexRoutingTable.Builder.readFrom(in);
397                builder.add(index);
398            }
399
400            return builder.build();
401        }
402
403        public static void writeTo(RoutingTable table, StreamOutput out) throws IOException {
404            out.writeLong(table.version);
405            out.writeVInt(table.indicesRouting.size());
406            for (IndexRoutingTable index : table.indicesRouting.values()) {
407                IndexRoutingTable.Builder.writeTo(index, out);
408            }
409        }
