020package org.elasticsearch.cluster.routing;
021
022import org.elasticsearch.ElasticSearchIllegalStateException;
023import org.elasticsearch.cluster.node.DiscoveryNode;
024
025import java.util.ArrayList;
026import java.util.Iterator;
027import java.util.List;
028
029import static com.google.common.collect.Lists.newArrayList;
030
031/**
032 */
033public class RoutingNode implements Iterable<MutableShardRouting> {
034
035    private final String nodeId;
036
037    private final DiscoveryNode node;
038
039    private final List<MutableShardRouting> shards;
040
041    public RoutingNode(String nodeId, DiscoveryNode node) {
042        this(nodeId, node, new ArrayList<MutableShardRouting>());
043    }
044
045    public RoutingNode(String nodeId, DiscoveryNode node, List<MutableShardRouting> shards) {
046        this.nodeId = nodeId;
047        this.node = node;
048        this.shards = shards;
049    }
050
051    @Override
052    public Iterator<MutableShardRouting> iterator() {
053        return shards.iterator();
054    }
055
056    public DiscoveryNode node() {
057        return this.node;
058    }
059
060    public String nodeId() {
061        return this.nodeId;
062    }
063
064    public List<MutableShardRouting> shards() {
065        return this.shards;
066    }
067
068    public void add(MutableShardRouting shard) {
069        for (MutableShardRouting shardRouting : shards) {
070            if (shardRouting.shardId().equals(shard.shardId())) {
071                throw new ElasticSearchIllegalStateException("Trying to add a shard [" + shard.shardId().index().name() + "][" + shard.shardId().id() + "] to a node [" + nodeId + "] where it already exists");
072            }
073        }
074        shards.add(shard);
075        shard.assignToNode(node.id());
076    }
077
078    public void removeByShardId(int shardId) {
079        for (Iterator<MutableShardRouting> it = shards.iterator(); it.hasNext(); ) {
080            MutableShardRouting shard = it.next();
081            if (shard.id() == shardId) {
082                it.remove();
083            }
084        }
085    }
086
087    public int numberOfShardsWithState(ShardRoutingState... states) {
088        int count = 0;
089        for (MutableShardRouting shardEntry : this) {
090            for (ShardRoutingState state : states) {
091                if (shardEntry.state() == state) {
092                    count++;
093                }
094            }
095        }
096        return count;
097    }
098
099    public List<MutableShardRouting> shardsWithState(ShardRoutingState... states) {
100        List<MutableShardRouting> shards = newArrayList();
101        for (MutableShardRouting shardEntry : this) {
102            for (ShardRoutingState state : states) {
103                if (shardEntry.state() == state) {
104                    shards.add(shardEntry);
105                }
106            }
107        }
108        return shards;
109    }
110
111    public List<MutableShardRouting> shardsWithState(String index, ShardRoutingState... states) {
112        List<MutableShardRouting> shards = newArrayList();
113        for (MutableShardRouting shardEntry : this) {
114            if (!shardEntry.index().equals(index)) {
115                continue;
116            }
117            for (ShardRoutingState state : states) {
118                if (shardEntry.state() == state) {
119                    shards.add(shardEntry);
120                }
121            }
122        }
123        return shards;
124    }
125
126    public int numberOfShardsNotWithState(ShardRoutingState state) {
127        int count = 0;
128        for (MutableShardRouting shardEntry : this) {
129            if (shardEntry.state() != state) {
130                count++;
131            }
132        }
133        return count;
134    }
135
136    /**
137     * The number fo shards on this node that will not be eventually relocated.
138     */
139    public int numberOfOwningShards() {
140        int count = 0;
141        for (MutableShardRouting shardEntry : this) {
142            if (shardEntry.state() != ShardRoutingState.RELOCATING) {
143                count++;
144            }
145        }
146
147        return count;
148    }
149
150    public String prettyPrint() {
151        StringBuilder sb = new StringBuilder();
152        sb.append("-----node_id[").append(nodeId).append("][" + (node == null ? "X" : "V") + "]\n");
153        for (MutableShardRouting entry : shards) {
154            sb.append("--------").append(entry.shortSummary()).append('\n');
155        }
156        return sb.toString();
157    }
158}
