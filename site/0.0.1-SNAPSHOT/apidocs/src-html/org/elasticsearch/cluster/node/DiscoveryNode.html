020package org.elasticsearch.cluster.node;
021
022import com.google.common.collect.ImmutableList;
023import com.google.common.collect.ImmutableMap;
024import org.elasticsearch.Version;
025import org.elasticsearch.common.io.stream.StreamInput;
026import org.elasticsearch.common.io.stream.StreamOutput;
027import org.elasticsearch.common.io.stream.Streamable;
028import org.elasticsearch.common.settings.Settings;
029import org.elasticsearch.common.transport.TransportAddress;
030import org.elasticsearch.common.transport.TransportAddressSerializers;
031
032import java.io.IOException;
033import java.io.Serializable;
034import java.util.Map;
035
036import static org.elasticsearch.common.transport.TransportAddressSerializers.addressToStream;
037
038/**
039 * A discovery node represents a node that is part of the cluster.
040 */
041public class DiscoveryNode implements Streamable, Serializable {
042
043    public static boolean nodeRequiresLocalStorage(Settings settings) {
044        return !(settings.getAsBoolean("node.client", false) || (!settings.getAsBoolean("node.data", true) && !settings.getAsBoolean("node.master", true)));
045    }
046
047    public static boolean clientNode(Settings settings) {
048        String client = settings.get("node.client");
049        return client != null && client.equals("true");
050    }
051
052    public static boolean masterNode(Settings settings) {
053        String master = settings.get("node.master");
054        if (master == null) {
055            return !clientNode(settings);
056        }
057        return master.equals("true");
058    }
059
060    public static boolean dataNode(Settings settings) {
061        String data = settings.get("node.data");
062        if (data == null) {
063            return !clientNode(settings);
064        }
065        return data.equals("true");
066    }
067
068    public static final ImmutableList<DiscoveryNode> EMPTY_LIST = ImmutableList.of();
069
070    private String nodeName = "".intern();
071
072    private String nodeId;
073
074    private TransportAddress address;
075
076    private ImmutableMap<String, String> attributes;
077
078    private Version version = Version.CURRENT;
079
080    private DiscoveryNode() {
081    }
082
083    public DiscoveryNode(String nodeId, TransportAddress address) {
084        this("", nodeId, address, ImmutableMap.<String, String>of());
085    }
086
087    public DiscoveryNode(String nodeName, String nodeId, TransportAddress address, Map<String, String> attributes) {
088        if (nodeName == null) {
089            this.nodeName = "".intern();
090        } else {
091            this.nodeName = nodeName.intern();
092        }
093        ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
094        for (Map.Entry<String, String> entry : attributes.entrySet()) {
095            builder.put(entry.getKey().intern(), entry.getValue().intern());
096        }
097        this.attributes = builder.build();
098        this.nodeId = nodeId.intern();
099        this.address = address;
100    }
101
102    /**
103     * Should this node form a connection to the provided node.
104     */
105    public boolean shouldConnectTo(DiscoveryNode otherNode) {
106        if (clientNode() && otherNode.clientNode()) {
107            return false;
108        }
109        return true;
110    }
111
112    /**
113     * The address that the node can be communicated with.
114     */
115    public TransportAddress address() {
116        return address;
117    }
118
119    /**
120     * The address that the node can be communicated with.
121     */
122    public TransportAddress getAddress() {
123        return address();
124    }
125
126    /**
127     * The unique id of the node.
128     */
129    public String id() {
130        return nodeId;
131    }
132
133    /**
134     * The unique id of the node.
135     */
136    public String getId() {
137        return id();
138    }
139
140    /**
141     * The name of the node.
142     */
143    public String name() {
144        return this.nodeName;
145    }
146
147    /**
148     * The name of the node.
149     */
150    public String getName() {
151        return name();
152    }
153
154    /**
155     * The node attributes.
156     */
157    public ImmutableMap<String, String> attributes() {
158        return this.attributes;
159    }
160
161    /**
162     * The node attributes.
163     */
164    public ImmutableMap<String, String> getAttributes() {
165        return attributes();
166    }
167
168    /**
169     * Should this node hold data (shards) or not.
170     */
171    public boolean dataNode() {
172        String data = attributes.get("data");
173        if (data == null) {
174            return !clientNode();
175        }
176        return data.equals("true");
177    }
178
179    /**
180     * Should this node hold data (shards) or not.
181     */
182    public boolean isDataNode() {
183        return dataNode();
184    }
185
186    /**
187     * Is the node a client node or not.
188     */
189    public boolean clientNode() {
190        String client = attributes.get("client");
191        return client != null && client.equals("true");
192    }
193
194    public boolean isClientNode() {
195        return clientNode();
196    }
197
198    /**
199     * Can this node become master or not.
200     */
201    public boolean masterNode() {
202        String master = attributes.get("master");
203        if (master == null) {
204            return !clientNode();
205        }
206        return master.equals("true");
207    }
208
209    /**
210     * Can this node become master or not.
211     */
212    public boolean isMasterNode() {
213        return masterNode();
214    }
215
216    public Version version() {
217        return this.version;
218    }
219
220    public Version getVersion() {
221        return this.version;
222    }
223
224    public static DiscoveryNode readNode(StreamInput in) throws IOException {
225        DiscoveryNode node = new DiscoveryNode();
226        node.readFrom(in);
227        return node;
228    }
229
230    @Override
231    public void readFrom(StreamInput in) throws IOException {
232        nodeName = in.readString().intern();
233        nodeId = in.readString().intern();
234        address = TransportAddressSerializers.addressFromStream(in);
235        int size = in.readVInt();
236        ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
237        for (int i = 0; i < size; i++) {
238            builder.put(in.readString().intern(), in.readString().intern());
239        }
240        attributes = builder.build();
241        version = Version.readVersion(in);
242    }
243
244    @Override
245    public void writeTo(StreamOutput out) throws IOException {
246        out.writeString(nodeName);
247        out.writeString(nodeId);
248        addressToStream(out, address);
249        out.writeVInt(attributes.size());
250        for (Map.Entry<String, String> entry : attributes.entrySet()) {
251            out.writeString(entry.getKey());
252            out.writeString(entry.getValue());
253        }
254        Version.writeVersion(version, out);
255    }
256
257    @Override
258    public boolean equals(Object obj) {
259        if (!(obj instanceof DiscoveryNode))
260            return false;
261
262        DiscoveryNode other = (DiscoveryNode) obj;
263        return this.nodeId.equals(other.nodeId);
264    }
265
266    @Override
267    public int hashCode() {
268        return nodeId.hashCode();
269    }
270
271    @Override
272    public String toString() {
273        StringBuilder sb = new StringBuilder();
274        if (nodeName.length() > 0) {
275            sb.append('[').append(nodeName).append(']');
276        }
277        if (nodeId != null) {
278            sb.append('[').append(nodeId).append(']');
279        }
280        if (address != null) {
281            sb.append('[').append(address).append(']');
282        }
283        if (!attributes.isEmpty()) {
284            sb.append(attributes);
285        }
286        return sb.toString();
287    }
