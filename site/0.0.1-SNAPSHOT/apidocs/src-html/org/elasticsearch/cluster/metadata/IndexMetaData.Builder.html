143    public static enum State {
144        OPEN((byte) 0),
145        CLOSE((byte) 1);
146
147        private final byte id;
148
149        State(byte id) {
150            this.id = id;
151        }
152
153        public byte id() {
154            return this.id;
155        }
156
157        public static State fromId(byte id) {
158            if (id == 0) {
159                return OPEN;
160            } else if (id == 1) {
161                return CLOSE;
162            }
163            throw new ElasticSearchIllegalStateException("No state match for id [" + id + "]");
164        }
165
166        public static State fromString(String state) {
167            if ("open".equals(state)) {
168                return OPEN;
169            } else if ("close".equals(state)) {
170                return CLOSE;
171            }
172            throw new ElasticSearchIllegalStateException("No state match for [" + state + "]");
173        }
174    }

203    private IndexMetaData(String index, long version, State state, Settings settings, ImmutableMap<String, MappingMetaData> mappings, ImmutableMap<String, AliasMetaData> aliases, ImmutableMap<String, Custom> customs) {
204        Preconditions.checkArgument(settings.getAsInt(SETTING_NUMBER_OF_SHARDS, -1) != -1, "must specify numberOfShards for index [" + index + "]");
205        Preconditions.checkArgument(settings.getAsInt(SETTING_NUMBER_OF_REPLICAS, -1) != -1, "must specify numberOfReplicas for index [" + index + "]");
206        this.index = index;
207        this.version = version;
208        this.state = state;
209        this.settings = settings;
210        this.mappings = mappings;
211        this.customs = customs;
212        this.totalNumberOfShards = numberOfShards() * (numberOfReplicas() + 1);
213
214        this.aliases = aliases;
215
216        ImmutableMap<String, String> includeMap = settings.getByPrefix("index.routing.allocation.include.").getAsMap();
217        if (includeMap.isEmpty()) {
218            includeFilters = null;
219        } else {
220            includeFilters = DiscoveryNodeFilters.buildFromKeyValue(includeMap);
221        }
222        ImmutableMap<String, String> excludeMap = settings.getByPrefix("index.routing.allocation.exclude.").getAsMap();
223        if (excludeMap.isEmpty()) {
224            excludeFilters = null;
225        } else {
226            excludeFilters = DiscoveryNodeFilters.buildFromKeyValue(excludeMap);
227        }
228    }

345    @Override
346    public boolean equals(Object o) {
347        if (this == o) return true;
348        if (o == null || getClass() != o.getClass()) return false;
349
350        IndexMetaData that = (IndexMetaData) o;
351
352        if (!aliases.equals(that.aliases)) return false;
353        if (!index.equals(that.index)) return false;
354        if (!mappings.equals(that.mappings)) return false;
355        if (!settings.equals(that.settings)) return false;
356        if (state != that.state) return false;
357
358        return true;
359    }

513        public IndexMetaData build() {
514            MapBuilder<String, AliasMetaData> tmpAliases = aliases;
515            Settings tmpSettings = settings;
516
517            // For backward compatibility
518            String[] legacyAliases = settings.getAsArray("index.aliases");
519            if (legacyAliases.length > 0) {
520                tmpAliases = MapBuilder.newMapBuilder();
521                for (String alias : legacyAliases) {
522                    AliasMetaData aliasMd = AliasMetaData.newAliasMetaDataBuilder(alias).build();
523                    tmpAliases.put(alias, aliasMd);
524                }
525                tmpAliases.putAll(aliases.immutableMap());
526                // Remove index.aliases from settings once they are migrated to the new data structure
527                tmpSettings = ImmutableSettings.settingsBuilder().put(settings).putArray("index.aliases").build();
528            }
529
530            // update default mapping on the MappingMetaData
531            if (mappings.containsKey("_default_"/*MapperService.DEFAULT_MAPPING*/)) {
532                MappingMetaData defaultMapping = mappings.get("_default_"/*MapperService.DEFAULT_MAPPING*/);
533                for (MappingMetaData mappingMetaData : mappings.map().values()) {
534                    mappingMetaData.updateDefaultMapping(defaultMapping);
535                }
536            }
537
538            return new IndexMetaData(index, version, state, tmpSettings, mappings.immutableMap(), tmpAliases.immutableMap(), customs.immutableMap());
539        }

541        public static void toXContent(IndexMetaData indexMetaData, XContentBuilder builder, ToXContent.Params params) throws IOException {
542            builder.startObject(indexMetaData.index(), XContentBuilder.FieldCaseConversion.NONE);
543
544            builder.field("version", indexMetaData.version());
545            builder.field("state", indexMetaData.state().toString().toLowerCase(Locale.ENGLISH));
546
547            boolean binary = params.paramAsBoolean("binary", false);
548
549            builder.startObject("settings");
550            for (Map.Entry<String, String> entry : indexMetaData.settings().getAsMap().entrySet()) {
551                builder.field(entry.getKey(), entry.getValue());
552            }
553            builder.endObject();
554
555            builder.startArray("mappings");
556            for (Map.Entry<String, MappingMetaData> entry : indexMetaData.mappings().entrySet()) {
557                if (binary) {
558                    builder.value(entry.getValue().source().compressed());
559                } else {
560                    byte[] data = entry.getValue().source().uncompressed();
561                    XContentParser parser = XContentFactory.xContent(data).createParser(data);
562                    Map<String, Object> mapping = parser.mapOrdered();
563                    parser.close();
564                    builder.map(mapping);
565                }
566            }
567            builder.endArray();
568
569            for (Map.Entry<String, Custom> entry : indexMetaData.customs().entrySet()) {
570                builder.startObject(entry.getKey(), XContentBuilder.FieldCaseConversion.NONE);
571                lookupFactorySafe(entry.getKey()).toXContent(entry.getValue(), builder, params);
572                builder.endObject();
573            }
574
575            builder.startObject("aliases");
576            for (AliasMetaData alias : indexMetaData.aliases().values()) {
577                AliasMetaData.Builder.toXContent(alias, builder, params);
578            }
579            builder.endObject();
580
581
582            builder.endObject();
583        }

585        public static IndexMetaData fromXContent(XContentParser parser) throws IOException {
586            if (parser.currentToken() == XContentParser.Token.START_OBJECT) {
587                parser.nextToken();
588            }
589            Builder builder = new Builder(parser.currentName());
590
591            String currentFieldName = null;
592            XContentParser.Token token = parser.nextToken();
593            while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
594                if (token == XContentParser.Token.FIELD_NAME) {
595                    currentFieldName = parser.currentName();
596                } else if (token == XContentParser.Token.START_OBJECT) {
597                    if ("settings".equals(currentFieldName)) {
598                        builder.settings(ImmutableSettings.settingsBuilder().put(SettingsLoader.Helper.loadNestedFromMap(parser.mapOrdered())));
599                    } else if ("mappings".equals(currentFieldName)) {
600                        while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
601                            if (token == XContentParser.Token.FIELD_NAME) {
602                                currentFieldName = parser.currentName();
603                            } else if (token == XContentParser.Token.START_OBJECT) {
604                                String mappingType = currentFieldName;
605                                Map<String, Object> mappingSource = MapBuilder.<String, Object>newMapBuilder().put(mappingType, parser.mapOrdered()).map();
606                                builder.putMapping(new MappingMetaData(mappingType, mappingSource));
607                            }
608                        }
609                    } else if ("aliases".equals(currentFieldName)) {
610                        while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
611                            builder.putAlias(AliasMetaData.Builder.fromXContent(parser));
612                        }
613                    } else {
614                        // check if its a custom index metadata
615                        Custom.Factory<Custom> factory = lookupFactory(currentFieldName);
616                        if (factory == null) {
617                            //TODO warn
618                            parser.skipChildren();
619                        } else {
620                            builder.putCustom(factory.type(), factory.fromXContent(parser));
621                        }
622                    }
623                } else if (token == XContentParser.Token.START_ARRAY) {
624                    if ("mappings".equals(currentFieldName)) {
625                        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
626                            if (token == XContentParser.Token.VALUE_EMBEDDED_OBJECT) {
627                                builder.putMapping(new MappingMetaData(new CompressedString(parser.binaryValue())));
628                            } else {
629                                Map<String, Object> mapping = parser.mapOrdered();
630                                if (mapping.size() == 1) {
631                                    String mappingType = mapping.keySet().iterator().next();
632                                    builder.putMapping(new MappingMetaData(mappingType, mapping));
633                                }
634                            }
635                        }
636                    }
637                } else if (token.isValue()) {
638                    if ("state".equals(currentFieldName)) {
639                        builder.state(State.fromString(parser.text()));
640                    } else if ("version".equals(currentFieldName)) {
641                        builder.version(parser.longValue());
642                    }
643                }
644            }
645            return builder.build();
646        }

648        public static IndexMetaData readFrom(StreamInput in) throws IOException {
649            Builder builder = new Builder(in.readUTF());
650            builder.version(in.readLong());
651            builder.state(State.fromId(in.readByte()));
652            builder.settings(readSettingsFromStream(in));
653            int mappingsSize = in.readVInt();
654            for (int i = 0; i < mappingsSize; i++) {
655                MappingMetaData mappingMd = MappingMetaData.readFrom(in);
656                builder.putMapping(mappingMd);
657            }
658            int aliasesSize = in.readVInt();
659            for (int i = 0; i < aliasesSize; i++) {
660                AliasMetaData aliasMd = AliasMetaData.Builder.readFrom(in);
661                builder.putAlias(aliasMd);
662            }
663            int customSize = in.readVInt();
664            for (int i = 0; i < customSize; i++) {
665                String type = in.readUTF();
666                Custom customIndexMetaData = lookupFactorySafe(type).readFrom(in);
667                builder.putCustom(type, customIndexMetaData);
668            }
669            return builder.build();
670        }

672        public static void writeTo(IndexMetaData indexMetaData, StreamOutput out) throws IOException {
673            out.writeUTF(indexMetaData.index());
674            out.writeLong(indexMetaData.version());
675            out.writeByte(indexMetaData.state().id());
676            writeSettingsToStream(indexMetaData.settings(), out);
677            out.writeVInt(indexMetaData.mappings().size());
678            for (MappingMetaData mappingMd : indexMetaData.mappings().values()) {
679                MappingMetaData.writeTo(mappingMd, out);
680            }
681            out.writeVInt(indexMetaData.aliases().size());
682            for (AliasMetaData aliasMd : indexMetaData.aliases().values()) {
683                AliasMetaData.Builder.writeTo(aliasMd, out);
684            }
685            out.writeVInt(indexMetaData.customs().size());
686            for (Map.Entry<String, Custom> entry : indexMetaData.customs().entrySet()) {
687                out.writeUTF(entry.getKey());
688                lookupFactorySafe(entry.getKey()).writeTo(entry.getValue(), out);
689            }
690        }
