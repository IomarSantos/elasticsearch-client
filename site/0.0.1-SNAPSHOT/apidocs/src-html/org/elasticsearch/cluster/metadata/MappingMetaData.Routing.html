159
160        public static String parseStringTimestamp(String timestampAsString, FormatDateTimeFormatter dateTimeFormatter) throws TimestampParsingException {
161            long ts;
162            try {
163                // if we manage to parse it, its a millisecond timestamp, just return the string as is
164                ts = Long.parseLong(timestampAsString);
165                return timestampAsString;
166            } catch (NumberFormatException e) {
167                try {
168                    ts = dateTimeFormatter.parser().parseMillis(timestampAsString);
169                } catch (RuntimeException e1) {
170                    throw new TimestampParsingException(timestampAsString);
171                }
172            }
173            return Long.toString(ts);
174        }

270    public MappingMetaData(String type, Map<String, Object> mapping) throws IOException {
271        this.type = type;
272        XContentBuilder mappingBuilder = XContentFactory.jsonBuilder().map(mapping);
273        this.source = new CompressedString(mappingBuilder.bytes());
274        Map<String, Object> withoutType = mapping;
275        if (mapping.size() == 1 && mapping.containsKey(type)) {
276            withoutType = (Map<String, Object>) mapping.get(type);
277        }
278        initMappers(withoutType);
279    }

281    public MappingMetaData(CompressedString mapping) throws IOException {
282        this.source = mapping;
283        Map<String, Object> mappingMap = XContentHelper.createParser(mapping.compressed(), 0, mapping.compressed().length).mapOrderedAndClose();
284        if (mappingMap.size() != 1) {
285            throw new ElasticSearchIllegalStateException("Can't derive type from mapping, no root type: " + mapping.string());
286        }
287        this.type = mappingMap.keySet().iterator().next();
288        initMappers((Map<String, Object>) mappingMap.get(this.type));
289    }

291    private void initMappers(Map<String, Object> withoutType) {
292        if (withoutType.containsKey("_id")) {
293            String path = null;
294            Map<String, Object> routingNode = (Map<String, Object>) withoutType.get("_id");
295            for (Map.Entry<String, Object> entry : routingNode.entrySet()) {
296                String fieldName = Strings.toUnderscoreCase(entry.getKey());
297                Object fieldNode = entry.getValue();
298                if (fieldName.equals("path")) {
299                    path = fieldNode.toString();
300                }
301            }
302            this.id = new Id(path);
303        } else {
304            this.id = Id.EMPTY;
305        }
306        if (withoutType.containsKey("_routing")) {
307            boolean required = false;
308            String path = null;
309            Map<String, Object> routingNode = (Map<String, Object>) withoutType.get("_routing");
310            for (Map.Entry<String, Object> entry : routingNode.entrySet()) {
311                String fieldName = Strings.toUnderscoreCase(entry.getKey());
312                Object fieldNode = entry.getValue();
313                if (fieldName.equals("required")) {
314                    required = nodeBooleanValue(fieldNode);
315                } else if (fieldName.equals("path")) {
316                    path = fieldNode.toString();
317                }
318            }
319            this.routing = new Routing(required, path);
320        } else {
321            this.routing = Routing.EMPTY;
322        }
323        if (withoutType.containsKey("_timestamp")) {
324            boolean enabled = false;
325            String path = null;
326            String format = "dateOptionalTime"; //TimestampFieldMapper.DEFAULT_DATE_TIME_FORMAT;
327            Map<String, Object> timestampNode = (Map<String, Object>) withoutType.get("_timestamp");
328            for (Map.Entry<String, Object> entry : timestampNode.entrySet()) {
329                String fieldName = Strings.toUnderscoreCase(entry.getKey());
330                Object fieldNode = entry.getValue();
331                if (fieldName.equals("enabled")) {
332                    enabled = nodeBooleanValue(fieldNode);
333                } else if (fieldName.equals("path")) {
334                    path = fieldNode.toString();
335                } else if (fieldName.equals("format")) {
336                    format = fieldNode.toString();
337                }
338            }
339            this.timestamp = new Timestamp(enabled, path, format);
340        } else {
341            this.timestamp = Timestamp.EMPTY;
342        }
343    }

412    public void parse(XContentParser parser, ParseContext parseContext) throws IOException {
413        innerParse(parser, parseContext);
414    }
415
416    private void innerParse(XContentParser parser, ParseContext context) throws IOException {
417        if (!context.parsingStillNeeded()) {
418            return;
419        }
420
421        XContentParser.Token t = parser.currentToken();
422        if (t == null) {
423            t = parser.nextToken();
424        }
425        if (t == XContentParser.Token.START_OBJECT) {
426            t = parser.nextToken();
427        }
428        String idPart = context.idParsingStillNeeded() ? id().pathElements()[context.locationId] : null;
429        String routingPart = context.routingParsingStillNeeded() ? routing().pathElements()[context.locationRouting] : null;
430        String timestampPart = context.timestampParsingStillNeeded() ? timestamp().pathElements()[context.locationTimestamp] : null;
431
432        for (; t == XContentParser.Token.FIELD_NAME; t = parser.nextToken()) {
433            // Must point to field name
434            String fieldName = parser.currentName();
435            // And then the value...
436            t = parser.nextToken();
437            boolean incLocationId = false;
438            boolean incLocationRouting = false;
439            boolean incLocationTimestamp = false;
440            if (context.idParsingStillNeeded() && fieldName.equals(idPart)) {
441                if (context.locationId + 1 == id.pathElements().length) {
442                    context.id = parser.textOrNull();
443                    context.idResolved = true;
444                } else {
445                    incLocationId = true;
446                }
447            }
448            if (context.routingParsingStillNeeded() && fieldName.equals(routingPart)) {
449                if (context.locationRouting + 1 == routing.pathElements().length) {
450                    context.routing = parser.textOrNull();
451                    context.routingResolved = true;
452                } else {
453                    incLocationRouting = true;
454                }
455            }
456            if (context.timestampParsingStillNeeded() && fieldName.equals(timestampPart)) {
457                if (context.locationTimestamp + 1 == timestamp.pathElements().length) {
458                    context.timestamp = parser.textOrNull();
459                    context.timestampResolved = true;
460                } else {
461                    incLocationTimestamp = true;
462                }
463            }
464
465            if (incLocationId || incLocationRouting || incLocationTimestamp) {
466                if (t == XContentParser.Token.START_OBJECT) {
467                    context.locationId += incLocationId ? 1 : 0;
468                    context.locationRouting += incLocationRouting ? 1 : 0;
469                    context.locationTimestamp += incLocationTimestamp ? 1 : 0;
470                    innerParse(parser, context);
471                    context.locationId -= incLocationId ? 1 : 0;
472                    context.locationRouting -= incLocationRouting ? 1 : 0;
473                    context.locationTimestamp -= incLocationTimestamp ? 1 : 0;
474                }
475            } else {
476                parser.skipChildren();
477            }
478
479            if (!context.parsingStillNeeded()) {
480                return;
481            }
482        }
483    }

485    public static void writeTo(MappingMetaData mappingMd, StreamOutput out) throws IOException {
486        out.writeUTF(mappingMd.type());
487        mappingMd.source().writeTo(out);
488        // id
489        if (mappingMd.id().hasPath()) {
490            out.writeBoolean(true);
491            out.writeUTF(mappingMd.id().path());
492        } else {
493            out.writeBoolean(false);
494        }
495        // routing
496        out.writeBoolean(mappingMd.routing().required());
497        if (mappingMd.routing().hasPath()) {
498            out.writeBoolean(true);
499            out.writeUTF(mappingMd.routing().path());
500        } else {
501            out.writeBoolean(false);
502        }
503        // timestamp
504        out.writeBoolean(mappingMd.timestamp().enabled());
505        if (mappingMd.timestamp().hasPath()) {
506            out.writeBoolean(true);
507            out.writeUTF(mappingMd.timestamp().path());
508        } else {
509            out.writeBoolean(false);
510        }
511        out.writeUTF(mappingMd.timestamp().format());
512    }

540    public static MappingMetaData readFrom(StreamInput in) throws IOException {
541        String type = in.readUTF();
542        CompressedString source = CompressedString.readCompressedString(in);
543        // id
544        Id id = new Id(in.readBoolean() ? in.readUTF() : null);
545        // routing
546        Routing routing = new Routing(in.readBoolean(), in.readBoolean() ? in.readUTF() : null);
547        // timestamp
548        Timestamp timestamp = new Timestamp(in.readBoolean(), in.readBoolean() ? in.readUTF() : null, in.readUTF());
549        return new MappingMetaData(type, source, id, routing, timestamp);
550    }
