067
068        interface Factory<T extends Custom> {
069
070            String type();
071
072            T readFrom(StreamInput in) throws IOException;
073
074            void writeTo(T customState, StreamOutput out) throws IOException;
075
076            void toXContent(T customState, XContentBuilder builder, ToXContent.Params params);
077        }
078    }

080    public static Map<String, Custom.Factory> customFactories = new HashMap<String, Custom.Factory>();
081
082    /**
083     * Register a custom index meta data factory. Make sure to call it from a static block.
084     */
085    public static void registerFactory(String type, Custom.Factory factory) {
086        customFactories.put(type, factory);
087    }
088
089    @Nullable
090    public static <T extends Custom> Custom.Factory<T> lookupFactory(String type) {
091        return customFactories.get(type);
092    }
093
094    public static <T extends Custom> Custom.Factory<T> lookupFactorySafe(String type) throws ElasticSearchIllegalArgumentException {
095        Custom.Factory<T> factory = customFactories.get(type);
096        if (factory == null) {
097            throw new ElasticSearchIllegalArgumentException("No custom state factory registered for type [" + type + "]");
098        }
099        return factory;
100    }

217    @Override
218    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
219        if (!params.paramAsBoolean("filter_nodes", false)) {
220            builder.field("master_node", nodes().masterNodeId());
221        }
222
223        // blocks
224        if (!params.paramAsBoolean("filter_blocks", false)) {
225            builder.startObject("blocks");
226
227            if (!blocks().global().isEmpty()) {
228                builder.startObject("global");
229                for (ClusterBlock block : blocks().global()) {
230                    block.toXContent(builder, params);
231                }
232                builder.endObject();
233            }
234
235            if (!blocks().indices().isEmpty()) {
236                builder.startObject("indices");
237                for (Map.Entry<String, ImmutableSet<ClusterBlock>> entry : blocks().indices().entrySet()) {
238                    builder.startObject(entry.getKey());
239                    for (ClusterBlock block : entry.getValue()) {
240                        block.toXContent(builder, params);
241                    }
242                    builder.endObject();
243                }
244                builder.endObject();
245            }
246
247            builder.endObject();
248        }
249
250        // nodes
251        if (!params.paramAsBoolean("filter_nodes", false)) {
252            builder.startObject("nodes");
253            for (DiscoveryNode node : nodes()) {
254                builder.startObject(node.id(), XContentBuilder.FieldCaseConversion.NONE);
255                builder.field("name", node.name());
256                builder.field("transport_address", node.address().toString());
257
258                builder.startObject("attributes");
259                for (Map.Entry<String, String> attr : node.attributes().entrySet()) {
260                    builder.field(attr.getKey(), attr.getValue());
261                }
262                builder.endObject();
263
264                builder.endObject();
265            }
266            builder.endObject();
267        }
268
269        // meta data
270        if (!params.paramAsBoolean("filter_metadata", false)) {
271            builder.startObject("metadata");
272
273            builder.startObject("templates");
274            for (IndexTemplateMetaData templateMetaData : metaData().templates().values()) {
275                builder.startObject(templateMetaData.name(), XContentBuilder.FieldCaseConversion.NONE);
276
277                builder.field("template", templateMetaData.template());
278                builder.field("order", templateMetaData.order());
279
280                builder.startObject("settings");
281                Settings settings = settingsFilter.filterSettings(templateMetaData.settings());
282                for (Map.Entry<String, String> entry : settings.getAsMap().entrySet()) {
283                    builder.field(entry.getKey(), entry.getValue());
284                }
285                builder.endObject();
286
287                builder.startObject("mappings");
288                for (Map.Entry<String, CompressedString> entry : templateMetaData.mappings().entrySet()) {
289                    byte[] mappingSource = entry.getValue().uncompressed();
290                    XContentParser parser = XContentFactory.xContent(mappingSource).createParser(mappingSource);
291                    Map<String, Object> mapping = parser.map();
292                    if (mapping.size() == 1 && mapping.containsKey(entry.getKey())) {
293                        // the type name is the root value, reduce it
294                        mapping = (Map<String, Object>) mapping.get(entry.getKey());
295                    }
296                    builder.field(entry.getKey());
297                    builder.map(mapping);
298                }
299                builder.endObject();
300
301
302                builder.endObject();
303            }
304            builder.endObject();
305
306            builder.startObject("indices");
307            for (IndexMetaData indexMetaData : metaData()) {
308                builder.startObject(indexMetaData.index(), XContentBuilder.FieldCaseConversion.NONE);
309
310                builder.field("state", indexMetaData.state().toString().toLowerCase(Locale.ENGLISH));
311
312                builder.startObject("settings");
313                Settings settings = settingsFilter.filterSettings(indexMetaData.settings());
314                for (Map.Entry<String, String> entry : settings.getAsMap().entrySet()) {
315                    builder.field(entry.getKey(), entry.getValue());
316                }
317                builder.endObject();
318
319                builder.startObject("mappings");
320                for (Map.Entry<String, MappingMetaData> entry : indexMetaData.mappings().entrySet()) {
321                    byte[] mappingSource = entry.getValue().source().uncompressed();
322                    XContentParser parser = XContentFactory.xContent(mappingSource).createParser(mappingSource);
323                    Map<String, Object> mapping = parser.map();
324                    if (mapping.size() == 1 && mapping.containsKey(entry.getKey())) {
325                        // the type name is the root value, reduce it
326                        mapping = (Map<String, Object>) mapping.get(entry.getKey());
327                    }
328                    builder.field(entry.getKey());
329                    builder.map(mapping);
330                }
331                builder.endObject();
332
333                builder.startArray("aliases");
334                for (String alias : indexMetaData.aliases().keySet()) {
335                    builder.value(alias);
336                }
337                builder.endArray();
338
339                builder.endObject();
340            }
341            builder.endObject();
342
343            builder.endObject();
344        }
345
346        // routing table
347        if (!params.paramAsBoolean("filter_routing_table", false)) {
348            builder.startObject("routing_table");
349            builder.startObject("indices");
350            for (IndexRoutingTable indexRoutingTable : routingTable()) {
351                builder.startObject(indexRoutingTable.index(), XContentBuilder.FieldCaseConversion.NONE);
352                builder.startObject("shards");
353                for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
354                    builder.startArray(Integer.toString(indexShardRoutingTable.shardId().id()));
355                    for (ShardRouting shardRouting : indexShardRoutingTable) {
356                        shardRouting.toXContent(builder, params);
357                    }
358                    builder.endArray();
359                }
360                builder.endObject();
361                builder.endObject();
362            }
363            builder.endObject();
364            builder.endObject();
365        }
366
367        // routing nodes
368        if (!params.paramAsBoolean("filter_routing_table", false)) {
369            builder.startObject("routing_nodes");
370            builder.startArray("unassigned");
371            for (ShardRouting shardRouting : readOnlyRoutingNodes().unassigned()) {
372                shardRouting.toXContent(builder, params);
373            }
374            builder.endArray();
375
376            builder.startObject("nodes");
377            for (RoutingNode routingNode : readOnlyRoutingNodes()) {
378                builder.startArray(routingNode.nodeId(), XContentBuilder.FieldCaseConversion.NONE);
379                for (ShardRouting shardRouting : routingNode) {
380                    shardRouting.toXContent(builder, params);
381                }
382                builder.endArray();
383            }
384            builder.endObject();
385
386            builder.endObject();
387        }
388
389        if (!params.paramAsBoolean("filter_routing_table", false)) {
390            builder.startArray("allocations");
391            for (Map.Entry<ShardId, List<AllocationExplanation.NodeExplanation>> entry : allocationExplanation().explanations().entrySet()) {
392                builder.startObject();
393                builder.field("index", entry.getKey().index().name());
394                builder.field("shard", entry.getKey().id());
395                builder.startArray("explanations");
396                for (AllocationExplanation.NodeExplanation nodeExplanation : entry.getValue()) {
397                    builder.field("desc", nodeExplanation.description());
398                    if (nodeExplanation.node() != null) {
399                        builder.startObject("node");
400                        builder.field("id", nodeExplanation.node().id());
401                        builder.field("name", nodeExplanation.node().name());
402                        builder.endObject();
403                    }
404                }
405                builder.endArray();
406                builder.endObject();
407            }
408            builder.endArray();
409        }
410
411        if (!params.paramAsBoolean("filter_customs", false)) {
412            for (Map.Entry<String, Custom> entry : customs().entrySet()) {
413                builder.startObject(entry.getKey());
414                lookupFactorySafe(entry.getKey()).toXContent(entry.getValue(), builder, params);
415                builder.endObject();
416            }
417        }
418
419        return builder;
420    }

527        public static byte[] toBytes(ClusterState state) throws IOException {
528            BasicCachedStreamOutput.Entry cachedEntry = BasicCachedStreamOutput.popEntry();
529            try {
530                BytesStreamOutput os = cachedEntry.bytes();
531                writeTo(state, os);
532                return os.bytes().copyBytesArray().toBytes();
533            } finally {
534                BasicCachedStreamOutput.pushEntry(cachedEntry);
535            }
536        }
537
538        public static ClusterState fromBytes(byte[] data, DiscoveryNode localNode) throws IOException {
539            return readFrom(new BytesStreamInput(data, false), localNode);
540        }

542        public static void writeTo(ClusterState state, StreamOutput out) throws IOException {
543            out.writeLong(state.version());
544            MetaData.Builder.writeTo(state.metaData(), out);
545            RoutingTable.Builder.writeTo(state.routingTable(), out);
546            DiscoveryNodes.Builder.writeTo(state.nodes(), out);
547            ClusterBlocks.Builder.writeClusterBlocks(state.blocks(), out);
548            state.allocationExplanation().writeTo(out);
549            out.writeVInt(state.customs().size());
550            for (Map.Entry<String, Custom> entry : state.customs().entrySet()) {
551                out.writeString(entry.getKey());
552                lookupFactorySafe(entry.getKey()).writeTo(entry.getValue(), out);
553            }
554        }
555
556        public static ClusterState readFrom(StreamInput in, @Nullable DiscoveryNode localNode) throws IOException {
557            Builder builder = new Builder();
558            builder.version = in.readLong();
559            builder.metaData = MetaData.Builder.readFrom(in);
560            builder.routingTable = RoutingTable.Builder.readFrom(in);
561            builder.nodes = DiscoveryNodes.Builder.readFrom(in, localNode);
562            builder.blocks = ClusterBlocks.Builder.readClusterBlocks(in);
563            builder.allocationExplanation = AllocationExplanation.readAllocationExplanation(in);
564            int customSize = in.readVInt();
565            for (int i = 0; i < customSize; i++) {
566                String type = in.readString();
567                Custom customIndexMetaData = lookupFactorySafe(type).readFrom(in);
568                builder.putCustom(type, customIndexMetaData);
569            }
570            return builder.build();
571        }
