020package org.elasticsearch.cluster;
021
022import com.google.common.collect.ImmutableList;
023import com.google.common.collect.Lists;
024import org.elasticsearch.cluster.metadata.IndexMetaData;
025import org.elasticsearch.cluster.metadata.MetaData;
026import org.elasticsearch.cluster.node.DiscoveryNodes;
027
028import java.util.List;
029
030/**
031 *
032 */
033public class ClusterChangedEvent {
034
035    private final String source;
036
037    private final ClusterState previousState;
038
039    private final ClusterState state;
040
041    private final DiscoveryNodes.Delta nodesDelta;
042
043    public ClusterChangedEvent(String source, ClusterState state, ClusterState previousState) {
044        this.source = source;
045        this.state = state;
046        this.previousState = previousState;
047        this.nodesDelta = state.nodes().delta(previousState.nodes());
048    }
049
050    /**
051     * The source that caused this cluster event to be raised.
052     */
053    public String source() {
054        return this.source;
055    }
056
057    public ClusterState state() {
058        return this.state;
059    }
060
061    public ClusterState previousState() {
062        return this.previousState;
063    }
064
065    public boolean routingTableChanged() {
066        return state.routingTable() != previousState.routingTable();
067    }
068
069    public boolean indexRoutingTableChanged(String index) {
070        if (!state.routingTable().hasIndex(index) && !previousState.routingTable().hasIndex(index)) {
071            return false;
072        }
073        if (state.routingTable().hasIndex(index) && previousState.routingTable().hasIndex(index)) {
074            return state.routingTable().index(index) != previousState.routingTable().index(index);
075        }
076        return true;
077    }
078
079    /**
080     * Returns the indices created in this event
081     */
082    public List<String> indicesCreated() {
083        if (previousState == null) {
084            return Lists.newArrayList(state.metaData().indices().keySet());
085        }
086        if (!metaDataChanged()) {
087            return ImmutableList.of();
088        }
089        List<String> created = null;
090        for (String index : state.metaData().indices().keySet()) {
091            if (!previousState.metaData().hasIndex(index)) {
092                if (created == null) {
093                    created = Lists.newArrayList();
094                }
095                created.add(index);
096            }
097        }
098        return created == null ? ImmutableList.<String>of() : created;
099    }
100
101    /**
102     * Returns the indices deleted in this event
103     */
104    public List<String> indicesDeleted() {
105        if (previousState == null) {
106            return ImmutableList.of();
107        }
108        if (!metaDataChanged()) {
109            return ImmutableList.of();
110        }
111        List<String> deleted = null;
112        for (String index : previousState.metaData().indices().keySet()) {
113            if (!state.metaData().hasIndex(index)) {
114                if (deleted == null) {
115                    deleted = Lists.newArrayList();
116                }
117                deleted.add(index);
118            }
119        }
120        return deleted == null ? ImmutableList.<String>of() : deleted;
121    }
122
123    public boolean metaDataChanged() {
124        return state.metaData() != previousState.metaData();
125    }
126
127    public boolean indexMetaDataChanged(IndexMetaData current) {
128        MetaData previousMetaData = previousState.metaData();
129        if (previousMetaData == null) {
130            return true;
131        }
132        IndexMetaData previousIndexMetaData = previousMetaData.index(current.index());
133        // no need to check on version, since disco modules will make sure to use the
134        // same instance if its a version match
135        if (previousIndexMetaData == current) {
136            return false;
137        }
138        return true;
139    }
140
141    public boolean blocksChanged() {
142        return state.blocks() != previousState.blocks();
143    }
144
145    public boolean localNodeMaster() {
146        return state.nodes().localNodeMaster();
147    }
148
149    public DiscoveryNodes.Delta nodesDelta() {
150        return this.nodesDelta;
151    }
152
153    public boolean nodesRemoved() {
154        return nodesDelta.removed();
155    }
156
157    public boolean nodesAdded() {
158        return nodesDelta.added();
159    }
160
161    public boolean nodesChanged() {
162        return nodesRemoved() || nodesAdded();
163    }
164}
