160    public final int id;
161    public final byte major;
162    public final byte minor;
163    public final byte revision;
164    public final byte build;
165    public final Boolean snapshot;
166
167    Version(int id, @Nullable Boolean snapshot) {
168        this.id = id;
169        this.major = (byte) ((id / 1000000) % 100);
170        this.minor = (byte) ((id / 10000) % 100);
171        this.revision = (byte) ((id / 100) % 100);
172        this.build = (byte) (id % 100);
173        this.snapshot = snapshot;
174    }
175
176    public boolean snapshot() {
177        return snapshot != null && snapshot;
178    }
179
180    public boolean after(Version version) {
181        return version.id > id;
182    }
183
184    public boolean onOrAfter(Version version) {
185        return version.id >= id;
186    }
187
188    public boolean before(Version version) {
189        return version.id < id;
190    }
191
192    public boolean onOrBefore(Version version) {
193        return version.id <= id;
194    }
195
196    /**
197     * Just the version number (without -SNAPSHOT if snapshot).
198     */
199    public String number() {
200        StringBuilder sb = new StringBuilder();
201        sb.append(major).append('.').append(minor).append('.').append(revision);
202        if (build < 50) {
203            sb.append(".Beta").append(build);
204        } else if (build < 99) {
205            sb.append(".RC").append(build - 50);
206        }
207        return sb.toString();
208    }
209
210    @Override
211    public String toString() {
212        StringBuilder sb = new StringBuilder();
213        sb.append(number());
214        if (snapshot()) {
215            sb.append("-SNAPSHOT");
216        }
217        return sb.toString();
218    }

220    @Override
221    public boolean equals(Object o) {
222        if (this == o) return true;
223        if (o == null || getClass() != o.getClass()) return false;
224
225        Version version = (Version) o;
226
227        if (id != version.id) return false;
228
229        return true;
230    }
