020package org.elasticsearch.rest;
021
022import org.elasticsearch.common.io.stream.StreamInput;
023import org.elasticsearch.common.io.stream.StreamOutput;
024
025import java.io.IOException;
026
027public enum RestStatus {
028    /**
029     * The client SHOULD continue with its request. This interim response is used to inform the client that the
030     * initial part of the request has been received and has not yet been rejected by the server. The client
031     * SHOULD continue by sending the remainder of the request or, if the request has already been completed,
032     * ignore this response. The server MUST send a final response after the request has been completed.
033     */
034    CONTINUE(100),
035    /**
036     * The server understands and is willing to comply with the client's request, via the Upgrade message header field
037     * (section 14.42), for a change in the application protocol being used on this connection. The server will
038     * switch protocols to those defined by the response's Upgrade header field immediately after the empty line
039     * which terminates the 101 response.
040     */
041    SWITCHING_PROTOCOLS(101),
042    /**
043     * The request has succeeded. The information returned with the response is dependent on the method
044     * used in the request, for example:
045     * <ul>
046     * <li>GET: an entity corresponding to the requested resource is sent in the response;</li>
047     * <li>HEAD: the entity-header fields corresponding to the requested resource are sent in the response without any message-body;</li>
048     * <li>POST: an entity describing or containing the result of the action;</li>
049     * <li>TRACE: an entity containing the request message as received by the end server.</li>
050     * </ul>
051     */
052    OK(200),
053    /**
054     * The request has been fulfilled and resulted in a new resource being created. The newly created resource can
055     * be referenced by the URI(s) returned in the entity of the response, with the most specific URI for the
056     * resource given by a Location header field. The response SHOULD include an entity containing a list of resource
057     * characteristics and location(s) from which the user or user agent can choose the one most appropriate. The
058     * entity format is specified by the media type given in the Content-Type header field. The origin server MUST
059     * create the resource before returning the 201 status code. If the action cannot be carried out immediately, the
060     * server SHOULD respond with 202 (Accepted) response instead.
061     * <p/>
062     * <p>A 201 response MAY contain an ETag response header field indicating the current value of the entity tag
063     * for the requested variant just created, see section 14.19.
064     */
065    CREATED(201),
066    /**
067     * The request has been accepted for processing, but the processing has not been completed.  The request might
068     * or might not eventually be acted upon, as it might be disallowed when processing actually takes place. There
069     * is no facility for re-sending a status code from an asynchronous operation such as this.
070     * <p/>
071     * <p>The 202 response is intentionally non-committal. Its purpose is to allow a server to accept a request for
072     * some other process (perhaps a batch-oriented process that is only run once per day) without requiring that
073     * the user agent's connection to the server persist until the process is completed. The entity returned with
074     * this response SHOULD include an indication of the request's current status and either a pointer to a status
075     * monitor or some estimate of when the user can expect the request to be fulfilled.
076     */
077    ACCEPTED(202),
078    /**
079     * The returned meta information in the entity-header is not the definitive set as available from the origin
080     * server, but is gathered from a local or a third-party copy. The set presented MAY be a subset or super set
081     * of the original version. For example, including local annotation information about the resource might
082     * result in a super set of the meta information known by the origin server. Use of this response code
083     * is not required and is only appropriate when the response would otherwise be 200 (OK).
084     */
085    NON_AUTHORITATIVE_INFORMATION(203),
086    /**
087     * The server has fulfilled the request but does not need to return an entity-body, and might want to return
088     * updated meta information. The response MAY include new or updated meta information in the form of
089     * entity-headers, which if present SHOULD be associated with the requested variant.
090     * <p/>
091     * <p>If the client is a user agent, it SHOULD NOT change its document view from that which caused the request
092     * to be sent. This response is primarily intended to allow input for actions to take place without causing a
093     * change to the user agent's active document view, although any new or updated meta information SHOULD be
094     * applied to the document currently in the user agent's active view.
095     * <p/>
096     * <p>The 204 response MUST NOT include a message-body, and thus is always terminated by the first empty
097     * line after the header fields.
098     */
099    NO_CONTENT(204),
100    /**
101     * The server has fulfilled the request and the user agent SHOULD reset the document view which caused the
102     * request to be sent. This response is primarily intended to allow input for actions to take place via user
103     * input, followed by a clearing of the form in which the input is given so that the user can easily initiate
104     * another input action. The response MUST NOT include an entity.
105     */
106    RESET_CONTENT(205),
107    /**
108     * The server has fulfilled the partial GET request for the resource. The request MUST have included a Range
109     * header field (section 14.35) indicating the desired range, and MAY have included an If-Range header
110     * field (section 14.27) to make the request conditional.
111     * <p/>
112     * <p>The response MUST include the following header fields:
113     * <ul>
114     * <li>Either a Content-Range header field (section 14.16) indicating the range included with this response,
115     * or a multipart/byteranges Content-Type including Content-Range fields for each part. If a Content-Length
116     * header field is present in the response, its value MUST match the actual number of OCTETs transmitted in
117     * the message-body.</li>
118     * <li>Date</li>
119     * <li>ETag and/or Content-Location, if the header would have been sent in a 200 response to the same request</li>
120     * <li>Expires, Cache-Control, and/or Vary, if the field-value might differ from that sent in any previous
121     * response for the same variant</li>
122     * </ul>
123     * <p/>
124     * <p>If the 206 response is the result of an If-Range request that used a strong cache validator
125     * (see section 13.3.3), the response SHOULD NOT include other entity-headers. If the response is the result
126     * of an If-Range request that used a weak validator, the response MUST NOT include other entity-headers;
127     * this prevents inconsistencies between cached entity-bodies and updated headers. Otherwise, the response MUST
128     * include all of the entity-headers that would have been returned with a 200 (OK) response to the same request.
129     * <p/>
130     * <p>A cache MUST NOT combine a 206 response with other previously cached content if the ETag or Last-Modified
131     * headers do not match exactly, see 13.5.4.
132     * <p/>
133     * <p>A cache that does not support the Range and Content-Range headers MUST NOT cache 206 (Partial) responses.
134     */
135    PARTIAL_CONTENT(206),
136    /**
137     * The 207 (Multi-Status) status code provides status for multiple independent operations (see Section 13 for
138     * more information).
139     * <p/>
140     * <p>A Multi-Status response conveys information about multiple resources in situations where multiple status
141     * codes might be appropriate. The default Multi-Status response body is a text/xml or application/xml HTTP
142     * entity with a 'multistatus' root element. Further elements contain 200, 300, 400, and 500 series status codes
143     * generated during the method invocation. 100 series status codes SHOULD NOT be recorded in a 'response'
144     * XML element.
145     * <p/>
146     * <p>Although '207' is used as the overall response status code, the recipient needs to consult the contents
147     * of the multistatus response body for further information about the success or failure of the method execution.
148     * The response MAY be used in success, partial success and also in failure situations.
149     * <p/>
150     * <p>The 'multistatus' root element holds zero or more 'response' elements in any order, each with
151     * information about an individual resource. Each 'response' element MUST have an 'href' element
152     * to identify the resource.
153     */
154    MULTI_STATUS(207),
155    /**
156     * The requested resource corresponds to any one of a set of representations, each with its own specific
157     * location, and agent-driven negotiation information (section 12) is being provided so that the user (or user
158     * agent) can select a preferred representation and redirect its request to that location.
159     * <p/>
160     * <p>Unless it was a HEAD request, the response SHOULD include an entity containing a list of resource
161     * characteristics and location(s) from which the user or user agent can choose the one most appropriate.
162     * The entity format is specified by the media type given in the Content-Type header field. Depending upon the
163     * format and the capabilities of the user agent, selection of the most appropriate choice MAY be performed
164     * automatically. However, this specification does not define any standard for such automatic selection.
165     * <p/>
166     * <p>If the server has a preferred choice of representation, it SHOULD include the specific URI for that
167     * representation in the Location field; user agents MAY use the Location field value for automatic redirection.
168     * This response is cacheable unless indicated otherwise.
169     */
170    MULTIPLE_CHOICES(300),
171    /**
172     * The requested resource has been assigned a new permanent URI and any future references to this resource
173     * SHOULD use one of the returned URIs.  Clients with link editing capabilities ought to automatically re-link
174     * references to the Request-URI to one or more of the new references returned by the server, where possible.
175     * This response is cacheable unless indicated otherwise.
176     * <p/>
177     * <p>The new permanent URI SHOULD be given by the Location field in the response. Unless the request method
178     * was HEAD, the entity of the response SHOULD contain a short hypertext note with a hyperlink to the new URI(s).
179     * <p/>
180     * <p>If the 301 status code is received in response to a request other than GET or HEAD, the user agent
181     * MUST NOT automatically redirect the request unless it can be confirmed by the user, since this might change
182     * the conditions under which the request was issued.
183     */
184    MOVED_PERMANENTLY(301),
185    /**
186     * The requested resource resides temporarily under a different URI. Since the redirection might be altered on
187     * occasion, the client SHOULD continue to use the Request-URI for future requests.  This response is only
188     * cacheable if indicated by a Cache-Control or Expires header field.
189     * <p/>
190     * <p>The temporary URI SHOULD be given by the Location field in the response. Unless the request method was
191     * HEAD, the entity of the response SHOULD contain a short hypertext note with a hyperlink to the new URI(s).
192     * <p/>
193     * <p>If the 302 status code is received in response to a request other than GET or HEAD, the user agent
194     * MUST NOT automatically redirect the request unless it can be confirmed by the user, since this might change
195     * the conditions under which the request was issued.
196     */
197    FOUND(302),
198    /**
199     * The response to the request can be found under a different URI and SHOULD be retrieved using a GET method on
200     * that resource. This method exists primarily to allow the output of a POST-activated script to redirect the
201     * user agent to a selected resource. The new URI is not a substitute reference for the originally requested
202     * resource. The 303 response MUST NOT be cached, but the response to the second (redirected) request might be
203     * cacheable.
204     * <p/>
205     * <p>The different URI SHOULD be given by the Location field in the response. Unless the request method was
206     * HEAD, the entity of the response SHOULD contain a short hypertext note with a hyperlink to the new URI(s).
207     */
208    SEE_OTHER(303),
209    /**
210     * If the client has performed a conditional GET request and access is allowed, but the document has not been
211     * modified, the server SHOULD respond with this status code. The 304 response MUST NOT contain a message-body,
212     * and thus is always terminated by the first empty line after the header fields.
213     * <p/>
214     * <p>The response MUST include the following header fields:
215     * <ul>
216     * <li>Date, unless its omission is required by section 14.18.1
217     * If a clockless origin server obeys these rules, and proxies and clients add their own Date to any
218     * response received without one (as already specified by [RFC 2068], section 14.19), caches will operate
219     * correctly.
220     * </li>
221     * <li>ETag and/or Content-Location, if the header would have been sent in a 200 response to the same request</li>
222     * <li>Expires, Cache-Control, and/or Vary, if the field-value might differ from that sent in any previous
223     * response for the same variant</li>
224     * </ul>
225     * <p/>
226     * <p>If the conditional GET used a strong cache validator (see section 13.3.3), the response SHOULD NOT include
227     * other entity-headers. Otherwise (i.e., the conditional GET used a weak validator), the response MUST NOT
228     * include other entity-headers; this prevents inconsistencies between cached entity-bodies and updated headers.
229     * <p/>
230     * <p>If a 304 response indicates an entity not currently cached, then the cache MUST disregard the response
231     * and repeat the request without the conditional.
232     * <p/>
233     * <p>If a cache uses a received 304 response to update a cache entry, the cache MUST update the entry to
234     * reflect any new field values given in the response.
235     */
236    NOT_MODIFIED(304),
237    /**
238     * The requested resource MUST be accessed through the proxy given by the Location field. The Location field
239     * gives the URI of the proxy. The recipient is expected to repeat this single request via the proxy.
240     * 305 responses MUST only be generated by origin servers.
241     */
242    USE_PROXY(305),
243    /**
244     * The requested resource resides temporarily under a different URI. Since the redirection MAY be altered on
245     * occasion, the client SHOULD continue to use the Request-URI for future requests.  This response is only
246     * cacheable if indicated by a Cache-Control or Expires header field.
247     * <p/>
248     * <p>The temporary URI SHOULD be given by the Location field in the response. Unless the request method was
249     * HEAD, the entity of the response SHOULD contain a short hypertext note with a hyperlink to the new URI(s) ,
250     * since many pre-HTTP/1.1 user agents do not understand the 307 status. Therefore, the note SHOULD contain
251     * the information necessary for a user to repeat the original request on the new URI.
252     * <p/>
253     * <p>If the 307 status code is received in response to a request other than GET or HEAD, the user agent MUST NOT
254     * automatically redirect the request unless it can be confirmed by the user, since this might change the
255     * conditions under which the request was issued.
256     */
257    TEMPORARY_REDIRECT(307),
258    /**
259     * The request could not be understood by the server due to malformed syntax. The client SHOULD NOT repeat the
260     * request without modifications.
261     */
262    BAD_REQUEST(400),
263    /**
264     * The request requires user authentication. The response MUST include a WWW-Authenticate header field
265     * (section 14.47) containing a challenge applicable to the requested resource. The client MAY repeat the request
266     * with a suitable Authorization header field (section 14.8). If the request already included Authorization
267     * credentials, then the 401 response indicates that authorization has been refused for those credentials.
268     * If the 401 response contains the same challenge as the prior response, and the user agent has already attempted
269     * authentication at least once, then the user SHOULD be presented the entity that was given in the response,
270     * since that entity might include relevant diagnostic information. HTTP access authentication is explained in
271     * "HTTP Authentication: Basic and Digest Access Authentication" [43].
272     */
273    UNAUTHORIZED(401),
274    /**
275     * This code is reserved for future use.
276     */
277    PAYMENT_REQUIRED(402),
278    /**
279     * The server understood the request, but is refusing to fulfill it. Authorization will not help and the request
280     * SHOULD NOT be repeated. If the request method was not HEAD and the server wishes to make public why the
281     * request has not been fulfilled, it SHOULD describe the reason for the refusal in the entity.  If the server
282     * does not wish to make this information available to the client, the status code 404 (Not Found) can be used
283     * instead.
284     */
285    FORBIDDEN(403),
286    /**
287     * The server has not found anything matching the Request-URI. No indication is given of whether the condition
288     * is temporary or permanent. The 410 (Gone) status code SHOULD be used if the server knows, through some
289     * internally configurable mechanism, that an old resource is permanently unavailable and has no forwarding
290     * address. This status code is commonly used when the server does not wish to reveal exactly why the request
291     * has been refused, or when no other response is applicable.
292     */
293    NOT_FOUND(404),
294    /**
295     * The method specified in the Request-Line is not allowed for the resource identified by the Request-URI.
296     * The response MUST include an Allow header containing a list of valid methods for the requested resource.
297     */
298    METHOD_NOT_ALLOWED(405),
299    /**
300     * The resource identified by the request is only capable of generating response entities which have content
301     * characteristics not acceptable according to the accept headers sent in the request.
302     * <p/>
303     * <p>Unless it was a HEAD request, the response SHOULD include an entity containing a list of available entity
304     * characteristics and location(s) from which the user or user agent can choose the one most appropriate.
305     * The entity format is specified by the media type given in the Content-Type header field. Depending upon the
306     * format and the capabilities of the user agent, selection of the most appropriate choice MAY be performed
307     * automatically. However, this specification does not define any standard for such automatic selection.
308     * <p/>
309     * <p>Note: HTTP/1.1 servers are allowed to return responses which are not acceptable according to the accept
310     * headers sent in the request. In some cases, this may even be preferable to sending a 406 response. User
311     * agents are encouraged to inspect the headers of an incoming response to determine if it is acceptable.
312     * <p/>
313     * <p>If the response could be unacceptable, a user agent SHOULD temporarily stop receipt of more data and query
314     * the user for a decision on further actions.
315     */
316    NOT_ACCEPTABLE(406),
317    /**
318     * This code is similar to 401 (Unauthorized), but indicates that the client must first authenticate itself with
319     * the proxy. The proxy MUST return a Proxy-Authenticate header field (section 14.33) containing a challenge
320     * applicable to the proxy for the requested resource. The client MAY repeat the request with a suitable
321     * Proxy-Authorization header field (section 14.34). HTTP access authentication is explained in
322     * "HTTP Authentication: Basic and Digest Access Authentication" [43].
323     */
324    PROXY_AUTHENTICATION(407),
325    /**
326     * The client did not produce a request within the time that the server was prepared to wait. The client MAY
327     * repeat the request without modifications at any later time.
328     */
329    REQUEST_TIMEOUT(408),
330    /**
331     * The request could not be completed due to a conflict with the current state of the resource. This code is
332     * only allowed in situations where it is expected that the user might be able to resolve the conflict and
333     * resubmit the request. The response body SHOULD include enough information for the user to recognize the
334     * source of the conflict. Ideally, the response entity would include enough information for the user or user
335     * agent to fix the problem; however, that might not be possible and is not required.
336     * <p/>
337     * <p>Conflicts are most likely to occur in response to a PUT request. For example, if versioning were being
338     * used and the entity being PUT included changes to a resource which conflict with those made by an earlier
339     * (third-party) request, the server might use the 409 response to indicate that it can't complete the request.
340     * In this case, the response entity would likely contain a list of the differences between the two versions in
341     * a format defined by the response Content-Type.
342     */
343    CONFLICT(409),
344    /**
345     * The requested resource is no longer available at the server and no forwarding address is known. This condition
346     * is expected to be considered permanent. Clients with link editing capabilities SHOULD delete references to
347     * the Request-URI after user approval. If the server does not know, or has no facility to determine, whether or
348     * not the condition is permanent, the status code 404 (Not Found) SHOULD be used instead. This response is
349     * cacheable unless indicated otherwise.
350     * <p/>
351     * <p>The 410 response is primarily intended to assist the task of web maintenance by notifying the recipient
352     * that the resource is intentionally unavailable and that the server owners desire that remote links to that
353     * resource be removed. Such an event is common for limited-time, promotional services and for resources belonging
354     * to individuals no longer working at the server's site. It is not necessary to mark all permanently unavailable
355     * resources as "gone" or to keep the mark for any length of time -- that is left to the discretion of the server
356     * owner.
357     */
358    GONE(410),
359    /**
360     * The server refuses to accept the request without a defined Content-Length. The client MAY repeat the request
361     * if it adds a valid Content-Length header field containing the length of the message-body in the request message.
362     */
363    LENGTH_REQUIRED(411),
364    /**
365     * The precondition given in one or more of the request-header fields evaluated to false when it was tested on
366     * the server. This response code allows the client to place preconditions on the current resource metainformation
367     * (header field data) and thus prevent the requested method from being applied to a resource other than the one
368     * intended.
369     */
370    PRECONDITION_FAILED(412),
371    /**
372     * The server is refusing to process a request because the request entity is larger than the server is willing
373     * or able to process. The server MAY close the connection to prevent the client from continuing the request.
374     * <p/>
375     * <p>If the condition is temporary, the server SHOULD include a Retry-After header field to indicate that it
376     * is temporary and after what time the client MAY try again.
377     */
378    REQUEST_ENTITY_TOO_LARGE(413),
379    /**
380     * The server is refusing to service the request because the Request-URI is longer than the server is willing
381     * to interpret. This rare condition is only likely to occur when a client has improperly converted a POST
382     * request to a GET request with long query information, when the client has descended into a URI "black hole"
383     * of redirection (e.g., a redirected URI prefix that points to a suffix of itself), or when the server is
384     * under attack by a client attempting to exploit security holes present in some servers using fixed-length
385     * buffers for reading or manipulating the Request-URI.
386     */
387    REQUEST_URI_TOO_LONG(414),
388    /**
389     * The server is refusing to service the request because the entity of the request is in a format not supported
390     * by the requested resource for the requested method.
391     */
392    UNSUPPORTED_MEDIA_TYPE(415),
393    /**
394     * A server SHOULD return a response with this status code if a request included a Range request-header field
395     * (section 14.35), and none of the range-specifier values in this field overlap the current extent of the
396     * selected resource, and the request did not include an If-Range request-header field. (For byte-ranges, this
397     * means that the first-byte-pos of all of the byte-range-spec values were greater than the current length of
398     * the selected resource.)
399     * <p/>
400     * <p>When this status code is returned for a byte-range request, the response SHOULD include a Content-Range
401     * entity-header field specifying the current length of the selected resource (see section 14.16). This
402     * response MUST NOT use the multipart/byteranges content-type.
403     */
404    REQUESTED_RANGE_NOT_SATISFIED(416),
405    /**
406     * The expectation given in an Expect request-header field (see section 14.20) could not be met by this server,
407     * or, if the server is a proxy, the server has unambiguous evidence that the request could not be met by the
408     * next-hop server.
409     */
410    EXPECTATION_FAILED(417),
411    /**
412     * The 422 (Unprocessable Entity) status code means the server understands the content type of the request
413     * entity (hence a 415(Unsupported Media Type) status code is inappropriate), and the syntax of the request
414     * entity is correct (thus a 400 (Bad Request) status code is inappropriate) but was unable to process the
415     * contained instructions. For example, this error condition may occur if an XML request body contains
416     * well-formed (i.e., syntactically correct), but semantically erroneous, XML instructions.
417     */
418    UNPROCESSABLE_ENTITY(422),
419    /**
420     * The 423 (Locked) status code means the source or destination resource of a method is locked. This response
421     * SHOULD contain an appropriate precondition or postcondition code, such as 'lock-token-submitted' or
422     * 'no-conflicting-lock'.
423     */
424    LOCKED(423),
425    /**
426     * The 424 (Failed Dependency) status code means that the method could not be performed on the resource because
427     * the requested action depended on another action and that action failed. For example, if a command in a
428     * PROPPATCH method fails, then, at minimum, the rest of the commands will also fail with 424 (Failed Dependency).
429     */
430    FAILED_DEPENDENCY(424),
431    /**
432     * The server encountered an unexpected condition which prevented it from fulfilling the request.
433     */
434    INTERNAL_SERVER_ERROR(500),
435    /**
436     * The server does not support the functionality required to fulfill the request. This is the appropriate
437     * response when the server does not recognize the request method and is not capable of supporting it for any
438     * resource.
439     */
440    NOT_IMPLEMENTED(501),
441    /**
442     * The server, while acting as a gateway or proxy, received an invalid response from the upstream server it
443     * accessed in attempting to fulfill the request.
444     */
445    BAD_GATEWAY(502),
446    /**
447     * The server is currently unable to handle the request due to a temporary overloading or maintenance of the
448     * server. The implication is that this is a temporary condition which will be alleviated after some delay.
449     * If known, the length of the delay MAY be indicated in a Retry-After header. If no Retry-After is given,
450     * the client SHOULD handle the response as it would for a 500 response.
451     */
452    SERVICE_UNAVAILABLE(503),
453    /**
454     * The server, while acting as a gateway or proxy, did not receive a timely response from the upstream server
455     * specified by the URI (e.g. HTTP, FTP, LDAP) or some other auxiliary server (e.g. DNS) it needed to access
456     * in attempting to complete the request.
457     */
458    GATEWAY_TIMEOUT(504),
459    /**
460     * The server does not support, or refuses to support, the HTTP protocol version that was used in the request
461     * message. The server is indicating that it is unable or unwilling to complete the request using the same major
462     * version as the client, as described in section 3.1, other than with this error message. The response SHOULD
463     * contain an entity describing why that version is not supported and what other protocols are supported by
464     * that server.
465     */
466    HTTP_VERSION_NOT_SUPPORTED(505),
467    /**
468     * The 507 (Insufficient Storage) status code means the method could not be performed on the resource because
469     * the server is unable to store the representation needed to successfully complete the request. This condition
470     * is considered to be temporary. If the request that received this status code was the result of a user action,
471     * the request MUST NOT be repeated until it is requested by a separate user action.
472     */
473    INSUFFICIENT_STORAGE(506);
474
475
476    private int status;
477
478    RestStatus(int status) {
479        this.status = (short) status;
480    }
481
482    public int getStatus() {
483        return status;
484    }
485
486    public static RestStatus readFrom(StreamInput in) throws IOException {
487        return RestStatus.valueOf(in.readUTF());
