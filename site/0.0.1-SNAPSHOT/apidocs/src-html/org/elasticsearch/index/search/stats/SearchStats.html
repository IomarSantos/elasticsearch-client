020package org.elasticsearch.index.search.stats;

022import org.elasticsearch.common.Nullable;
023import org.elasticsearch.common.io.stream.StreamInput;

035/**
036 */
037public class SearchStats implements Streamable, ToXContent {
038
039    public static class Stats implements Streamable, ToXContent {
040
041        private long queryCount;
042        private long queryTimeInMillis;
043        private long queryCurrent;
044
045        private long fetchCount;
046        private long fetchTimeInMillis;
047        private long fetchCurrent;
048
049        Stats() {
050
051        }
052
053        public Stats(long queryCount, long queryTimeInMillis, long queryCurrent, long fetchCount, long fetchTimeInMillis, long fetchCurrent) {
054            this.queryCount = queryCount;
055            this.queryTimeInMillis = queryTimeInMillis;
056            this.queryCurrent = queryCurrent;
057            this.fetchCount = fetchCount;
058            this.fetchTimeInMillis = fetchTimeInMillis;
059            this.fetchCurrent = fetchCurrent;
060        }
061
062        public void add(Stats stats) {
063            queryCount += stats.queryCount;
064            queryTimeInMillis += stats.queryTimeInMillis;
065            queryCurrent += stats.queryCurrent;
066
067            fetchCount += stats.fetchCount;
068            fetchTimeInMillis += stats.fetchTimeInMillis;
069            fetchCurrent += stats.fetchCurrent;
070        }
071
072        public long queryCount() {
073            return queryCount;
074        }
075
076        public long getQueryCount() {
077            return queryCount;
078        }
079
080        public TimeValue queryTime() {
081            return new TimeValue(queryTimeInMillis);
082        }
083
084        public long queryTimeInMillis() {
085            return queryTimeInMillis;
086        }
087
088        public long getQueryTimeInMillis() {
089            return queryTimeInMillis;
090        }
091
092        public long queryCurrent() {
093            return queryCurrent;
094        }
095
096        public long getQueryCurrent() {
097            return queryCurrent;
098        }
099
100        public long fetchCount() {
101            return fetchCount;
102        }
103
104        public long getFetchCount() {
105            return fetchCount;
106        }
107
108        public TimeValue fetchTime() {
109            return new TimeValue(fetchTimeInMillis);
110        }
111
112        public long fetchTimeInMillis() {
113            return fetchTimeInMillis;
114        }
115
116        public long getFetchTimeInMillis() {
117            return fetchTimeInMillis;
118        }
119
120        public long fetchCurrent() {
121            return fetchCurrent;
122        }
123
124        public long getFetchCurrent() {
125            return fetchCurrent;
126        }
127
128
129        public static Stats readStats(StreamInput in) throws IOException {
130            Stats stats = new Stats();
131            stats.readFrom(in);
132            return stats;
133        }
134
135        @Override
136        public void readFrom(StreamInput in) throws IOException {
137            queryCount = in.readVLong();
138            queryTimeInMillis = in.readVLong();
139            queryCurrent = in.readVLong();
140
141            fetchCount = in.readVLong();
142            fetchTimeInMillis = in.readVLong();
143            fetchCurrent = in.readVLong();
144        }
145
146        @Override
147        public void writeTo(StreamOutput out) throws IOException {
148            out.writeVLong(queryCount);
149            out.writeVLong(queryTimeInMillis);
150            out.writeVLong(queryCurrent);
151
152            out.writeVLong(fetchCount);
153            out.writeVLong(fetchTimeInMillis);
154            out.writeVLong(fetchCurrent);
155        }
156
157        @Override
158        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
159            builder.field(Fields.QUERY_TOTAL, queryCount);
160            builder.field(Fields.QUERY_TIME, queryTime().toString());
161            builder.field(Fields.QUERY_TIME_IN_MILLIS, queryTimeInMillis);
162            builder.field(Fields.QUERY_CURRENT, queryCurrent);
163
164            builder.field(Fields.FETCH_TOTAL, fetchCount);
165            builder.field(Fields.FETCH_TIME, fetchTime().toString());
166            builder.field(Fields.FETCH_TIME_IN_MILLIS, fetchTimeInMillis);
167            builder.field(Fields.FETCH_CURRENT, fetchCurrent);
168
169            return builder;
170        }
171    }
172
173    private Stats totalStats;
174
175    @Nullable
176    Map<String, Stats> groupStats;
177
178    public SearchStats() {
179        totalStats = new Stats();
180    }
181
182    public SearchStats(Stats totalStats, @Nullable Map<String, Stats> groupStats) {
183        this.totalStats = totalStats;
184        this.groupStats = groupStats;
185    }
186
187    public void add(SearchStats searchStats) {
188        add(searchStats, true);
189    }
190
191    public void add(SearchStats searchStats, boolean includeTypes) {
192        if (searchStats == null) {
193            return;
194        }
195        totalStats.add(searchStats.totalStats);
196        if (includeTypes && searchStats.groupStats != null && !searchStats.groupStats.isEmpty()) {
197            if (groupStats == null) {
198                groupStats = new HashMap<String, Stats>(searchStats.groupStats.size());
199            }
200            for (Map.Entry<String, Stats> entry : searchStats.groupStats.entrySet()) {
201                Stats stats = groupStats.get(entry.getKey());
202                if (stats == null) {
203                    groupStats.put(entry.getKey(), entry.getValue());
204                } else {
205                    stats.add(entry.getValue());
206                }
207            }
208        }
209    }
210
211    public Stats total() {
212        return this.totalStats;
213    }
214
215    @Nullable
216    public Map<String, Stats> groupStats() {
217        return this.groupStats;
218    }
219
220    @Override
221    public XContentBuilder toXContent(XContentBuilder builder, ToXContent.Params params) throws IOException {
222        builder.startObject(Fields.SEARCH);
223        totalStats.toXContent(builder, params);
224        if (groupStats != null && !groupStats.isEmpty()) {
225            builder.startObject(Fields.GROUPS);
226            for (Map.Entry<String, Stats> entry : groupStats.entrySet()) {
227                builder.startObject(entry.getKey(), XContentBuilder.FieldCaseConversion.NONE);
228                entry.getValue().toXContent(builder, params);
229                builder.endObject();
230            }
231            builder.endObject();
232        }
233        builder.endObject();
234        return builder;
235    }
236
237    static final class Fields {
238        static final XContentBuilderString SEARCH = new XContentBuilderString("search");
239        static final XContentBuilderString GROUPS = new XContentBuilderString("groups");
240        static final XContentBuilderString QUERY_TOTAL = new XContentBuilderString("query_total");
241        static final XContentBuilderString QUERY_TIME = new XContentBuilderString("query_time");
242        static final XContentBuilderString QUERY_TIME_IN_MILLIS = new XContentBuilderString("query_time_in_millis");
243        static final XContentBuilderString QUERY_CURRENT = new XContentBuilderString("query_current");
244        static final XContentBuilderString FETCH_TOTAL = new XContentBuilderString("fetch_total");
245        static final XContentBuilderString FETCH_TIME = new XContentBuilderString("fetch_time");
246        static final XContentBuilderString FETCH_TIME_IN_MILLIS = new XContentBuilderString("fetch_time_in_millis");
247        static final XContentBuilderString FETCH_CURRENT = new XContentBuilderString("fetch_current");
248    }
249
250    public static SearchStats readSearchStats(StreamInput in) throws IOException {
251        SearchStats searchStats = new SearchStats();
252        searchStats.readFrom(in);
253        return searchStats;
254    }
255
256    @Override
257    public void readFrom(StreamInput in) throws IOException {
258        totalStats = Stats.readStats(in);
259        if (in.readBoolean()) {
260            int size = in.readVInt();
261            groupStats = new HashMap<String, Stats>(size);
262            for (int i = 0; i < size; i++) {
263                groupStats.put(in.readUTF(), Stats.readStats(in));
264            }
265        }
266    }
267
268    @Override
269    public void writeTo(StreamOutput out) throws IOException {
270        totalStats.writeTo(out);
271        if (groupStats == null || groupStats.isEmpty()) {
272            out.writeBoolean(false);
273        } else {
274            out.writeBoolean(true);
275            out.writeVInt(groupStats.size());
276            for (Map.Entry<String, Stats> entry : groupStats.entrySet()) {
277                out.writeUTF(entry.getKey());
278                entry.getValue().writeTo(out);
279            }
280        }
281    }
