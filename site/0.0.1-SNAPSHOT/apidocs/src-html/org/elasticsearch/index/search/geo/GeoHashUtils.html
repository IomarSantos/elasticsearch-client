020import gnu.trove.map.hash.TIntIntHashMap;

022/**
023 * Utilities for encoding and decoding geohashes. Based on
024 * http://en.wikipedia.org/wiki/Geohash.
025 */
026// LUCENE MONITOR: monitor against spatial package
027// replaced with native DECODE_MAP
028public class GeoHashUtils {
029
030    private static final char[] BASE_32 = {'0', '1', '2', '3', '4', '5', '6',
031            '7', '8', '9', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'j', 'k', 'm', 'n',
032            'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'};
033
034//    private final static Map<Character, Integer> DECODE_MAP = new HashMap<Character, Integer>();
035
036    private final static TIntIntHashMap DECODE_MAP = new TIntIntHashMap();
037
038    public static final int PRECISION = 12;
039    private static final int[] BITS = {16, 8, 4, 2, 1};
040
041    static {
042        for (int i = 0; i < BASE_32.length; i++) {
043            DECODE_MAP.put(BASE_32[i], i);
044        }
045    }
046
047    private GeoHashUtils() {
048    }
049
050    public static String encode(double latitude, double longitude) {
051        return encode(latitude, longitude, PRECISION);
052    }
053
054    /**
055     * Encodes the given latitude and longitude into a geohash
056     *
057     * @param latitude  Latitude to encode
058     * @param longitude Longitude to encode
059     * @return Geohash encoding of the longitude and latitude
060     */
061    public static String encode(double latitude, double longitude, int precision) {
062//        double[] latInterval = {-90.0, 90.0};
063//        double[] lngInterval = {-180.0, 180.0};
064        double latInterval0 = -90.0;
065        double latInterval1 = 90.0;
066        double lngInterval0 = -180.0;
067        double lngInterval1 = 180.0;
068
069        final StringBuilder geohash = new StringBuilder();
070        boolean isEven = true;
071
072        int bit = 0;
073        int ch = 0;
074
075        while (geohash.length() < precision) {
076            double mid = 0.0;
077            if (isEven) {
078//                mid = (lngInterval[0] + lngInterval[1]) / 2D;
079                mid = (lngInterval0 + lngInterval1) / 2D;
080                if (longitude > mid) {
081                    ch |= BITS[bit];
082//                    lngInterval[0] = mid;
083                    lngInterval0 = mid;
084                } else {
085//                    lngInterval[1] = mid;
086                    lngInterval1 = mid;
087                }
088            } else {
089//                mid = (latInterval[0] + latInterval[1]) / 2D;
090                mid = (latInterval0 + latInterval1) / 2D;
091                if (latitude > mid) {
092                    ch |= BITS[bit];
093//                    latInterval[0] = mid;
094                    latInterval0 = mid;
095                } else {
096//                    latInterval[1] = mid;
097                    latInterval1 = mid;
098                }
099            }
100
101            isEven = !isEven;
102
103            if (bit < 4) {
104                bit++;
105            } else {
106                geohash.append(BASE_32[ch]);
107                bit = 0;
108                ch = 0;
109            }
110        }
111
112        return geohash.toString();
113    }
114
115    public static double[] decode(String geohash) {
116        double[] ret = new double[2];
117        decode(geohash, ret);
118        return ret;
119    }
120
121    /**
122     * Decodes the given geohash into a latitude and longitude
123     *
124     * @param geohash Geohash to deocde
125     * @return Array with the latitude at index 0, and longitude at index 1
126     */
127    public static void decode(String geohash, double[] ret) {
128//        double[] latInterval = {-90.0, 90.0};
129//        double[] lngInterval = {-180.0, 180.0};
130        double latInterval0 = -90.0;
131        double latInterval1 = 90.0;
132        double lngInterval0 = -180.0;
133        double lngInterval1 = 180.0;
134
135        boolean isEven = true;
136
137        for (int i = 0; i < geohash.length(); i++) {
138            final int cd = DECODE_MAP.get(geohash.charAt(i));
139
140            for (int mask : BITS) {
141                if (isEven) {
142                    if ((cd & mask) != 0) {
143//                        lngInterval[0] = (lngInterval[0] + lngInterval[1]) / 2D;
144                        lngInterval0 = (lngInterval0 + lngInterval1) / 2D;
145                    } else {
146//                        lngInterval[1] = (lngInterval[0] + lngInterval[1]) / 2D;
147                        lngInterval1 = (lngInterval0 + lngInterval1) / 2D;
148                    }
149                } else {
150                    if ((cd & mask) != 0) {
151//                        latInterval[0] = (latInterval[0] + latInterval[1]) / 2D;
152                        latInterval0 = (latInterval0 + latInterval1) / 2D;
153                    } else {
154//                        latInterval[1] = (latInterval[0] + latInterval[1]) / 2D;
155                        latInterval1 = (latInterval0 + latInterval1) / 2D;
156                    }
157                }
158                isEven = !isEven;
159            }
160
161        }
162//        latitude = (latInterval[0] + latInterval[1]) / 2D;
163        ret[0] = (latInterval0 + latInterval1) / 2D;
164//        longitude = (lngInterval[0] + lngInterval[1]) / 2D;
165        ret[1] = (lngInterval0 + lngInterval1) / 2D;
166
167//        return ret;
168    }
