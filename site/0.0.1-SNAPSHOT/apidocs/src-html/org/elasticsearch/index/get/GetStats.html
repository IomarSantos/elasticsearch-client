020package org.elasticsearch.index.get;
021
022import org.elasticsearch.common.io.stream.StreamInput;
023import org.elasticsearch.common.io.stream.StreamOutput;
024import org.elasticsearch.common.io.stream.Streamable;
025import org.elasticsearch.common.unit.TimeValue;
026import org.elasticsearch.common.xcontent.ToXContent;
027import org.elasticsearch.common.xcontent.XContentBuilder;
028import org.elasticsearch.common.xcontent.XContentBuilderString;
029
030import java.io.IOException;
031
032/**
033 */
034public class GetStats implements Streamable, ToXContent {
035
036    private long existsCount;
037    private long existsTimeInMillis;
038    private long missingCount;
039    private long missingTimeInMillis;
040    private long current;
041
042    public GetStats() {
043    }
044
045    public GetStats(long existsCount, long existsTimeInMillis, long missingCount, long missingTimeInMillis, long current) {
046        this.existsCount = existsCount;
047        this.existsTimeInMillis = existsTimeInMillis;
048        this.missingCount = missingCount;
049        this.missingTimeInMillis = missingTimeInMillis;
050        this.current = current;
051    }
052
053    public void add(GetStats stats) {
054        if (stats == null) {
055            return;
056        }
057        existsCount += stats.existsCount;
058        existsTimeInMillis += stats.existsTimeInMillis;
059        missingCount += stats.missingCount;
060        missingTimeInMillis += stats.missingTimeInMillis;
061        current += stats.current;
062    }
063
064    public long count() {
065        return existsCount + missingCount;
066    }
067
068    public long getCount() {
069        return count();
070    }
071
072    public long timeInMillis() {
073        return existsTimeInMillis + missingTimeInMillis;
074    }
075
076    public long getTimeInMillis() {
077        return timeInMillis();
078    }
079
080    public TimeValue time() {
081        return new TimeValue(timeInMillis());
082    }
083
084    public TimeValue getTime() {
085        return time();
086    }
087
088    public long existsCount() {
089        return this.existsCount;
090    }
091
092    public long getExistsCount() {
093        return this.existsCount;
094    }
095
096    public long existsTimeInMillis() {
097        return this.existsTimeInMillis;
098    }
099
100    public long getExistsTimeInMillis() {
101        return this.existsTimeInMillis;
102    }
103
104    public TimeValue existsTime() {
105        return new TimeValue(existsTimeInMillis);
106    }
107
108    public TimeValue getExistsTime() {
109        return existsTime();
110    }
111
112    public long missingCount() {
113        return this.missingCount;
114    }
115
116    public long getMissingCount() {
117        return this.missingCount;
118    }
119
120    public long missingTimeInMillis() {
121        return this.missingTimeInMillis;
122    }
123
124    public long getMissingTimeInMillis() {
125        return this.missingTimeInMillis;
126    }
127
128    public TimeValue missingTime() {
129        return new TimeValue(missingTimeInMillis);
130    }
131
132    public TimeValue getMissingTime() {
133        return missingTime();
134    }
135
136    public long current() {
137        return this.current;
138    }
139
140    public long getCurrent() {
141        return this.current;
142    }
143
144    @Override
145    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
146        builder.startObject(Fields.GET);
147        builder.field(Fields.TOTAL, count());
148        builder.field(Fields.TIME, time().toString());
149        builder.field(Fields.TIME_IN_MILLIS, timeInMillis());
150        builder.field(Fields.EXISTS_TOTAL, existsCount);
151        builder.field(Fields.EXISTS_TIME, existsTime().toString());
152        builder.field(Fields.EXISTS_TIME_IN_MILLIS, existsTimeInMillis);
153        builder.field(Fields.MISSING_TOTAL, missingCount);
154        builder.field(Fields.MISSING_TIME, missingTime().toString());
155        builder.field(Fields.MISSING_TIME_IN_MILLIS, missingTimeInMillis);
156        builder.field(Fields.CURRENT, current);
157        builder.endObject();
158        return builder;
159    }
160
161    static final class Fields {
162        static final XContentBuilderString GET = new XContentBuilderString("get");
163        static final XContentBuilderString TOTAL = new XContentBuilderString("total");
164        static final XContentBuilderString TIME = new XContentBuilderString("time");
165        static final XContentBuilderString TIME_IN_MILLIS = new XContentBuilderString("time_in_millis");
166        static final XContentBuilderString EXISTS_TOTAL = new XContentBuilderString("exists_total");
167        static final XContentBuilderString EXISTS_TIME = new XContentBuilderString("exists_time");
168        static final XContentBuilderString EXISTS_TIME_IN_MILLIS = new XContentBuilderString("exists_time_in_millis");
169        static final XContentBuilderString MISSING_TOTAL = new XContentBuilderString("missing_total");
170        static final XContentBuilderString MISSING_TIME = new XContentBuilderString("missing_time");
171        static final XContentBuilderString MISSING_TIME_IN_MILLIS = new XContentBuilderString("missing_time_in_millis");
172        static final XContentBuilderString CURRENT = new XContentBuilderString("current");
173    }
174
175    public static GetStats readGetStats(StreamInput in) throws IOException {
176        GetStats stats = new GetStats();
177        stats.readFrom(in);
178        return stats;
179    }
180
181    @Override
182    public void readFrom(StreamInput in) throws IOException {
183        existsCount = in.readVLong();
184        existsTimeInMillis = in.readVLong();
185        missingCount = in.readVLong();
186        missingTimeInMillis = in.readVLong();
187        current = in.readVLong();
188    }
189
190    @Override
191    public void writeTo(StreamOutput out) throws IOException {
192        out.writeVLong(existsCount);
193        out.writeVLong(existsTimeInMillis);
194        out.writeVLong(missingCount);
195        out.writeVLong(missingTimeInMillis);
196        out.writeVLong(current);
197    }
198}
