020package org.elasticsearch.index.query;
021
022import org.elasticsearch.common.xcontent.XContentBuilder;
023import org.elasticsearch.index.search.geo.GeoDistance;
024
025import java.io.IOException;
026
027/**
028 *
029 */
030public class GeoDistanceRangeFilterBuilder extends BaseFilterBuilder {
031
032    public static final String NAME = "geo_distance_range";
033
034    private final String name;
035
036    private Object from;
037    private Object to;
038    private boolean includeLower = true;
039    private boolean includeUpper = true;
040
041    private double lat;
042
043    private double lon;
044
045    private String geohash;
046
047    private GeoDistance geoDistance;
048
049    private Boolean cache;
050    private String cacheKey;
051
052    private String filterName;
053
054    private String optimizeBbox;
055
056    public GeoDistanceRangeFilterBuilder(String name) {
057        this.name = name;
058    }
059
060    public GeoDistanceRangeFilterBuilder point(double lat, double lon) {
061        this.lat = lat;
062        this.lon = lon;
063        return this;
064    }
065
066    public GeoDistanceRangeFilterBuilder lat(double lat) {
067        this.lat = lat;
068        return this;
069    }
070
071    public GeoDistanceRangeFilterBuilder lon(double lon) {
072        this.lon = lon;
073        return this;
074    }
075
076    public GeoDistanceRangeFilterBuilder from(Object from) {
077        this.from = from;
078        return this;
079    }
080
081    public GeoDistanceRangeFilterBuilder to(Object to) {
082        this.to = to;
083        return this;
084    }
085
086    public GeoDistanceRangeFilterBuilder gt(Object from) {
087        this.from = from;
088        this.includeLower = false;
089        return this;
090    }
091
092    public GeoDistanceRangeFilterBuilder gte(Object from) {
093        this.from = from;
094        this.includeLower = true;
095        return this;
096    }
097
098    public GeoDistanceRangeFilterBuilder lt(Object to) {
099        this.to = to;
100        this.includeUpper = false;
101        return this;
102    }
103
104    public GeoDistanceRangeFilterBuilder lte(Object to) {
105        this.to = to;
106        this.includeUpper = true;
107        return this;
108    }
109
110    public GeoDistanceRangeFilterBuilder includeLower(boolean includeLower) {
111        this.includeLower = includeLower;
112        return this;
113    }
114
115    public GeoDistanceRangeFilterBuilder includeUpper(boolean includeUpper) {
116        this.includeUpper = includeUpper;
117        return this;
118    }
119
120    public GeoDistanceRangeFilterBuilder geohash(String geohash) {
121        this.geohash = geohash;
122        return this;
123    }
124
125    public GeoDistanceRangeFilterBuilder geoDistance(GeoDistance geoDistance) {
126        this.geoDistance = geoDistance;
127        return this;
128    }
129
130    public GeoDistanceRangeFilterBuilder optimizeBbox(String optimizeBbox) {
131        this.optimizeBbox = optimizeBbox;
132        return this;
133    }
134
135    /**
136     * Sets the filter name for the filter that can be used when searching for matched_filters per hit.
137     */
138    public GeoDistanceRangeFilterBuilder filterName(String filterName) {
139        this.filterName = filterName;
140        return this;
141    }
142
143    /**
144     * Should the filter be cached or not. Defaults to <tt>false</tt>.
145     */
146    public GeoDistanceRangeFilterBuilder cache(boolean cache) {
147        this.cache = cache;
148        return this;
149    }
150
151    public GeoDistanceRangeFilterBuilder cacheKey(String cacheKey) {
152        this.cacheKey = cacheKey;
153        return this;
154    }
155
156    @Override
157    protected void doXContent(XContentBuilder builder, Params params) throws IOException {
158        builder.startObject(NAME);
159        if (geohash != null) {
160            builder.field(name, geohash);
161        } else {
162            builder.startArray(name).value(lon).value(lat).endArray();
163        }
164        builder.field("from", from);
165        builder.field("to", to);
166        builder.field("include_lower", includeLower);
167        builder.field("include_upper", includeUpper);
168        if (geoDistance != null) {
169            builder.field("distance_type", geoDistance.name().toLowerCase());
170        }
171        if (optimizeBbox != null) {
172            builder.field("optimize_bbox", optimizeBbox);
173        }
174        if (filterName != null) {
175            builder.field("_name", filterName);
176        }
177        if (cache != null) {
178            builder.field("_cache", cache);
179        }
180        if (cacheKey != null) {
181            builder.field("_cache_key", cacheKey);
182        }
183        builder.endObject();
184    }
185}
