020package org.elasticsearch.index.query;
021
022import org.elasticsearch.common.xcontent.XContentBuilder;
023
024import java.io.IOException;
025import java.util.Locale;
026
027/**
028 * Match query is a query that analyzes the text and constructs a query as the result of the analysis. It
029 * can construct different queries based on the type provided.
030 */
031public class MatchQueryBuilder extends BaseQueryBuilder implements BoostableQueryBuilder<MatchQueryBuilder> {
032
033    public static final String NAME = "match";
034
035    public static enum Operator {
036        OR,
037        AND
038    }
039
040    public static enum Type {
041        /**
042         * The text is analyzed and terms are added to a boolean query.
043         */
044        BOOLEAN,
045        /**
046         * The text is analyzed and used as a phrase query.
047         */
048        PHRASE,
049        /**
050         * The text is analyzed and used in a phrase query, with the last term acting as a prefix.
051         */
052        PHRASE_PREFIX
053    }
054
055    private final String name;
056
057    private final Object text;
058
059    private Type type;
060
061    private Operator operator;
062
063    private String analyzer;
064
065    private Float boost;
066
067    private Integer slop;
068
069    private String fuzziness;
070
071    private Integer prefixLength;
072
073    private Integer maxExpansions;
074
075    private String minimumShouldMatch;
076
077    private String rewrite = null;
078
079    private String fuzzyRewrite = null;
080
081    private Boolean lenient;
082
083    /**
084     * Constructs a new text query.
085     */
086    public MatchQueryBuilder(String name, Object text) {
087        this.name = name;
088        this.text = text;
089    }
090
091    /**
092     * Sets the type of the text query.
093     */
094    public MatchQueryBuilder type(Type type) {
095        this.type = type;
096        return this;
097    }
098
099    /**
100     * Sets the operator to use when using a boolean query. Defaults to <tt>OR</tt>.
101     */
102    public MatchQueryBuilder operator(Operator operator) {
103        this.operator = operator;
104        return this;
105    }
106
107    /**
108     * Explicitly set the analyzer to use. Defaults to use explicit mapping config for the field, or, if not
109     * set, the default search analyzer.
110     */
111    public MatchQueryBuilder analyzer(String analyzer) {
112        this.analyzer = analyzer;
113        return this;
114    }
115
116    /**
117     * Set the boost to apply to the query.
118     */
119    public MatchQueryBuilder boost(float boost) {
120        this.boost = boost;
121        return this;
122    }
123
124    /**
125     * Set the phrase slop if evaluated to a phrase query type.
126     */
127    public MatchQueryBuilder slop(int slop) {
128        this.slop = slop;
129        return this;
130    }
131
132    /**
133     * Sets the minimum similarity used when evaluated to a fuzzy query type. Defaults to "0.5".
134     */
135    public MatchQueryBuilder fuzziness(Object fuzziness) {
136        this.fuzziness = fuzziness.toString();
137        return this;
138    }
139
140    public MatchQueryBuilder prefixLength(int prefixLength) {
141        this.prefixLength = prefixLength;
142        return this;
143    }
144
145    /**
146     * When using fuzzy or prefix type query, the number of term expansions to use. Defaults to unbounded
147     * so its recommended to set it to a reasonable value for faster execution.
148     */
149    public MatchQueryBuilder maxExpansions(int maxExpansions) {
150        this.maxExpansions = maxExpansions;
151        return this;
152    }
153
154    public MatchQueryBuilder minimumShouldMatch(String minimumShouldMatch) {
155        this.minimumShouldMatch = minimumShouldMatch;
156        return this;
157    }
158
159    public MatchQueryBuilder rewrite(String rewrite) {
160        this.rewrite = rewrite;
161        return this;
162    }
163
164    public MatchQueryBuilder fuzzyRewrite(String fuzzyRewrite) {
165        this.fuzzyRewrite = fuzzyRewrite;
166        return this;
167    }
168
169    /**
170     * Sets whether format based failures will be ignored.
171     */
172    public MatchQueryBuilder setLenient(boolean lenient) {
173        this.lenient = lenient;
174        return this;
175    }
176
177    @Override
178    public void doXContent(XContentBuilder builder, Params params) throws IOException {
179        builder.startObject(NAME);
180        builder.startObject(name);
181
182        builder.field("query", text);
183        if (type != null) {
184            builder.field("type", type.toString().toLowerCase(Locale.ENGLISH));
185        }
186        if (operator != null) {
187            builder.field("operator", operator.toString());
188        }
189        if (analyzer != null) {
190            builder.field("analyzer", analyzer);
191        }
192        if (boost != null) {
193            builder.field("boost", boost);
194        }
195        if (slop != null) {
196            builder.field("slop", slop);
197        }
198        if (fuzziness != null) {
199            builder.field("fuzziness", fuzziness);
200        }
201        if (prefixLength != null) {
202            builder.field("prefix_length", prefixLength);
203        }
204        if (maxExpansions != null) {
205            builder.field("max_expansions", maxExpansions);
206        }
207        if (minimumShouldMatch != null) {
208            builder.field("minimum_should_match", minimumShouldMatch);
209        }
210        if (rewrite != null) {
211            builder.field("rewrite", rewrite);
212        }
213        if (fuzzyRewrite != null) {
214            builder.field("fuzzy_rewrite", fuzzyRewrite);
215        }
216
217        if (lenient != null) {
218            builder.field("lenient", lenient);
219        }
220
221        builder.endObject();
222        builder.endObject();
223    }
224}
