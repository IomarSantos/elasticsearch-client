020package org.elasticsearch.index.query;
021
022import org.elasticsearch.common.xcontent.XContentBuilder;
023import org.elasticsearch.index.search.geo.Point;
024
025import java.io.IOException;
026
027/**
028 *
029 */
030public class GeoBoundingBoxFilterBuilder extends BaseFilterBuilder {
031
032    public static final String NAME = "geo_bbox";
033
034    private final String name;
035
036    private Point topLeft;
037
038    private String topLeftGeohash;
039
040    private Point bottomRight;
041
042    private String bottomRightGeohash;
043
044    private Boolean cache;
045    private String cacheKey;
046
047    private String filterName;
048
049    private String type;
050
051    public GeoBoundingBoxFilterBuilder(String name) {
052        this.name = name;
053    }
054
055    /**
056     * Adds top left point.
057     *
058     * @param lat The latitude
059     * @param lon The longitude
060     */
061    public GeoBoundingBoxFilterBuilder topLeft(double lat, double lon) {
062        topLeft = new Point();
063        topLeft.lat = lat;
064        topLeft.lon = lon;
065        return this;
066    }
067
068    /**
069     * Adds bottom right point.
070     *
071     * @param lat The latitude
072     * @param lon The longitude
073     */
074    public GeoBoundingBoxFilterBuilder bottomRight(double lat, double lon) {
075        bottomRight = new Point();
076        bottomRight.lat = lat;
077        bottomRight.lon = lon;
078        return this;
079    }
080
081    public GeoBoundingBoxFilterBuilder topLeft(String geohash) {
082        this.topLeftGeohash = geohash;
083        return this;
084    }
085
086    public GeoBoundingBoxFilterBuilder bottomRight(String geohash) {
087        this.bottomRightGeohash = geohash;
088        return this;
089    }
090
091    /**
092     * Sets the filter name for the filter that can be used when searching for matched_filters per hit.
093     */
094    public GeoBoundingBoxFilterBuilder filterName(String filterName) {
095        this.filterName = filterName;
096        return this;
097    }
098
099    /**
100     * Should the filter be cached or not. Defaults to <tt>false</tt>.
101     */
102    public GeoBoundingBoxFilterBuilder cache(boolean cache) {
103        this.cache = cache;
104        return this;
105    }
106
107    public GeoBoundingBoxFilterBuilder cacheKey(String cacheKey) {
108        this.cacheKey = cacheKey;
109        return this;
110    }
111
112    /**
113     * Sets the type of executing of the geo bounding box. Can be either `memory` or `indexed`. Defaults
114     * to `memory`.
115     */
116    public GeoBoundingBoxFilterBuilder type(String type) {
117        this.type = type;
118        return this;
119    }
120
121    @Override
122    protected void doXContent(XContentBuilder builder, Params params) throws IOException {
123        builder.startObject(NAME);
124
125        builder.startObject(name);
126        if (topLeftGeohash != null) {
127            builder.field("top_left", topLeftGeohash);
128        } else if (topLeft != null) {
129            builder.startArray("top_left").value(topLeft.lon).value(topLeft.lat).endArray();
130        } else {
131            throw new QueryBuilderException("geo_bounding_box requires 'top_left' to be set");
132        }
133
134        if (bottomRightGeohash != null) {
135            builder.field("bottom_right", bottomRightGeohash);
136        } else if (bottomRight != null) {
137            builder.startArray("bottom_right").value(bottomRight.lon).value(bottomRight.lat).endArray();
138        } else {
139            throw new QueryBuilderException("geo_bounding_box requires 'bottom_right' to be set");
140        }
141        builder.endObject();
142
143        if (filterName != null) {
144            builder.field("_name", filterName);
145        }
146        if (cache != null) {
147            builder.field("_cache", cache);
148        }
149        if (cacheKey != null) {
150            builder.field("_cache_key", cacheKey);
151        }
152        if (type != null) {
153            builder.field("type", type);
154        }
155
156        builder.endObject();
157    }
158}
