020package org.elasticsearch.index.query;
021
022import org.elasticsearch.common.xcontent.XContentBuilder;
023
024import java.io.IOException;
025
026/**
027 * Constructs a filter that only match on documents that the field has a value in them.
028 */
029public class MissingFilterBuilder extends BaseFilterBuilder {
030
031    public static final String NAME = "missing";
032
033    private String name;
034
035    private String filterName;
036
037    private Boolean nullValue;
038
039    private Boolean existence;
040
041    public MissingFilterBuilder(String name) {
042        this.name = name;
043    }
044
045    /**
046     * Should the missing filter automatically include fields with null value configured in the
047     * mappings. Defaults to <tt>false</tt>.
048     */
049    public MissingFilterBuilder nullValue(boolean nullValue) {
050        this.nullValue = nullValue;
051        return this;
052    }
053
054    /**
055     * Should hte missing filter include documents where the field doesn't exists in the docs.
056     * Defaults to <tt>true</tt>.
057     */
058    public MissingFilterBuilder existence(boolean existence) {
059        this.existence = existence;
060        return this;
061    }
062
063    /**
064     * Sets the filter name for the filter that can be used when searching for matched_filters per hit.
065     */
066    public MissingFilterBuilder filterName(String filterName) {
067        this.filterName = filterName;
068        return this;
069    }
070
071    @Override
072    protected void doXContent(XContentBuilder builder, Params params) throws IOException {
073        builder.startObject(NAME);
074        builder.field("field", name);
075        if (nullValue != null) {
076            builder.field("null_value", nullValue);
077        }
078        if (existence != null) {
079            builder.field("existence", existence);
080        }
081        if (filterName != null) {
082            builder.field("_name", filterName);
083        }
084        builder.endObject();
085    }
086}
