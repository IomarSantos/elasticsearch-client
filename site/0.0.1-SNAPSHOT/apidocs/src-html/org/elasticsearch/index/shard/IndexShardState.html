020package org.elasticsearch.index.shard;
021
022import org.elasticsearch.ElasticSearchIllegalArgumentException;
023
024/**
025 *
026 */
027public enum IndexShardState {
028    CREATED((byte) 0),
029    RECOVERING((byte) 1),
030    STARTED((byte) 2),
031    RELOCATED((byte) 3),
032    CLOSED((byte) 4);
033
034    private final byte id;
035
036    IndexShardState(byte id) {
037        this.id = id;
038    }
039
040    public byte id() {
041        return this.id;
042    }
043
044    public static IndexShardState fromId(byte id) throws ElasticSearchIllegalArgumentException {
045        if (id == 0) {
046            return CREATED;
047        } else if (id == 1) {
048            return RECOVERING;
049        } else if (id == 2) {
050            return STARTED;
051        } else if (id == 3) {
052            return RELOCATED;
053        } else if (id == 4) {
054            return CLOSED;
055        }
056        throw new ElasticSearchIllegalArgumentException("No mapping for id [" + id + "]");
057    }
058}
