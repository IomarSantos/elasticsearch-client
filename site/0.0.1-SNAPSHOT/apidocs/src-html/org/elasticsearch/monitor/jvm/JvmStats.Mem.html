074    static {
075        runtimeMXBean = ManagementFactory.getRuntimeMXBean();
076        memoryMXBean = ManagementFactory.getMemoryMXBean();
077        threadMXBean = ManagementFactory.getThreadMXBean();
078
079        try {
080            managementFactoryPlatformMXBeansMethod = ManagementFactory.class.getMethod("getPlatformMXBeans", Class.class);
081        } catch (Throwable e) {
082            managementFactoryPlatformMXBeansMethod = null;
083        }
084
085
086        try {
087            bufferPoolMXBeanClass = Class.forName("java.lang.management.BufferPoolMXBean");
088            bufferPoolMXBeanNameMethod = bufferPoolMXBeanClass.getMethod("getName");
089            bufferPoolMXBeanCountMethod = bufferPoolMXBeanClass.getMethod("getCount");
090            bufferPoolMXBeanTotalCapacityMethod = bufferPoolMXBeanClass.getMethod("getTotalCapacity");
091            bufferPoolMXBeanMemoryUsedMethod = bufferPoolMXBeanClass.getMethod("getMemoryUsed");
092            bufferPoolsEnabled = true;
093        } catch (Throwable t) {
094            bufferPoolsEnabled = false;
095        }
096
097        JvmInfo info = JvmInfo.jvmInfo();
098        boolean defaultEnableLastGc = false;
099        if (info.versionAsInteger() == 170) {
100            defaultEnableLastGc = info.versionUpdatePack() >= 4;
101        } else if (info.versionAsInteger() > 170) {
102            defaultEnableLastGc = true;
103        }
104        // always disable lastG, some reports it gives are strange...
105        defaultEnableLastGc = false;
106
107        boolean enableLastGc = Booleans.parseBoolean(System.getProperty("monitor.jvm.enable_last_gc"), defaultEnableLastGc);
108        if (enableLastGc) {
109            try {
110                Class sunGcClass = Class.forName("com.sun.management.GarbageCollectorMXBean");
111                Class gcInfoClass = Class.forName("com.sun.management.GcInfo");
112
113                getLastGcInfoMethod = sunGcClass.getDeclaredMethod("getLastGcInfo");
114                getLastGcInfoMethod.setAccessible(true);
115
116                getMemoryUsageBeforeGcMethod = gcInfoClass.getDeclaredMethod("getMemoryUsageBeforeGc");
117                getMemoryUsageBeforeGcMethod.setAccessible(true);
118                getMemoryUsageAfterGcMethod = gcInfoClass.getDeclaredMethod("getMemoryUsageAfterGc");
119                getMemoryUsageAfterGcMethod.setAccessible(true);
120                getStartTimeMethod = gcInfoClass.getDeclaredMethod("getStartTime");
121                getStartTimeMethod.setAccessible(true);
122                getEndTimeMethod = gcInfoClass.getDeclaredMethod("getEndTime");
123                getEndTimeMethod.setAccessible(true);
124                getDurationMethod = gcInfoClass.getDeclaredMethod("getDuration");
125                getDurationMethod.setAccessible(true);
126
127            } catch (Throwable ex) {
128                enableLastGc = false;
129            }
130        }
131
132        JvmStats.enableLastGc = enableLastGc;
133    }

135    public static JvmStats jvmStats() {
136        JvmStats stats = new JvmStats(System.currentTimeMillis(), runtimeMXBean.getUptime());
137        stats.mem = new Mem();
138        MemoryUsage memUsage = memoryMXBean.getHeapMemoryUsage();
139        stats.mem.heapUsed = memUsage.getUsed() < 0 ? 0 : memUsage.getUsed();
140        stats.mem.heapCommitted = memUsage.getCommitted() < 0 ? 0 : memUsage.getCommitted();
141        memUsage = memoryMXBean.getNonHeapMemoryUsage();
142        stats.mem.nonHeapUsed = memUsage.getUsed() < 0 ? 0 : memUsage.getUsed();
143        stats.mem.nonHeapCommitted = memUsage.getCommitted() < 0 ? 0 : memUsage.getCommitted();
144
145        List<MemoryPoolMXBean> memoryPoolMXBeans = ManagementFactory.getMemoryPoolMXBeans();
146        stats.mem.pools = new MemoryPool[memoryPoolMXBeans.size()];
147        for (int i = 0; i < memoryPoolMXBeans.size(); i++) {
148            MemoryPoolMXBean memoryPoolMXBean = memoryPoolMXBeans.get(i);
149            MemoryUsage usage = memoryPoolMXBean.getUsage();
150            MemoryUsage peakUsage = memoryPoolMXBean.getPeakUsage();
151            stats.mem.pools[i] = new MemoryPool(memoryPoolMXBean.getName(),
152                    usage.getUsed() < 0 ? 0 : usage.getUsed(),
153                    usage.getMax() < 0 ? 0 : usage.getMax(),
154                    peakUsage.getUsed() < 0 ? 0 : peakUsage.getUsed(),
155                    peakUsage.getMax() < 0 ? 0 : peakUsage.getMax()
156            );
157        }
158
159        stats.threads = new Threads();
160        stats.threads.count = threadMXBean.getThreadCount();
161        stats.threads.peakCount = threadMXBean.getPeakThreadCount();
162
163        List<GarbageCollectorMXBean> gcMxBeans = ManagementFactory.getGarbageCollectorMXBeans();
164        stats.gc = new GarbageCollectors();
165        stats.gc.collectors = new GarbageCollector[gcMxBeans.size()];
166        for (int i = 0; i < stats.gc.collectors.length; i++) {
167            GarbageCollectorMXBean gcMxBean = gcMxBeans.get(i);
168            stats.gc.collectors[i] = new GarbageCollector();
169            stats.gc.collectors[i].name = gcMxBean.getName();
170            stats.gc.collectors[i].collectionCount = gcMxBean.getCollectionCount();
171            stats.gc.collectors[i].collectionTime = gcMxBean.getCollectionTime();
172            if (enableLastGc) {
173                try {
174                    Object lastGcInfo = getLastGcInfoMethod.invoke(gcMxBean);
175                    if (lastGcInfo != null) {
176                        Map<String, MemoryUsage> usageBeforeGc = (Map<String, MemoryUsage>) getMemoryUsageBeforeGcMethod.invoke(lastGcInfo);
177                        Map<String, MemoryUsage> usageAfterGc = (Map<String, MemoryUsage>) getMemoryUsageAfterGcMethod.invoke(lastGcInfo);
178                        long startTime = (Long) getStartTimeMethod.invoke(lastGcInfo);
179                        long endTime = (Long) getEndTimeMethod.invoke(lastGcInfo);
180                        long duration = (Long) getDurationMethod.invoke(lastGcInfo);
181
182                        long previousMemoryUsed = 0;
183                        long memoryUsed = 0;
184                        long memoryMax = 0;
185                        for (Map.Entry<String, MemoryUsage> entry : usageBeforeGc.entrySet()) {
186                            previousMemoryUsed += entry.getValue().getUsed();
187                        }
188                        for (Map.Entry<String, MemoryUsage> entry : usageAfterGc.entrySet()) {
189                            MemoryUsage mu = entry.getValue();
190                            memoryUsed += mu.getUsed();
191                            memoryMax += mu.getMax();
192                        }
193
194                        stats.gc.collectors[i].lastGc = new GarbageCollector.LastGc(startTime, endTime, memoryMax, previousMemoryUsed, memoryUsed, duration);
195                    }
196                } catch (Exception e) {
197//                    e.printStackTrace();
198                }
199            }
200        }
201
202
203        if (bufferPoolsEnabled) {
204            try {
205                List bufferPools = (List) managementFactoryPlatformMXBeansMethod.invoke(null, bufferPoolMXBeanClass);
206                stats.bufferPools = new ArrayList<BufferPool>(bufferPools.size());
207                for (Object bufferPool : bufferPools) {
208                    String name = (String) bufferPoolMXBeanNameMethod.invoke(bufferPool);
209                    Long count = (Long) bufferPoolMXBeanCountMethod.invoke(bufferPool);
210                    Long totalCapacity = (Long) bufferPoolMXBeanTotalCapacityMethod.invoke(bufferPool);
211                    Long memoryUsed = (Long) bufferPoolMXBeanMemoryUsedMethod.invoke(bufferPool);
212                    stats.bufferPools.add(new BufferPool(name, count, totalCapacity, memoryUsed));
213                }
214            } catch (Throwable t) {
215                //t.printStackTrace();
216            }
217        }
218
219        return stats;
220    }

282    @Override
283    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
284        builder.startObject(Fields.JVM);
285        builder.field(Fields.TIMESTAMP, timestamp);
286        builder.field(Fields.UPTIME, uptime().format());
287        builder.field(Fields.UPTIME_IN_MILLIS, uptime().millis());
288        if (mem != null) {
289            builder.startObject(Fields.MEM);
290            builder.field(Fields.HEAP_USED, mem.heapUsed().toString());
291            builder.field(Fields.HEAP_USED_IN_BYTES, mem.heapUsed().bytes());
292            builder.field(Fields.HEAP_COMMITTED, mem.heapCommitted().toString());
293            builder.field(Fields.HEAP_COMMITTED_IN_BYTES, mem.heapCommitted().bytes());
294
295            builder.field(Fields.NON_HEAP_USED, mem.nonHeapUsed().toString());
296            builder.field(Fields.NON_HEAP_USED_IN_BYTES, mem.nonHeapUsed);
297            builder.field(Fields.NON_HEAP_COMMITTED, mem.nonHeapCommitted().toString());
298            builder.field(Fields.NON_HEAP_COMMITTED_IN_BYTES, mem.nonHeapCommitted);
299
300            builder.startObject(Fields.POOLS);
301            for (MemoryPool pool : mem) {
302                builder.startObject(pool.name(), XContentBuilder.FieldCaseConversion.NONE);
303                builder.field(Fields.USED, pool.used().toString());
304                builder.field(Fields.USED_IN_BYTES, pool.used);
305                builder.field(Fields.MAX, pool.max().toString());
306                builder.field(Fields.MAX_IN_BYTES, pool.max);
307
308                builder.field(Fields.PEAK_USED, pool.peakUsed().toString());
309                builder.field(Fields.PEAK_USED_IN_BYTES, pool.peakUsed);
310                builder.field(Fields.PEAK_MAX, pool.peakMax().toString());
311                builder.field(Fields.PEAK_MAX_IN_BYTES, pool.peakMax);
312
313                builder.endObject();
314            }
315            builder.endObject();
316
317            builder.endObject();
318        }
319        if (threads != null) {
320            builder.startObject(Fields.THREADS);
321            builder.field(Fields.COUNT, threads.count());
322            builder.field(Fields.PEAK_COUNT, threads.peakCount());
323            builder.endObject();
324        }
325        if (gc != null) {
326            builder.startObject(Fields.GC);
327            builder.field(Fields.COLLECTION_COUNT, gc.collectionCount());
328            builder.field(Fields.COLLECTION_TIME, gc.collectionTime().format());
329            builder.field(Fields.COLLECTION_TIME_IN_MILLIS, gc.collectionTime().millis());
330
331            builder.startObject(Fields.COLLECTORS);
332            for (GarbageCollector collector : gc) {
333                builder.startObject(collector.name(), XContentBuilder.FieldCaseConversion.NONE);
334                builder.field(Fields.COLLECTION_COUNT, collector.collectionCount());
335                builder.field(Fields.COLLECTION_TIME, collector.collectionTime().format());
336                builder.field(Fields.COLLECTION_TIME_IN_MILLIS, collector.collectionTime().millis());
337                builder.endObject();
338            }
339            builder.endObject();
340
341            builder.endObject();
342        }
343
344        if (bufferPools != null) {
345            builder.startObject(Fields.BUFFER_POOLS);
346            for (BufferPool bufferPool : bufferPools) {
347                builder.startObject(bufferPool.name(), XContentBuilder.FieldCaseConversion.NONE);
348                builder.field(Fields.COUNT, bufferPool.count());
349                builder.field(Fields.USED, bufferPool.used().toString());
350                builder.field(Fields.USED_IN_BYTES, bufferPool.used);
351                builder.field(Fields.TOTAL_CAPACITY, bufferPool.totalCapacity().toString());
352                builder.field(Fields.TOTAL_CAPACITY_IN_BYTES, bufferPool.totalCapacity);
353                builder.endObject();
354            }
355            builder.endObject();
356        }
357
358        builder.endObject();
359        return builder;
360    }

412    @Override
413    public void readFrom(StreamInput in) throws IOException {
414        timestamp = in.readVLong();
415        uptime = in.readVLong();
416
417        mem = Mem.readMem(in);
418        threads = Threads.readThreads(in);
419        gc = GarbageCollectors.readGarbageCollectors(in);
420
421        if (in.readBoolean()) {
422            int size = in.readVInt();
423            bufferPools = new ArrayList<BufferPool>(size);
424            for (int i = 0; i < size; i++) {
425                BufferPool bufferPool = new BufferPool();
426                bufferPool.readFrom(in);
427                bufferPools.add(bufferPool);
428            }
429        }
430    }
431
432    @Override
433    public void writeTo(StreamOutput out) throws IOException {
434        out.writeVLong(timestamp);
435        out.writeVLong(uptime);
436
437        mem.writeTo(out);
438        threads.writeTo(out);
439        gc.writeTo(out);
440
441        if (bufferPools == null) {
442            out.writeBoolean(false);
443        } else {
444            out.writeBoolean(true);
445            out.writeVInt(bufferPools.size());
446            for (BufferPool bufferPool : bufferPools) {
447                bufferPool.writeTo(out);
448            }
449        }

813
814        long heapCommitted;
815        long heapUsed;
816        long nonHeapCommitted;
817        long nonHeapUsed;
818
819        MemoryPool[] pools = new MemoryPool[0];
820
821        Mem() {
822        }
823
824        public static Mem readMem(StreamInput in) throws IOException {
825            Mem mem = new Mem();
826            mem.readFrom(in);
827            return mem;
828        }
829
830        @Override
831        public Iterator<MemoryPool> iterator() {
832            return Iterators.forArray(pools);
833        }
834
835        @Override
836        public void readFrom(StreamInput in) throws IOException {
837            heapCommitted = in.readVLong();
838            heapUsed = in.readVLong();
839            nonHeapCommitted = in.readVLong();
840            nonHeapUsed = in.readVLong();
841
842            pools = new MemoryPool[in.readVInt()];
843            for (int i = 0; i < pools.length; i++) {
844                pools[i] = MemoryPool.readMemoryPool(in);
845            }
846        }
847
848        @Override
849        public void writeTo(StreamOutput out) throws IOException {
850            out.writeVLong(heapCommitted);
851            out.writeVLong(heapUsed);
852            out.writeVLong(nonHeapCommitted);
853            out.writeVLong(nonHeapUsed);
854
855            out.writeVInt(pools.length);
856            for (MemoryPool pool : pools) {
857                pool.writeTo(out);
858            }
859        }
860
