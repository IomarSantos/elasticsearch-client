106    static final class Fields {
107        static final XContentBuilderString OS = new XContentBuilderString("os");
108        static final XContentBuilderString TIMESTAMP = new XContentBuilderString("timestamp");
109        static final XContentBuilderString UPTIME = new XContentBuilderString("uptime");
110        static final XContentBuilderString UPTIME_IN_MILLIS = new XContentBuilderString("uptime_in_millis");
111        static final XContentBuilderString LOAD_AVERAGE = new XContentBuilderString("load_average");
112
113        static final XContentBuilderString CPU = new XContentBuilderString("cpu");
114        static final XContentBuilderString SYS = new XContentBuilderString("sys");
115        static final XContentBuilderString USER = new XContentBuilderString("user");
116        static final XContentBuilderString IDLE = new XContentBuilderString("idle");
117
118        static final XContentBuilderString MEM = new XContentBuilderString("mem");
119        static final XContentBuilderString SWAP = new XContentBuilderString("swap");
120        static final XContentBuilderString FREE = new XContentBuilderString("free");
121        static final XContentBuilderString FREE_IN_BYTES = new XContentBuilderString("free_in_bytes");
122        static final XContentBuilderString USED = new XContentBuilderString("used");
123        static final XContentBuilderString USED_IN_BYTES = new XContentBuilderString("used_in_bytes");
124
125        static final XContentBuilderString FREE_PERCENT = new XContentBuilderString("free_percent");
126        static final XContentBuilderString USED_PERCENT = new XContentBuilderString("used_percent");
127
128        static final XContentBuilderString ACTUAL_FREE = new XContentBuilderString("actual_free");
129        static final XContentBuilderString ACTUAL_FREE_IN_BYTES = new XContentBuilderString("actual_free_in_bytes");
130        static final XContentBuilderString ACTUAL_USED = new XContentBuilderString("actual_used");
131        static final XContentBuilderString ACTUAL_USED_IN_BYTES = new XContentBuilderString("actual_used_in_bytes");
132    }

134    @Override
135    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
136        builder.startObject(Fields.OS);
137        builder.field(Fields.TIMESTAMP, timestamp);
138
139        if (uptime != -1) {
140            builder.field(Fields.UPTIME, uptime().format());
141            builder.field(Fields.UPTIME_IN_MILLIS, uptime().millis());
142        }
143
144        if (loadAverage.length > 0) {
145            builder.startArray(Fields.LOAD_AVERAGE);
146            for (double value : loadAverage) {
147                builder.value(value);
148            }
149            builder.endArray();
150        }
151
152        if (cpu != null) {
153            builder.startObject(Fields.CPU);
154            builder.field(Fields.SYS, cpu.sys());
155            builder.field(Fields.USER, cpu.user());
156            builder.field(Fields.IDLE, cpu.idle());
157            builder.endObject();
158        }
159
160        if (mem != null) {
161            builder.startObject(Fields.MEM);
162            builder.field(Fields.FREE, mem.free().toString());
163            builder.field(Fields.FREE_IN_BYTES, mem.free);
164            builder.field(Fields.USED, mem.used().toString());
165            builder.field(Fields.USED_IN_BYTES, mem.used);
166
167            builder.field(Fields.FREE_PERCENT, mem.freePercent());
168            builder.field(Fields.USED_PERCENT, mem.usedPercent());
169
170            builder.field(Fields.ACTUAL_FREE, mem.actualFree().toString());
171            builder.field(Fields.ACTUAL_FREE_IN_BYTES, mem.actualFree);
172            builder.field(Fields.ACTUAL_USED, mem.actualUsed().toString());
173            builder.field(Fields.ACTUAL_USED_IN_BYTES, mem.actualUsed);
174
175            builder.endObject();
176        }
177
178        if (swap != null) {
179            builder.startObject(Fields.SWAP);
180            builder.field(Fields.USED, swap.used().toString());
181            builder.field(Fields.USED_IN_BYTES, swap.used);
182            builder.field(Fields.FREE, swap.free().toString());
183            builder.field(Fields.FREE_IN_BYTES, swap.free);
184            builder.endObject();
185        }
186
187        builder.endObject();
188        return builder;
189    }

191    public static OsStats readOsStats(StreamInput in) throws IOException {
192        OsStats stats = new OsStats();
193        stats.readFrom(in);
194        return stats;
195    }
196
197    @Override
198    public void readFrom(StreamInput in) throws IOException {
199        timestamp = in.readVLong();
200        loadAverage = new double[in.readVInt()];
201        for (int i = 0; i < loadAverage.length; i++) {
202            loadAverage[i] = in.readDouble();
203        }
204        uptime = in.readLong();
205        if (in.readBoolean()) {
206            cpu = Cpu.readCpu(in);
207        }
208        if (in.readBoolean()) {
209            mem = Mem.readMem(in);
210        }
211        if (in.readBoolean()) {
212            swap = Swap.readSwap(in);
213        }
214    }
215
216    @Override
217    public void writeTo(StreamOutput out) throws IOException {
218        out.writeVLong(timestamp);
219        out.writeVInt(loadAverage.length);
220        for (double val : loadAverage) {
221            out.writeDouble(val);
222        }
223        out.writeLong(uptime);
224        if (cpu == null) {
225            out.writeBoolean(false);
226        } else {
227            out.writeBoolean(true);
228            cpu.writeTo(out);
229        }
230        if (mem == null) {
231            out.writeBoolean(false);
232        } else {
233            out.writeBoolean(true);
234            mem.writeTo(out);
235        }
236        if (swap == null) {
237            out.writeBoolean(false);
238        } else {
239            out.writeBoolean(true);
240            swap.writeTo(out);
241        }
242    }
