020package org.elasticsearch.monitor.network;
021
022import org.elasticsearch.common.io.stream.StreamInput;
023import org.elasticsearch.common.io.stream.StreamOutput;
024import org.elasticsearch.common.io.stream.Streamable;
025import org.elasticsearch.common.xcontent.ToXContent;
026import org.elasticsearch.common.xcontent.XContentBuilder;
027import org.elasticsearch.common.xcontent.XContentBuilderString;
028
029import java.io.IOException;
030import java.io.Serializable;
031
032/**
033 *
034 */
035public class NetworkStats implements Streamable, Serializable, ToXContent {
036
037    long timestamp;
038
039    Tcp tcp = null;
040
041    NetworkStats() {
042
043    }
044
045    static final class Fields {
046        static final XContentBuilderString NETWORK = new XContentBuilderString("network");
047        static final XContentBuilderString TCP = new XContentBuilderString("tcp");
048        static final XContentBuilderString ACTIVE_OPENS = new XContentBuilderString("active_opens");
049        static final XContentBuilderString PASSIVE_OPENS = new XContentBuilderString("passive_opens");
050        static final XContentBuilderString CURR_ESTAB = new XContentBuilderString("curr_estab");
051        static final XContentBuilderString IN_SEGS = new XContentBuilderString("in_segs");
052        static final XContentBuilderString OUT_SEGS = new XContentBuilderString("out_segs");
053        static final XContentBuilderString RETRANS_SEGS = new XContentBuilderString("retrans_segs");
054        static final XContentBuilderString ESTAB_RESETS = new XContentBuilderString("estab_resets");
055        static final XContentBuilderString ATTEMPT_FAILS = new XContentBuilderString("attempt_fails");
056        static final XContentBuilderString IN_ERRS = new XContentBuilderString("in_errs");
057        static final XContentBuilderString OUT_RSTS = new XContentBuilderString("out_rsts");
058    }
059
060    @Override
061    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
062        builder.startObject(Fields.NETWORK);
063        if (tcp != null) {
064            builder.startObject(Fields.TCP);
065            builder.field(Fields.ACTIVE_OPENS, tcp.getActiveOpens());
066            builder.field(Fields.PASSIVE_OPENS, tcp.getPassiveOpens());
067            builder.field(Fields.CURR_ESTAB, tcp.getCurrEstab());
068            builder.field(Fields.IN_SEGS, tcp.getInSegs());
069            builder.field(Fields.OUT_SEGS, tcp.getOutSegs());
070            builder.field(Fields.RETRANS_SEGS, tcp.getRetransSegs());
071            builder.field(Fields.ESTAB_RESETS, tcp.getEstabResets());
072            builder.field(Fields.ATTEMPT_FAILS, tcp.getAttemptFails());
073            builder.field(Fields.IN_ERRS, tcp.getInErrs());
074            builder.field(Fields.OUT_RSTS, tcp.getOutRsts());
075            builder.endObject();
076        }
077        builder.endObject();
078        return builder;
079    }
080
081    public static NetworkStats readNetworkStats(StreamInput in) throws IOException {
082        NetworkStats stats = new NetworkStats();
083        stats.readFrom(in);
084        return stats;
085    }
086
087    @Override
088    public void readFrom(StreamInput in) throws IOException {
089        timestamp = in.readVLong();
090        if (in.readBoolean()) {
091            tcp = Tcp.readNetworkTcp(in);
092        }
093    }
094
095    @Override
096    public void writeTo(StreamOutput out) throws IOException {
097        out.writeVLong(timestamp);
098        if (tcp == null) {
099            out.writeBoolean(false);
100        } else {
101            out.writeBoolean(true);
102            tcp.writeTo(out);
103        }
104    }
105
106    public long timestamp() {
107        return timestamp;
108    }
109
110    public long getTimestamp() {
111        return timestamp();
112    }
113
114    public Tcp tcp() {
115        return tcp;
116    }
117
118    public Tcp getTcp() {
119        return tcp();
120    }
121
122    public static class Tcp implements Serializable, Streamable {
123
124        long activeOpens;
125        long passiveOpens;
126        long attemptFails;
127        long estabResets;
128        long currEstab;
129        long inSegs;
130        long outSegs;
131        long retransSegs;
132        long inErrs;
133        long outRsts;
134
135        public static Tcp readNetworkTcp(StreamInput in) throws IOException {
136            Tcp tcp = new Tcp();
137            tcp.readFrom(in);
138            return tcp;
139        }
140
141        @Override
142        public void readFrom(StreamInput in) throws IOException {
143            activeOpens = in.readLong();
144            passiveOpens = in.readLong();
145            attemptFails = in.readLong();
146            estabResets = in.readLong();
147            currEstab = in.readLong();
148            inSegs = in.readLong();
149            outSegs = in.readLong();
150            retransSegs = in.readLong();
151            inErrs = in.readLong();
152            outRsts = in.readLong();
153        }
154
155        @Override
156        public void writeTo(StreamOutput out) throws IOException {
157            out.writeLong(activeOpens);
158            out.writeLong(passiveOpens);
159            out.writeLong(attemptFails);
160            out.writeLong(estabResets);
161            out.writeLong(currEstab);
162            out.writeLong(inSegs);
163            out.writeLong(outSegs);
164            out.writeLong(retransSegs);
165            out.writeLong(inErrs);
166            out.writeLong(outRsts);
167        }
168
169        public long activeOpens() {
170            return this.activeOpens;
171        }
172
173        public long getActiveOpens() {
174            return activeOpens();
175        }
176
177        public long passiveOpens() {
178            return passiveOpens;
179        }
180
181        public long getPassiveOpens() {
182            return passiveOpens();
183        }
184
185        public long attemptFails() {
186            return attemptFails;
187        }
188
189        public long getAttemptFails() {
190            return attemptFails();
191        }
192
193        public long estabResets() {
194            return estabResets;
195        }
196
197        public long getEstabResets() {
198            return estabResets();
199        }
200
201        public long currEstab() {
202            return currEstab;
203        }
204
205        public long getCurrEstab() {
206            return currEstab();
207        }
208
209        public long inSegs() {
210            return inSegs;
211        }
212
213        public long getInSegs() {
214            return inSegs();
215        }
216
217        public long outSegs() {
218            return outSegs;
219        }
220
221        public long getOutSegs() {
222            return outSegs();
223        }
224
225        public long retransSegs() {
226            return retransSegs;
227        }
228
229        public long getRetransSegs() {
230            return retransSegs();
231        }
232
233        public long inErrs() {
234            return inErrs;
235        }
236
237        public long getInErrs() {
238            return inErrs();
239        }
240
241        public long outRsts() {
242            return outRsts;
243        }
244
245        public long getOutRsts() {
246            return outRsts();
247        }
248    }
249}
