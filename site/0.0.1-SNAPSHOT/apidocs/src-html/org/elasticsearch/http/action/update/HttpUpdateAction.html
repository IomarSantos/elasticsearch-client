020package org.elasticsearch.http.action.update;
021
022import org.elasticsearch.action.ActionListener;
023import org.elasticsearch.action.support.HttpAction;
024import org.elasticsearch.action.support.HttpClient;
025import org.elasticsearch.action.support.HttpRequest;
026import org.elasticsearch.action.support.HttpResponse;
027import org.elasticsearch.action.update.UpdateRequest;
028import org.elasticsearch.action.update.UpdateResponse;
029import org.elasticsearch.common.xcontent.XContentHelper;
030
031import java.util.Map;
032
033public class HttpUpdateAction extends HttpAction<UpdateRequest, UpdateResponse> {
034
035    public static final String NAME = "update";
036    private static final String METHOD = "POST";
037    private static final String ENDPOINT = "_update";
038
039    @Override
040    protected void doExecute(final HttpClient client, final UpdateRequest request, final ActionListener<UpdateResponse> listener) {
041        HttpRequest httpRequest = new HttpRequest(METHOD, ENDPOINT)
042                .index(request.index())
043                .type(request.type())
044                .id(request.id())
045                .param("routing", request.routing())
046                .param("parent", request.parent())
047                .param("refresh", request.refresh())
048                .param("percolate", request.percolate())
049                .param("replication", request.replicationType().name().toLowerCase())
050                .param("consistency", request.consistencyLevel().name().toLowerCase())
051                .param("routing", request.routing())
052                .param("script", request.script())
053                .param("lang", request.scriptLang())
054                .param("fields", request.fields())
055                .param("retry_on_conflict", request.retryOnConflict());
056        if (request.doc() != null) {
057            httpRequest.body(request.doc().source());
058        } else {
059            if (request.scriptParams() != null) {
060                for (Map.Entry<String, Object> me : request.scriptParams().entrySet()) {
061                    httpRequest.param("sp_" + me.getKey(), me.getValue().toString());
062                }
063            }
064        }
065        submit(client, httpRequest, listener);
066    }
067
068    @Override
069    protected UpdateResponse toResponse(HttpResponse response) {
070        Map<String, Object> map = XContentHelper.convertToMap(response.getBody(), false).v2();
071        UpdateResponse updateResponse = new UpdateResponse(
072                map.get("_index").toString(),
073                map.get("_type").toString(),
074                map.get("_id").toString(),
075                Long.parseLong(map.get("_version").toString()));
076        return updateResponse;
077    }
078}
