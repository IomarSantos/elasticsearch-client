171    /**
172     * Set to <tt>true</tt> to indicate that a {@link ConnectTransportException} should be thrown when
173     * sending a message (otherwise, it will be passed to the response handler). Defaults to <tt>false</tt>.
174     * <p/>
175     * <p>This is useful when logic based on connect failure is needed without having to wrap the handler,
176     * for example, in case of retries across several nodes.
177     */

194    public <T extends TransportResponse> void sendRequest(final DiscoveryNode node, final String action, final TransportRequest request,
195                                                          final TransportResponseHandler<T> handler) throws TransportException {
196        sendRequest(node, action, request, TransportRequestOptions.EMPTY, handler);
197    }

199    public <T extends TransportResponse> void sendRequest(final DiscoveryNode node, final String action, final TransportRequest request,
200                                                          final TransportRequestOptions options, final TransportResponseHandler<T> handler) throws TransportException {
201        final long requestId = newRequestId();
202        TimeoutHandler timeoutHandler = null;
203        try {
204            if (options.timeout() != null) {
205                timeoutHandler = new TimeoutHandler(requestId);
206                timeoutHandler.future = threadPool.schedule(options.timeout(), ThreadPool.Names.GENERIC, timeoutHandler);
207            }
208            clientHandlers.put(requestId, new RequestHolder<T>(handler, node, action, timeoutHandler));
209            transport.sendRequest(node, requestId, action, request, options);
210        } catch (final Exception e) {
211            // usually happen either because we failed to connect to the node
212            // or because we failed serializing the message
213            clientHandlers.remove(requestId);
214            if (timeoutHandler != null) {
215                timeoutHandler.future.cancel(false);
216            }
217            if (throwConnectException) {
218                if (e instanceof ConnectTransportException) {
219                    throw (ConnectTransportException) e;
220                }
221            }
222            // callback that an exception happened, but on a different thread since we don't
223            // want handlers to worry about stack overflows
224            final SendRequestTransportException sendRequestException = new SendRequestTransportException(node, action, e);
225            threadPool.executor(ThreadPool.Names.GENERIC).execute(new Runnable() {
226                @Override
227                public void run() {
228                    handler.handleException(sendRequestException);
229                }
230            });
231        }
232    }
233
234    private long newRequestId() {
235        return requestIds.getAndIncrement();

242    public void registerHandler(String action, TransportRequestHandler handler) {
243        synchronized (serverHandlersMutex) {
244            TransportRequestHandler handlerReplaced = serverHandlers.get(action);
245            serverHandlers = MapBuilder.newMapBuilder(serverHandlers).put(action, handler).immutableMap();
246            if (handlerReplaced != null) {
247                logger.warn("Registered two transport handlers for action {}, handlers: {}, {}", action, handler, handlerReplaced);
248            }
249        }
250    }
251
252    public void removeHandler(String action) {
253        synchronized (serverHandlersMutex) {
254            serverHandlers = MapBuilder.newMapBuilder(serverHandlers).remove(action).immutableMap();
255        }
256    }

258    class Adapter implements TransportServiceAdapter {
259
260        final MeanMetric rxMetric = new MeanMetric();
261        final MeanMetric txMetric = new MeanMetric();
262
263        @Override
264        public void received(long size) {
265            rxMetric.inc(size);
266        }
267
268        @Override
269        public void sent(long size) {
270            txMetric.inc(size);
271        }
272
273        @Override
274        public TransportRequestHandler handler(String action) {
275            return serverHandlers.get(action);
276        }
277
278        @Override
279        public TransportResponseHandler remove(long requestId) {
280            RequestHolder holder = clientHandlers.remove(requestId);
281            if (holder == null) {
282                // lets see if its in the timeout holder
283                TimeoutInfoHolder timeoutInfoHolder = timeoutInfoHandlers.remove(requestId);
284                if (timeoutInfoHolder != null) {
285                    long time = System.currentTimeMillis();
286                    logger.warn("Received response for a request that has timed out, sent [{}ms] ago, timed out [{}ms] ago, action [{}], node [{}], id [{}]", time - timeoutInfoHolder.sentTime(), time - timeoutInfoHolder.timeoutTime(), timeoutInfoHolder.action(), timeoutInfoHolder.node(), requestId);
287                } else {
288                    logger.warn("Transport response handler not found of id [{}]", requestId);
289                }
290                return null;
291            }
292            holder.cancel();
293            return holder.handler();
294        }
295
296        @Override
297        public void raiseNodeConnected(final DiscoveryNode node) {
298            threadPool.generic().execute(new Runnable() {
299                @Override
300                public void run() {
301                    for (TransportConnectionListener connectionListener : connectionListeners) {
302                        connectionListener.onNodeConnected(node);
303                    }
304                }
305            });
306        }
307
308        @Override
309        public void raiseNodeDisconnected(final DiscoveryNode node) {
310            if (stoppedOrClosed) {
311                return;
312            }
313            threadPool.generic().execute(new Runnable() {
314                @Override
315                public void run() {
316                    for (TransportConnectionListener connectionListener : connectionListeners) {
317                        connectionListener.onNodeDisconnected(node);
318                    }
319                    // node got disconnected, raise disconnection on possible ongoing handlers
320                    for (Map.Entry<Long, RequestHolder> entry : clientHandlers.entrySet()) {
321                        RequestHolder holder = entry.getValue();
322                        if (holder.node().equals(node)) {
323                            final RequestHolder holderToNotify = clientHandlers.remove(entry.getKey());
324                            if (holderToNotify != null) {
325                                // callback that an exception happened, but on a different thread since we don't
326                                // want handlers to worry about stack overflows
327                                threadPool.generic().execute(new Runnable() {
328                                    @Override
329                                    public void run() {
330                                        holderToNotify.handler().handleException(new NodeDisconnectedException(node, holderToNotify.action()));
331                                    }
332                                });
333                            }
334                        }
335                    }
336                }
337            });
338        }
339    }

341    class TimeoutHandler implements Runnable {
342
343        private final long requestId;
344
345        private final long sentTime = System.currentTimeMillis();
346
347        ScheduledFuture future;
348
349        TimeoutHandler(long requestId) {
350            this.requestId = requestId;
351        }
352
353        public long sentTime() {
354            return sentTime;
355        }
356
357        @Override
358        public void run() {
359            if (future.isCancelled()) {
360                return;
361            }
362            final RequestHolder holder = clientHandlers.remove(requestId);
363            if (holder != null) {
364                // add it to the timeout information holder, in case we are going to get a response later
365                long timeoutTime = System.currentTimeMillis();
366                timeoutInfoHandlers.put(requestId, new TimeoutInfoHolder(holder.node(), holder.action(), sentTime, timeoutTime));
367                holder.handler().handleException(new ReceiveTimeoutTransportException(holder.node(), holder.action(), "request_id [" + requestId + "] timed out after [" + (timeoutTime - sentTime) + "ms]"));
368            }
369        }
370    }

373    static class TimeoutInfoHolder {
374
375        private final DiscoveryNode node;
376
377        private final String action;
378
379        private final long sentTime;
380
381        private final long timeoutTime;
382
383        TimeoutInfoHolder(DiscoveryNode node, String action, long sentTime, long timeoutTime) {
384            this.node = node;
385            this.action = action;
386            this.sentTime = sentTime;
387            this.timeoutTime = timeoutTime;
388        }

390        public DiscoveryNode node() {
391            return node;
392        }
393
394        public String action() {
395            return action;
396        }

400        }

407    static class RequestHolder<T extends TransportResponse> {
408
409        private final TransportResponseHandler<T> handler;
410
411        private final DiscoveryNode node;
412
413        private final String action;
414
415        private final TimeoutHandler timeout;
416
417        RequestHolder(TransportResponseHandler<T> handler, DiscoveryNode node, String action, TimeoutHandler timeout) {
418            this.handler = handler;
419            this.node = node;
420            this.action = action;
421            this.timeout = timeout;
422        }
423
424        public TransportResponseHandler<T> handler() {
425            return handler;
426        }
427
428        public DiscoveryNode node() {
429            return this.node;
430        }
431
432        public String action() {
433            return this.action;
434        }
435
436        public void cancel() {
437            if (timeout != null) {
438                timeout.future.cancel(false);
439            }
440        }
