020package org.elasticsearch.transport;
021
022import org.elasticsearch.common.io.stream.StreamInput;
023import org.elasticsearch.common.io.stream.StreamOutput;
024import org.elasticsearch.common.io.stream.Streamable;
025import org.elasticsearch.common.unit.ByteSizeValue;
026import org.elasticsearch.common.xcontent.ToXContent;
027import org.elasticsearch.common.xcontent.XContentBuilder;
028import org.elasticsearch.common.xcontent.XContentBuilderString;
029
030import java.io.IOException;
031
032public class TransportStats implements Streamable, ToXContent {
033
034    private long serverOpen;
035    private long rxCount;
036    private long rxSize;
037    private long txCount;
038    private long txSize;
039
040    TransportStats() {
041
042    }
043
044    public TransportStats(long serverOpen, long rxCount, long rxSize, long txCount, long txSize) {
045        this.serverOpen = serverOpen;
046        this.rxCount = rxCount;
047        this.rxSize = rxSize;
048        this.txCount = txCount;
049        this.txSize = txSize;
050    }
051
052    public long serverOpen() {
053        return this.serverOpen;
054    }
055
056    public long getServerOpen() {
057        return serverOpen();
058    }
059
060    public long rxCount() {
061        return rxCount;
062    }
063
064    public long getRxCount() {
065        return rxCount();
066    }
067
068    public ByteSizeValue rxSize() {
069        return new ByteSizeValue(rxSize);
070    }
071
072    public ByteSizeValue getRxSize() {
073        return rxSize();
074    }
075
076    public long txCount() {
077        return txCount;
078    }
079
080    public long getTxCount() {
081        return txCount();
082    }
083
084    public ByteSizeValue txSize() {
085        return new ByteSizeValue(txSize);
086    }
087
088    public ByteSizeValue getTxSize() {
089        return txSize();
090    }
091
092    public static TransportStats readTransportStats(StreamInput in) throws IOException {
093        TransportStats stats = new TransportStats();
094        stats.readFrom(in);
095        return stats;
096    }
097
098    @Override
099    public void readFrom(StreamInput in) throws IOException {
100        serverOpen = in.readVLong();
101        rxCount = in.readVLong();
102        rxSize = in.readVLong();
103        txCount = in.readVLong();
104        txSize = in.readVLong();
105    }
106
107    @Override
108    public void writeTo(StreamOutput out) throws IOException {
109        out.writeVLong(serverOpen);
110        out.writeVLong(rxCount);
111        out.writeVLong(rxSize);
112        out.writeVLong(txCount);
113        out.writeVLong(txSize);
114    }
115
116    @Override
117    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
118        builder.startObject(Fields.TRANSPORT);
119        builder.field(Fields.SERVER_OPEN, serverOpen);
120        builder.field(Fields.RX_COUNT, rxCount);
121        builder.field(Fields.RX_SIZE, rxSize().toString());
122        builder.field(Fields.RX_SIZE_IN_BYTES, rxSize);
123        builder.field(Fields.TX_COUNT, txCount);
124        builder.field(Fields.TX_SIZE, txSize().toString());
125        builder.field(Fields.TX_SIZE_IN_BYTES, txSize);
126        builder.endObject();
127        return builder;
128    }
129
130    static final class Fields {
131        static final XContentBuilderString TRANSPORT = new XContentBuilderString("transport");
132        static final XContentBuilderString SERVER_OPEN = new XContentBuilderString("server_open");
133        static final XContentBuilderString RX_COUNT = new XContentBuilderString("rx_count");
134        static final XContentBuilderString RX_SIZE = new XContentBuilderString("rx_size");
135        static final XContentBuilderString RX_SIZE_IN_BYTES = new XContentBuilderString("rx_size_in_bytes");
136        static final XContentBuilderString TX_COUNT = new XContentBuilderString("tx_count");
137        static final XContentBuilderString TX_SIZE = new XContentBuilderString("tx_size");
138        static final XContentBuilderString TX_SIZE_IN_BYTES = new XContentBuilderString("tx_size_in_bytes");
139    }
140}
