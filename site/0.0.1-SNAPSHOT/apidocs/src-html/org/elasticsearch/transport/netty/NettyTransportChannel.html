020package org.elasticsearch.transport.netty;
021
022import org.elasticsearch.Version;
023import org.elasticsearch.common.bytes.NettyBytesArray;
024import org.elasticsearch.common.compress.NettyCompressorFactory;
025import org.elasticsearch.common.io.ThrowableObjectOutputStream;
026import org.elasticsearch.common.io.stream.NettyBytesStreamOutput;
027import org.elasticsearch.common.io.stream.NettyCachedStreamOutput;
028import org.elasticsearch.common.io.stream.StreamOutput;
029import org.elasticsearch.transport.*;
030import org.elasticsearch.transport.support.TransportStatus;
031import org.jboss.netty.buffer.ChannelBuffer;
032import org.jboss.netty.channel.Channel;
033import org.jboss.netty.channel.ChannelFuture;
034
035import java.io.IOException;
036import java.io.NotSerializableException;
037
038/**
039 *
040 */
041public class NettyTransportChannel implements TransportChannel {
042
043    private final NettyTransport transport;
044
045    private final Version version;
046
047    private final String action;
048
049    private final Channel channel;
050
051    private final long requestId;
052
053    public NettyTransportChannel(NettyTransport transport, String action, Channel channel, long requestId, Version version) {
054        this.version = version;
055        this.transport = transport;
056        this.action = action;
057        this.channel = channel;
058        this.requestId = requestId;
059    }
060
061    @Override
062    public String action() {
063        return this.action;
064    }
065
066    @Override
067    public void sendResponse(TransportResponse response) throws IOException {
068        sendResponse(response, TransportResponseOptions.EMPTY);
069    }
070
071    @Override
072    public void sendResponse(TransportResponse response, TransportResponseOptions options) throws IOException {
073        if (transport.compress()) {
074            options.withCompress(true);
075        }
076        NettyCachedStreamOutput.Entry cachedEntry = NettyCachedStreamOutput.popEntry();
077
078        byte status = 0;
079        status = TransportStatus.setResponse(status);
080
081        if (options.compress()) {
082            status = TransportStatus.setCompress(status);
083            cachedEntry.bytes().skip(NettyHeader.HEADER_SIZE);
084            StreamOutput stream = cachedEntry.handles(NettyCompressorFactory.defaultCompressor());
085            stream.setVersion(version);
086            response.writeTo(stream);
087            stream.close();
088        } else {
089            StreamOutput stream = cachedEntry.handles();
090            stream.setVersion(version);
091            cachedEntry.bytes().skip(NettyHeader.HEADER_SIZE);
092            response.writeTo(stream);
093            stream.close();
094        }
095        ChannelBuffer buffer = ((NettyBytesArray)cachedEntry.bytes().bytes()).toChannelBuffer();
096        NettyHeader.writeHeader(buffer, requestId, status, version);
097        ChannelFuture future = channel.write(buffer);
098        future.addListener(new ClientNettyTransport.CacheFutureListener(cachedEntry));
099    }
100
101    @Override
102    public void sendResponse(Throwable error) throws IOException {
103        NettyCachedStreamOutput.Entry cachedEntry = NettyCachedStreamOutput.popEntry();
104        NettyBytesStreamOutput stream;
105        try {
106            stream = cachedEntry.bytes();
107            cachedEntry.bytes().skip(NettyHeader.HEADER_SIZE);
108            RemoteTransportException tx = new RemoteTransportException(""/*transport.nodeName()*/, transport.wrapAddress(channel.getLocalAddress()), action, error);
109            ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
110            too.writeObject(tx);
111            too.close();
112        } catch (NotSerializableException e) {
113            cachedEntry.reset();
114            stream = cachedEntry.bytes();
115            cachedEntry.bytes().skip(NettyHeader.HEADER_SIZE);
116            RemoteTransportException tx = new RemoteTransportException(""/*transport.nodeName()*/, transport.wrapAddress(channel.getLocalAddress()), action, new NotSerializableTransportException(error));
117            ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
118            too.writeObject(tx);
119            too.close();
120        }
121
122        byte status = 0;
123        status = TransportStatus.setResponse(status);
124        status = TransportStatus.setError(status);
125
126        ChannelBuffer buffer = ((NettyBytesArray)cachedEntry.bytes().bytes()).toChannelBuffer();
127        NettyHeader.writeHeader(buffer, requestId, status, version);
128        ChannelFuture future = channel.write(buffer);
129        future.addListener(new ClientNettyTransport.CacheFutureListener(cachedEntry));
130    }
131}
