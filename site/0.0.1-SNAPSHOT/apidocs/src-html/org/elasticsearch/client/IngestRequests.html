020package org.elasticsearch.client;
021
022import org.elasticsearch.action.bulk.BulkRequest;
023import org.elasticsearch.action.delete.DeleteRequest;
024import org.elasticsearch.action.get.GetRequest;
025import org.elasticsearch.action.index.IndexRequest;
026import org.elasticsearch.common.xcontent.XContentType;
027
028/**
029 * A handy one stop shop for creating indexer requests (make sure to import static this class).
030 */
031public class IngestRequests {
032
033    /**
034     * The content type used to generate request builders (query / search).
035     */
036    public static XContentType CONTENT_TYPE = XContentType.SMILE;
037
038    /**
039     * The default content type to use to generate source documents when indexing.
040     */
041    public static XContentType INDEX_CONTENT_TYPE = XContentType.JSON;
042
043    public static IndexRequest indexRequest() {
044        return new IndexRequest();
045    }
046
047    /**
048     * Create an index request against a specific index. Note the {@link IndexRequest#type(String)} must be
049     * set as well and optionally the {@link IndexRequest#id(String)}.
050     *
051     * @param index The index name to index the request against
052     * @return The index request
053     * @see org.elasticsearch.client.Client#index(org.elasticsearch.action.index.IndexRequest)
054     */
055    public static IndexRequest indexRequest(String index) {
056        return new IndexRequest(index);
057    }
058
059    /**
060     * Creates a delete request against a specific index. Note the {@link DeleteRequest#type(String)} and
061     * {@link DeleteRequest#id(String)} must be set.
062     *
063     * @param index The index name to delete from
064     * @return The delete request
065     * @see org.elasticsearch.client.Client#delete(org.elasticsearch.action.delete.DeleteRequest)
066     */
067    public static DeleteRequest deleteRequest(String index) {
068        return new DeleteRequest(index);
069    }
070
071    /**
072     * Creats a new bulk request.
073     */
074    public static BulkRequest bulkRequest() {
075        return new BulkRequest();
076    }
077
078    /**
079     * Creates a get request to get the JSON source from an index based on a type and id. Note, the
080     * {@link GetRequest#type(String)} and {@link GetRequest#id(String)} must be set.
081     *
082     * @param index The index to get the JSON source from
083     * @return The get request
084     * @see org.elasticsearch.client.Client#get(org.elasticsearch.action.get.GetRequest)
085     */
086    public static GetRequest getRequest(String index) {
087        return new GetRequest(index);
088    }
089
090}
