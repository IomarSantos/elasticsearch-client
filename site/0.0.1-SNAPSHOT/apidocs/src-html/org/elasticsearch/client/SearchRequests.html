020package org.elasticsearch.client;
021
022import org.elasticsearch.action.count.CountRequest;
023import org.elasticsearch.action.deletebyquery.DeleteByQueryRequest;
024import org.elasticsearch.action.mlt.MoreLikeThisRequest;
025import org.elasticsearch.action.search.SearchRequest;
026import org.elasticsearch.action.search.SearchScrollRequest;
027import org.elasticsearch.common.xcontent.XContentType;
028
029/**
030 * A handy one stop shop for creating requests (make sure to import static this class).
031 */
032public class SearchRequests {
033
034    /**
035     * The content type used to generate request builders (query / search).
036     */
037    public static XContentType CONTENT_TYPE = XContentType.SMILE;
038
039    /**
040     * The default content type to use to generate source documents when indexing.
041     */
042    public static XContentType INDEX_CONTENT_TYPE = XContentType.JSON;
043
044    /**
045     * Creates a delete by query request. Note, the query itself must be set either by setting the JSON source
046     * of the query, or by using a {@link org.elasticsearch.index.query.QueryBuilder} (using {@link org.elasticsearch.index.query.QueryBuilders}).
047     *
048     * @param indices The indices the delete by query against. Use <tt>null</tt> or <tt>_all</tt> to execute against all indices
049     * @return The delete by query request
050     * @see org.elasticsearch.client.Client#deleteByQuery(org.elasticsearch.action.deletebyquery.DeleteByQueryRequest)
051     */
052    public static DeleteByQueryRequest deleteByQueryRequest(String... indices) {
053        return new DeleteByQueryRequest(indices);
054    }
055
056    /**
057     * Creates a count request which counts the hits matched against a query. Note, the query itself must be set
058     * either using the JSON source of the query, or using a {@link org.elasticsearch.index.query.QueryBuilder} (using {@link org.elasticsearch.index.query.QueryBuilders}).
059     *
060     * @param indices The indices to count matched documents against a query. Use <tt>null</tt> or <tt>_all</tt> to execute against all indices
061     * @return The count request
062     * @see org.elasticsearch.client.Client#count(org.elasticsearch.action.count.CountRequest)
063     */
064    public static CountRequest countRequest(String... indices) {
065        return new CountRequest(indices);
066    }
067
068    /**
069     * More like this request represents a request to search for documents that are "like" the provided (fetched)
070     * document.
071     *
072     * @param index The index to load the document from
073     * @return The more like this request
074     * @see org.elasticsearch.client.Client#moreLikeThis(org.elasticsearch.action.mlt.MoreLikeThisRequest)
075     */
076    public static MoreLikeThisRequest moreLikeThisRequest(String index) {
077        return new MoreLikeThisRequest(index);
078    }
079
080    /**
081     * Creates a search request against one or more indices. Note, the search source must be set either using the
082     * actual JSON search source, or the {@link org.elasticsearch.search.builder.SearchSourceBuilder}.
083     *
084     * @param indices The indices to search against. Use <tt>null</tt> or <tt>_all</tt> to execute against all indices
085     * @return The search request
086     * @see org.elasticsearch.client.Client#search(org.elasticsearch.action.search.SearchRequest)
087     */
088    public static SearchRequest searchRequest(String... indices) {
089        return new SearchRequest(indices);
090    }
091
092    /**
093     * Creates a search scroll request allowing to continue searching a previous search request.
094     *
095     * @param scrollId The scroll id representing the scrollable search
096     * @return The search scroll request
097     * @see org.elasticsearch.client.Client#searchScroll(org.elasticsearch.action.search.SearchScrollRequest)
098     */
099    public static SearchScrollRequest searchScrollRequest(String scrollId) {
100        return new SearchScrollRequest(scrollId);
101    }
102
103}
