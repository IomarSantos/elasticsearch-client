020package org.elasticsearch.client.support;
021
022import org.elasticsearch.action.Action;
023import org.elasticsearch.action.ActionFuture;
024import org.elasticsearch.action.ActionListener;
025import org.elasticsearch.action.ActionRequest;
026import org.elasticsearch.action.ActionRequestBuilder;
027import org.elasticsearch.action.ActionResponse;
028import org.elasticsearch.action.count.CountAction;
029import org.elasticsearch.action.count.CountRequest;
030import org.elasticsearch.action.count.CountRequestBuilder;
031import org.elasticsearch.action.count.CountResponse;
032import org.elasticsearch.action.deletebyquery.DeleteByQueryAction;
033import org.elasticsearch.action.deletebyquery.DeleteByQueryRequest;
034import org.elasticsearch.action.deletebyquery.DeleteByQueryRequestBuilder;
035import org.elasticsearch.action.deletebyquery.DeleteByQueryResponse;
036import org.elasticsearch.action.explain.ExplainAction;
037import org.elasticsearch.action.explain.ExplainRequest;
038import org.elasticsearch.action.explain.ExplainRequestBuilder;
039import org.elasticsearch.action.explain.ExplainResponse;
040import org.elasticsearch.action.mlt.MoreLikeThisAction;
041import org.elasticsearch.action.mlt.MoreLikeThisRequest;
042import org.elasticsearch.action.mlt.MoreLikeThisRequestBuilder;
043import org.elasticsearch.action.search.MultiSearchAction;
044import org.elasticsearch.action.search.MultiSearchRequest;
045import org.elasticsearch.action.search.MultiSearchRequestBuilder;
046import org.elasticsearch.action.search.MultiSearchResponse;
047import org.elasticsearch.action.search.SearchAction;
048import org.elasticsearch.action.search.SearchRequest;
049import org.elasticsearch.action.search.SearchRequestBuilder;
050import org.elasticsearch.action.search.SearchResponse;
051import org.elasticsearch.action.search.SearchScrollAction;
052import org.elasticsearch.action.search.SearchScrollRequest;
053import org.elasticsearch.action.search.SearchScrollRequestBuilder;
054import org.elasticsearch.client.GenericClient;
055import org.elasticsearch.client.SearchClient;
056
057public abstract class AbstractSearchClient implements SearchClient {
058
059    @Override
060    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, SearchClient extends GenericClient> 
061            RequestBuilder prepareExecute(final Action<Request, Response, RequestBuilder, SearchClient> action) {
062        return action.newRequestBuilder((SearchClient)this);
063    }
064
065    @Override
066    public ActionFuture<DeleteByQueryResponse> deleteByQuery(final DeleteByQueryRequest request) {
067        return execute(DeleteByQueryAction.INSTANCE, request);
068    }
069
070    @Override
071    public void deleteByQuery(final DeleteByQueryRequest request, final ActionListener<DeleteByQueryResponse> listener) {
072        execute(DeleteByQueryAction.INSTANCE, request, listener);
073    }
074
075    @Override
076    public DeleteByQueryRequestBuilder prepareDeleteByQuery(String... indices) {
077        return new DeleteByQueryRequestBuilder(this).setIndices(indices);
078    }
079
080    @Override
081    public ActionFuture<SearchResponse> search(final SearchRequest request) {
082        return execute(SearchAction.INSTANCE, request);
083    }
084
085    @Override
086    public void search(final SearchRequest request, final ActionListener<SearchResponse> listener) {
087        execute(SearchAction.INSTANCE, request, listener);
088    }
089
090    @Override
091    public SearchRequestBuilder prepareSearch(String... indices) {
092        return new SearchRequestBuilder(this).setIndices(indices);
093    }
094
095    @Override
096    public ActionFuture<SearchResponse> searchScroll(final SearchScrollRequest request) {
097        return execute(SearchScrollAction.INSTANCE, request);
098    }
099
100    @Override
101    public void searchScroll(final SearchScrollRequest request, final ActionListener<SearchResponse> listener) {
102        execute(SearchScrollAction.INSTANCE, request, listener);
103    }
104
105    @Override
106    public SearchScrollRequestBuilder prepareSearchScroll(String scrollId) {
107        return new SearchScrollRequestBuilder(this, scrollId);
108    }
109
110    @Override
111    public ActionFuture<MultiSearchResponse> multiSearch(MultiSearchRequest request) {
112        return execute(MultiSearchAction.INSTANCE, request);
113    }
114
115    @Override
116    public void multiSearch(MultiSearchRequest request, ActionListener<MultiSearchResponse> listener) {
117        execute(MultiSearchAction.INSTANCE, request, listener);
118    }
119
120    @Override
121    public MultiSearchRequestBuilder prepareMultiSearch() {
122        return new MultiSearchRequestBuilder(this);
123    }
124
125    @Override
126    public ActionFuture<CountResponse> count(final CountRequest request) {
127        return execute(CountAction.INSTANCE, request);
128    }
129
130    @Override
131    public void count(final CountRequest request, final ActionListener<CountResponse> listener) {
132        execute(CountAction.INSTANCE, request, listener);
133    }
134
135    @Override
136    public CountRequestBuilder prepareCount(String... indices) {
137        return new CountRequestBuilder(this).setIndices(indices);
138    }
139
140    @Override
141    public ActionFuture<SearchResponse> moreLikeThis(final MoreLikeThisRequest request) {
142        return execute(MoreLikeThisAction.INSTANCE, request);
143    }
144
145    @Override
146    public void moreLikeThis(final MoreLikeThisRequest request, final ActionListener<SearchResponse> listener) {
147        execute(MoreLikeThisAction.INSTANCE, request, listener);
148    }
149
150    @Override
151    public MoreLikeThisRequestBuilder prepareMoreLikeThis(String index, String type, String id) {
152        return new MoreLikeThisRequestBuilder(this, index, type, id);
153    }
154
155    @Override
156    public ExplainRequestBuilder prepareExplain(String index, String type, String id) {
157        return new ExplainRequestBuilder(this, index, type, id);
158    }
159
160    @Override
161    public ActionFuture<ExplainResponse> explain(ExplainRequest request) {
162        return execute(ExplainAction.INSTANCE, request);
163    }
164
165    @Override
166    public void explain(ExplainRequest request, ActionListener<ExplainResponse> listener) {
167        execute(ExplainAction.INSTANCE, request, listener);
168    }
169}
