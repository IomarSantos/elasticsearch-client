058    /**
059     * The content type used to generate request builders (query / search).

068    /**
069     * Creates a delete by query request. Note, the query itself must be set either by setting the JSON source
070     * of the query, or by using a {@link org.elasticsearch.index.query.QueryBuilder} (using {@link org.elasticsearch.index.query.QueryBuilders}).
071     *
072     * @param indices The indices the delete by query against. Use <tt>null</tt> or <tt>_all</tt> to execute against all indices
073     * @return The delete by query request
074     * @see org.elasticsearch.client.Client#deleteByQuery(org.elasticsearch.action.deletebyquery.DeleteByQueryRequest)
075     */
076    public static DeleteByQueryRequest deleteByQueryRequest(String... indices) {
077        return new DeleteByQueryRequest(indices);
078    }
079
080    /**
081     * Creates a count request which counts the hits matched against a query. Note, the query itself must be set
082     * either using the JSON source of the query, or using a {@link org.elasticsearch.index.query.QueryBuilder} (using {@link org.elasticsearch.index.query.QueryBuilders}).
083     *
084     * @param indices The indices to count matched documents against a query. Use <tt>null</tt> or <tt>_all</tt> to execute against all indices
085     * @return The count request

087     */

089        return new CountRequest(indices);
090    }
