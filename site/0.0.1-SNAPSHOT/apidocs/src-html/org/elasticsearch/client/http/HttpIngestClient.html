020package org.elasticsearch.client.http;
021
022import com.google.common.collect.ImmutableList;
023import com.google.common.collect.Sets;
024
025import org.elasticsearch.ElasticSearchException;
026import org.elasticsearch.action.Action;
027import org.elasticsearch.action.ActionFuture;
028import org.elasticsearch.action.ActionListener;
029import org.elasticsearch.action.ActionRequest;
030import org.elasticsearch.action.ActionRequestBuilder;
031import org.elasticsearch.action.ActionResponse;
032import org.elasticsearch.action.support.HttpClient;
033import org.elasticsearch.client.GenericClient;
034import org.elasticsearch.client.http.support.InternalHttpIngestClient;
035import org.elasticsearch.client.support.AbstractIngestClient;
036import org.elasticsearch.client.internal.InternalClientSettingsPreparer;
037import org.elasticsearch.common.collect.Tuple;
038import org.elasticsearch.common.compress.BasicCompressorFactory;
039import org.elasticsearch.common.io.CachedStreams;
040import org.elasticsearch.common.settings.ImmutableSettings;
041import org.elasticsearch.common.settings.Settings;
042import org.elasticsearch.common.transport.TransportAddress;
043import org.elasticsearch.common.util.concurrent.ThreadLocals;
044import org.elasticsearch.env.ClientEnvironment;
045import org.elasticsearch.threadpool.ThreadPool;
046import org.elasticsearch.threadpool.client.ClientThreadPool;
047
048import java.util.Arrays;
049import java.util.Set;
050import java.util.concurrent.TimeUnit;
051
052import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
053
054
055public class HttpIngestClient extends AbstractIngestClient {
056
057    private final Settings settings;
058
059    private final ClientEnvironment environment;
060
061    private final ThreadPool threadPool;
062
063    
064    private final HttpClient internalClient;
065    
066    private Set<TransportAddress> addresses;
067    
068    public HttpIngestClient() throws ElasticSearchException {
069        this(ImmutableSettings.Builder.EMPTY_SETTINGS, true);
070    }
071
072    public HttpIngestClient(Settings settings) {
073        this(settings, true);
074    }
075
076    public HttpIngestClient(Settings.Builder settings) {
077        this(settings.build(), true);
078    }
079
080    public HttpIngestClient(Settings.Builder settings, boolean loadConfigSettings) throws ElasticSearchException {
081        this(settings.build(), loadConfigSettings);
082    }
083
084    public HttpIngestClient(Settings pSettings, boolean loadConfigSettings) throws ElasticSearchException {
085        Tuple<Settings, ClientEnvironment> tuple = InternalClientSettingsPreparer.prepareSettings(pSettings, loadConfigSettings);
086        // some defaults, not really needed, just for TransportClient compatibility
087        this.settings = settingsBuilder().put(tuple.v1())
088                .put("network.server", false)
089                .put("node.client", true)
090                .build();
091        this.environment = tuple.v2();
092        this.threadPool = new ClientThreadPool();
093        this.addresses = Sets.newHashSet();
094        this.internalClient = new InternalHttpIngestClient(settings);
095        BasicCompressorFactory.configure(settings);
096    }
097
098    public ImmutableList<TransportAddress> transportAddresses() {
099        return ImmutableList.copyOf(addresses);
100    }
101
102    public HttpIngestClient addTransportAddress(TransportAddress address) {
103        addresses.add(address);
104        return this;
105    }
106
107    public HttpIngestClient addTransportAddresses(TransportAddress... address) {
108        addresses.addAll(Arrays.asList(address));
109        return this;
110    }
111
112    public HttpIngestClient removeTransportAddress(TransportAddress address) {
113        addresses.remove(address);
114        return this;
115    }
116
117    @Override
118    public ThreadPool threadPool() {
119        return threadPool;
120    }
121    
122    @Override
123    public void close() {
124        internalClient.close();
125        threadPool.shutdown();
126        try {
127            threadPool.awaitTermination(10, TimeUnit.SECONDS);
128        } catch (InterruptedException e) {
129            // ignore
130        }
131        try {
132            threadPool.shutdownNow();
133        } catch (Exception e) {
134            // ignore
135        }
136        CachedStreams.clear();
137        ThreadLocals.clearReferencesThreadLocals();
138    }
139
140    @Override
141    public Settings settings() {
142        return this.settings;
143    }
144
145    @Override
146    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
147            ActionFuture<Response> execute(Action<Request, Response, RequestBuilder, Client> action, Request request) {
148        return internalClient.execute(action, request);
149    }
150
151    @Override
152    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, SearchClient extends GenericClient> 
153            void execute(Action<Request, Response, RequestBuilder, SearchClient> action, Request request, ActionListener<Response> listener) {
154       internalClient.execute(action, request);
155    }
156}
