001package org.elasticsearch.client.http.support;
002
003import org.elasticsearch.action.Action;
004import org.elasticsearch.action.ActionFuture;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.ActionRequest;
007import org.elasticsearch.action.ActionRequestBuilder;
008import org.elasticsearch.action.ActionResponse;
009import org.elasticsearch.action.HttpIngestActionModule;
010import org.elasticsearch.action.support.HttpAction;
011import org.elasticsearch.action.support.HttpClient;
012import org.elasticsearch.client.GenericClient;
013import org.elasticsearch.common.settings.Settings;
014
015public class InternalHttpIngestClient extends HttpClient {
016
017    private final HttpIngestActionModule actions = new HttpIngestActionModule();
018    
019    public InternalHttpIngestClient(Settings settings) {
020        super(settings);
021    }
022    
023    @Override
024    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
025            ActionFuture<Response> execute(Action<Request, Response, RequestBuilder, Client> action, Request request) {
026        HttpAction<Request,Response> httpAction = actions.getAction(action.name());
027        return httpAction.execute(this, request);
028    }
029
030    @Override
031    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
032            void execute(Action<Request, Response, RequestBuilder, Client> action, Request request, ActionListener<Response> listener) {
033        HttpAction<Request, Response> httpAction = actions.getAction(action.name());
034        httpAction.execute(this, request, listener);
035    }
036
037    @Override
038    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
039            RequestBuilder prepareExecute(Action<Request, Response, RequestBuilder, Client> action) {
040        return action.newRequestBuilder((Client)this);
041    }
042    
043}
