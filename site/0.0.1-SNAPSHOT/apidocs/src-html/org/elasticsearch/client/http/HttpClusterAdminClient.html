020package org.elasticsearch.client.http;
021
022import com.google.common.collect.ImmutableList;
023import com.google.common.collect.Sets;
024
025import org.elasticsearch.ElasticSearchException;
026import org.elasticsearch.action.ActionFuture;
027import org.elasticsearch.action.ActionListener;
028import org.elasticsearch.action.ActionRequest;
029import org.elasticsearch.action.ActionRequestBuilder;
030import org.elasticsearch.action.ActionResponse;
031import org.elasticsearch.action.HttpClusterAdminActionModule;
032import org.elasticsearch.action.admin.cluster.ClusterAction;
033import org.elasticsearch.action.support.HttpAction;
034import org.elasticsearch.action.support.HttpClient;
035import org.elasticsearch.client.http.support.InternalHttpClusterAdminClient;
036import org.elasticsearch.client.internal.InternalClientSettingsPreparer;
037import org.elasticsearch.client.support.AbstractClusterAdminClient;
038import org.elasticsearch.common.collect.Tuple;
039import org.elasticsearch.common.compress.BasicCompressorFactory;
040import org.elasticsearch.common.io.CachedStreams;
041import org.elasticsearch.common.settings.ImmutableSettings;
042import org.elasticsearch.common.settings.Settings;
043import org.elasticsearch.common.transport.TransportAddress;
044import org.elasticsearch.common.util.concurrent.ThreadLocals;
045import org.elasticsearch.env.ClientEnvironment;
046import org.elasticsearch.threadpool.ThreadPool;
047import org.elasticsearch.threadpool.client.ClientThreadPool;
048
049import java.util.Arrays;
050import java.util.Set;
051import java.util.concurrent.TimeUnit;
052
053import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
054
055public class HttpClusterAdminClient extends AbstractClusterAdminClient {
056
057    private final Settings settings;
058
059    private final ClientEnvironment environment;
060
061    private final ThreadPool threadPool;
062
063    private final HttpClusterAdminActionModule actions = new HttpClusterAdminActionModule();
064    
065    private final HttpClient internalClient;
066    
067    private Set<TransportAddress> addresses;
068    
069    public HttpClusterAdminClient() throws ElasticSearchException {
070        this(ImmutableSettings.Builder.EMPTY_SETTINGS, true);
071    }
072
073    public HttpClusterAdminClient(Settings settings) {
074        this(settings, true);
075    }
076
077    public HttpClusterAdminClient(Settings.Builder settings) {
078        this(settings.build(), true);
079    }
080
081    public HttpClusterAdminClient(Settings.Builder settings, boolean loadConfigSettings) throws ElasticSearchException {
082        this(settings.build(), loadConfigSettings);
083    }
084
085    public HttpClusterAdminClient(Settings pSettings, boolean loadConfigSettings) throws ElasticSearchException {
086        Tuple<Settings, ClientEnvironment> tuple = InternalClientSettingsPreparer.prepareSettings(pSettings, loadConfigSettings);
087        // some defaults, not really needed, just for TransportClient compatibility
088        this.settings = settingsBuilder().put(tuple.v1())
089                .put("network.server", false)
090                .put("node.client", true)
091                .build();
092        this.environment = tuple.v2();
093        this.threadPool = new ClientThreadPool();
094        this.addresses = Sets.newHashSet();
095        this.internalClient = new InternalHttpClusterAdminClient(settings, actions);
096        BasicCompressorFactory.configure(settings);
097    }
098
099    public ImmutableList<TransportAddress> transportAddresses() {
100        return ImmutableList.copyOf(addresses);
101    }
102
103    public HttpClusterAdminClient addTransportAddress(TransportAddress address) {
104        addresses.add(address);
105        return this;
106    }
107
108    public HttpClusterAdminClient addTransportAddresses(TransportAddress... address) {
109        addresses.addAll(Arrays.asList(address));
110        return this;
111    }
112
113    public HttpClusterAdminClient removeTransportAddress(TransportAddress address) {
114        addresses.remove(address);
115        return this;
116    }
117
118    @Override
119    public ThreadPool threadPool() {
120        return threadPool;
121    }
122    
123    @Override
124    public void close() {
125        internalClient.close();
126        threadPool.shutdown();
127        try {
128            threadPool.awaitTermination(10, TimeUnit.SECONDS);
129        } catch (InterruptedException e) {
130            // ignore
131        }
132        try {
133            threadPool.shutdownNow();
134        } catch (Exception e) {
135            // ignore
136        }
137        CachedStreams.clear();
138        ThreadLocals.clearReferencesThreadLocals();
139    }
140
141    @Override
142    public Settings settings() {
143        return this.settings;
144    }
145
146    @Override
147    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>> 
148            ActionFuture<Response> execute(ClusterAction<Request, Response, RequestBuilder> action, Request request) {
149        HttpAction<Request,Response> httpAction = actions.getAction(action.name());
150        return httpAction.execute(internalClient, request);
151    }
152
153    @Override
154    public <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>> 
155            void execute(ClusterAction<Request, Response, RequestBuilder> action, Request request, ActionListener<Response> listener) {
156        HttpAction<Request,Response> httpAction = actions.getAction(action.name());
157        httpAction.execute(internalClient, request);
158    }
159}
