020package org.elasticsearch.client.internal;
021
022import org.elasticsearch.common.collect.Tuple;
023import org.elasticsearch.common.settings.ImmutableSettings;
024import org.elasticsearch.common.settings.Settings;
025import org.elasticsearch.env.ClientEnvironment;
026import org.elasticsearch.env.FailedToResolveConfigException;
027
028import static org.elasticsearch.common.Strings.cleanPath;
029import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
030
031/**
032 *
033 */
034public class InternalClientSettingsPreparer {
035
036    public static Tuple<Settings, ClientEnvironment> prepareSettings(Settings pSettings, boolean loadConfigSettings) {
037        // ignore this prefixes when getting properties from es. and elasticsearch.
038        String[] ignorePrefixes = new String[]{"es.default.", "elasticsearch.default."};
039        // just create enough settings to build the environment
040        ImmutableSettings.Builder settingsBuilder = settingsBuilder()
041                .put(pSettings)
042                .putProperties("elasticsearch.default.", System.getProperties())
043                .putProperties("es.default.", System.getProperties())
044                .putProperties("elasticsearch.", System.getProperties(), ignorePrefixes)
045                .putProperties("es.", System.getProperties(), ignorePrefixes)
046                .replacePropertyPlaceholders();
047
048        ClientEnvironment environment = new ClientEnvironment(settingsBuilder.build());
049
050        if (loadConfigSettings) {
051            boolean loadFromEnv = true;
052            // if its default, then load it, but also load form env
053            if (System.getProperty("es.default.config") != null) {
054                loadFromEnv = true;
055                settingsBuilder.loadFromUrl(environment.resolveConfig(System.getProperty("es.default.config")));
056            }
057            // if explicit, just load it and don't load from env
058            if (System.getProperty("es.config") != null) {
059                loadFromEnv = false;
060                settingsBuilder.loadFromUrl(environment.resolveConfig(System.getProperty("es.config")));
061            }
062            if (System.getProperty("elasticsearch.config") != null) {
063                loadFromEnv = false;
064                settingsBuilder.loadFromUrl(environment.resolveConfig(System.getProperty("elasticsearch.config")));
065            }
066            if (loadFromEnv) {
067                try {
068                    settingsBuilder.loadFromUrl(environment.resolveConfig("elasticsearch.yml"));
069                } catch (FailedToResolveConfigException e) {
070                    // ignore
071                } catch (NoClassDefFoundError e) {
072                    // ignore, no yaml
073                }
074                try {
075                    settingsBuilder.loadFromUrl(environment.resolveConfig("elasticsearch.json"));
076                } catch (FailedToResolveConfigException e) {
077                    // ignore
078                }
079                try {
080                    settingsBuilder.loadFromUrl(environment.resolveConfig("elasticsearch.properties"));
081                } catch (FailedToResolveConfigException e) {
082                    // ignore
083                }
084            }
085        }
086
087        settingsBuilder.put(pSettings)
088                .putProperties("elasticsearch.", System.getProperties(), ignorePrefixes)
089                .putProperties("es.", System.getProperties(), ignorePrefixes)
090                .replacePropertyPlaceholders();
091
092        Settings v1 = settingsBuilder.build();
093        environment = new ClientEnvironment(v1);
094
095        // put back the env settings
096        settingsBuilder = settingsBuilder().put(v1);
097        // we put back the path.logs so we can use it in the logging configuration file
098        settingsBuilder.put("path.logs", cleanPath(environment.logsFile().getAbsolutePath()));
099
100        v1 = settingsBuilder.build();
101
102        return new Tuple<Settings, ClientEnvironment>(v1, environment);
103    }
104}
