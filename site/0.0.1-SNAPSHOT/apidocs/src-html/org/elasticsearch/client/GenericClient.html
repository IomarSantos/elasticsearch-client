020package org.elasticsearch.client;
021
022import org.elasticsearch.action.Action;
023import org.elasticsearch.action.ActionFuture;
024import org.elasticsearch.action.ActionListener;
025import org.elasticsearch.action.ActionRequest;
026import org.elasticsearch.action.ActionRequestBuilder;
027import org.elasticsearch.action.ActionResponse;
028import org.elasticsearch.client.internal.InternalGenericClient;
029
030/**
031 * A client provides an interface for performing generic actions/operations against the server.
032 */
033public interface GenericClient extends InternalGenericClient {
034
035    /**
036     * Executes a generic action, denoted by an {@link Action}.
037     *
038     * @param action           The action type to execute.
039     * @param request          The action request.
040     * @param <Request>        The request type.
041     * @param <Response>       the response type.
042     * @param <RequestBuilder> The request builder type.
043     * @return A future allowing to get back the response.
044     */
045    <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
046            ActionFuture<Response> execute(final Action<Request, Response, RequestBuilder, Client> action, final Request request);
047
048    /**
049     * Executes a generic action, denoted by an {@link Action}.
050     *
051     * @param action           The action type to execute.
052     * @param request          Teh action request.
053     * @param listener         The listener to receive the response back.
054     * @param <Request>        The request type.
055     * @param <Response>       The response type.
056     * @param <RequestBuilder> The request builder type.
057     */
058    <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
059            void execute(final Action<Request, Response, RequestBuilder, Client> action, final Request request, ActionListener<Response> listener);
060
061    /**
062     * Prepares a request builder to execute, specified by {@link Action}.
063     *
064     * @param action           The action type to execute.
065     * @param <Request>        The request type.
066     * @param <Response>       The response type.
067     * @param <RequestBuilder> The request builder.
068     * @return The request builder, that can, at a later stage, execute the request.
069     */
070    <Request extends ActionRequest, Response extends ActionResponse, RequestBuilder extends ActionRequestBuilder<Request, Response, RequestBuilder>, Client extends GenericClient> 
071            RequestBuilder prepareExecute(final Action<Request, Response, RequestBuilder, Client> action);
072    
073}
