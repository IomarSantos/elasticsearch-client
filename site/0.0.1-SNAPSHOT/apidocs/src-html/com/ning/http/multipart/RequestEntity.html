016package com.ning.http.multipart;
017
018import java.io.IOException;
019import java.io.OutputStream;
020
021/**
022 * This class is an adaptation of the Apache HttpClient implementation
023 *
024 * @link http://hc.apache.org/httpclient-3.x/
025 */
026public interface RequestEntity {
027
028    /**
029     * Tests if {@link #writeRequest(java.io.OutputStream)} can be called more than once.
030     *
031     * @return <tt>true</tt> if the entity can be written to {@link java.io.OutputStream} more than once,
032     *         <tt>false</tt> otherwise.
033     */
034    boolean isRepeatable();
035
036    /**
037     * Writes the request entity to the given stream.
038     *
039     * @param out
040     * @throws java.io.IOException
041     */
042    void writeRequest(OutputStream out) throws IOException;
043
044    /**
045     * Gets the request entity's length. This method should return a non-negative value if the content
046     * length is known or a negative value if it is not. In the latter case the
047     * EntityEnclosingMethod will use chunk encoding to
048     * transmit the request entity.
049     *
050     * @return a non-negative value when content length is known or a negative value when content length
051     *         is not known
052     */
053    long getContentLength();
054
055    /**
056     * Gets the entity's content type.  This content type will be used as the value for the
057     * "Content-Type" header.
058     *
059     * @return the entity's content type
060     */
061    String getContentType();
062
063}
