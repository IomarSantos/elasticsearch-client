016package com.ning.http.multipart;
017
018import com.ning.http.client.FluentStringsMap;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022import java.io.IOException;
023import java.io.OutputStream;
024import java.util.Random;
025
026/**
027 * This class is an adaptation of the Apache HttpClient implementation
028 *
029 * @link http://hc.apache.org/httpclient-3.x/
030 */
031public class MultipartRequestEntity implements RequestEntity {
032
033    /**
034     * The Content-Type for multipart/form-data.
035     */
036    private static final String MULTIPART_FORM_CONTENT_TYPE = "multipart/form-data";
037
038    /**
039     * The pool of ASCII chars to be used for generating a multipart boundary.
040     */
041    private static byte[] MULTIPART_CHARS = MultipartEncodingUtil.getAsciiBytes(
042            "-_1234567890abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ");
043
044    /**
045     * Generates a random multipart boundary string.
046     *
047     * @return
048     */
049    private static byte[] generateMultipartBoundary() {
050        Random rand = new Random();
051        byte[] bytes = new byte[rand.nextInt(11) + 30]; // a random size from 30 to 40
052        for (int i = 0; i < bytes.length; i++) {
053            bytes[i] = MULTIPART_CHARS[rand.nextInt(MULTIPART_CHARS.length)];
054        }
055        return bytes;
056    }
057
058    private final Logger log = LoggerFactory.getLogger(MultipartRequestEntity.class);
059
060    /**
061     * The MIME parts as set by the constructor
062     */
063    protected Part[] parts;
064
065    private byte[] multipartBoundary;
066
067    private FluentStringsMap methodParams;
068
069    /**
070     * Creates a new multipart entity containing the given parts.
071     *
072     * @param parts        The parts to include.
073     * @param methodParams The params of the HttpMethod using this entity.
074     */
075    public MultipartRequestEntity(Part[] parts, FluentStringsMap methodParams) {
076        if (parts == null) {
077            throw new IllegalArgumentException("parts cannot be null");
078        }
079        if (methodParams == null) {
080            methodParams = new FluentStringsMap();
081        }
082        this.parts = parts;
083        this.methodParams = methodParams;
084    }
085
086    /**
087     * Returns the MIME boundary string that is used to demarcate boundaries of
088     * this part. The first call to this method will implicitly create a new
089     * boundary string. To create a boundary string first the
090     * HttpMethodParams.MULTIPART_BOUNDARY parameter is considered. Otherwise
091     * a random one is generated.
092     *
093     * @return The boundary string of this entity in ASCII encoding.
094     */
095    protected byte[] getMultipartBoundary() {
096        if (multipartBoundary == null) {
097            String temp = methodParams.get("") == null ? null : methodParams.get("").iterator().next();
098            if (temp != null) {
099                multipartBoundary = MultipartEncodingUtil.getAsciiBytes(temp);
100            } else {
101                multipartBoundary = generateMultipartBoundary();
102            }
103        }
104        return multipartBoundary;
105    }
106
107    /**
108     * Returns <code>true</code> if all parts are repeatable, <code>false</code> otherwise.
109     */
110    public boolean isRepeatable() {
111        for (int i = 0; i < parts.length; i++) {
112            if (!parts[i].isRepeatable()) {
113                return false;
114            }
115        }
116        return true;
117    }
118
119    /* (non-Javadoc)
120     * @see org.apache.commons.httpclient.methods.RequestEntity#writeRequest(java.io.OutputStream)
121     */
122    public void writeRequest(OutputStream out) throws IOException {
123        Part.sendParts(out, parts, getMultipartBoundary());
124    }
125
126    /* (non-Javadoc)
127     * @see org.apache.commons.httpclient.methods.RequestEntity#getContentLength()
128     */
129    public long getContentLength() {
130        try {
131            return Part.getLengthOfParts(parts, getMultipartBoundary());
132        } catch (Exception e) {
133            log.error("An exception occurred while getting the length of the parts", e);
134            return 0;
135        }
136    }
137
138    /* (non-Javadoc)
139     * @see org.apache.commons.httpclient.methods.RequestEntity#getContentType()
140     */
141    public String getContentType() {
142        StringBuffer buffer = new StringBuffer(MULTIPART_FORM_CONTENT_TYPE);
143        buffer.append("; boundary=");
144        buffer.append(MultipartEncodingUtil.getAsciiString(getMultipartBoundary()));
145        return buffer.toString();
146    }
147
148}
