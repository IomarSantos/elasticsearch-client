016package com.ning.http.multipart;
017
018import java.io.UnsupportedEncodingException;
019
020/**
021 * This class is an adaptation of the Apache HttpClient implementation
022 *
023 * @link http://hc.apache.org/httpclient-3.x/
024 */
025public class MultipartEncodingUtil {
026
027    public static byte[] getAsciiBytes(String data) {
028        try {
029            return data.getBytes("US-ASCII");
030        } catch (UnsupportedEncodingException e) {
031            throw new RuntimeException(e);
032        }
033    }
034
035    public static String getAsciiString(final byte[] data) {
036        if (data == null) {
037            throw new IllegalArgumentException("Parameter may not be null");
038        }
039
040        try {
041            return new String(data, "US-ASCII");
042        } catch (UnsupportedEncodingException e) {
043            throw new RuntimeException(e);
044        }
045    }
046
047    public static byte[] getBytes(final String data, String charset) {
048
049        if (data == null) {
050            throw new IllegalArgumentException("data may not be null");
051        }
052
053        if (charset == null || charset.length() == 0) {
054            throw new IllegalArgumentException("charset may not be null or empty");
055        }
056
057        try {
058            return data.getBytes(charset);
059        } catch (UnsupportedEncodingException e) {
060            throw new IllegalArgumentException(String.format("Unsupported encoding: %s", charset));
061        }
062    }
063}
