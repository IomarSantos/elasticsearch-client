033public class MultipartBody implements RandomAccessBody {

035    private byte[] boundary;
036    private long contentLength;
037    private List<com.ning.http.client.Part> parts;
038    private List<RandomAccessFile> files;
039    private int startPart;
040    private final static Logger logger = LoggerFactory.getLogger(MultipartBody.class);
041    ByteArrayInputStream currentStream;

043    boolean endWritten;

049    enum FileLocation {NONE, START, MIDDLE, END}

051    public MultipartBody(List<com.ning.http.client.Part> parts, String boundary, String contentLength) {
052        this.boundary = MultipartEncodingUtil.getAsciiBytes(boundary.substring("multipart/form-data; boundary=".length()));
053        this.contentLength = Long.parseLong(contentLength);
054        this.parts = parts;
055
056        files = new ArrayList<RandomAccessFile>();
057
058        startPart = 0;
059        currentStreamPosition = -1;
060        endWritten = false;
061        doneWritingParts = false;
062        fileLocation = FileLocation.NONE;
063        currentFilePart = null;
064    }
065
066    public void close() throws IOException {
067        for (RandomAccessFile file : files) {
068            file.close();
069        }
070    }
071
072    public long getContentLength() {
073        return contentLength;
074    }

076    public long read(ByteBuffer buffer) throws IOException {
077        try {
078            int overallLength = 0;
079
080            int maxLength = buffer.capacity();
081
082            if (startPart == parts.size() && endWritten) {
083                return overallLength;
084            }
085
086            boolean full = false;
087            while (!full && !doneWritingParts) {
088                com.ning.http.client.Part part = null;
089                if (startPart < parts.size()) {
090                    part = parts.get(startPart);
091                }
092                if (currentFileChannel != null) {
093                    overallLength += currentFileChannel.read(buffer);
094
095                    if (currentFileChannel.position() == currentFileChannel.size()) {
096                        currentFileChannel.close();
097                        currentFileChannel = null;
098                    }
099
100                    if (overallLength == maxLength) {
101                        full = true;
102                    }
103                } else if (currentStreamPosition > -1) {
104                    overallLength += writeToBuffer(buffer, maxLength - overallLength);
105
106                    if (overallLength == maxLength) {
107                        full = true;
108                    }
109                    if (startPart == parts.size() && currentStream.available() == 0) {
110                        doneWritingParts = true;
111                    }
112                } else if (part instanceof StringPart) {
113                    StringPart currentPart = (StringPart) part;
114
115                    initializeStringPart(currentPart);
116
117                    startPart++;
118                } else if (part instanceof com.ning.http.client.StringPart) {
119                    StringPart currentPart = generateClientStringpart(part);
120
121                    initializeStringPart(currentPart);
122
123                    startPart++;
124                } else if (part instanceof FilePart) {
125                    if (fileLocation == FileLocation.NONE) {
126                        currentFilePart = (FilePart) part;
127                        initializeFilePart(currentFilePart);
128                    } else if (fileLocation == FileLocation.START) {
129                        initializeFileBody(currentFilePart);
130                    } else if (fileLocation == FileLocation.MIDDLE) {
131                        initializeFileEnd(currentFilePart);
132                    } else if (fileLocation == FileLocation.END) {
133                        startPart++;
134                        if (startPart == parts.size() && currentStream.available() == 0) {
135                            doneWritingParts = true;
136                        }
137                    }
138                } else if (part instanceof com.ning.http.client.FilePart) {
139                    if (fileLocation == FileLocation.NONE) {
140                        currentFilePart = generateClientFilePart(part);
141                        initializeFilePart(currentFilePart);
142                    } else if (fileLocation == FileLocation.START) {
143                        initializeFileBody(currentFilePart);
144                    } else if (fileLocation == FileLocation.MIDDLE) {
145                        initializeFileEnd(currentFilePart);
146                    } else if (fileLocation == FileLocation.END) {
147                        startPart++;
148                        if (startPart == parts.size() && currentStream.available() == 0) {
149                            doneWritingParts = true;
150                        }
151                    }
152                } else if (part instanceof com.ning.http.client.ByteArrayPart) {
153                    com.ning.http.client.ByteArrayPart bytePart =
154                            (com.ning.http.client.ByteArrayPart) part;
155
156                    if (fileLocation == FileLocation.NONE) {
157                        currentFilePart =
158                                generateClientByteArrayPart(bytePart);
159
160                        initializeFilePart(currentFilePart);
161                    } else if (fileLocation == FileLocation.START) {
162                        initializeByteArrayBody(currentFilePart);
163                    } else if (fileLocation == FileLocation.MIDDLE) {
164                        initializeFileEnd(currentFilePart);
165                    } else if (fileLocation == FileLocation.END) {
166                        startPart++;
167                        if (startPart == parts.size() && currentStream.available() == 0) {
168                            doneWritingParts = true;
169                        }
170                    }
171                }
172            }
173
174            if (doneWritingParts) {
175                if (currentStreamPosition == -1) {
176                    ByteArrayOutputStream endWriter = new ByteArrayOutputStream();
177
178                    Part.sendMessageEnd(endWriter, boundary);
179
180                    initializeBuffer(endWriter);
181                }
182
183                if (currentStreamPosition > -1) {
184                    overallLength += writeToBuffer(buffer, maxLength - overallLength);
185
186                    if (currentStream.available() == 0) {
187                        currentStream.close();
188                        currentStreamPosition = -1;
189                        endWritten = true;
190                    }
191                }
192            }
193            return overallLength;
194
195        } catch (Exception e) {
196            logger.info("read exception", e);
197            return 0;
198        }
199    }

201    private void initializeByteArrayBody(FilePart filePart)
202            throws IOException {
203
204        ByteArrayOutputStream output = generateByteArrayBody(filePart);
205
206        initializeBuffer(output);
207
208        fileLocation = FileLocation.MIDDLE;
209    }
210
211    private void initializeFileEnd(FilePart currentPart)
212            throws IOException {
213
214        ByteArrayOutputStream output = generateFileEnd(currentPart);
215
216        initializeBuffer(output);
217
218        fileLocation = FileLocation.END;
219
220    }
221
222    private void initializeFileBody(FilePart currentPart)
223            throws IOException {
224
225        if (FilePartSource.class.isAssignableFrom(currentPart.getSource().getClass())) {
226
227            FilePartSource source = (FilePartSource) currentPart.getSource();
228
229            File file = source.getFile();
230
231            RandomAccessFile raf = new RandomAccessFile(file, "r");
232            files.add(raf);
233
234            currentFileChannel = raf.getChannel();
235
236        } else {
237            PartSource partSource = currentPart.getSource();
238
239            InputStream stream = partSource.createInputStream();
240
241            byte[] bytes = new byte[(int) partSource.getLength()];
242
243            stream.read(bytes);
244
245            currentStream = new ByteArrayInputStream(bytes);
246
247            currentStreamPosition = 0;
248        }
249
250        fileLocation = FileLocation.MIDDLE;
251    }
252
253    private void initializeFilePart(FilePart filePart)
254            throws IOException {
255
256        filePart.setPartBoundary(boundary);
257
258        ByteArrayOutputStream output = generateFileStart(filePart);
259
260        initializeBuffer(output);
261
262        fileLocation = FileLocation.START;
263    }
264
265    private void initializeStringPart(StringPart currentPart)
266            throws IOException {
267        currentPart.setPartBoundary(boundary);
268
269        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
270
271        Part.sendPart(outputStream, currentPart, boundary);
272
273        initializeBuffer(outputStream);
274    }

276    private int writeToBuffer(ByteBuffer buffer, int length)
277            throws IOException {
278
279        int available = currentStream.available();
280
281        int writeLength = Math.min(available, length);
282
283        byte[] bytes = new byte[writeLength];
284
285        currentStream.read(bytes);
286
287        buffer.put(bytes);
288
289        if (available <= length) {
290            currentStream.close();
291            currentStreamPosition = -1;
292        } else {
293            currentStreamPosition += writeLength;
294        }
295
296        return writeLength;
297    }
298
299    private void initializeBuffer(ByteArrayOutputStream outputStream)
300            throws IOException {
301
302        currentStream = new ByteArrayInputStream(outputStream.toByteArray());
303
304        currentStreamPosition = 0;
305
306    }

308    public long transferTo(long position, long count, WritableByteChannel target)
309            throws IOException {
310
311        long overallLength = 0;
312
313        if (startPart == parts.size()) {
314            return contentLength;
315        }
316
317        int tempPart = startPart;
318
319        for (com.ning.http.client.Part part : parts) {
320            if (part instanceof Part) {
321                overallLength += handleMultiPart(target, (Part) part);
322            } else {
323                overallLength += handleClientPart(target, part);
324            }
325
326            tempPart++;
327        }
328        ByteArrayOutputStream endWriter =
329                new ByteArrayOutputStream();
330
331        Part.sendMessageEnd(endWriter, boundary);
332
333        overallLength += writeToTarget(target, endWriter);
334
335        startPart = tempPart;
336
337        return overallLength;
338    }
339
340    private long handleClientPart(
341            WritableByteChannel target, com.ning.http.client.Part part) throws IOException {
342
343        if (part.getClass().equals(com.ning.http.client.StringPart.class)) {
344            StringPart currentPart = generateClientStringpart(part);
345
346            return handleStringPart(target, currentPart);
347        } else if (part.getClass().equals(com.ning.http.client.FilePart.class)) {
348            FilePart filePart = generateClientFilePart(part);
349
350            return handleFilePart(target, filePart);
351        } else if (part.getClass().equals(com.ning.http.client.ByteArrayPart.class)) {
352            com.ning.http.client.ByteArrayPart bytePart = (com.ning.http.client.ByteArrayPart) part;
353
354            FilePart filePart = generateClientByteArrayPart(bytePart);
355
356            return handleByteArrayPart(target, filePart, bytePart.getData());
357        }
358
359        return 0;
360    }

362    private FilePart generateClientByteArrayPart(
363            com.ning.http.client.ByteArrayPart bytePart) {
364        ByteArrayPartSource source = new ByteArrayPartSource(bytePart.getFileName(), bytePart.getData());
365
366        FilePart filePart = new FilePart(bytePart.getName(), source, bytePart.getMimeType(), bytePart.getCharSet());
367        return filePart;
368    }

370    private FilePart generateClientFilePart(com.ning.http.client.Part part)
371            throws FileNotFoundException {
372        com.ning.http.client.FilePart currentPart = (com.ning.http.client.FilePart) part;
373
374        FilePart filePart = new FilePart(currentPart.getName(), currentPart.getFile(), currentPart.getMimeType(), currentPart.getCharSet());
375        return filePart;
376    }

382        return currentPart;

385    private long handleByteArrayPart(WritableByteChannel target,
386                                     FilePart filePart, byte[] data) throws IOException {
387
388        ByteArrayOutputStream output = generateByteArrayBody(filePart);
389        return writeToTarget(target, output);

415    private long handleFileHeaders(WritableByteChannel target, FilePart filePart) throws IOException {
416        filePart.setPartBoundary(boundary);
417
418        ByteArrayOutputStream overhead = generateFileStart(filePart);
419
420        return writeToTarget(target, overhead);
421    }
422
423    private ByteArrayOutputStream generateFileStart(FilePart filePart)
424            throws IOException {
425        ByteArrayOutputStream overhead = new ByteArrayOutputStream();
426
427        filePart.setPartBoundary(boundary);
428
429        filePart.sendStart(overhead);
430        filePart.sendDispositionHeader(overhead);
431        filePart.sendContentTypeHeader(overhead);
432        filePart.sendTransferEncodingHeader(overhead);
433        filePart.sendEndOfHeader(overhead);
434        return overhead;
435    }

437    private long handleFilePart(WritableByteChannel target, FilePart filePart) throws IOException {
438        FilePartStallHandler handler = new FilePartStallHandler(
439                filePart.getStalledTime(), filePart);
440        
441        handler.start();
442        
443        if (FilePartSource.class.isAssignableFrom(filePart.getSource().getClass())) {
444            int length = 0;
445
446            length += handleFileHeaders(target, filePart);
447            FilePartSource source = (FilePartSource) filePart.getSource();
448
449            File file = source.getFile();
450
451            RandomAccessFile raf = new RandomAccessFile(file, "r");
452            files.add(raf);
453
454            FileChannel fc = raf.getChannel();
455
456            long l = file.length();
457            int fileLength = 0;
458            long nWrite = 0;
459            synchronized (fc) {
460                while (fileLength != l) {
461                        if(handler.isFailed()) {
462                                logger.debug("Stalled error");
463                        throw new FileUploadStalledException();
464                        }
465                    try {
466                        nWrite = fc.transferTo(fileLength, l, target);
467                       
468                        if (nWrite == 0) {
469                            logger.info("Waiting for writing...");
470                            try {
471                                fc.wait(50);
472                            } catch (InterruptedException e) {
473                                logger.trace(e.getMessage(), e);
474                            }
475                        }
476                        else {
477                                handler.writeHappened();
478                        }
479                    } catch (IOException ex) {
480                        String message = ex.getMessage();
481
482                        // http://bugs.sun.com/view_bug.do?bug_id=5103988
483                        if (message != null && message.equalsIgnoreCase("Resource temporarily unavailable")) {
484                            try {
485                                fc.wait(1000);
486                            } catch (InterruptedException e) {
487                                logger.trace(e.getMessage(), e);
488                            }
489                            logger.warn("Experiencing NIO issue http://bugs.sun.com/view_bug.do?bug_id=5103988. Retrying");
490                            continue;
491                        } else {
492                            throw ex;
493                        }
494                    }
495                    fileLength += nWrite;
496                }
497            }
498            handler.completed();
499            
500            fc.close();
501
502            length += handleFileEnd(target, filePart);
503
504            return length;
505        } else {
506            return handlePartSource(target, filePart);
507        }
508    }
509
510    private long handlePartSource(WritableByteChannel target, FilePart filePart) throws IOException {
511
512        int length = 0;
513
514        length += handleFileHeaders(target, filePart);
515
516        PartSource partSource = filePart.getSource();
517
518        InputStream stream = partSource.createInputStream();
519
520        try {
521            int nRead = 0;
522            while (nRead != -1) {
523                // Do not buffer the entire monster in memory.
524                byte[] bytes = new byte[8192];
525                nRead = stream.read(bytes);
526                if (nRead > 0) {
527                    ByteArrayOutputStream bos = new ByteArrayOutputStream(nRead);
528                    bos.write(bytes, 0, nRead);
529                    writeToTarget(target, bos);
530                }
531            }
532        } finally {
533            stream.close();
534        }
535        length += handleFileEnd(target, filePart);
536
537        return length;
538    }

540    private long handleStringPart(WritableByteChannel target, StringPart currentPart) throws IOException {
541
542        currentPart.setPartBoundary(boundary);
543
544        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
545
546        Part.sendPart(outputStream, currentPart, boundary);
547
548        return writeToTarget(target, outputStream);
549    }
550
551    private long handleMultiPart(WritableByteChannel target, Part currentPart) throws IOException {
552
553        currentPart.setPartBoundary(boundary);
554
555        if (currentPart.getClass().equals(StringPart.class)) {
556            return handleStringPart(target, (StringPart) currentPart);
557        } else if (currentPart.getClass().equals(FilePart.class)) {
558            FilePart filePart = (FilePart) currentPart;
559            
560            return handleFilePart(target, filePart);
561        }
562        return 0;
563    }
564
565    private long writeToTarget(WritableByteChannel target, ByteArrayOutputStream byteWriter)
566            throws IOException {
567
568        int written = 0;
569        int maxSpin = 0;
570        synchronized (byteWriter) {
571            ByteBuffer message = ByteBuffer.wrap(byteWriter.toByteArray());
572            while ((target.isOpen()) && (written < byteWriter.size())) {
573                long nWrite = target.write(message);
574                written += nWrite;
575                if (nWrite == 0 && maxSpin++ < 10) {
576                    logger.info("Waiting for writing...");
577                    try {
578                        byteWriter.wait(1000);
579                    } catch (InterruptedException e) {
580                        logger.trace(e.getMessage(), e);
581                    }
582                } else {
583                    if (maxSpin >= 10) {
584                        throw new IOException("Unable to write on channel " + target);
585                    }
586                    maxSpin = 0;
587                }
588            }
589        }
590        return written;
591    }
