013package com.ning.http.util;
014
015import com.ning.http.client.ProxyServer;
016import com.ning.http.client.ProxyServer.Protocol;
017import com.ning.http.client.Request;
018
019import java.net.URI;
020import java.util.List;
021import java.util.Properties;
022
023/**
024 * Utilities for Proxy handling.
025 *
026 * @author cstamas
027 */
028public class ProxyUtils {
029
030    private static final String PROPERTY_PREFIX = "com.ning.http.client.AsyncHttpClientConfig.proxy.";
031
032    /**
033     * The host to use as proxy.
034     */
035    public static final String PROXY_HOST = "http.proxyHost";
036
037    /**
038     * The port to use for the proxy.
039     */
040    public static final String PROXY_PORT = "http.proxyPort";
041
042    /**
043     * The protocol to use. Is mapped to the {@link Protocol} enum.
044     */
045    public static final String PROXY_PROTOCOL = PROPERTY_PREFIX + "protocol";
046
047    /**
048     * A specification of non-proxy hosts. See http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html
049     */
050    public static final String PROXY_NONPROXYHOSTS = "http.nonProxyHosts";
051
052    /**
053     * The username to use for authentication for the proxy server.
054     */
055    public static final String PROXY_USER = PROPERTY_PREFIX + "user";
056
057    /**
058     * The password to use for authentication for the proxy server.
059     */
060    public static final String PROXY_PASSWORD = PROPERTY_PREFIX + "password";
061
062    /**
063     * Checks whether proxy should be used according to nonProxyHosts settings of it, or we want to go directly to
064     * target host. If <code>null</code> proxy is passed in, this method returns true -- since there is NO proxy, we
065     * should avoid to use it. Simple hostname pattern matching using "*" are supported, but only as prefixes.
066     * See http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html
067     *
068     * @param proxyServer
069     * @param request
070     * @return true if we have to avoid proxy use (obeying non-proxy hosts settings), false otherwise.
071     */
072    public static boolean avoidProxy(final ProxyServer proxyServer, final Request request) {
073        return avoidProxy(proxyServer, AsyncHttpProviderUtils.getHost(URI.create(request.getUrl())));
074    }
075
076    /**
077     * Checks whether proxy should be used according to nonProxyHosts settings of it, or we want to go directly to
078     * target host. If <code>null</code> proxy is passed in, this method returns true -- since there is NO proxy, we
079     * should avoid to use it. Simple hostname pattern matching using "*" are supported, but only as prefixes.
080     * See http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html
081     *
082     * @param proxyServer
083     * @param target      the hostname
084     * @return true if we have to avoid proxy use (obeying non-proxy hosts settings), false otherwise.
085     */
086    public static boolean avoidProxy(final ProxyServer proxyServer, final String target) {
087        if (proxyServer != null) {
088            final String targetHost = target.toLowerCase();
089
090            List<String> nonProxyHosts = proxyServer.getNonProxyHosts();
091
092            if (nonProxyHosts != null && nonProxyHosts.size() > 0) {
093                for (String nonProxyHost : nonProxyHosts) {
094                    if (nonProxyHost.startsWith("*") && nonProxyHost.length() > 1
095                            && targetHost.endsWith(nonProxyHost.substring(1).toLowerCase())) {
096                        return true;
097                    } else if (nonProxyHost.equalsIgnoreCase(targetHost)) {
098                        return true;
099                    }
100                }
101            }
102
103            return false;
104        } else {
105            return true;
106        }
107    }
108
109    /**
110     * Creates a proxy server instance from the given properties.
111     * <p/>
112     * Currently the default http.* proxy properties are supported as well as properties specific for AHC.
113     *
114     * @param properties the properties to evaluate. Must not be null.
115     * @return a ProxyServer instance or null, if no valid properties were set.
116     * @see <a href="http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html">Networking Properties</a>
117     * @see #PROXY_HOST
118     * @see #PROXY_PORT
119     * @see #PROXY_PROTOCOL
120     * @see #PROXY_NONPROXYHOSTS
121     */
122    public static ProxyServer createProxy(Properties properties) {
123        String host = System.getProperty(PROXY_HOST);
124
125        if (host != null) {
126            int port = Integer.valueOf(System.getProperty(PROXY_PORT, "80"));
127
128            Protocol protocol;
129            try {
130                protocol = Protocol.valueOf(System.getProperty(PROXY_PROTOCOL, "HTTP"));
131            } catch (IllegalArgumentException e) {
132                protocol = Protocol.HTTP;
133            }
134
135            ProxyServer proxyServer = new ProxyServer(protocol, host, port, System.getProperty(PROXY_USER), System.getProperty(PROXY_PASSWORD));
136
137            String nonProxyHosts = System.getProperties().getProperty(PROXY_NONPROXYHOSTS);
138            if (nonProxyHosts != null) {
139                for (String spec : nonProxyHosts.split("\\|")) {
140                    proxyServer.addNonProxyHost(spec);
141                }
142            }
143
144            return proxyServer;
145        }
146
147        return null;
148    }
149}
