013package com.ning.http.util;
014
015import com.ning.http.client.ProxyServer;
016import com.ning.http.client.Realm;
017
018import java.io.UnsupportedEncodingException;
019import java.security.NoSuchAlgorithmException;
020
021public final class AuthenticatorUtils {
022
023    public static String computeBasicAuthentication(Realm realm) throws UnsupportedEncodingException {
024        String s = realm.getPrincipal() + ":" + realm.getPassword();
025        return "Basic " + Base64.encode(s.getBytes(realm.getEncoding()));
026    }
027
028    public static String computeBasicAuthentication(ProxyServer proxyServer) throws UnsupportedEncodingException {
029        String s = proxyServer.getPrincipal() + ":" + proxyServer.getPassword();
030        return "Basic " + Base64.encode(s.getBytes(proxyServer.getEncoding()));
031    }
032
033    public static String computeDigestAuthentication(Realm realm) throws NoSuchAlgorithmException, UnsupportedEncodingException {
034
035        StringBuilder builder = new StringBuilder().append("Digest ");
036        construct(builder, "username", realm.getPrincipal());
037        construct(builder, "realm", realm.getRealmName());
038        construct(builder, "nonce", realm.getNonce());
039        construct(builder, "uri", realm.getUri());
040        builder.append("algorithm").append('=').append(realm.getAlgorithm()).append(", ");
041
042        construct(builder, "response", realm.getResponse());
043        if (realm.getOpaque() != null && realm.getOpaque() != null && realm.getOpaque().equals("") == false)
044            construct(builder, "opaque", realm.getOpaque());
045        builder.append("qop").append('=').append(realm.getQop()).append(", ");
046        builder.append("nc").append('=').append(realm.getNc()).append(", ");
047        construct(builder, "cnonce", realm.getCnonce(), true);
048
049        return new String(builder.toString().getBytes("ISO_8859_1"));
050    }
051
052    private static StringBuilder construct(StringBuilder builder, String name, String value) {
053        return construct(builder, name, value, false);
054    }
055
056    private static StringBuilder construct(StringBuilder builder, String name, String value, boolean tail) {
057        return builder.append(name).append('=').append('"').append(value).append(tail ? "\"" : "\", ");
058    }
059}
