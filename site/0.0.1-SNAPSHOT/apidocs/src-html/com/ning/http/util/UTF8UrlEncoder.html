016package com.ning.http.util;
017
018/**
019 * Convenience class that encapsulates details of "percent encoding"
020 * (as per RFC-3986, see [http://www.ietf.org/rfc/rfc3986.txt]).
021 */
022public class UTF8UrlEncoder {
023    private static final boolean encodeSpaceUsingPlus = System.getProperty("com.com.ning.http.util.UTF8UrlEncoder.encodeSpaceUsingPlus") == null ? false : true;
024
025    /**
026     * Encoding table used for figuring out ascii characters that must be escaped
027     * (all non-Ascii characers need to be encoded anyway)
028     */
029    private final static int[] SAFE_ASCII = new int[128];
030
031    static {
032        for (int i = 'a'; i <= 'z'; ++i) {
033            SAFE_ASCII[i] = 1;
034        }
035        for (int i = 'A'; i <= 'Z'; ++i) {
036            SAFE_ASCII[i] = 1;
037        }
038        for (int i = '0'; i <= '9'; ++i) {
039            SAFE_ASCII[i] = 1;
040        }
041        SAFE_ASCII['-'] = 1;
042        SAFE_ASCII['.'] = 1;
043        SAFE_ASCII['_'] = 1;
044        SAFE_ASCII['~'] = 1;
045    }
046
047    private final static char[] HEX = "0123456789ABCDEF".toCharArray();
048
049    private UTF8UrlEncoder() {
050    }
051
052    public static String encode(String input) {
053        StringBuilder sb = new StringBuilder(input.length() + 16);
054        appendEncoded(sb, input);
055        return sb.toString();
056    }
057
058    public static StringBuilder appendEncoded(StringBuilder sb, String input) {
059        final int[] safe = SAFE_ASCII;
060
061        for (int i = 0, len = input.length(); i < len; ++i) {
062            char c = input.charAt(i);
063            if (c <= 127) {
064                if (safe[c] != 0) {
065                    sb.append(c);
066                } else {
067                    appendSingleByteEncoded(sb, c);
068                }
069            } else {
070                appendMultiByteEncoded(sb, c);
071            }
072        }
073        return sb;
074    }
075
076    private final static void appendSingleByteEncoded(StringBuilder sb, int value) {
077
078        if (encodeSpaceUsingPlus && value == 32) {
079            sb.append('+');
080            return;
081        }
082
083        sb.append('%');
084        sb.append(HEX[value >> 4]);
085        sb.append(HEX[value & 0xF]);
086    }
087
088    private final static void appendMultiByteEncoded(StringBuilder sb, int value) {
089        // two or three bytes? (ignoring surrogate pairs for now, which would yield 4 bytes)
090        if (value < 0x800) {
091            appendSingleByteEncoded(sb, (0xc0 | (value >> 6)));
092            appendSingleByteEncoded(sb, (0x80 | (value & 0x3f)));
093        } else {
094            appendSingleByteEncoded(sb, (0xe0 | (value >> 12)));
095            appendSingleByteEncoded(sb, (0x80 | ((value >> 6) & 0x3f)));
096            appendSingleByteEncoded(sb, (0x80 | (value & 0x3f)));
097        }
098    }
099
100}
