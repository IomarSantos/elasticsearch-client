013package com.ning.http.util;
014
015/**
016 * Implements the "base64" binary encoding scheme as defined by
017 * <a href="http://tools.ietf.org/html/rfc2045">RFC 2045</a>.
018 * <p/>
019 * Portions of code here are taken from Apache Pivot
020 */
021public final class Base64 {
022    private static final char[] lookup = new char[64];
023    private static final byte[] reverseLookup = new byte[256];
024
025    static {
026        // Populate the lookup array
027
028        for (int i = 0; i < 26; i++) {
029            lookup[i] = (char) ('A' + i);
030        }
031
032        for (int i = 26, j = 0; i < 52; i++, j++) {
033            lookup[i] = (char) ('a' + j);
034        }
035
036        for (int i = 52, j = 0; i < 62; i++, j++) {
037            lookup[i] = (char) ('0' + j);
038        }
039
040        lookup[62] = '+';
041        lookup[63] = '/';
042
043        // Populate the reverse lookup array
044
045        for (int i = 0; i < 256; i++) {
046            reverseLookup[i] = -1;
047        }
048
049        for (int i = 'Z'; i >= 'A'; i--) {
050            reverseLookup[i] = (byte) (i - 'A');
051        }
052
053        for (int i = 'z'; i >= 'a'; i--) {
054            reverseLookup[i] = (byte) (i - 'a' + 26);
055        }
056
057        for (int i = '9'; i >= '0'; i--) {
058            reverseLookup[i] = (byte) (i - '0' + 52);
059        }
060
061        reverseLookup['+'] = 62;
062        reverseLookup['/'] = 63;
063        reverseLookup['='] = 0;
064    }
065
066    /**
067     * This class is not instantiable.
068     */
069    private Base64() {
070    }
071
072    /**
073     * Encodes the specified data into a base64 string.
074     *
075     * @param bytes The unencoded raw data.
076     */
077    public static String encode(byte[] bytes) {
078        // always sequence of 4 characters for each 3 bytes; padded with '='s as necessary:
079        StringBuilder buf = new StringBuilder(((bytes.length + 2) / 3) * 4);
080
081        // first, handle complete chunks (fast loop)
082        int i = 0;
083        for (int end = bytes.length - 2; i < end; ) {
084            int chunk = ((bytes[i++] & 0xFF) << 16)
085                    | ((bytes[i++] & 0xFF) << 8)
086                    | (bytes[i++] & 0xFF);
087            buf.append(lookup[chunk >> 18]);
088            buf.append(lookup[(chunk >> 12) & 0x3F]);
089            buf.append(lookup[(chunk >> 6) & 0x3F]);
090            buf.append(lookup[chunk & 0x3F]);
091        }
092
093        // then leftovers, if any
094        int len = bytes.length;
095        if (i < len) { // 1 or 2 extra bytes?
096            int chunk = ((bytes[i++] & 0xFF) << 16);
097            buf.append(lookup[chunk >> 18]);
098            if (i < len) { // 2 bytes
099                chunk |= ((bytes[i] & 0xFF) << 8);
100                buf.append(lookup[(chunk >> 12) & 0x3F]);
101                buf.append(lookup[(chunk >> 6) & 0x3F]);
102            } else { // 1 byte
103                buf.append(lookup[(chunk >> 12) & 0x3F]);
104                buf.append('=');
105            }
106            buf.append('=');
107        }
108        return buf.toString();
109    }
110
111    /**
112     * Decodes the specified base64 string back into its raw data.
113     *
114     * @param encoded The base64 encoded string.
115     */
116    public static byte[] decode(String encoded) {
117        int padding = 0;
118
119        for (int i = encoded.length() - 1; encoded.charAt(i) == '='; i--) {
120            padding++;
121        }
122
123        int length = encoded.length() * 6 / 8 - padding;
124        byte[] bytes = new byte[length];
125
126        for (int i = 0, index = 0, n = encoded.length(); i < n; i += 4) {
127            int word = reverseLookup[encoded.charAt(i)] << 18;
128            word += reverseLookup[encoded.charAt(i + 1)] << 12;
129            word += reverseLookup[encoded.charAt(i + 2)] << 6;
130            word += reverseLookup[encoded.charAt(i + 3)];
131
132            for (int j = 0; j < 3 && index + j < length; j++) {
133                bytes[index + j] = (byte) (word >> (8 * (2 - j)));
134            }
135
136            index += 3;
137        }
138
139        return bytes;
140    }
141}
