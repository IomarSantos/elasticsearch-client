143    public final static URI createUri(String u) {
144        URI uri = URI.create(u);
145        final String scheme = uri.getScheme();
146        if (scheme == null || !scheme.equalsIgnoreCase("http") && !scheme.equalsIgnoreCase("https") && !scheme.equalsIgnoreCase("ws")
147                && !scheme.equalsIgnoreCase("wss")) {
148            throw new IllegalArgumentException("The URI scheme, of the URI " + u
149                    + ", must be equal (ignoring case) to 'http', 'https', 'ws', or 'wss'");
150        }
151
152        String path = uri.getPath();
153        if (path == null) {
154            throw new IllegalArgumentException("The URI path, of the URI " + uri
155                    + ", must be non-null");
156        } else if (path.length() > 0 && path.charAt(0) != '/') {
157            throw new IllegalArgumentException("The URI path, of the URI " + uri
158                    + ". must start with a '/'");
159        } else if (path.length() == 0) {
160            return URI.create(u + "/");
161        }
162
163        return uri;
164    }

173        if (port == -1) {

194    public final static byte[] contentToBytes(List<HttpResponseBodyPart> bodyParts) throws UnsupportedEncodingException {
195        final int partCount = bodyParts.size();
196        if (partCount == 0) {
197            return NO_BYTES;
198        }
199        if (partCount == 1) {
200            return bodyParts.get(0).getBodyPartBytes();
201        }
202        int size = 0;
203        ArrayList<byte[]> chunks = new ArrayList<byte[]>(partCount);
204        for (HttpResponseBodyPart part : bodyParts) {
205            byte[] chunk = part.getBodyPartBytes();
206            size += chunk.length;
207            chunks.add(chunk);
208        }
209        byte[] bytes = new byte[size];
210        int offset = 0;
211        for (byte[] chunk : chunks) {
212            System.arraycopy(chunk, 0, bytes, offset, chunk.length);
213            offset += chunk.length;
214        }
215        return bytes;
216    }

218    public final static byte[] contentToBytes(List<HttpResponseBodyPart> bodyParts, int maxLen) throws UnsupportedEncodingException {
219        final int partCount = bodyParts.size();
220        if (partCount == 0) {
221            return NO_BYTES;
222        }
223        if (partCount == 1) {
224            byte[] chunk = bodyParts.get(0).getBodyPartBytes();
225            if (chunk.length <= maxLen) {
226                return chunk;
227            }
228            byte[] result = new byte[maxLen];
229            System.arraycopy(chunk, 0, result, 0, maxLen);
230            return result;
231        }
232        int size = 0;
233        byte[] result = new byte[maxLen];
234        for (HttpResponseBodyPart part : bodyParts) {
235            byte[] chunk = part.getBodyPartBytes();
236            int amount = Math.min(maxLen-size, chunk.length);
237            System.arraycopy(chunk, 0, result, size, amount);
238            size += amount;
239            if (size == maxLen) {
240                return result;
241            }
242        }
243        if (size < maxLen) {
244            byte[] old = result;
245            result = new byte[old.length];
246            System.arraycopy(old, 0, result, 0, old.length);
247        }
248        return result;
249    }

251    public final static InputStream contentAsStream(List<HttpResponseBodyPart> bodyParts)
252    {
253        switch (bodyParts.size()) {
254        case 0:
255            return new ByteArrayInputStream(NO_BYTES);
256        case 1:
257            return bodyParts.get(0).readBodyPartBytes();
258        }
259        Vector<InputStream> streams = new Vector<InputStream>(bodyParts.size());
260        for (HttpResponseBodyPart part : bodyParts) {
261            streams.add(part.readBodyPartBytes());
262        }
263        return new SequenceInputStream(streams.elements());
264    }

274    public final static URI getRedirectUri(URI uri, String location) {
275        if(location == null)
276            throw new IllegalArgumentException("URI " + uri + " was redirected to null location");
277        URI newUri = uri.resolve(location);
278
279        String scheme = newUri.getScheme();
280
281        if (scheme == null || !scheme.equalsIgnoreCase("http")
282                && !scheme.equalsIgnoreCase("https")
283                && !scheme.equals("ws")
284                && !scheme.equals("wss")) {
285            throw new IllegalArgumentException("The URI scheme, of the URI " + newUri
286                    + ", must be equal (ignoring case) to 'ws, 'wss', 'http', or 'https'");
287        }
288
289        return newUri;
290    }

307    public final static MultipartRequestEntity createMultipartRequestEntity(List<Part> params, FluentStringsMap methodParams) throws FileNotFoundException {
308        com.ning.http.multipart.Part[] parts = new com.ning.http.multipart.Part[params.size()];
309        int i = 0;
310
311        for (Part part : params) {
312            if (part instanceof com.ning.http.multipart.Part) {
313                parts[i] = (com.ning.http.multipart.Part) part;
314            } else if (part instanceof StringPart) {
315                parts[i] = new com.ning.http.multipart.StringPart(part.getName(),
316                        ((StringPart) part).getValue(),
317                        ((StringPart) part).getCharset());
318            } else if (part instanceof FilePart) {
319                parts[i] = new com.ning.http.multipart.FilePart(part.getName(),
320                        ((FilePart) part).getFile(),
321                        ((FilePart) part).getMimeType(),
322                        ((FilePart) part).getCharSet());
323
324            } else if (part instanceof ByteArrayPart) {
325                PartSource source = new ByteArrayPartSource(((ByteArrayPart) part).getFileName(), ((ByteArrayPart) part).getData());
326                parts[i] = new com.ning.http.multipart.FilePart(part.getName(),
327                        source,
328                        ((ByteArrayPart) part).getMimeType(),
329                        ((ByteArrayPart) part).getCharSet());
330
331            } else if (part == null) {
332                throw new NullPointerException("Part cannot be null");
333            } else {
334                throw new IllegalArgumentException(String.format("Unsupported part type for multipart parameter %s",
335                        part.getName()));
336            }
337            ++i;
338        }
339        return new MultipartRequestEntity(parts, methodParams);
340    }

342    public final static byte[] readFully(InputStream in, int[] lengthWrapper) throws IOException {
343        // just in case available() returns bogus (or -1), allocate non-trivial chunk
344        byte[] b = new byte[Math.max(512, in.available())];
345        int offset = 0;
346        while (true) {
347            int left = b.length - offset;
348            int count = in.read(b, offset, left);
349            if (count < 0) { // EOF
350                break;
351            }
352            offset += count;
353            if (count == left) { // full buffer, need to expand
354                b = doubleUp(b);
355            }
356        }
357        // wish Java had Tuple return type...
358        lengthWrapper[0] = offset;
359        return b;
360    }

362    private static byte[] doubleUp(byte[] b) {
363        int len = b.length;
364        byte[] b2 = new byte[len + len];
365        System.arraycopy(b, 0, b2, 0, len);
366        return b2;
367    }

369    public static String encodeCookies(Collection<Cookie> cookies) {
370        StringBuilder sb = new StringBuilder();
371
372        for (Cookie cookie : cookies) {
373            if (cookie.getVersion() >= 1) {
374                add(sb, '$' + VERSION, 1);
375            }
376
377            add(sb, cookie.getName(), cookie.getValue());
378
379            if (cookie.getPath() != null) {
380                add(sb, '$' + PATH, cookie.getPath());
381            }
382
383            if (cookie.getDomain() != null) {
384                add(sb, '$' + DOMAIN, cookie.getDomain());
385            }
386
387            if (cookie.getVersion() >= 1) {
388                if (!cookie.getPorts().isEmpty()) {
389                    sb.append('$');
390                    sb.append(PORT);
391                    sb.append((char) EQUALS);
392                    sb.append((char) DOUBLE_QUOTE);
393                    for (int port : cookie.getPorts()) {
394                        sb.append(port);
395                        sb.append((char) COMMA);
396                    }
397                    sb.setCharAt(sb.length() - 1, (char) DOUBLE_QUOTE);
398                    sb.append((char) SEMICOLON);
399                }
400            }
401        }
402
403        sb.setLength(sb.length() - 1);
404        return sb.toString();
405    }

407    private static void add(StringBuilder sb, String name, String val) {
408        if (val == null) {
409            addQuoted(sb, name, "");
410            return;
411        }
412
413        for (int i = 0; i < val.length(); i++) {
414            char c = val.charAt(i);
415            switch (c) {
416                case '\t':
417                case ' ':
418                case '"':
419                case '(':
420                case ')':
421                case ',':
422                case '/':
423                case ':':
424                case ';':
425                case '<':
426                case '=':
427                case '>':
428                case '?':
429                case '@':
430                case '[':
431                case '\\':
432                case ']':
433                case '{':
434                case '}':
435                    addQuoted(sb, name, val);
436                    return;
437            }
438        }
439
440        addUnquoted(sb, name, val);
441    }

443    private static void addUnquoted(StringBuilder sb, String name, String val) {
444        sb.append(name);
445        sb.append((char) EQUALS);
446        sb.append(val);
447        sb.append((char) SEMICOLON);
448    }

450    private static void addQuoted(StringBuilder sb, String name, String val) {
451        if (val == null) {
452            val = "";
453        }
454
455        sb.append(name);
456        sb.append((char) EQUALS);
457        sb.append((char) DOUBLE_QUOTE);
458        sb.append(val.replace("\\", "\\\\").replace("\"", "\\\""));
459        sb.append((char) DOUBLE_QUOTE);
460        sb.append((char) SEMICOLON);
461    }

487    public static String parseCharset(String contentType) {
488        for (String part : contentType.split(";")) {
489            if (part.trim().startsWith("charset=")) {
490                String[] val = part.split("=");
491                if (val.length > 1) {
492                    String charset = val[1].trim();
493                    // Quite a lot of sites have charset="CHARSET",
494                    // e.g. charset="utf-8". Note the quotes. This is 
495                    // not correct, but client should be able to handle
496                    // it (all browsers do, Apache HTTP Client and Grizzly 
497                    // strip it by default)
498                    // This is a poor man's trim("\"").trim("'")
499                    return charset.replaceAll("\"", "").replaceAll("'", "");
500                }
501            }
502        }
503        return null;
504    }

506    public static Cookie parseCookie(String value) {
507        String[] fields = value.split(";\\s*");
508        String[] cookie = fields[0].split("=");
509        String cookieName = cookie[0];
510        String cookieValue = (cookie.length == 1) ? null : cookie[1];
511
512        int maxAge = -1;
513        String path = null;
514        String domain = null;
515        boolean secure = false;
516
517        boolean maxAgeSet = false;
518        boolean expiresSet = false;
519
520        for (int j = 1; j < fields.length; j++) {
521            if ("secure".equalsIgnoreCase(fields[j])) {
522                secure = true;
523            } else if (fields[j].indexOf('=') > 0) {
524                String[] f = fields[j].split("=");
525                if (f.length == 1) continue; // Add protection against null field values
526
527                // favor 'max-age' field over 'expires'
528                if (!maxAgeSet && "max-age".equalsIgnoreCase(f[0])) {
529                    try {
530                        maxAge = Math.max(Integer.valueOf(removeQuote(f[1])), 0);
531                    } catch (NumberFormatException e1) {
532                        // ignore failure to parse -> treat as session cookie
533                        // invalidate a previously parsed expires-field
534                        maxAge = -1;
535                    }
536                    maxAgeSet = true;
537                } else if (!maxAgeSet && !expiresSet && "expires".equalsIgnoreCase(f[0])) {
538                    try {
539                        maxAge = Math.max(convertExpireField(f[1]), 0);
540                    } catch (Exception e) {
541                        // original behavior, is this correct at all (expires field with max-age semantics)?
542                        try {
543                            maxAge = Math.max(Integer.valueOf(f[1]), 0);
544                        } catch (NumberFormatException e1) {
545                            // ignore failure to parse -> treat as session cookie
546                        }
547                    }
548                    expiresSet = true;
549                } else if ("domain".equalsIgnoreCase(f[0])) {
550                    domain = f[1];
551                } else if ("path".equalsIgnoreCase(f[0])) {
552                    path = f[1];
553                }
554            }
555        }
556
557        return new Cookie(domain, cookieName, cookieValue, path, maxAge, secure);
558    }

560    public static int convertExpireField(String timestring) throws Exception {
561        Exception exception = null;
562        String trimmedTimeString = removeQuote(timestring.trim());
563        long now = System.currentTimeMillis();
564        for (SimpleDateFormat sdf : simpleDateFormat.get()) {
565            try {
566                long expire = sdf.parse(trimmedTimeString).getTime();
567                return (int) ((expire - now) / 1000);
568            } catch (ParseException e) {
569                exception = e;
570            } catch (NumberFormatException e) {
571                exception = e;
572            }
573        }
574
575        throw exception;
576    }

578    private final static String removeQuote(String s) {
579        if (s.startsWith("\"")) {
580            s = s.substring(1);
581        }
582
583        if (s.endsWith("\"")) {
584            s = s.substring(0, s.length() - 1);
585        }
586        return s;
587    }

589    public static void checkBodyParts(int statusCode, Collection<HttpResponseBodyPart> bodyParts) {
590        if (bodyParts == null || bodyParts.size() == 0) {
591
592            // We allow empty body on 204
593            if (statusCode == 204) return;
594
595            throw new IllegalStateException(BODY_NOT_COMPUTED);
596        }
