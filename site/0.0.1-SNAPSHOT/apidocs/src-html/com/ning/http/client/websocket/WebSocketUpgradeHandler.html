013package com.ning.http.client.websocket;
014
015import com.ning.http.client.AsyncHandler;
016import com.ning.http.client.HttpResponseBodyPart;
017import com.ning.http.client.HttpResponseHeaders;
018import com.ning.http.client.HttpResponseStatus;
019import com.ning.http.client.UpgradeHandler;
020
021import java.util.concurrent.ConcurrentLinkedQueue;
022import java.util.concurrent.atomic.AtomicBoolean;
023
024/**
025 * An {@link AsyncHandler} which is able to execute WebSocket upgrade. Use the Builder for configuring WebSocket options.
026 */
027public class WebSocketUpgradeHandler implements UpgradeHandler<WebSocket>, AsyncHandler<WebSocket> {
028
029    private WebSocket webSocket;
030    private final ConcurrentLinkedQueue<WebSocketListener> l;
031    @SuppressWarnings("unused")
032    private final String protocol;
033    @SuppressWarnings("unused")
034    private final long maxByteSize;
035    @SuppressWarnings("unused")
036    private final long maxTextSize;
037    private final AtomicBoolean ok = new AtomicBoolean(false);
038
039    private WebSocketUpgradeHandler(Builder b) {
040        l = b.l;
041        protocol = b.protocol;
042        maxByteSize = b.maxByteSize;
043        maxTextSize = b.maxTextSize;
044    }
045
046    /**
047     * {@inheritDoc}
048     */
049    @Override
050    public final void onThrowable(Throwable t) {
051        onFailure(t);
052    }
053
054    /**
055     * {@inheritDoc}
056     */
057    @Override
058    public final STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
059        return STATE.CONTINUE;
060    }
061
062    /**
063     * {@inheritDoc}
064     */
065    @Override
066    public final STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception {
067        if (responseStatus.getStatusCode() == 101) {
068            return STATE.UPGRADE;
069        } else {
070            return STATE.ABORT;
071        }
072    }
073
074    /**
075     * {@inheritDoc}
076     */
077    @Override
078    public final STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
079        return STATE.CONTINUE;
080    }
081
082    /**
083     * {@inheritDoc}
084     */
085    @Override
086    public final WebSocket onCompleted() throws Exception {
087        if (webSocket == null) {
088            throw new IllegalStateException("WebSocket is null");
089        }
090        return webSocket;
091    }
092
093    /**
094     * {@inheritDoc}
095     */
096    @Override
097    public final void onSuccess(WebSocket webSocket) {
098        this.webSocket = webSocket;
099        for (WebSocketListener w : l) {
100            webSocket.addWebSocketListener(w);
101            w.onOpen(webSocket);
102        }
103        ok.set(true);
104    }
105
106    /**
107     * {@inheritDoc}
108     */
109    @Override
110    public final void onFailure(Throwable t) {
111        for (WebSocketListener w : l) {
112            if (!ok.get() && webSocket != null) {
113                webSocket.addWebSocketListener(w);
114            }
115            w.onError(t);
116        }
117    }
118
119    public final void onClose(WebSocket webSocket, int status, String reasonPhrase) {
120        // Connect failure
121        if (this.webSocket == null) this.webSocket = webSocket;
122
123        for (WebSocketListener w : l) {
124            if (webSocket != null) {
125                webSocket.addWebSocketListener(w);
126            }
127            w.onClose(webSocket);
128            if (WebSocketCloseCodeReasonListener.class.isAssignableFrom(w.getClass())) {
129                WebSocketCloseCodeReasonListener.class.cast(w).onClose(webSocket, status, reasonPhrase);
130            }
131        }
132    }
133
134    /**
135     * Build a {@link WebSocketUpgradeHandler}
136     */
137    public final static class Builder {
138        private ConcurrentLinkedQueue<WebSocketListener> l = new ConcurrentLinkedQueue<WebSocketListener>();
139        private String protocol = "";
140        private long maxByteSize = 8192;
141        private long maxTextSize = 8192;
142
143        /**
144         * Add a {@link WebSocketListener} that will be added to the {@link WebSocket}
145         *
146         * @param listener a {@link WebSocketListener}
147         * @return this
148         */
149        public Builder addWebSocketListener(WebSocketListener listener) {
150            l.add(listener);
151            return this;
152        }
153
154        /**
155         * Remove a {@link WebSocketListener}
156         *
157         * @param listener a {@link WebSocketListener}
158         * @return this
159         */
160        public Builder removeWebSocketListener(WebSocketListener listener) {
161            l.remove(listener);
162            return this;
163        }
164
165        /**
166         * Set the WebSocket protocol.
167         *
168         * @param protocol the WebSocket protocol.
169         * @return this
170         */
171        public Builder setProtocol(String protocol) {
172            this.protocol = protocol;
173            return this;
174        }
175
176        /**
177         * Set the max size of the WebSocket byte message that will be sent.
178         *
179         * @param maxByteSize max size of the WebSocket byte message
180         * @return this
181         */
182        public Builder setMaxByteSize(long maxByteSize) {
183            this.maxByteSize = maxByteSize;
184            return this;
185        }
186
187        /**
188         * Set the max size of the WebSocket text message that will be sent.
189         *
190         * @param maxTextSize max size of the WebSocket byte message
191         * @return this
192         */
193        public Builder setMaxTextSize(long maxTextSize) {
194            this.maxTextSize = maxTextSize;
195            return this;
196        }
197
198        /**
199         * Build a {@link WebSocketUpgradeHandler}
200         * @return a {@link WebSocketUpgradeHandler}
201         */
202        public WebSocketUpgradeHandler build() {
203            return new WebSocketUpgradeHandler(this);
204        }
205    }
206}
