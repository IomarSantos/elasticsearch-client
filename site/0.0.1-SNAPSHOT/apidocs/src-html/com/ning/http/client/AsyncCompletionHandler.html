017package com.ning.http.client;
018
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * An {@link AsyncHandler} augmented with an {@link #onCompleted(Response)} convenience method which gets called
024 * when the {@link Response} processing is finished.  This class also implement the {@link ProgressAsyncHandler} callback,
025 * all doing nothing except returning {@link com.ning.http.client.AsyncHandler.STATE#CONTINUE}
026 *
027 * @param <T> Type of the value that will be returned by the associated {@link java.util.concurrent.Future}
028 */
029public abstract class AsyncCompletionHandler<T> implements AsyncHandler<T>, ProgressAsyncHandler<T> {
030
031    private final Logger log = LoggerFactory.getLogger(AsyncCompletionHandlerBase.class);
032    private final Response.ResponseBuilder builder = new Response.ResponseBuilder();
033
034    /**
035     * {@inheritDoc}
036     */
037    public STATE onBodyPartReceived(final HttpResponseBodyPart content) throws Exception {
038        builder.accumulate(content);
039        return STATE.CONTINUE;
040    }
041
042    /**
043     * {@inheritDoc}
044     */
045    public STATE onStatusReceived(final HttpResponseStatus status) throws Exception {
046        builder.reset();
047        builder.accumulate(status);
048        return STATE.CONTINUE;
049    }
050
051    /**
052     * {@inheritDoc}
053     */
054    public STATE onHeadersReceived(final HttpResponseHeaders headers) throws Exception {
055        builder.accumulate(headers);
056        return STATE.CONTINUE;
057    }
058
059    /**
060     * {@inheritDoc}
061     */
062    public final T onCompleted() throws Exception {
063        return onCompleted(builder.build());
064    }
065
066    /**
067     * {@inheritDoc}
068     */
069    public void onThrowable(Throwable t) {
070        log.debug(t.getMessage(), t);
071    }
072
073    /**
074     * Invoked once the HTTP response processing is finished.
075     * <p/>
076     * <p/>
077     * Gets always invoked as last callback method.
078     *
079     * @param response The {@link Response}
080     * @return T Value that will be returned by the associated {@link java.util.concurrent.Future}
081     * @throws Exception if something wrong happens
082     */
083    abstract public T onCompleted(Response response) throws Exception;
084
085    /**
086     * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
087     * written on the I/O socket.
088     *
089     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
090     */
091    public STATE onHeaderWriteCompleted() {
092        return STATE.CONTINUE;
093    }
094
095    /**
096     * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
097     * written on the I/O socket.
098     *
099     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
100     */
101    public STATE onContentWriteCompleted() {
102        return STATE.CONTINUE;
103    }
104
105    /**
106     * Invoked when the I/O operation associated with the {@link Request} body as been progressed.
107     *
108     * @param amount  The amount of bytes to transfer.
109     * @param current The amount of bytes transferred
110     * @param total   The total number of bytes transferred
111     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
112     */
113    public STATE onContentWriteProgress(long amount, long current, long total) {
114        return STATE.CONTINUE;
115    }
116}
