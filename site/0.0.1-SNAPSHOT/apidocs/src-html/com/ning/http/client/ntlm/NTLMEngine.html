111    /**
112     * Returns the response for the given message.
113     *
114     * @param message  the message that was received from the server.
115     * @param username the username to authenticate with.
116     * @param password the password to authenticate with.
117     * @param host     The host.
118     * @param domain   the NT domain to authenticate in.
119     * @return The response.
120     * @throws NTLMEngineException If the messages cannot be retrieved.
121     */
122    final String getResponseFor(String message, String username, String password,
123                                String host, String domain) throws NTLMEngineException {
124
125        final String response;
126        if (message == null || message.trim().equals("")) {
127            response = getType1Message(host, domain);
128        } else {
129            Type2Message t2m = new Type2Message(message);
130            response = getType3Message(username, password, host, domain, t2m.getChallenge(), t2m
131                    .getFlags(), t2m.getTarget(), t2m.getTargetInfo());
132        }
133        return response;
134    }

216    private static int readULong(byte[] src, int index) throws NTLMEngineException {
217        if (src.length < index + 4)
218            throw new NTLMEngineException("NTLM authentication - buffer too small for DWORD");
219        return (src[index] & 0xff) | ((src[index + 1] & 0xff) << 8)
220                | ((src[index + 2] & 0xff) << 16) | ((src[index + 3] & 0xff) << 24);
221    }
222
223    private static int readUShort(byte[] src, int index) throws NTLMEngineException {
224        if (src.length < index + 2)
225            throw new NTLMEngineException("NTLM authentication - buffer too small for WORD");
226        return (src[index] & 0xff) | ((src[index + 1] & 0xff) << 8);
227    }
228
229    private static byte[] readSecurityBuffer(byte[] src, int index) throws NTLMEngineException {
230        int length = readUShort(src, index);
231        int offset = readULong(src, index + 4);
232        if (src.length < offset + length)
233            throw new NTLMEngineException(
234                    "NTLM authentication - buffer too small for data item");
235        byte[] buffer = new byte[length];
236        System.arraycopy(src, offset, buffer, 0, length);
237        return buffer;
238    }

240    /**
241     * Calculate a challenge block
242     */
243    private static byte[] makeRandomChallenge() throws NTLMEngineException {
244        if (RND_GEN == null) {
245            throw new NTLMEngineException("Random generator not available");
246        }
247        byte[] rval = new byte[8];
248        synchronized (RND_GEN) {
249            RND_GEN.nextBytes(rval);
250        }
251        return rval;
252    }

254    /**
255     * Calculate an NTLM2 challenge block
256     */
257    private static byte[] makeNTLM2RandomChallenge() throws NTLMEngineException {
258        if (RND_GEN == null) {
259            throw new NTLMEngineException("Random generator not available");
260        }
261        byte[] rval = new byte[24];
262        synchronized (RND_GEN) {
263            RND_GEN.nextBytes(rval);
264        }
265        // 8-byte challenge, padded with zeros to 24 bytes.
266        Arrays.fill(rval, 8, 24, (byte) 0x00);
267        return rval;
268    }

337     * Calculates the NTLM2 Session Response for the given challenge, using the
338     * specified password and client challenge.
339     *
340     * @param password        The user's password.
341     * @param challenge       The Type 2 challenge from the server.
342     * @param clientChallenge The random 8-byte client challenge.
343     * @return The NTLM2 Session Response. This is placed in the NTLM response
344     *         field of the Type 3 message; the LM response field contains the
345     *         client challenge, null-padded to 24 bytes.

347    static byte[] getNTLM2SessionResponse(String password, byte[] challenge,
348                                          byte[] clientChallenge) throws NTLMEngineException {
349        try {
350            byte[] ntlmHash = ntlmHash(password);
351
352            // Look up MD5 algorithm (was necessary on jdk 1.4.2)
353            // This used to be needed, but java 1.5.0_07 includes the MD5
354            // algorithm (finally)
355            // Class x = Class.forName("gnu.crypto.hash.MD5");
356            // Method updateMethod = x.getMethod("update",new
357            // Class[]{byte[].class});
358            // Method digestMethod = x.getMethod("digest",new Class[0]);
359            // Object mdInstance = x.newInstance();
360            // updateMethod.invoke(mdInstance,new Object[]{challenge});
361            // updateMethod.invoke(mdInstance,new Object[]{clientChallenge});
362            // byte[] digest = (byte[])digestMethod.invoke(mdInstance,new
363            // Object[0]);
364
365            MessageDigest md5 = MessageDigest.getInstance("MD5");
366            md5.update(challenge);
367            md5.update(clientChallenge);
368            byte[] digest = md5.digest();
369

371            System.arraycopy(digest, 0, sessionHash, 0, 8);
372            return lmResponse(ntlmHash, sessionHash);
373        } catch (Exception e) {

380    /**
381     * Creates the LM Hash of the user's password.
382     *
383     * @param password The password.
384     * @return The LM Hash of the given password, used in the calculation of the
385     *         LM Response.
386     */
387    private static byte[] lmHash(String password) throws NTLMEngineException {
388        try {
389            byte[] oemPassword = password.toUpperCase(Locale.ENGLISH).getBytes("US-ASCII");
390            int length = Math.min(oemPassword.length, 14);
391            byte[] keyBytes = new byte[14];
392            System.arraycopy(oemPassword, 0, keyBytes, 0, length);
393            Key lowKey = createDESKey(keyBytes, 0);
394            Key highKey = createDESKey(keyBytes, 7);
395            byte[] magicConstant = "KGS!@#$%".getBytes("US-ASCII");
396            Cipher des = Cipher.getInstance("DES/ECB/NoPadding");
397            des.init(Cipher.ENCRYPT_MODE, lowKey);
398            byte[] lowHash = des.doFinal(magicConstant);
399            des.init(Cipher.ENCRYPT_MODE, highKey);
400            byte[] highHash = des.doFinal(magicConstant);
401            byte[] lmHash = new byte[16];
402            System.arraycopy(lowHash, 0, lmHash, 0, 8);
403            System.arraycopy(highHash, 0, lmHash, 8, 8);
404            return lmHash;
405        } catch (Exception e) {
406            throw new NTLMEngineException(e.getMessage(), e);
407        }
408    }

410    /**
411     * Creates the NTLM Hash of the user's password.
412     *
413     * @param password The password.
414     * @return The NTLM Hash of the given password, used in the calculation of
415     *         the NTLM Response and the NTLMv2 and LMv2 Hashes.
416     */
417    private static byte[] ntlmHash(String password) throws NTLMEngineException {
418        try {
419            byte[] unicodePassword = password.getBytes("UnicodeLittleUnmarked");
420            MD4 md4 = new MD4();
421            md4.update(unicodePassword);
422            return md4.getOutput();
423        } catch (java.io.UnsupportedEncodingException e) {
424            throw new NTLMEngineException("Unicode not supported: " + e.getMessage(), e);
425        }
426    }

428    /**
429     * Creates the NTLMv2 Hash of the user's password.
430     *
431     * @param target   The authentication target (i.e., domain).
432     * @param user     The username.
433     * @param password The password.
434     * @return The NTLMv2 Hash, used in the calculation of the NTLMv2 and LMv2
435     *         Responses.
436     */
437    private static byte[] ntlmv2Hash(String target, String user, String password)
438            throws NTLMEngineException {
439        try {
440            byte[] ntlmHash = ntlmHash(password);
441            HMACMD5 hmacMD5 = new HMACMD5(ntlmHash);
442            // Upper case username, mixed case target!!
443            hmacMD5.update(user.toUpperCase(Locale.ENGLISH).getBytes("UnicodeLittleUnmarked"));
444            hmacMD5.update(target.getBytes("UnicodeLittleUnmarked"));
445            return hmacMD5.getOutput();
446        } catch (java.io.UnsupportedEncodingException e) {
447            throw new NTLMEngineException("Unicode not supported! " + e.getMessage(), e);
448        }
449    }

451    /**
452     * Creates the LM Response from the given hash and Type 2 challenge.
453     *
454     * @param hash      The LM or NTLM Hash.
455     * @param challenge The server challenge from the Type 2 message.
456     * @return The response (either LM or NTLM, depending on the provided hash).
457     */
458    private static byte[] lmResponse(byte[] hash, byte[] challenge) throws NTLMEngineException {
459        try {
460            byte[] keyBytes = new byte[21];
461            System.arraycopy(hash, 0, keyBytes, 0, 16);
462            Key lowKey = createDESKey(keyBytes, 0);
463            Key middleKey = createDESKey(keyBytes, 7);
464            Key highKey = createDESKey(keyBytes, 14);
465            Cipher des = Cipher.getInstance("DES/ECB/NoPadding");
466            des.init(Cipher.ENCRYPT_MODE, lowKey);
467            byte[] lowResponse = des.doFinal(challenge);
468            des.init(Cipher.ENCRYPT_MODE, middleKey);
469            byte[] middleResponse = des.doFinal(challenge);
470            des.init(Cipher.ENCRYPT_MODE, highKey);
471            byte[] highResponse = des.doFinal(challenge);
472            byte[] lmResponse = new byte[24];
473            System.arraycopy(lowResponse, 0, lmResponse, 0, 8);
474            System.arraycopy(middleResponse, 0, lmResponse, 8, 8);
475            System.arraycopy(highResponse, 0, lmResponse, 16, 8);
476            return lmResponse;
477        } catch (Exception e) {
478            throw new NTLMEngineException(e.getMessage(), e);
479        }
480    }

482    /**
483     * Creates the LMv2 Response from the given hash, client data, and Type 2
484     * challenge.
485     *
486     * @param hash       The NTLMv2 Hash.
487     * @param clientData The client data (blob or client challenge).
488     * @param challenge  The server challenge from the Type 2 message.
489     * @return The response (either NTLMv2 or LMv2, depending on the client
490     *         data).
491     */
492    private static byte[] lmv2Response(byte[] hash, byte[] challenge, byte[] clientData)
493            throws NTLMEngineException {
494        HMACMD5 hmacMD5 = new HMACMD5(hash);
495        hmacMD5.update(challenge);
496        hmacMD5.update(clientData);
497        byte[] mac = hmacMD5.getOutput();
498        byte[] lmv2Response = new byte[mac.length + clientData.length];
499        System.arraycopy(mac, 0, lmv2Response, 0, mac.length);
500        System.arraycopy(clientData, 0, lmv2Response, mac.length, clientData.length);
501        return lmv2Response;
502    }

504    /**
505     * Creates the NTLMv2 blob from the given target information block and
506     * client challenge.
507     *
508     * @param targetInformation The target information block from the Type 2 message.
509     * @param clientChallenge   The random 8-byte client challenge.
510     * @return The blob, used in the calculation of the NTLMv2 Response.
511     */
512    private static byte[] createBlob(byte[] clientChallenge, byte[] targetInformation) {
513        byte[] blobSignature = new byte[]{(byte) 0x01, (byte) 0x01, (byte) 0x00, (byte) 0x00};
514        byte[] reserved = new byte[]{(byte) 0x00, (byte) 0x00, (byte) 0x00, (byte) 0x00};
515        byte[] unknown1 = new byte[]{(byte) 0x00, (byte) 0x00, (byte) 0x00, (byte) 0x00};
516        long time = System.currentTimeMillis();
517        time += 11644473600000l; // milliseconds from January 1, 1601 -> epoch.
518        time *= 10000; // tenths of a microsecond.
519        // convert to little-endian byte array.
520        byte[] timestamp = new byte[8];
521        for (int i = 0; i < 8; i++) {
522            timestamp[i] = (byte) time;
523            time >>>= 8;
524        }
525        byte[] blob = new byte[blobSignature.length + reserved.length + timestamp.length + 8
526                + unknown1.length + targetInformation.length];
527        int offset = 0;
528        System.arraycopy(blobSignature, 0, blob, offset, blobSignature.length);
529        offset += blobSignature.length;
530        System.arraycopy(reserved, 0, blob, offset, reserved.length);
531        offset += reserved.length;
532        System.arraycopy(timestamp, 0, blob, offset, timestamp.length);
533        offset += timestamp.length;
534        System.arraycopy(clientChallenge, 0, blob, offset, 8);
535        offset += 8;
536        System.arraycopy(unknown1, 0, blob, offset, unknown1.length);
537        offset += unknown1.length;
538        System.arraycopy(targetInformation, 0, blob, offset, targetInformation.length);
539        return blob;
540    }

542    /**
543     * Creates a DES encryption key from the given key material.
544     *
545     * @param bytes  A byte array containing the DES key material.
546     * @param offset The offset in the given byte array at which the 7-byte key
547     *               material starts.
548     * @return A DES encryption key created from the key material starting at
549     *         the specified offset in the given byte array.
550     */
551    private static Key createDESKey(byte[] bytes, int offset) {
552        byte[] keyBytes = new byte[7];
553        System.arraycopy(bytes, offset, keyBytes, 0, 7);
554        byte[] material = new byte[8];
555        material[0] = keyBytes[0];
556        material[1] = (byte) (keyBytes[0] << 7 | (keyBytes[1] & 0xff) >>> 1);
557        material[2] = (byte) (keyBytes[1] << 6 | (keyBytes[2] & 0xff) >>> 2);
558        material[3] = (byte) (keyBytes[2] << 5 | (keyBytes[3] & 0xff) >>> 3);
559        material[4] = (byte) (keyBytes[3] << 4 | (keyBytes[4] & 0xff) >>> 4);
560        material[5] = (byte) (keyBytes[4] << 3 | (keyBytes[5] & 0xff) >>> 5);
561        material[6] = (byte) (keyBytes[5] << 2 | (keyBytes[6] & 0xff) >>> 6);
562        material[7] = (byte) (keyBytes[6] << 1);
563        oddParity(material);
564        return new SecretKeySpec(material, "DES");
565    }

567    /**
568     * Applies odd parity to the given byte array.
569     *
570     * @param bytes The data whose parity bits are to be adjusted for odd parity.
571     */
572    private static void oddParity(byte[] bytes) {
573        for (int i = 0; i < bytes.length; i++) {
574            byte b = bytes[i];
575            boolean needsParity = (((b >>> 7) ^ (b >>> 6) ^ (b >>> 5) ^ (b >>> 4) ^ (b >>> 3)
576                    ^ (b >>> 2) ^ (b >>> 1)) & 0x01) == 0;
577            if (needsParity) {
578                bytes[i] |= (byte) 0x01;
579            } else {
580                bytes[i] &= (byte) 0xfe;
581            }
582        }
583    }

585    /**
586     * NTLM message generation, base class
587     */
588    static class NTLMMessage {
589        /**
590         * The current response
591         */
592        private byte[] messageContents = null;
593
594        /**
595         * The current output position
596         */
597        private int currentOutputPosition = 0;
598
599        /**
600         * Constructor to use when message contents are not yet known
601         */
602        NTLMMessage() {
603        }
604
605        /**
606         * Constructor to use when message contents are known
607         */
608        NTLMMessage(String messageBody, int expectedType) throws NTLMEngineException {
609            messageContents = Base64.decode(messageBody);
610
611            // Look for NTLM message
612            if (messageContents.length < SIGNATURE.length)
613                throw new NTLMEngineException("NTLM message decoding error - packet too short");
614            int i = 0;
615            while (i < SIGNATURE.length) {
616                if (messageContents[i] != SIGNATURE[i])
617                    throw new NTLMEngineException(
618                            "NTLM message expected - instead got unrecognized bytes");
619                i++;
620            }
621
622            // Check to be sure there's a type 2 message indicator next
623            int type = readULong(SIGNATURE.length);
624            if (type != expectedType)
625                throw new NTLMEngineException("NTLM type " + Integer.toString(expectedType)
626                        + " message expected - instead got type " + Integer.toString(type));
627
628            currentOutputPosition = messageContents.length;
629        }
630

632         * Get the length of the signature and flags, so calculations can adjust
633         * offsets accordingly.
634         */
635        protected int getPreambleLength() {
636            return SIGNATURE.length + 4;
637        }
638
639        /**
640         * Get the message length
641         */
642        protected int getMessageLength() {
643            return currentOutputPosition;
644        }
645
646        /**
647         * Read a byte from a position within the message buffer
648         */
649        protected byte readByte(int position) throws NTLMEngineException {
650            if (messageContents.length < position + 1)
651                throw new NTLMEngineException("NTLM: Message too short");
652            return messageContents[position];
653        }
654
655        /**
656         * Read a bunch of bytes from a position in the message buffer
657         */
658        protected void readBytes(byte[] buffer, int position) throws NTLMEngineException {
659            if (messageContents.length < position + buffer.length)
660                throw new NTLMEngineException("NTLM: Message too short");
661            System.arraycopy(messageContents, position, buffer, 0, buffer.length);
662        }

665         * Read a ushort from a position within the message buffer
666         */
667        protected int readUShort(int position) throws NTLMEngineException {
668            return NTLMEngine.readUShort(messageContents, position);
669        }
670
671        /**
672         * Read a ulong from a position within the message buffer
673         */
674        protected int readULong(int position) throws NTLMEngineException {
675            return NTLMEngine.readULong(messageContents, position);
676        }

678        /**
679         * Read a security buffer from a position within the message buffer
680         */
681        protected byte[] readSecurityBuffer(int position) throws NTLMEngineException {
682            return NTLMEngine.readSecurityBuffer(messageContents, position);
683        }
684
685        /**
686         * Prepares the object to create a response of the given length.
687         *
688         * @param maxlength the maximum length of the response to prepare, not
689         *                  including the type and the signature (which this method
690         *                  adds).
691         */
692        protected void prepareResponse(int maxlength, int messageType) {
693            messageContents = new byte[maxlength];
694            currentOutputPosition = 0;
695            addBytes(SIGNATURE);
696            addULong(messageType);
697        }
698
699        /**
700         * Adds the given byte to the response.
701         *
702         * @param b the byte to add.
703         */
704        protected void addByte(byte b) {
705            messageContents[currentOutputPosition] = b;
706            currentOutputPosition++;
707        }
708
709        /**
710         * Adds the given bytes to the response.
711         *
712         * @param bytes the bytes to add.
713         */
714        protected void addBytes(byte[] bytes) {
715            for (int i = 0; i < bytes.length; i++) {
716                messageContents[currentOutputPosition] = bytes[i];
717                currentOutputPosition++;
718            }
719        }
720
721        /**
722         * Adds a USHORT to the response
723         */
724        protected void addUShort(int value) {
725            addByte((byte) (value & 0xff));
726            addByte((byte) (value >> 8 & 0xff));
727        }
728
729        /**
730         * Adds a ULong to the response
731         */
732        protected void addULong(int value) {
733            addByte((byte) (value & 0xff));
734            addByte((byte) (value >> 8 & 0xff));
735            addByte((byte) (value >> 16 & 0xff));
736            addByte((byte) (value >> 24 & 0xff));
737        }
738
739        /**
740         * Returns the response that has been generated after shrinking the
741         * array if required and base64 encodes the response.
742         *
743         * @return The response as above.
744         */
745        String getResponse() throws UnsupportedEncodingException {
746            byte[] resp;
747            if (messageContents.length > currentOutputPosition) {
748                byte[] tmp = new byte[currentOutputPosition];
749                for (int i = 0; i < currentOutputPosition; i++) {
750                    tmp[i] = messageContents[i];
751                }
752                resp = tmp;
753            } else {
754                resp = messageContents;
755            }
756            return Base64.encode(resp);
757        }

770         */

793            // signature or type.
794            int finalLength = 32 + hostBytes.length + domainBytes.length;
795
796            // Set up the response. This will initialize the signature, message
797            // type, and flags.
798            prepareResponse(finalLength, 1);

810            addUShort(domainBytes.length);

813            addULong(hostBytes.length + 32);

817            addUShort(hostBytes.length);
818

820            addULong(32);

826            addBytes(domainBytes);

829        }

837        protected byte[] challenge;
838        protected String target;
839        protected byte[] targetInfo;
840        protected int flags;
841
842        Type2Message(String message) throws NTLMEngineException {
843            super(message, 2);
844
845            // Parse out the rest of the info we need from the message
846            // The nonce is the 8 bytes starting from the byte in position 24.
847            challenge = new byte[8];
848            readBytes(challenge, 24);
849
850            flags = readULong(20);
851            if ((flags & FLAG_UNICODE_ENCODING) == 0)
852                throw new NTLMEngineException(
853                        "NTLM type 2 message has flags that make no sense: "
854                                + Integer.toString(flags));
855            // Do the target!
856            target = null;
857            // The TARGET_DESIRED flag is said to not have understood semantics
858            // in Type2 messages, so use the length of the packet to decide
859            // how to proceed instead
860            if (getMessageLength() >= 12 + 8) {
861                byte[] bytes = readSecurityBuffer(12);
862                if (bytes.length != 0) {
863                    try {
864                        target = new String(bytes, "UnicodeLittleUnmarked");
865                    } catch (java.io.UnsupportedEncodingException e) {
866                        throw new NTLMEngineException(e.getMessage(), e);
867                    }
868                }
869            }
870
871            // Do the target info!
872            targetInfo = null;
873            // TARGET_DESIRED flag cannot be relied on, so use packet length
874            if (getMessageLength() >= 40 + 8) {
875                byte[] bytes = readSecurityBuffer(40);
876                if (bytes.length != 0) {
877                    targetInfo = bytes;
878                }
879            }
880        }
881

884         */

886            return challenge;
887        }

892        String getTarget() {

898         */
899        byte[] getTargetInfo() {
900            return targetInfo;
901        }
902
903        /**
904         * Retrieve the response flags
905         */
906        int getFlags() {
907            return flags;
908        }

915    static class Type3Message extends NTLMMessage {
916        // Response flags from the type2 message
917        protected int type2Flags;
918
919        protected byte[] domainBytes;
920        protected byte[] hostBytes;
921        protected byte[] userBytes;
922
923        protected byte[] lmResp;
924        protected byte[] ntResp;
925
926        /**
927         * Constructor. Pass the arguments we will need
928         */
929        Type3Message(String domain, String host, String user, String password, byte[] nonce,
930                     int type2Flags, String target, byte[] targetInformation)
931                throws NTLMEngineException {
932            // Save the flags
933            this.type2Flags = type2Flags;
934
935            // Strip off domain name from the host!
936            host = convertHost(host);
937            // Use only the base domain name!
938            domain = convertDomain(domain);
939
940            // Use the new code to calculate the responses, including v2 if that
941            // seems warranted.
942            try {
943                if (targetInformation != null && target != null) {
944                    byte[] clientChallenge = makeRandomChallenge();
945                    ntResp = getNTLMv2Response(target, user, password, nonce, clientChallenge,
946                            targetInformation);
947                    lmResp = getLMv2Response(target, user, password, nonce, clientChallenge);
948                } else {
949                    if ((type2Flags & FLAG_NEGOTIATE_NTLM2) != 0) {
950                        // NTLM2 session stuff is requested
951                        byte[] clientChallenge = makeNTLM2RandomChallenge();
952
953                        ntResp = getNTLM2SessionResponse(password, nonce, clientChallenge);
954                        lmResp = clientChallenge;
955
956                        // All the other flags we send (signing, sealing, key
957                        // exchange) are supported, but they don't do anything
958                        // at all in an
959                        // NTLM2 context! So we're done at this point.
960                    } else {
961                        ntResp = getNTLMResponse(password, nonce);
962                        lmResp = getLMResponse(password, nonce);
963                    }
964                }
965            } catch (NTLMEngineException e) {
966                // This likely means we couldn't find the MD4 hash algorithm -
967                // fail back to just using LM
968                ntResp = new byte[0];
969                lmResp = getLMResponse(password, nonce);
970            }
971
972            try {
973                domainBytes = domain.toUpperCase(Locale.ENGLISH).getBytes("UnicodeLittleUnmarked");
974                hostBytes = host.getBytes("UnicodeLittleUnmarked");
975                userBytes = user.getBytes("UnicodeLittleUnmarked");
976            } catch (java.io.UnsupportedEncodingException e) {
977                throw new NTLMEngineException("Unicode not supported: " + e.getMessage(), e);
978            }
979        }
980
981        /**
982         * Assemble the response
983         */
984        @Override
985        String getResponse() throws UnsupportedEncodingException {
986            int ntRespLen = ntResp.length;
987            int lmRespLen = lmResp.length;
988
989            int domainLen = domainBytes.length;
990            int hostLen = hostBytes.length;
991            int userLen = userBytes.length;
992
993            // Calculate the layout within the packet
994            int lmRespOffset = 64;
995            int ntRespOffset = lmRespOffset + lmRespLen;
996            int domainOffset = ntRespOffset + ntRespLen;
997            int userOffset = domainOffset + domainLen;
998            int hostOffset = userOffset + userLen;
999            int finalLength = hostOffset + hostLen;
1000
1001            // Start the response. Length includes signature and type
1002            prepareResponse(finalLength, 3);
1003
1004            // LM Resp Length (twice)
1005            addUShort(lmRespLen);
1006            addUShort(lmRespLen);
1007
1008            // LM Resp Offset
1009            addULong(lmRespOffset);
1010
1011            // NT Resp Length (twice)
1012            addUShort(ntRespLen);
1013            addUShort(ntRespLen);
1014
1015            // NT Resp Offset
1016            addULong(ntRespOffset);
1017
1018            // Domain length (twice)
1019            addUShort(domainLen);
1020            addUShort(domainLen);
1021
1022            // Domain offset.
1023            addULong(domainOffset);
1024
1025            // User Length (twice)
1026            addUShort(userLen);
1027            addUShort(userLen);
1028
1029            // User offset
1030            addULong(userOffset);
1031
1032            // Host length (twice)
1033            addUShort(hostLen);
1034            addUShort(hostLen);
1035
1036            // Host offset
1037            addULong(hostOffset);
1038
1039            // 4 bytes of zeros - not sure what this is
1040            addULong(0);
1041
1042            // Message length
1043            addULong(finalLength);
1044
1045            // Flags. Currently: NEGOTIATE_NTLM + UNICODE_ENCODING +
1046            // TARGET_DESIRED + NEGOTIATE_128
1047            addULong(FLAG_NEGOTIATE_NTLM | FLAG_UNICODE_ENCODING | FLAG_TARGET_DESIRED
1048                    | FLAG_NEGOTIATE_128 | (type2Flags & FLAG_NEGOTIATE_NTLM2)
1049                    | (type2Flags & FLAG_NEGOTIATE_SIGN) | (type2Flags & FLAG_NEGOTIATE_SEAL)
1050                    | (type2Flags & FLAG_NEGOTIATE_KEY_EXCH)
1051                    | (type2Flags & FLAG_NEGOTIATE_ALWAYS_SIGN));
1052
1053            // Add the actual data
1054            addBytes(lmResp);
1055            addBytes(ntResp);
1056            addBytes(domainBytes);
1057            addBytes(userBytes);
1058            addBytes(hostBytes);
1059
1060            return super.getResponse();
1061        }

1088     * Cryptography support - MD4. The following class was based loosely on the
1089     * RFC and on code found at http://www.cs.umd.edu/~harry/jotp/src/md.java.
1090     * Code correctness was verified by looking at MD4.java from the jcifs
1091     * library (http://jcifs.samba.org). It was massaged extensively to the
1092     * final form found here by Karl Wright (kwright@metacarta.com).
1093     */
1094    static class MD4 {
1095        protected int A = 0x67452301;
1096        protected int B = 0xefcdab89;
1097        protected int C = 0x98badcfe;
1098        protected int D = 0x10325476;
1099        protected long count = 0L;
1100        protected byte[] dataBuffer = new byte[64];
1101
1102        MD4() {
1103        }
1104
1105        void update(byte[] input) {
1106            // We always deal with 512 bits at a time. Correspondingly, there is
1107            // a buffer 64 bytes long that we write data into until it gets
1108            // full.
1109            int curBufferPos = (int) (count & 63L);
1110            int inputIndex = 0;
1111            while (input.length - inputIndex + curBufferPos >= dataBuffer.length) {
1112                // We have enough data to do the next step. Do a partial copy
1113                // and a transform, updating inputIndex and curBufferPos
1114                // accordingly
1115                int transferAmt = dataBuffer.length - curBufferPos;
1116                System.arraycopy(input, inputIndex, dataBuffer, curBufferPos, transferAmt);
1117                count += transferAmt;
1118                curBufferPos = 0;
1119                inputIndex += transferAmt;
1120                processBuffer();
1121            }
1122
1123            // If there's anything left, copy it into the buffer and leave it.
1124            // We know there's not enough left to process.
1125            if (inputIndex < input.length) {
1126                int transferAmt = input.length - inputIndex;
1127                System.arraycopy(input, inputIndex, dataBuffer, curBufferPos, transferAmt);
1128                count += transferAmt;
1129            }
1130        }
1131
1132        byte[] getOutput() {
1133            // Feed pad/length data into engine. This must round out the input
1134            // to a multiple of 512 bits.
1135            int bufferIndex = (int) (count & 63L);
1136            int padLen = (bufferIndex < 56) ? (56 - bufferIndex) : (120 - bufferIndex);
1137            byte[] postBytes = new byte[padLen + 8];
1138            // Leading 0x80, specified amount of zero padding, then length in
1139            // bits.
1140            postBytes[0] = (byte) 0x80;
1141            // Fill out the last 8 bytes with the length
1142            for (int i = 0; i < 8; i++) {
1143                postBytes[padLen + i] = (byte) ((count * 8) >>> (8 * i));
1144            }
1145
1146            // Update the engine
1147            update(postBytes);
1148
1149            // Calculate final result
1150            byte[] result = new byte[16];
1151            writeULong(result, A, 0);
1152            writeULong(result, B, 4);
1153            writeULong(result, C, 8);
1154            writeULong(result, D, 12);
1155            return result;
1156        }
1157
1158        protected void processBuffer() {
1159            // Convert current buffer to 16 ulongs
1160            int[] d = new int[16];
1161
1162            for (int i = 0; i < 16; i++) {
1163                d[i] = (dataBuffer[i * 4] & 0xff) + ((dataBuffer[i * 4 + 1] & 0xff) << 8)
1164                        + ((dataBuffer[i * 4 + 2] & 0xff) << 16)
1165                        + ((dataBuffer[i * 4 + 3] & 0xff) << 24);
1166            }
1167
1168            // Do a round of processing
1169            int AA = A;
1170            int BB = B;
1171            int CC = C;
1172            int DD = D;
1173            round1(d);
1174            round2(d);
1175            round3(d);
1176            A += AA;
1177            B += BB;
1178            C += CC;
1179            D += DD;
1180
1181        }

1254     * Cryptography support - HMACMD5 - algorithmically based on various web
1255     * resources by Karl Wright
1256     */
1257    static class HMACMD5 {
1258        protected byte[] ipad;
1259        protected byte[] opad;
1260        protected MessageDigest md5;
1261
1262        HMACMD5(byte[] key) throws NTLMEngineException {
1263            try {
1264                md5 = MessageDigest.getInstance("MD5");
1265            } catch (Exception ex) {
1266                // Umm, the algorithm doesn't exist - throw an
1267                // NTLMEngineException!
1268                throw new NTLMEngineException(
1269                        "Error getting md5 message digest implementation: " + ex.getMessage(), ex);
1270            }
1271
1272            // Initialize the pad buffers with the key
1273            ipad = new byte[64];
1274            opad = new byte[64];
1275
1276            int keyLength = key.length;
1277            if (keyLength > 64) {
1278                // Use MD5 of the key instead, as described in RFC 2104
1279                md5.update(key);
1280                key = md5.digest();
1281                keyLength = key.length;
1282            }
1283            int i = 0;
1284            while (i < keyLength) {
1285                ipad[i] = (byte) (key[i] ^ (byte) 0x36);
1286                opad[i] = (byte) (key[i] ^ (byte) 0x5c);
1287                i++;
1288            }
1289            while (i < 64) {
1290                ipad[i] = (byte) 0x36;
1291                opad[i] = (byte) 0x5c;
1292                i++;
1293            }
1294
1295            // Very important: update the digest with the ipad buffer
1296            md5.reset();
1297            md5.update(ipad);
1298
1299        }
1300
1301        /**
1302         * Grab the current digest. This is the "answer".
1303         */
1304        byte[] getOutput() {
1305            byte[] digest = md5.digest();
1306            md5.update(opad);
1307            return md5.digest(digest);
1308        }
1309
1310        /**
1311         * Update by adding a complete array
1312         */
1313        void update(byte[] input) {
1314            md5.update(input);
1315        }
1316
1317        /**
1318         * Update the algorithm
1319         */
1320        void update(byte[] input, int offset, int length) {
1321            md5.update(input, offset, length);
1322        }
1323
1324    }
