013package com.ning.http.client;
014
015/**
016 * An extended {@link AsyncHandler} with two extra callback who get invoked during the content upload to a remote server.
017 * This {@link AsyncHandler} must be used only with PUT and POST request.
018 */
019public interface ProgressAsyncHandler<T> extends AsyncHandler<T> {
020
021    /**
022     * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
023     * written on the I/O socket.
024     *
025     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
026     */
027    STATE onHeaderWriteCompleted();
028
029    /**
030     * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
031     * written on the I/O socket.
032     *
033     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
034     */
035    STATE onContentWriteCompleted();
036
037    /**
038     * Invoked when the I/O operation associated with the {@link Request} body wasn't fully written in a single I/O write
039     * operation. This method is never invoked if the write operation complete in a sinfle I/O write.
040     *
041     * @param amount  The amount of bytes to transfer.
042     * @param current The amount of bytes transferred
043     * @param total   The total number of bytes transferred
044     * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
045     */
046    STATE onContentWriteProgress(long amount, long current, long total);
047
048}
