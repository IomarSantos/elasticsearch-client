013package com.ning.http.client;
014
015import com.ning.http.client.Response.ResponseBuilder;
016
017import java.io.FilterInputStream;
018import java.io.IOException;
019import java.io.InputStream;
020import java.io.OutputStream;
021import java.util.concurrent.CountDownLatch;
022import java.util.concurrent.ExecutionException;
023import java.util.concurrent.Future;
024import java.util.concurrent.Semaphore;
025
026/**
027 * An AsyncHandler that returns Response (without body, so status code and
028 * headers only) as fast as possible for inspection, but leaves you the option
029 * to defer body consumption.
030 * <p/>
031 * This class introduces new call: getResponse(), that blocks caller thread as
032 * long as headers are received, and return Response as soon as possible, but
033 * still pouring response body into supplied output stream. This handler is
034 * meant for situations when the "recommended" way (using
035 * <code>client.prepareGet("http://foo.com/aResource").execute().get()</code>
036 * would not work for you, since a potentially large response body is about to
037 * be GETted, but you need headers first, or you don't know yet (depending on
038 * some logic, maybe coming from headers) where to save the body, or you just
039 * want to leave body stream to some other component to consume it.
040 * <p/>
041 * All these above means that this AsyncHandler needs a bit of different
042 * handling than "recommended" way. Some examples:
043 * <p/>
044 * <pre>
045 *     FileOutputStream fos = ...
046 *     BodyDeferringAsyncHandler bdah = new BodyDeferringAsyncHandler(fos);
047 *     // client executes async
048 *     Future&lt;Response&gt; fr = client.prepareGet(&quot;http://foo.com/aresource&quot;).execute(
049 *      bdah);
050 *     // main thread will block here until headers are available
051 *     Response response = bdah.getResponse();
052 *     // you can continue examine headers while actual body download happens
053 *     // in separate thread
054 *     // ...
055 *     // finally &quot;join&quot; the download
056 *     fr.get();
057 * </pre>
058 * <p/>
059 * <pre>
060 *     PipedOutputStream pout = new PipedOutputStream();
061 *     BodyDeferringAsyncHandler bdah = new BodyDeferringAsyncHandler(pout);
062 *     // client executes async
063 *     Future&lt;Response&gt; fr = client.prepareGet(&quot;http://foo.com/aresource&quot;).execute(bdah);
064 *     // main thread will block here until headers are available
065 *     Response response = bdah.getResponse();
066 *     if (response.getStatusCode() == 200) {
067 *      InputStream pin = new BodyDeferringInputStream(fr,new PipedInputStream(pout));
068 *      // consume InputStream
069 *      ...
070 *     } else {
071 *      // handle unexpected response status code
072 *      ...
073 *     }
074 * </pre>
075 */
076public class BodyDeferringAsyncHandler implements AsyncHandler<Response> {
077    private final ResponseBuilder responseBuilder = new ResponseBuilder();
078
079    private final CountDownLatch headersArrived = new CountDownLatch(1);
080
081    private final OutputStream output;
082
083    private volatile boolean responseSet;
084
085    private volatile Response response;
086
087    private volatile Throwable throwable;
088
089    private final Semaphore semaphore = new Semaphore(1);
090
091    public BodyDeferringAsyncHandler(final OutputStream os) {
092        this.output = os;
093        this.responseSet = false;
094    }
095
096    public void onThrowable(Throwable t) {
097        this.throwable = t;
098        // Counting down to handle error cases too.
099        // In "premature exceptions" cases, the onBodyPartReceived() and
100        // onCompleted()
101        // methods will never be invoked, leaving caller of getResponse() method
102        // blocked forever.
103        try {
104            semaphore.acquire();
105        } catch (InterruptedException e) {
106            // Ignore
107        } finally {
108            headersArrived.countDown();
109            semaphore.release();
110        }
111
112        try {
113            closeOut();
114        } catch (IOException e) {
115            // ignore
116        }
117    }
118
119    public STATE onStatusReceived(HttpResponseStatus responseStatus)
120            throws Exception {
121        responseBuilder.reset();
122        responseBuilder.accumulate(responseStatus);
123        return STATE.CONTINUE;
124    }
125
126    public STATE onHeadersReceived(HttpResponseHeaders headers)
127            throws Exception {
128        responseBuilder.accumulate(headers);
129        return STATE.CONTINUE;
130    }
131
132    public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart)
133            throws Exception {
134        // body arrived, flush headers
135        if (!responseSet) {
136            response = responseBuilder.build();
137            responseSet = true;
138            headersArrived.countDown();
139        }
140
141        bodyPart.writeTo(output);
142        return STATE.CONTINUE;
143    }
144
145    protected void closeOut() throws IOException {
146        try {
147            output.flush();
148        } finally {
149            output.close();
150        }
151    }
152
153    public Response onCompleted() throws IOException {
154        // Counting down to handle error cases too.
155        // In "normal" cases, latch is already at 0 here
156        // But in other cases, for example when because of some error
157        // onBodyPartReceived() is never called, the caller
158        // of getResponse() would remain blocked infinitely.
159        // By contract, onCompleted() is always invoked, even in case of errors
160        headersArrived.countDown();
161
162        closeOut();
163
164        try {
165            semaphore.acquire();
166            if (throwable != null) {
167                IOException ioe = new IOException(throwable.getMessage());
168                ioe.initCause(throwable);
169                throw ioe;
170            } else {
171                // sending out current response
172                return responseBuilder.build();
173            }
174        } catch (InterruptedException e) {
175            return null;
176        } finally {
177            semaphore.release();
178        }
179    }
180
181    /**
182     * This method -- unlike Future<Reponse>.get() -- will block only as long,
183     * as headers arrive. This is useful for large transfers, to examine headers
184     * ASAP, and defer body streaming to it's fine destination and prevent
185     * unneeded bandwidth consumption. The response here will contain the very
186     * 1st response from server, so status code and headers, but it might be
187     * incomplete in case of broken servers sending trailing headers. In that
188     * case, the "usual" Future<Response>.get() method will return complete
189     * headers, but multiple invocations of getResponse() will always return the
190     * 1st cached, probably incomplete one. Note: the response returned by this
191     * method will contain everything <em>except</em> the response body itself,
192     * so invoking any method like Response.getResponseBodyXXX() will result in
193     * error! Also, please not that this method might return <code>null</code>
194     * in case of some errors.
195     *
196     * @return a {@link Response}
197     * @throws InterruptedException
198     */
199    public Response getResponse() throws InterruptedException, IOException {
200        // block here as long as headers arrive
201        headersArrived.await();
202
203        try {
204            semaphore.acquire();
205            if (throwable != null) {
206                IOException ioe = new IOException(throwable.getMessage());
207                ioe.initCause(throwable);
208                throw ioe;
209            } else {
210                return response;
211            }
212        } finally {
213            semaphore.release();
214        }
215    }
216
217    // ==
218
219    /**
220     * A simple helper class that is used to perform automatic "join" for async
221     * download and the error checking of the Future of the request.
222     */
223    public static class BodyDeferringInputStream extends FilterInputStream {
224        private final Future<Response> future;
225
226        private final BodyDeferringAsyncHandler bdah;
227
228        public BodyDeferringInputStream(final Future<Response> future,
229                                        final BodyDeferringAsyncHandler bdah, final InputStream in) {
230            super(in);
231            this.future = future;
232            this.bdah = bdah;
233        }
234
235        /**
236         * Closes the input stream, and "joins" (wait for complete execution
237         * together with potential exception thrown) of the async request.
238         */
239        public void close() throws IOException {
240            // close
241            super.close();
242            // "join" async request
243            try {
244                getLastResponse();
245            } catch (Exception e) {
246                IOException ioe = new IOException(e.getMessage());
247                ioe.initCause(e);
248                throw ioe;
249            }
250        }
251
252        /**
253         * Delegates to {@link BodyDeferringAsyncHandler#getResponse()}. Will
254         * blocks as long as headers arrives only. Might return
255         * <code>null</code>. See
256         * {@link BodyDeferringAsyncHandler#getResponse()} method for details.
257         *
258         * @return a {@link Response}
259         * @throws InterruptedException
260         */
261        public Response getAsapResponse() throws InterruptedException,
262                IOException {
263            return bdah.getResponse();
264        }
265
266        /**
267         * Delegates to <code>Future<Response>#get()</code> method. Will block
268         * as long as complete response arrives.
269         *
270         * @return a {@link Response}
271         * @throws InterruptedException
272         * @throws ExecutionException
273         */
274        public Response getLastResponse() throws InterruptedException,
275                ExecutionException {
276            return future.get();
277        }
278    }
