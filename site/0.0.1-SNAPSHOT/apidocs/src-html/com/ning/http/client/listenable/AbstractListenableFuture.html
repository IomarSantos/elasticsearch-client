029package com.ning.http.client.listenable;
030
031import com.ning.http.client.ListenableFuture;
032
033import java.util.concurrent.Executor;
034
035/**
036 * <p>An abstract base implementation of the listener support provided by
037 * {@link ListenableFuture}. This class uses an {@link ExecutionList} to
038 * guarantee that all registered listeners will be executed. Listener/Executor
039 * pairs are stored in the execution list and executed in the order in which
040 * they were added, but because of thread scheduling issues there is no
041 * guarantee that the JVM will execute them in order. In addition, listeners
042 * added after the task is complete will be executed immediately, even if some
043 * previously added listeners have not yet been executed.
044 *
045 * @author Sven Mawson
046 * @since 1
047 */
048public abstract class AbstractListenableFuture<V> implements ListenableFuture<V> {
049
050    // The execution list to hold our executors.
051    private final ExecutionList executionList = new ExecutionList();
052
053    /*
054    * Adds a listener/executor pair to execution list to execute when this task
055    * is completed.
056    */
057
058    public ListenableFuture<V> addListener(Runnable listener, Executor exec) {
059        executionList.add(listener, exec);
060        return this;
061    }
062
063    /*
064    * Override the done method to execute the execution list.
065    */
066    protected void done() {
067        executionList.run();
068    }
069}
