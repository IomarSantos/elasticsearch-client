013package com.ning.http.client.filter;
014
015/**
016 * A Filter interface that gets invoked before making the processing of the response bytes. {@link ResponseFilter} are invoked
017 * before the actual response's status code get processed. That means authorization, proxy authentication and redirects
018 * processing hasn't occured when {@link ResponseFilter} gets invoked.
019 */
020public interface ResponseFilter {
021
022    /**
023     * An {@link com.ning.http.client.AsyncHttpProvider} will invoke {@link ResponseFilter#filter} and will use the
024     * returned {@link FilterContext#replayRequest()} and {@link FilterContext#getAsyncHandler()} to decide if the response
025     * processing can continue. If {@link FilterContext#replayRequest()} return true, a new request will be made
026     * using {@link FilterContext#getRequest()} and the current response processing will be ignored.
027     *
028     * @param ctx a {@link FilterContext}
029     * @return {@link FilterContext}. The {@link FilterContext} instance may not the same as the original one.
030     * @throws FilterException to interrupt the filter processing.
031     */
032    public <T> FilterContext<T> filter(FilterContext<T> ctx) throws FilterException;
033}
