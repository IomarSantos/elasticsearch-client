029package com.ning.http.client.providers.netty.util;
030
031import org.jboss.netty.channel.Channel;
032import org.jboss.netty.channel.ChannelFuture;
033import org.jboss.netty.channel.group.ChannelGroup;
034import org.jboss.netty.channel.group.ChannelGroupFuture;
035import org.jboss.netty.channel.group.DefaultChannelGroup;
036import org.jboss.netty.channel.group.DefaultChannelGroupFuture;
037import org.slf4j.Logger;
038import org.slf4j.LoggerFactory;
039
040import java.util.ArrayList;
041import java.util.Collection;
042import java.util.concurrent.atomic.AtomicBoolean;
043import java.util.concurrent.locks.ReentrantReadWriteLock;
044
045/**
046 * Extension of {@link DefaultChannelGroup} that's used mainly as a cleanup container, where {@link #close()} is only
047 * supposed to be called once.
048 *
049 * @author <a href="http://bruno.biasedbit.com/">Bruno de Carvalho</a>
050 */
051public class CleanupChannelGroup extends DefaultChannelGroup {
052
053    private final static Logger logger = LoggerFactory.getLogger(CleanupChannelGroup.class);
054
055    // internal vars --------------------------------------------------------------------------------------------------
056
057    private final AtomicBoolean closed;
058    private final ReentrantReadWriteLock lock;
059
060    // constructors ---------------------------------------------------------------------------------------------------
061
062    public CleanupChannelGroup() {
063        this.closed = new AtomicBoolean(false);
064        this.lock = new ReentrantReadWriteLock();
065    }
066
067    public CleanupChannelGroup(String name) {
068        super(name);
069        this.closed = new AtomicBoolean(false);
070        this.lock = new ReentrantReadWriteLock();
071    }
072
073    // DefaultChannelGroup --------------------------------------------------------------------------------------------
074
075    @Override
076    public ChannelGroupFuture close() {
077        this.lock.writeLock().lock();
078        try {
079            if (!this.closed.getAndSet(true)) {
080                // First time close() is called.
081                return super.close();
082            } else {
083                Collection<ChannelFuture> futures = new ArrayList<ChannelFuture>();
084                logger.debug("CleanupChannelGroup Already closed");
085                return new DefaultChannelGroupFuture(ChannelGroup.class.cast(this), futures);
086            }
087        } finally {
088            this.lock.writeLock().unlock();
089        }
090    }
091
092    @Override
093    public boolean add(Channel channel) {
094        // Synchronization must occur to avoid add() and close() overlap (thus potentially leaving one channel open).
095        // This could also be done by synchronizing the method itself but using a read lock here (rather than a
096        // synchronized() block) allows multiple concurrent calls to add().
097        this.lock.readLock().lock();
098        try {
099            if (this.closed.get()) {
100                // Immediately close channel, as close() was already called.
101                channel.close();
102                return false;
103            }
104
105            return super.add(channel);
106        } finally {
107            this.lock.readLock().unlock();
108        }
109    }
110}
