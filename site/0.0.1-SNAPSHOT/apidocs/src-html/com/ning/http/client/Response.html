017package com.ning.http.client;
018
019import java.io.IOException;
020import java.io.InputStream;
021import java.net.MalformedURLException;
022import java.net.URI;
023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.List;
026
027/**
028 * Represents the asynchronous HTTP response callback for an {@link com.ning.http.client.AsyncCompletionHandler}
029 */
030public interface Response {
031    /**
032     * Returns the status code for the request.
033     *
034     * @return The status code
035     */
036    public int getStatusCode();
037
038    /**
039     * Returns the status text for the request.
040     *
041     * @return The status text
042     */
043    public String getStatusText();
044
045    /**
046     * Return the entire response body as a byte[].
047     *
048     * @return the entire response body as a byte[].
049     * @throws IOException
050     */
051    public byte[] getResponseBodyAsBytes() throws IOException;
052
053    /**
054     * Returns an input stream for the response body. Note that you should not try to get this more than once,
055     * and that you should not close the stream.
056     *
057     * @return The input stream
058     * @throws java.io.IOException
059     */
060    public InputStream getResponseBodyAsStream() throws IOException;
061
062    /**
063     * Returns the first maxLength bytes of the response body as a string. Note that this does not check
064     * whether the content type is actually a textual one, but it will use the charset if present in the content
065     * type header.
066     *
067     * @param maxLength The maximum number of bytes to read
068     * @param charset   the charset to use when decoding the stream
069     * @return The response body
070     * @throws java.io.IOException
071     */
072    public String getResponseBodyExcerpt(int maxLength, String charset) throws IOException;
073
074    /**
075     * Return the entire response body as a String.
076     *
077     * @param charset the charset to use when decoding the stream
078     * @return the entire response body as a String.
079     * @throws IOException
080     */
081    public String getResponseBody(String charset) throws IOException;
082
083    /**
084     * Returns the first maxLength bytes of the response body as a string. Note that this does not check
085     * whether the content type is actually a textual one, but it will use the charset if present in the content
086     * type header.
087     *
088     * @param maxLength The maximum number of bytes to read
089     * @return The response body
090     * @throws java.io.IOException
091     */
092    public String getResponseBodyExcerpt(int maxLength) throws IOException;
093
094    /**
095     * Return the entire response body as a String.
096     *
097     * @return the entire response body as a String.
098     * @throws IOException
099     */
100    public String getResponseBody() throws IOException;
101
102    /**
103     * Return the request {@link URI}. Note that if the request got redirected, the value of the {@link URI} will be
104     * the last valid redirect url.
105     *
106     * @return the request {@link URI}.
107     * @throws MalformedURLException
108     */
109    public URI getUri() throws MalformedURLException;
110
111    /**
112     * Return the content-type header value.
113     *
114     * @return the content-type header value.
115     */
116    public String getContentType();
117
118    /**
119     * Return the response header
120     *
121     * @return the response header
122     */
123    public String getHeader(String name);
124
125    /**
126     * Return a {@link List} of the response header value.
127     *
128     * @return the response header
129     */
130    public List<String> getHeaders(String name);
131
132    public FluentCaseInsensitiveStringsMap getHeaders();
133
134    /**
135     * Return true if the response redirects to another object.
136     *
137     * @return True if the response redirects to another object.
138     */
139    boolean isRedirected();
140
141    /**
142     * Subclasses SHOULD implement toString() in a way that identifies the request for logging.
143     *
144     * @return The textual representation
145     */
146    public String toString();
147
148    /**
149     * Return the list of {@link Cookie}.
150     */
151    public List<Cookie> getCookies();
152
153    /**
154     * Return true if the response's status has been computed by an {@link AsyncHandler}
155     *
156     * @return true if the response's status has been computed by an {@link AsyncHandler}
157     */
158    public boolean hasResponseStatus();
159
160    /**
161     * Return true if the response's headers has been computed by an {@link AsyncHandler} It will return false if the
162     * either {@link com.ning.http.client.AsyncHandler#onStatusReceived(HttpResponseStatus)}
163     * or {@link AsyncHandler#onHeadersReceived(HttpResponseHeaders)} returned {@link com.ning.http.client.AsyncHandler.STATE#ABORT}
164     *
165     * @return true if the response's headers has been computed by an {@link AsyncHandler}
166     */
167    public boolean hasResponseHeaders();
168
169    /**
170     * Return true if the response's body has been computed by an {@link AsyncHandler}. It will return false if the
171     * either {@link com.ning.http.client.AsyncHandler#onStatusReceived(HttpResponseStatus)}
172     * or {@link AsyncHandler#onHeadersReceived(HttpResponseHeaders)} returned {@link com.ning.http.client.AsyncHandler.STATE#ABORT}
173     *
174     * @return true if the response's body has been computed by an {@link AsyncHandler}
175     */
176    public boolean hasResponseBody();
177
178
179    public static class ResponseBuilder {
180        private final List<HttpResponseBodyPart> bodies =
181                Collections.synchronizedList(new ArrayList<HttpResponseBodyPart>());
182        private HttpResponseStatus status;
183        private HttpResponseHeaders headers;
184
185        /**
186         * Accumulate {@link HttpContent} in order to build a {@link Response}
187         *
188         * @param httpContent {@link HttpContent}
189         * @return this
190         */
191        public ResponseBuilder accumulate(HttpContent httpContent) {
192            if (httpContent instanceof HttpResponseStatus) {
193                status = (HttpResponseStatus) httpContent;
194            } else if (httpContent instanceof HttpResponseHeaders) {
195                headers = (HttpResponseHeaders) httpContent;
196            } else if (httpContent instanceof HttpResponseBodyPart) {
197                bodies.add((HttpResponseBodyPart) httpContent);
198            }
199            return this;
200        }
201
202        /**
203         * Build a {@link Response} instance
204         *
205         * @return a {@link Response} instance
206         */
207        public Response build() {
208            return status == null ? null : status.provider().prepareResponse(status, headers, bodies);
209        }
210
211        /**
212         * Reset the internal state of this builder.
213         */
214        public void reset() {
215            bodies.clear();
216            status = null;
217            headers = null;
218        }
219    }
220
221}
