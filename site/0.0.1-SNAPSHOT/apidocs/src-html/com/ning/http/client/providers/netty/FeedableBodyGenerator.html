013package com.ning.http.client.providers.netty;
014
015import java.io.IOException;
016import java.nio.ByteBuffer;
017import java.util.Queue;
018import java.util.concurrent.ConcurrentLinkedQueue;
019import java.util.concurrent.atomic.AtomicInteger;
020
021import com.ning.http.client.Body;
022import com.ning.http.client.BodyGenerator;
023
024/**
025 * {@link BodyGenerator} which may return just part of the payload at the time
026 * handler is requesting it. If it happens - PartialBodyGenerator becomes responsible
027 * for finishing payload transferring asynchronously.
028 */
029public class FeedableBodyGenerator implements BodyGenerator {
030    private final static byte[] END_PADDING = "\r\n".getBytes();
031    private final static byte[] ZERO = "0".getBytes();
032    private final Queue<BodyPart> queue = new ConcurrentLinkedQueue<BodyPart>();
033    private final AtomicInteger queueSize = new AtomicInteger();
034    private FeedListener listener;
035
036    @Override
037    public Body createBody() throws IOException {
038        return new PushBody();
039    }
040
041    public void feed(final ByteBuffer buffer, final boolean isLast) throws IOException {
042        queue.offer(new BodyPart(buffer, isLast));
043        queueSize.incrementAndGet();
044        if (listener != null) {
045            listener.onContentAdded();
046        }
047    }
048
049    public static interface FeedListener {
050        public void onContentAdded();
051    }
052
053    public void setListener(FeedListener listener) {
054        this.listener = listener;
055    }
056
057    private final class PushBody implements Body {
058        private final int ONGOING = 0;
059        private final int CLOSING = 1;
060        private final int FINISHED = 2;
061
062        private int finishState = 0;
063
064        @Override
065        public long getContentLength() {
066            return -1;
067        }
068
069        @Override
070        public long read(final ByteBuffer buffer) throws IOException {
071            BodyPart nextPart = queue.peek();
072            if (nextPart == null) {
073                // Nothing in the queue
074                switch (finishState) {
075                case ONGOING:
076                    return 0;
077                case CLOSING:
078                    buffer.put(ZERO);
079                    buffer.put(END_PADDING);
080                    finishState = FINISHED;
081                    return buffer.position();
082                case FINISHED:
083                    buffer.put(END_PADDING);
084                    return -1;
085                }
086            }
087            int capacity = buffer.remaining() - 10; // be safe (we'll have to add size, ending, etc.)
088            int size = Math.min(nextPart.buffer.remaining(), capacity);
089            buffer.put(Integer.toHexString(size).getBytes());
090            buffer.put(END_PADDING);
091            for (int i=0; i < size; i++) {
092              buffer.put(nextPart.buffer.get());
093            }
094            buffer.put(END_PADDING);
095            if (!nextPart.buffer.hasRemaining()) {
096                if (nextPart.isLast) {
097                    finishState = CLOSING;
098                }
099                queue.remove();
100            }
101            return size;
102        }
103
104        @Override
105        public void close() throws IOException {
106        }
107
108    }
109
110    private final static class BodyPart {
111        private final boolean isLast;
112        private final ByteBuffer buffer;
113
114        public BodyPart(final ByteBuffer buffer, final boolean isLast) {
115            this.buffer = buffer;
116            this.isLast = isLast;
117        }
118    }
119}
