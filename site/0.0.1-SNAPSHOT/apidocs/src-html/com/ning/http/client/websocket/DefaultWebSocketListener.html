040package com.ning.http.client.websocket;
041
042/**
043 * Default WebSocketListener implementation.  Most methods are no-ops.  This 
044 * allows for quick override customization without clutter of methods that the
045 * developer isn't interested in dealing with.
046 * 
047 * @since 1.7.0
048 */
049public class DefaultWebSocketListener implements  WebSocketByteListener, WebSocketTextListener, WebSocketPingListener, WebSocketPongListener {
050
051    protected WebSocket webSocket;
052    
053    // -------------------------------------- Methods from WebSocketByteListener
054
055    /**
056     * {@inheritDoc}
057     */
058    @Override
059    public void onMessage(byte[] message) {
060    }
061
062    /**
063     * {@inheritDoc}
064     */
065    @Override
066    public void onFragment(byte[] fragment, boolean last) {
067    }
068
069    
070    // -------------------------------------- Methods from WebSocketPingListener
071    
072    /**
073     * {@inheritDoc}
074     */
075    @Override
076    public void onPing(byte[] message) {
077    }
078
079    
080    // -------------------------------------- Methods from WebSocketPongListener
081    
082    /**
083     * {@inheritDoc}
084     */
085    @Override
086    public void onPong(byte[] message) {
087    }
088    
089    
090    // -------------------------------------- Methods from WebSocketTextListener
091
092
093    /**
094     * {@inheritDoc}
095     */
096    @Override
097    public void onMessage(String message) {
098    }
099
100    /**
101     * {@inheritDoc}
102     */
103    @Override
104    public void onFragment(String fragment, boolean last) {
105    }
106    
107    
108    // ------------------------------------------ Methods from WebSocketListener
109
110    /**
111     * {@inheritDoc}
112     */
113    @Override
114    public void onOpen(WebSocket websocket) {
115        this.webSocket = websocket;
116    }
117
118    /**
119     * {@inheritDoc}
120     */
121    @Override
122    public void onClose(WebSocket websocket) {
123        this.webSocket = null;
124    }
125
126    /**
127     * {@inheritDoc}
128     */
129    @Override
130    public void onError(Throwable t) {
131    }
132}
