013package com.ning.http.client.generators;
014
015import com.ning.http.client.BodyGenerator;
016import com.ning.http.client.RandomAccessBody;
017
018import java.io.File;
019import java.io.IOException;
020import java.io.RandomAccessFile;
021import java.nio.ByteBuffer;
022import java.nio.channels.FileChannel;
023import java.nio.channels.WritableByteChannel;
024
025/**
026 * Creates a request body from the contents of a file.
027 */
028public class FileBodyGenerator
029        implements BodyGenerator {
030
031    private final File file;
032    private final long regionSeek;
033    private final long regionLength;
034
035    public FileBodyGenerator(File file) {
036        if (file == null) {
037            throw new IllegalArgumentException("no file specified");
038        }
039        this.file = file;
040        this.regionLength = file.length();
041        this.regionSeek = 0;
042    }
043
044    public FileBodyGenerator(File file, long regionSeek, long regionLength) {
045        if (file == null) {
046            throw new IllegalArgumentException("no file specified");
047        }
048        this.file = file;
049        this.regionLength = regionLength;
050        this.regionSeek = regionSeek;
051    }
052
053    /**
054     * {@inheritDoc}
055     */
056    /* @Override */
057    public RandomAccessBody createBody()
058            throws IOException {
059        return new FileBody(file, regionSeek, regionLength);
060    }
061
062    protected static class FileBody
063            implements RandomAccessBody {
064
065        private final RandomAccessFile file;
066
067        private final FileChannel channel;
068
069        private final long length;
070
071        public FileBody(File file)
072                throws IOException {
073            this.file = new RandomAccessFile(file, "r");
074            channel = this.file.getChannel();
075            length = file.length();
076        }
077
078        public FileBody(File file, long regionSeek, long regionLength)
079                throws IOException {
080            this.file = new RandomAccessFile(file, "r");
081            channel = this.file.getChannel();
082            length = regionLength;
083            if (regionSeek > 0) {
084                this.file.seek(regionSeek);
085            }
086        }
087
088        public long getContentLength() {
089            return length;
090        }
091
092        public long read(ByteBuffer buffer)
093                throws IOException {
094            return channel.read(buffer);
095        }
096
097        public long transferTo(long position, long count, WritableByteChannel target)
098                throws IOException {
099            if (count > length) {
100                count = length;
101            }
102            return channel.transferTo(position, count, target);
103        }
104
105        public void close()
106                throws IOException {
107            file.close();
108        }
109
110    }
111
112}
