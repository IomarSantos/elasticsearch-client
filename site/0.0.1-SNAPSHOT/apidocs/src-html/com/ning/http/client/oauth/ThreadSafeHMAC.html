017package com.ning.http.client.oauth;
018
019import com.ning.http.util.UTF8Codec;
020
021import javax.crypto.Mac;
022import javax.crypto.spec.SecretKeySpec;
023
024/**
025 * Since cloning (of MAC instances)  is not necessarily supported on all platforms
026 * (and specifically seems to fail on MacOS), let's wrap synchronization/reuse details here.
027 * Assumption is that this is bit more efficient (even considering synchronization)
028 * than locating and reconstructing instance each time.
029 * In future we may want to use soft references and thread local instance.
030 *
031 * @author tatu (tatu.saloranta@iki.fi)
032 */
033public class ThreadSafeHMAC {
034    private static final String HMAC_SHA1_ALGORITHM = "HmacSHA1";
035
036    private final Mac mac;
037
038    public ThreadSafeHMAC(ConsumerKey consumerAuth, RequestToken userAuth) {
039        byte[] keyBytes = UTF8Codec.toUTF8(consumerAuth.getSecret() + "&" + userAuth.getSecret());
040        SecretKeySpec signingKey = new SecretKeySpec(keyBytes, HMAC_SHA1_ALGORITHM);
041
042        // Get an hmac_sha1 instance and initialize with the signing key
043        try {
044            mac = Mac.getInstance(HMAC_SHA1_ALGORITHM);
045            mac.init(signingKey);
046        } catch (Exception e) {
047            throw new IllegalArgumentException(e);
048        }
049
050    }
051
052    public synchronized byte[] digest(byte[] message) {
053        mac.reset();
054        return mac.doFinal(message);
055    }
056}
