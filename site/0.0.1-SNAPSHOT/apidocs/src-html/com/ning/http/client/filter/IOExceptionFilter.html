013package com.ning.http.client.filter;
014
015/**
016 * This filter is invoked when an {@link java.io.IOException} occurs during an http transaction.
017 */
018public interface IOExceptionFilter {
019
020    /**
021     * An {@link com.ning.http.client.AsyncHttpProvider} will invoke {@link IOExceptionFilter#filter} and will
022     * use the returned {@link FilterContext} to replay the {@link com.ning.http.client.Request} or abort the processing.
023     *
024     * @param ctx a {@link FilterContext}
025     * @return {@link FilterContext}. The {@link FilterContext} instance may not the same as the original one.
026     * @throws FilterException to interrupt the filter processing.
027     */
028    public <T> FilterContext<T> filter(FilterContext<T> ctx) throws FilterException;
029}
