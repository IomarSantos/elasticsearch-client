017package com.ning.http.client;
018
019import java.io.File;
020import java.io.IOException;
021import java.io.InputStream;
022import java.io.OutputStream;
023import java.net.InetAddress;
024import java.util.Collection;
025import java.util.List;
026
027/**
028 * The Request class can be used to construct HTTP request:
029 * <blockquote><pre>
030 *   Request r = new RequestBuilder().setUrl("url")
031 *                      .setRealm((new Realm.RealmBuilder()).setPrincipal(user)
032 *                      .setPassword(admin)
033 *                      .setRealmName("MyRealm")
034 *                      .setScheme(Realm.AuthScheme.DIGEST).build());
035 *   r.execute();
036 * </pre></blockquote>
037 */
038public interface Request {
039
040    /**
041     * An entity that can be used to manipulate the Request's body output before it get sent.
042     */
043    public static interface EntityWriter {
044        public void writeEntity(OutputStream out) throws IOException;
045    }
046
047    /**
048     * Return the request's type (GET, POST, etc.)
049     *
050     * @return the request's type (GET, POST, etc.)
051     * @deprecated - use getMethod
052     */
053    public String getReqType();
054
055    /**
056     * Return the request's method name (GET, POST, etc.)
057     *
058     * @return the request's method name (GET, POST, etc.)
059     */
060    public String getMethod();
061
062    /**
063     * Return the decoded url
064     *
065     * @return the decoded url
066     */
067    public String getUrl();
068
069    /**
070     * Return the InetAddress to override
071     *
072     * @return the InetAddress
073     */
074    public InetAddress getInetAddress();
075
076    public InetAddress getLocalAddress();
077
078    /**
079     * Return the undecoded url
080     *
081     * @return the undecoded url
082     */
083    public String getRawUrl();
084
085    /**
086     * Return the current set of Headers.
087     *
088     * @return a {@link FluentCaseInsensitiveStringsMap} contains headers.
089     */
090    public FluentCaseInsensitiveStringsMap getHeaders();
091
092    /**
093     * Return Coookie.
094     *
095     * @return an unmodifiable Collection of Cookies
096     */
097    public Collection<Cookie> getCookies();
098
099    /**
100     * Return the current request's body as a byte array
101     *
102     * @return a byte array of the current request's body.
103     */
104    public byte[] getByteData();
105
106    /**
107     * Return the current request's body as a string
108     *
109     * @return an String representation of the current request's body.
110     */
111    public String getStringData();
112
113    /**
114     * Return the current request's body as an InputStream
115     *
116     * @return an InputStream representation of the current request's body.
117     */
118    public InputStream getStreamData();
119
120    /**
121     * Return the current request's body as an EntityWriter
122     *
123     * @return an EntityWriter representation of the current request's body.
124     */
125    public EntityWriter getEntityWriter();
126
127    /**
128     * Return the current request's body generator.
129     *
130     * @return A generator for the request body.
131     */
132    public BodyGenerator getBodyGenerator();
133
134    /**
135     * Return the current size of the content-lenght header based on the body's size.
136     *
137     * @return the current size of the content-lenght header based on the body's size.
138     * @deprecated
139     */
140    public long getLength();
141
142    /**
143     * Return the current size of the content-lenght header based on the body's size.
144     *
145     * @return the current size of the content-lenght header based on the body's size.
146     */
147    public long getContentLength();
148
149    /**
150     * Return the current parameters.
151     *
152     * @return a {@link FluentStringsMap} of parameters.
153     */
154    public FluentStringsMap getParams();
155
156    /**
157     * Return the current {@link Part}
158     *
159     * @return the current {@link Part}
160     */
161    public List<Part> getParts();
162
163    /**
164     * Return the virtual host value.
165     *
166     * @return the virtual host value.
167     */
168    public String getVirtualHost();
169
170    /**
171     * Return the query params.
172     *
173     * @return {@link FluentStringsMap} of query string
174     */
175    public FluentStringsMap getQueryParams();
176
177    /**
178     * Return the {@link ProxyServer}
179     *
180     * @return the {@link ProxyServer}
181     */
182    public ProxyServer getProxyServer();
183
184    /**
185     * Return the {@link Realm}
186     *
187     * @return the {@link Realm}
188     */
189    public Realm getRealm();
190
191    /**
192     * Return the {@link File} to upload.
193     *
194     * @return the {@link File} to upload.
195     */
196    public File getFile();
197
198    /**
199     * Return the <tt>true></tt> to follow redirect
200     *
201     * @return the <tt>true></tt> to follow redirect
202     */
203    public boolean isRedirectEnabled();
204
205    /**
206     *
207     * @return <tt>true></tt> if request's redirectEnabled setting
208     *          should be used in place of client's
209     */
210    public boolean isRedirectOverrideSet();
211
212    /**
213     * Return Per request configuration.
214     *
215     * @return Per request configuration.
216     */
217    public PerRequestConfig getPerRequestConfig();
218
219    /**
220     * Return the HTTP Range header value, or
221     *
222     * @return the range header value, or 0 is not set.
223     */
224    public long getRangeOffset();
225
226    /**
227     * Return the encoding value used when encoding the request's body.
228     *
229     * @return the encoding value used when encoding the request's body.
230     */
231    public String getBodyEncoding();
232
233    public boolean isUseRawUrl();
234
235}
