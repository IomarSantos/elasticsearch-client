013package com.ning.http.client.providers.jdk;
014
015import com.ning.http.client.AsyncHandler;
016import com.ning.http.client.listenable.AbstractListenableFuture;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020import java.net.HttpURLConnection;
021import java.util.concurrent.Callable;
022import java.util.concurrent.CancellationException;
023import java.util.concurrent.ExecutionException;
024import java.util.concurrent.Future;
025import java.util.concurrent.TimeUnit;
026import java.util.concurrent.TimeoutException;
027import java.util.concurrent.atomic.AtomicBoolean;
028import java.util.concurrent.atomic.AtomicLong;
029import java.util.concurrent.atomic.AtomicReference;
030
031
032public class JDKFuture<V> extends AbstractListenableFuture<V> {
033
034    private final static Logger logger = LoggerFactory.getLogger(JDKFuture.class);
035
036    protected Future<V> innerFuture;
037    protected final AsyncHandler<V> asyncHandler;
038    protected final int responseTimeoutInMs;
039    protected final AtomicBoolean cancelled = new AtomicBoolean(false);
040    protected final AtomicBoolean timedOut = new AtomicBoolean(false);
041    protected final AtomicBoolean isDone = new AtomicBoolean(false);
042    protected final AtomicReference<Throwable> exception = new AtomicReference<Throwable>();
043    protected final AtomicLong touch = new AtomicLong(System.currentTimeMillis());
044    protected final AtomicBoolean contentProcessed = new AtomicBoolean(false);
045    protected final HttpURLConnection urlConnection;
046    private boolean writeHeaders;
047    private boolean writeBody;
048
049    public JDKFuture(AsyncHandler<V> asyncHandler, int responseTimeoutInMs, HttpURLConnection urlConnection) {
050        this.asyncHandler = asyncHandler;
051        this.responseTimeoutInMs = responseTimeoutInMs;
052        this.urlConnection = urlConnection;
053        writeHeaders = true;
054        writeBody = true;
055    }
056
057    protected void setInnerFuture(Future<V> innerFuture) {
058        this.innerFuture = innerFuture;
059    }
060
061    public void done(Callable<?> callable) {
062        isDone.set(true);
063        super.done();
064    }
065
066    public void abort(Throwable t) {
067        exception.set(t);
068        if (innerFuture != null) {
069            innerFuture.cancel(true);
070        }
071        if (!timedOut.get() && !cancelled.get()) {
072            try {
073                asyncHandler.onThrowable(t);
074            } catch (Throwable te) {
075                logger.debug("asyncHandler.onThrowable", te);
076            }
077        }
078        super.done();
079    }
080
081    public void content(V v) {
082    }
083
084    public boolean cancel(boolean mayInterruptIfRunning) {
085        if (!cancelled.get() && innerFuture != null) {
086            urlConnection.disconnect();
087            try {
088                asyncHandler.onThrowable(new CancellationException());
089            } catch (Throwable te) {
090                logger.debug("asyncHandler.onThrowable", te);
091            }
092            cancelled.set(true);
093            super.done();
094            return innerFuture.cancel(mayInterruptIfRunning);
095        } else {
096            super.done();
097            return false;
098        }
099    }
100
101    public boolean isCancelled() {
102        if (innerFuture != null) {
103            return innerFuture.isCancelled();
104        } else {
105            return false;
106        }
107    }
108
109    public boolean isDone() {
110        contentProcessed.set(true);
111        return innerFuture.isDone();
112    }
113
114    public V get() throws InterruptedException, ExecutionException {
115        try {
116            return get(responseTimeoutInMs, TimeUnit.MILLISECONDS);
117        } catch (TimeoutException e) {
118            throw new ExecutionException(e);
119        }
120    }
121
122    public V get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
123        V content = null;
124        try {
125            if (innerFuture != null) {
126                content = innerFuture.get(timeout, unit);
127            }
128        } catch (TimeoutException t) {
129            if (!contentProcessed.get() && timeout != -1 && ((System.currentTimeMillis() - touch.get()) <= responseTimeoutInMs)) {
130                return get(timeout, unit);
131            }
132
133            if (exception.get() == null) {
134                timedOut.set(true);
135                throw new ExecutionException(new TimeoutException(String.format("No response received after %s", responseTimeoutInMs)));
136            }
137        } catch (CancellationException ce) {
138        }
139
140        if (exception.get() != null) {
141            throw new ExecutionException(exception.get());
142        }
143        return content;
144    }
145
146    /**
147     * Is the Future still valid
148     *
149     * @return <code>true</code> if response has expired and should be terminated.
150     */
151    public boolean hasExpired() {
152        return responseTimeoutInMs != -1 && ((System.currentTimeMillis() - touch.get()) > responseTimeoutInMs);
153    }
154
155    /**
156     * {@inheritDoc}
157     */
158    /* @Override  */
159    public void touch() {
160        touch.set(System.currentTimeMillis());
161    }
162
163    /**
164     * {@inheritDoc}
165     */
166    /* @Override  */
167    public boolean getAndSetWriteHeaders(boolean writeHeaders) {
168        boolean b = this.writeHeaders;
169        this.writeHeaders = writeHeaders;
170        return b;
171    }
172
173    /**
174     * {@inheritDoc}
175     */
176    /* @Override */
177    public boolean getAndSetWriteBody(boolean writeBody) {
178        boolean b = this.writeBody;
179        this.writeBody = writeBody;
180        return b;
181    }
182}
