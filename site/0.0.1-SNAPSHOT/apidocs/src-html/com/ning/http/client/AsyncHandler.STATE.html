016package com.ning.http.client;
017
018/**
019 * An asynchronous handler or callback which gets invoked as soon as some data is available when
020 * processing an asynchronous response.<br/>
021 * Callback methods get invoked in the following order:
022 * <ol>
023 * <li>{@link #onStatusReceived(HttpResponseStatus)},</li>
024 * <li>{@link #onHeadersReceived(HttpResponseHeaders)},</li>
025 * <li>{@link #onBodyPartReceived(HttpResponseBodyPart)}, which could be invoked multiple times,</li>
026 * <li>{@link #onCompleted()}, once the response has been fully read.</li>
027 * </ol>
028 * <p/>
029 * Returning a {@link AsyncHandler.STATE#ABORT} from any of those callback methods will interrupt asynchronous response
030 * processing, after that only {@link #onCompleted()} is going to be called.
031 * <p/>
032 * <p/>
033 * AsyncHandler aren't thread safe, hence you should avoid re-using the same instance when doing concurrent requests.
034 * As an exmaple, the following may produce unexpected results:
035 * <blockquote><pre>
036 *   AsyncHandler ah = new AsyncHandler() {....};
037 *   AsyncHttpClient client = new AsyncHttpClient();
038 *   client.prepareGet("http://...").execute(ah);
039 *   client.prepareGet("http://...").execute(ah);
040 * </pre></blockquote>
041 * It is recommended to create a new instance instead.
042 *
043 * @param <T> Type of object returned by the {@link java.util.concurrent.Future#get}
044 */
045public interface AsyncHandler<T> {
046
047    public static enum STATE {
048
049        /**
050         * Stop the processing.
051         */
052        ABORT,
053        /**
054         * Continue the processing
055         */
056        CONTINUE,
057        /**
058         * Upgrade the protocol.
059         */
060        UPGRADE
061    }
062
063    /**
064     * Invoked when an unexpected exception occurs during the processing of the response. The exception may have been
065     * produced by implementation of onXXXReceived method invocation.
066     *
067     * @param t a {@link Throwable}
068     */
069    void onThrowable(Throwable t);
070
071    /**
072     * Invoked as soon as some response body part are received. Could be invoked many times.
073     *
074     * @param bodyPart response's body part.
075     * @return a {@link STATE} telling to CONTINUE or ABORT the current processing.
076     * @throws Exception if something wrong happens
077     */
078    STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception;
079
080    /**
081     * Invoked as soon as the HTTP status line has been received
082     *
083     * @param responseStatus the status code and test of the response
084     * @return a {@link STATE} telling to CONTINUE or ABORT the current processing.
085     * @throws Exception if something wrong happens
086     */
087    STATE onStatusReceived(HttpResponseStatus responseStatus) throws Exception;
088
089    /**
090     * Invoked as soon as the HTTP headers has been received. Can potentially be invoked more than once if a broken server
091     * sent trailing headers.
092     *
093     * @param headers the HTTP headers.
094     * @return a {@link STATE} telling to CONTINUE or ABORT the current processing.
095     * @throws Exception if something wrong happens
096     */
097    STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception;
098
099    /**
100     * Invoked once the HTTP response processing is finished.
101     * <p/>
102     * <p/>
103     * Gets always invoked as last callback method.
104     *
105     * @return T Value that will be returned by the associated {@link java.util.concurrent.Future}
106     * @throws Exception if something wrong happens
107     */
108    T onCompleted() throws Exception;
109}
