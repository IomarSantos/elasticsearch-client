013package com.ning.http.client.providers.netty;
014
015import com.ning.http.client.websocket.WebSocket;
016import com.ning.http.client.websocket.WebSocketByteListener;
017import com.ning.http.client.websocket.WebSocketCloseCodeReasonListener;
018import com.ning.http.client.websocket.WebSocketListener;
019import com.ning.http.client.websocket.WebSocketTextListener;
020import org.jboss.netty.channel.Channel;
021import org.jboss.netty.handler.codec.http.websocketx.BinaryWebSocketFrame;
022import org.jboss.netty.handler.codec.http.websocketx.PingWebSocketFrame;
023import org.jboss.netty.handler.codec.http.websocketx.PongWebSocketFrame;
024import org.jboss.netty.handler.codec.http.websocketx.TextWebSocketFrame;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import java.util.concurrent.ConcurrentLinkedQueue;
029
030import static org.jboss.netty.buffer.ChannelBuffers.wrappedBuffer;
031
032public class NettyWebSocket implements WebSocket {
033    private final static Logger logger = LoggerFactory.getLogger(NettyWebSocket.class);
034
035    private final Channel channel;
036    private final ConcurrentLinkedQueue<WebSocketListener> listeners = new ConcurrentLinkedQueue<WebSocketListener>();
037
038    public NettyWebSocket(Channel channel) {
039        this.channel = channel;
040    }
041
042    // @Override
043    public WebSocket sendMessage(byte[] message) {
044        channel.write(new BinaryWebSocketFrame(wrappedBuffer(message)));
045        return this;
046    }
047
048    // @Override
049    public WebSocket stream(byte[] fragment, boolean last) {
050        throw new UnsupportedOperationException("Streaming currently only supported by the Grizzly provider.");
051    }
052
053    // @Override
054    public WebSocket stream(byte[] fragment, int offset, int len, boolean last) {
055        throw new UnsupportedOperationException("Streaming currently only supported by the Grizzly provider.");
056    }
057
058    // @Override
059    public WebSocket sendTextMessage(String message) {
060        channel.write(new TextWebSocketFrame(message));
061        return this;
062    }
063
064    // @Override
065    public WebSocket streamText(String fragment, boolean last) {
066        throw new UnsupportedOperationException("Streaming currently only supported by the Grizzly provider.");
067    }
068
069    // @Override
070    public WebSocket sendPing(byte[] payload) {
071        channel.write(new PingWebSocketFrame(wrappedBuffer(payload)));
072        return this;
073    }
074
075    // @Override
076    public WebSocket sendPong(byte[] payload) {
077        channel.write(new PongWebSocketFrame(wrappedBuffer(payload)));
078        return this;
079    }
080
081    // @Override
082    public WebSocket addWebSocketListener(WebSocketListener l) {
083        listeners.add(l);
084        return this;
085    }
086
087    // @Override
088    public WebSocket removeWebSocketListener(WebSocketListener l) {
089        listeners.remove(l);
090        return this;
091    }
092
093    // @Override
094    public boolean isOpen() {
095        return channel.isOpen();
096    }
097
098    // @Override
099    public void close() {
100        onClose();
101        listeners.clear();
102        channel.close();
103    }
104
105    protected void onMessage(byte[] message) {
106        for (WebSocketListener l : listeners) {
107            if (WebSocketByteListener.class.isAssignableFrom(l.getClass())) {
108                try {
109                    WebSocketByteListener.class.cast(l).onMessage(message);
110                } catch (Exception ex) {
111                    l.onError(ex);
112                }
113            }
114        }
115    }
116
117    protected void onTextMessage(String message) {
118        for (WebSocketListener l : listeners) {
119            if (WebSocketTextListener.class.isAssignableFrom(l.getClass())) {
120                try {
121                    WebSocketTextListener.class.cast(l).onMessage(message);
122                } catch (Exception ex) {
123                    l.onError(ex);
124                }
125            }
126        }
127    }
128
129    protected void onError(Throwable t) {
130        for (WebSocketListener l : listeners) {
131            try {
132                l.onError(t);
133            } catch (Throwable t2) {
134                logger.error("", t2);
135            }
136
137        }
138    }
139
140    protected void onClose() {
141        onClose(1000, "Normal closure; the connection successfully completed whatever purpose for which it was created.");
142    }
143
144    protected void onClose(int code, String reason) {
145        for (WebSocketListener l : listeners) {
146            try {
147                if (WebSocketCloseCodeReasonListener.class.isAssignableFrom(l.getClass())) {
148                    WebSocketCloseCodeReasonListener.class.cast(l).onClose(this, code, reason);
149                }
150                l.onClose(this);
151            } catch (Throwable t) {
152                l.onError(t);
153            }
154        }
155    }
156
157    @Override
158    public String toString() {
159        return "NettyWebSocket{" +
160                "channel=" + channel +
161                '}';
162    }
163}
