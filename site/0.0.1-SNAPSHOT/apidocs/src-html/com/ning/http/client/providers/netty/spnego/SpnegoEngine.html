038package com.ning.http.client.providers.netty.spnego;
039
040import com.ning.http.util.Base64;
041import org.ietf.jgss.GSSContext;
042import org.ietf.jgss.GSSException;
043import org.ietf.jgss.GSSManager;
044import org.ietf.jgss.GSSName;
045import org.ietf.jgss.Oid;
046import org.slf4j.Logger;
047import org.slf4j.LoggerFactory;
048
049import java.io.IOException;
050
051/**
052 * SPNEGO (Simple and Protected GSSAPI Negotiation Mechanism) authentication
053 * scheme.
054 *
055 * @since 4.1
056 */
057public class SpnegoEngine {
058    private static final String SPNEGO_OID = "1.3.6.1.5.5.2";
059    private static final String KERBEROS_OID = "1.2.840.113554.1.2.2";
060
061    private final Logger log = LoggerFactory.getLogger(getClass());
062
063    private final SpnegoTokenGenerator spnegoGenerator;
064
065    public SpnegoEngine(final SpnegoTokenGenerator spnegoGenerator) {
066        this.spnegoGenerator = spnegoGenerator;
067    }
068
069    public SpnegoEngine() {
070        this(null);
071    }
072
073    public String generateToken(String server) throws Throwable {
074        GSSContext gssContext = null;
075        byte[] token = null; // base64 decoded challenge
076        Oid negotiationOid = null;
077
078        try {
079            log.debug("init {}", server);
080            /* Using the SPNEGO OID is the correct method.
081             * Kerberos v5 works for IIS but not JBoss. Unwrapping
082             * the initial token when using SPNEGO OID looks like what is
083             * described here...
084             *
085             * http://msdn.microsoft.com/en-us/library/ms995330.aspx
086             *
087             * Another helpful URL...
088             *
089             * http://publib.boulder.ibm.com/infocenter/wasinfo/v7r0/index.jsp?topic=/com.ibm.websphere.express.doc/info/exp/ae/tsec_SPNEGO_token.html
090             *
091             * Unfortunately SPNEGO is JRE >=1.6.
092             */
093
094            /** Try SPNEGO by default, fall back to Kerberos later if error */
095            negotiationOid = new Oid(SPNEGO_OID);
096
097            boolean tryKerberos = false;
098            try {
099                GSSManager manager = GSSManager.getInstance();
100                GSSName serverName = manager.createName("HTTP@" + server, GSSName.NT_HOSTBASED_SERVICE);
101                gssContext = manager.createContext(
102                        serverName.canonicalize(negotiationOid), negotiationOid, null,
103                        GSSContext.DEFAULT_LIFETIME);
104                gssContext.requestMutualAuth(true);
105                gssContext.requestCredDeleg(true);
106            } catch (GSSException ex) {
107                log.error("generateToken", ex);
108                // BAD MECH means we are likely to be using 1.5, fall back to Kerberos MECH.
109                // Rethrow any other exception.
110                if (ex.getMajor() == GSSException.BAD_MECH) {
111                    log.debug("GSSException BAD_MECH, retry with Kerberos MECH");
112                    tryKerberos = true;
113                } else {
114                    throw ex;
115                }
116
117            }
118            if (tryKerberos) {
119                /* Kerberos v5 GSS-API mechanism defined in RFC 1964.*/
120                log.debug("Using Kerberos MECH {}", KERBEROS_OID);
121                negotiationOid = new Oid(KERBEROS_OID);
122                GSSManager manager = GSSManager.getInstance();
123                GSSName serverName = manager.createName("HTTP@" + server, GSSName.NT_HOSTBASED_SERVICE);
124                gssContext = manager.createContext(
125                        serverName.canonicalize(negotiationOid), negotiationOid, null,
126                        GSSContext.DEFAULT_LIFETIME);
127                gssContext.requestMutualAuth(true);
128                gssContext.requestCredDeleg(true);
129            }
130
131            // TODO suspicious: this will always be null because no value has been assigned before. Assign directly?
132            if (token == null) {
133                token = new byte[0];
134            }
135
136            token = gssContext.initSecContext(token, 0, token.length);
137            if (token == null) {
138                throw new Exception("GSS security context initialization failed");
139            }
140
141            /*
142             * IIS accepts Kerberos and SPNEGO tokens. Some other servers Jboss, Glassfish?
143             * seem to only accept SPNEGO. Below wraps Kerberos into SPNEGO token.
144             */
145            if (spnegoGenerator != null && negotiationOid.toString().equals(KERBEROS_OID)) {
146                token = spnegoGenerator.generateSpnegoDERObject(token);
147            }
148
149            gssContext.dispose();
150
151            String tokenstr = new String(Base64.encode(token));
152            log.debug("Sending response '{}' back to the server", tokenstr);
153
154            return tokenstr;
155        } catch (GSSException gsse) {
156            log.error("generateToken", gsse);
157            if (gsse.getMajor() == GSSException.DEFECTIVE_CREDENTIAL
158                    || gsse.getMajor() == GSSException.CREDENTIALS_EXPIRED)
159                throw new Exception(gsse.getMessage(), gsse);
160            if (gsse.getMajor() == GSSException.NO_CRED)
161                throw new Exception(gsse.getMessage(), gsse);
162            if (gsse.getMajor() == GSSException.DEFECTIVE_TOKEN
163                    || gsse.getMajor() == GSSException.DUPLICATE_TOKEN
164                    || gsse.getMajor() == GSSException.OLD_TOKEN)
165                throw new Exception(gsse.getMessage(), gsse);
166            // other error
167            throw new Exception(gsse.getMessage());
168        } catch (IOException ex) {
169            throw new Exception(ex.getMessage());
170        }
171    }
172}
