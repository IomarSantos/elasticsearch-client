013package com.ning.http.client;
014
015import com.ning.http.client.filter.IOExceptionFilter;
016import com.ning.http.client.filter.RequestFilter;
017import com.ning.http.client.filter.ResponseFilter;
018import com.ning.http.util.ProxyUtils;
019
020import javax.net.ssl.HostnameVerifier;
021import javax.net.ssl.SSLContext;
022import javax.net.ssl.SSLSession;
023import java.util.LinkedList;
024import java.util.concurrent.ExecutorService;
025import java.util.concurrent.Executors;
026import java.util.concurrent.ScheduledExecutorService;
027import java.util.concurrent.ThreadFactory;
028
029/**
030 * Simple JavaBean version of  {@link AsyncHttpClientConfig}
031 */
032public class AsyncHttpClientConfigBean extends AsyncHttpClientConfig {
033
034    public AsyncHttpClientConfigBean() {
035        configureExecutors();
036        configureDefaults();
037        configureFilters();
038    }
039
040    void configureFilters() {
041        requestFilters = new LinkedList<RequestFilter>();
042        responseFilters = new LinkedList<ResponseFilter>();
043        ioExceptionFilters = new LinkedList<IOExceptionFilter>();
044    }
045
046    void configureDefaults() {
047        maxTotalConnections = Integer.getInteger(ASYNC_CLIENT + "defaultMaxTotalConnections", -1);
048        maxConnectionPerHost = Integer.getInteger(ASYNC_CLIENT + "defaultMaxConnectionsPerHost", -1);
049        connectionTimeOutInMs = Integer.getInteger(ASYNC_CLIENT + "defaultConnectionTimeoutInMS", 60 * 1000);
050        idleConnectionInPoolTimeoutInMs = Integer.getInteger(ASYNC_CLIENT + "defaultIdleConnectionInPoolTimeoutInMS", 60 * 1000);
051        idleConnectionTimeoutInMs = Integer.getInteger(ASYNC_CLIENT + "defaultIdleConnectionTimeoutInMS", 60 * 1000);
052        requestTimeoutInMs = Integer.getInteger(ASYNC_CLIENT + "defaultRequestTimeoutInMS", 60 * 1000);
053        redirectEnabled = Boolean.getBoolean(ASYNC_CLIENT + "defaultRedirectsEnabled");
054        maxDefaultRedirects = Integer.getInteger(ASYNC_CLIENT + "defaultMaxRedirects", 5);
055        compressionEnabled = Boolean.getBoolean(ASYNC_CLIENT + "compressionEnabled");
056        userAgent = System.getProperty(ASYNC_CLIENT + "userAgent", "AsyncHttpClient/" + AHC_VERSION);
057
058        boolean useProxyProperties = Boolean.getBoolean(ASYNC_CLIENT + "useProxyProperties");
059        if (useProxyProperties) {
060            proxyServer = ProxyUtils.createProxy(System.getProperties());
061        }
062
063        allowPoolingConnection = true;
064        requestCompressionLevel = -1;
065        maxRequestRetry = 5;
066        allowSslConnectionPool = true;
067        useRawUrl = false;
068        removeQueryParamOnRedirect = true;
069        hostnameVerifier = new HostnameVerifier() {
070
071            public boolean verify(String s, SSLSession sslSession) {
072                return true;
073            }
074        };
075    }
076
077    void configureExecutors() {
078        reaper = Executors.newScheduledThreadPool(Runtime.getRuntime().availableProcessors(), new ThreadFactory() {
079            public Thread newThread(Runnable r) {
080                Thread t = new Thread(r, "AsyncHttpClient-Reaper");
081                t.setDaemon(true);
082                return t;
083            }
084        });
085        applicationThreadPool = Executors.newCachedThreadPool(new ThreadFactory() {
086            public Thread newThread(Runnable r) {
087                Thread t = new Thread(r, "AsyncHttpClient-Callback");
088                t.setDaemon(true);
089                return t;
090            }
091        });
092    }

149    public AsyncHttpClientConfigBean setReaper(ScheduledExecutorService reaper) {
150        if (this.reaper != null) {
151            this.reaper.shutdownNow();

153        this.reaper = reaper;
154        return this;
155    }
