013package com.ning.http.client.resumable;
014
015import java.io.IOException;
016import java.nio.ByteBuffer;
017
018/**
019 * A listener class that can be used to digest the bytes from an {@link ResumableAsyncHandler}
020 */
021public interface ResumableListener {
022
023    /**
024     * Invoked when some bytes are available to digest.
025     *
026     * @param byteBuffer the current bytes
027     * @throws IOException
028     */
029    public void onBytesReceived(ByteBuffer byteBuffer) throws IOException;
030
031    /**
032     * Invoked when all the bytes has been sucessfully transferred.
033     */
034    public void onAllBytesReceived();
035
036    /**
037     * Return the length of previously downloaded bytes.
038     *
039     * @return the length of previously downloaded bytes
040     */
041    public long length();
042
043}
