014package com.ning.http.client.generators;
015
016import com.ning.http.client.Body;
017import com.ning.http.client.BodyGenerator;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.nio.ByteBuffer;
024
025/**
026 * A {@link BodyGenerator} which use an {@link InputStream} for reading bytes, without having to read the entire
027 * stream in memory.
028 * <p/>
029 * NOTE: The {@link InputStream} must support the {@link InputStream#mark} and {@link java.io.InputStream#reset()} operation.
030 * If not, mechanisms like authentication, redirect, or resumable download will not works.
031 */
032public class InputStreamBodyGenerator implements BodyGenerator {
033
034    private final static byte[] END_PADDING = "\r\n".getBytes();
035    private final static byte[] ZERO = "0".getBytes();
036    private final InputStream inputStream;
037    private final static Logger logger = LoggerFactory.getLogger(InputStreamBodyGenerator.class);
038    private boolean patchNettyChunkingIssue = false;
039
040    public InputStreamBodyGenerator(InputStream inputStream) {
041        this.inputStream = inputStream;
042
043        if (inputStream.markSupported()) {
044            inputStream.mark(0);
045        } else {
046            logger.warn("inputStream.markSupported() not supported. Some features will not works");
047        }
048    }
049
050    /**
051     * {@inheritDoc}
052     */
053    /* @Override */
054    public Body createBody() throws IOException {
055        return new ISBody();
056    }
057
058    protected class ISBody implements Body {
059        private boolean eof = false;
060        private int endDataCount = 0;
061        private byte[] chunk;
062
063        public long getContentLength() {
064            return -1;
065        }
066
067        public long read(ByteBuffer buffer) throws IOException {
068
069            // To be safe.
070            chunk = new byte[buffer.remaining() - 10];
071
072
073            int read = -1;
074            try {
075                read = inputStream.read(chunk);
076            } catch (IOException ex) {
077                logger.warn("Unable to read", ex);
078            }
079
080            if (patchNettyChunkingIssue) {
081                if (read == -1) {
082                    // Since we are chuncked, we must output extra bytes before considering the input stream closed.
083                    // chunking requires to end the chunking:
084                    // - A Terminating chunk of  "0\r\n".getBytes(),
085                    // - Then a separate packet of "\r\n".getBytes()
086                    if (!eof) {
087                        endDataCount++;
088                        if (endDataCount == 2)
089                            eof = true;
090
091                        if (endDataCount == 1)
092                            buffer.put(ZERO);
093
094                        buffer.put(END_PADDING);
095
096
097                        return buffer.position();
098                    } else {
099                        if (inputStream.markSupported()) {
100                            inputStream.reset();
101                        }
102                        eof = false;
103                    }
104                    return -1;
105                }
106
107                /**
108                 * Netty 3.2.3 doesn't support chunking encoding properly, so we chunk encoding ourself.
109                 */
110
111                buffer.put(Integer.toHexString(read).getBytes());
112                // Chunking is separated by "<bytesreads>\r\n"
113                buffer.put(END_PADDING);
114                buffer.put(chunk, 0, read);
115                // Was missing the final chunk \r\n.
116                buffer.put(END_PADDING);
117            } else {
118                if (read > 0) {
119                    buffer.put(chunk, 0, read);
120                }
121            }
122            return read;
123        }
124
125        public void close() throws IOException {
126            inputStream.close();
127        }
128    }
129
130    /**
131     * HACK: This is required because Netty has issues with chunking.
132     *
133     * @param patchNettyChunkingIssue
134     */
135    public void patchNettyChunkingIssue(boolean patchNettyChunkingIssue) {
136        this.patchNettyChunkingIssue = patchNettyChunkingIssue;
137    }
138}
