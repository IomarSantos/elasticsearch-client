013package com.ning.http.client.providers.netty;
014
015import com.ning.http.client.ConnectionsPool;
016import org.jboss.netty.channel.Channel;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Set;
023import java.util.Timer;
024import java.util.TimerTask;
025import java.util.concurrent.ConcurrentHashMap;
026import java.util.concurrent.ConcurrentLinkedQueue;
027import java.util.concurrent.atomic.AtomicBoolean;
028
029/**
030 * A simple implementation of {@link com.ning.http.client.ConnectionsPool} based on a {@link java.util.concurrent.ConcurrentHashMap}
031 */
032public class NettyConnectionsPool implements ConnectionsPool<String, Channel> {
033
034    private final static Logger log = LoggerFactory.getLogger(NettyConnectionsPool.class);
035    private final ConcurrentHashMap<String, ConcurrentLinkedQueue<IdleChannel>> connectionsPool = new ConcurrentHashMap<String, ConcurrentLinkedQueue<IdleChannel>>();
036    private final ConcurrentHashMap<Channel, IdleChannel> channel2IdleChannel = new ConcurrentHashMap<Channel, IdleChannel>();
037    private final ConcurrentHashMap<Channel, Long> channel2CreationDate = new ConcurrentHashMap<Channel, Long>();
038    private final AtomicBoolean isClosed = new AtomicBoolean(false);
039    private final Timer idleConnectionDetector = new Timer(true);
040    private final boolean sslConnectionPoolEnabled;
041    private final int maxTotalConnections;
042    private final int maxConnectionPerHost;
043    private final int maxConnectionLifeTimeInMs;
044    private final long maxIdleTime;
045
046    public NettyConnectionsPool(NettyAsyncHttpProvider provider) {
047        this.maxTotalConnections = provider.getConfig().getMaxTotalConnections();
048        this.maxConnectionPerHost = provider.getConfig().getMaxConnectionPerHost();
049        this.sslConnectionPoolEnabled = provider.getConfig().isSslConnectionPoolEnabled();
050        this.maxIdleTime = provider.getConfig().getIdleConnectionInPoolTimeoutInMs();
051        this.maxConnectionLifeTimeInMs = provider.getConfig().getMaxConnectionLifeTimeInMs();
052        this.idleConnectionDetector.schedule(new IdleChannelDetector(), maxIdleTime, maxIdleTime);
053    }
054
055    private static class IdleChannel {
056        final String uri;
057        final Channel channel;
058        final long start;
059
060        IdleChannel(String uri, Channel channel) {
061            this.uri = uri;
062            this.channel = channel;
063            this.start = System.currentTimeMillis();
064        }
065
066        @Override
067        public boolean equals(Object o) {
068            if (this == o) return true;
069            if (!(o instanceof IdleChannel)) return false;
070
071            IdleChannel that = (IdleChannel) o;
072
073            if (channel != null ? !channel.equals(that.channel) : that.channel != null) return false;
074
075            return true;
076        }
077
078        @Override
079        public int hashCode() {
080            return channel != null ? channel.hashCode() : 0;
081        }
082    }
083
084    private class IdleChannelDetector extends TimerTask {
085        @Override
086        public void run() {
087            try {
088                if (isClosed.get()) return;
089
090                if (log.isDebugEnabled()) {
091                    Set<String> keys = connectionsPool.keySet();
092
093                    for (String s : keys) {
094                        log.debug("Entry count for : {} : {}", s, connectionsPool.get(s).size());
095                    }
096                }
097
098                List<IdleChannel> channelsInTimeout = new ArrayList<IdleChannel>();
099                long currentTime = System.currentTimeMillis();
100
101                for (IdleChannel idleChannel : channel2IdleChannel.values()) {
102                    long age = currentTime - idleChannel.start;
103                    if (age > maxIdleTime) {
104
105                        log.debug("Adding Candidate Idle Channel {}", idleChannel.channel);
106
107                        // store in an unsynchronized list to minimize the impact on the ConcurrentHashMap.
108                        channelsInTimeout.add(idleChannel);
109                    }
110                }
111                long endConcurrentLoop = System.currentTimeMillis();
112
113                for (IdleChannel idleChannel : channelsInTimeout) {
114                    Object attachment = idleChannel.channel.getPipeline().getContext(NettyAsyncHttpProvider.class).getAttachment();
115                    if (attachment != null) {
116                        if (NettyResponseFuture.class.isAssignableFrom(attachment.getClass())) {
117                            NettyResponseFuture<?> future = (NettyResponseFuture<?>) attachment;
118
119                            if (!future.isDone() && !future.isCancelled()) {
120                                log.debug("Future not in appropriate state %s\n", future);
121                                continue;
122                            }
123                        }
124                    }
125
126                    if (remove(idleChannel)) {
127                        log.debug("Closing Idle Channel {}", idleChannel.channel);
128                        close(idleChannel.channel);
129                    }
130                }
131
132                if (log.isTraceEnabled())
133                    log.trace(String.format("%d channel open, %d idle channels closed (times: 1st-loop=%d, 2nd-loop=%d).\n",
134                        connectionsPool.size(), channelsInTimeout.size(), endConcurrentLoop - currentTime, System.currentTimeMillis() - endConcurrentLoop));
135            } catch (Throwable t) {
136                log.error("uncaught exception!", t);
137            }
138        }
139    }
140
141    /**
142     * {@inheritDoc}
143     */
144    public boolean offer(String uri, Channel channel) {
145        if (isClosed.get()) return false;
146
147        if (!sslConnectionPoolEnabled && uri.startsWith("https")) {
148            return false;
149        }
150
151        Long createTime = channel2CreationDate.get(channel);
152        if (createTime == null){
153           channel2CreationDate.putIfAbsent(channel, System.currentTimeMillis());
154        }
155        else if (maxConnectionLifeTimeInMs != -1 && (createTime + maxConnectionLifeTimeInMs) < System.currentTimeMillis() ) {
156           log.debug("Channel {} expired", channel);
157           return false;
158        }
159
160        log.debug("Adding uri: {} for channel {}", uri, channel);
161        channel.getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(new NettyAsyncHttpProvider.DiscardEvent());
162
163        ConcurrentLinkedQueue<IdleChannel> idleConnectionForHost = connectionsPool.get(uri);
164        if (idleConnectionForHost == null) {
165            ConcurrentLinkedQueue<IdleChannel> newPool = new ConcurrentLinkedQueue<IdleChannel>();
166            idleConnectionForHost = connectionsPool.putIfAbsent(uri, newPool);
167            if (idleConnectionForHost == null) idleConnectionForHost = newPool;
168        }
169
170        boolean added;
171        int size = idleConnectionForHost.size();
172        if (maxConnectionPerHost == -1 || size < maxConnectionPerHost) {
173            IdleChannel idleChannel = new IdleChannel(uri, channel);
174            synchronized (idleConnectionForHost) {
175                added = idleConnectionForHost.add(idleChannel);
176
177                if (channel2IdleChannel.put(channel, idleChannel) != null) {
178                    log.error("Channel {} already exists in the connections pool!", channel);
179                }
180            }
181        } else {
182            log.debug("Maximum number of requests per host reached {} for {}", maxConnectionPerHost, uri);
183            added = false;
184        }
185        return added;
186    }
187
188    /**
189     * {@inheritDoc}
190     */
191    public Channel poll(String uri) {
192        if (!sslConnectionPoolEnabled && uri.startsWith("https")) {
193            return null;
194        }
195
196        IdleChannel idleChannel = null;
197        ConcurrentLinkedQueue<IdleChannel> idleConnectionForHost = connectionsPool.get(uri);
198        if (idleConnectionForHost != null) {
199            boolean poolEmpty = false;
200            while (!poolEmpty && idleChannel == null) {
201                if (idleConnectionForHost.size() > 0) {
202                    synchronized (idleConnectionForHost) {
203                        idleChannel = idleConnectionForHost.poll();
204                        if (idleChannel != null) {
205                            channel2IdleChannel.remove(idleChannel.channel);
206                        }
207                    }
208                }
209
210                if (idleChannel == null) {
211                    poolEmpty = true;
212                } else if (!idleChannel.channel.isConnected() || !idleChannel.channel.isOpen()) {
213                    idleChannel = null;
214                    log.trace("Channel not connected or not opened!");
215                }
216            }
217        }
218        return idleChannel != null ? idleChannel.channel : null;
219    }
220
221    private boolean remove(IdleChannel pooledChannel) {
222        if (pooledChannel == null || isClosed.get()) return false;
223
224        boolean isRemoved = false;
225        ConcurrentLinkedQueue<IdleChannel> pooledConnectionForHost = connectionsPool.get(pooledChannel.uri);
226        if (pooledConnectionForHost != null) {
227            isRemoved = pooledConnectionForHost.remove(pooledChannel);
228        }
229        isRemoved |= channel2IdleChannel.remove(pooledChannel.channel) != null;
230        return isRemoved;
231    }
232
233    /**
234     * {@inheritDoc}
235     */
236    public boolean removeAll(Channel channel) {
237        channel2CreationDate.remove(channel);
238        return !isClosed.get() && remove(channel2IdleChannel.get(channel));
239    }
240
241    /**
242     * {@inheritDoc}
243     */
244    public boolean canCacheConnection() {
245        if (!isClosed.get() && maxTotalConnections != -1 && channel2IdleChannel.size() >= maxTotalConnections) {
246            return false;
247        } else {
248            return true;
249        }
250    }
251
252    /**
253     * {@inheritDoc}
254     */
255    public void destroy() {
256        if (isClosed.getAndSet(true)) return;
257
258        // stop timer
259        idleConnectionDetector.cancel();
260        idleConnectionDetector.purge();
261
262        for (Channel channel : channel2IdleChannel.keySet()) {
263            close(channel);
264        }
265        connectionsPool.clear();
266        channel2IdleChannel.clear();
267        channel2CreationDate.clear();
268    }
269
270    private void close(Channel channel) {
271        try {
272            channel.getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(new NettyAsyncHttpProvider.DiscardEvent());
273            channel2CreationDate.remove(channel);
274            channel.close();
275        } catch (Throwable t) {
276            // noop
277        }
278    }
279
280    public final String toString() {
281        return String.format("NettyConnectionPool: {pool-size: %d}", channel2IdleChannel.size());
282    }
283}
