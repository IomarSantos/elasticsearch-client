013package com.ning.http.client.resumable;
014
015import com.ning.http.client.Request;
016import com.ning.http.client.filter.FilterContext;
017import com.ning.http.client.filter.FilterException;
018import com.ning.http.client.filter.IOExceptionFilter;
019
020/**
021 * Simple {@link IOExceptionFilter} that replay the current {@link com.ning.http.client.Request} using
022 * a {@link ResumableAsyncHandler}
023 */
024public class ResumableIOExceptionFilter implements IOExceptionFilter {
025    public FilterContext filter(FilterContext ctx) throws FilterException {
026        if (ctx.getIOException() != null && ResumableAsyncHandler.class.isAssignableFrom(ctx.getAsyncHandler().getClass())) {
027
028            Request request = ResumableAsyncHandler.class.cast(ctx.getAsyncHandler()).adjustRequestRange(ctx.getRequest());
029
030            return new FilterContext.FilterContextBuilder(ctx).request(request).replayRequest(true).build();
031        }
032        return ctx;
033    }
034}
