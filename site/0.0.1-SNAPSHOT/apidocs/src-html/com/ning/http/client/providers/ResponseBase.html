001package com.ning.http.client.providers;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.net.URI;
006import java.util.List;
007
008import com.ning.http.client.FluentCaseInsensitiveStringsMap;
009import com.ning.http.client.HttpResponseBodyPart;
010import com.ning.http.client.HttpResponseHeaders;
011import com.ning.http.client.HttpResponseStatus;
012import com.ning.http.client.Response;
013import com.ning.http.util.AsyncHttpProviderUtils;
014
015public abstract class ResponseBase implements Response
016{
017    protected final static String DEFAULT_CHARSET = "ISO-8859-1";
018    protected final static String HEADERS_NOT_COMPUTED = "Response's headers hasn't been computed by your AsyncHandler.";
019
020    protected final List<HttpResponseBodyPart> bodyParts;
021    protected final HttpResponseHeaders headers;
022    protected final HttpResponseStatus status;
023
024    protected ResponseBase(HttpResponseStatus status,
025            HttpResponseHeaders headers,
026            List<HttpResponseBodyPart> bodyParts)
027    {
028        this.bodyParts = bodyParts;
029        this.headers = headers;
030        this.status = status;
031    }
032
033    /* @Override */
034    public final int getStatusCode() {
035        return status.getStatusCode();
036    }
037
038    /* @Override */
039    public final String getStatusText() {
040        return status.getStatusText();
041    }
042
043
044    /* @Override */
045    public final URI getUri() /*throws MalformedURLException*/ {
046        return status.getUrl();
047    }
048
049    /* @Override */
050    public final String getContentType() {
051        return getHeader("Content-Type");
052    }
053
054    /* @Override */
055    public final String getHeader(String name) {
056        return getHeaders().getFirstValue(name);
057    }
058
059    /* @Override */
060    public final List<String> getHeaders(String name) {
061        return getHeaders().get(name);
062    }
063
064    /* @Override */
065    public final FluentCaseInsensitiveStringsMap getHeaders() {
066        if (headers == null) {
067            throw new IllegalStateException(HEADERS_NOT_COMPUTED);
068        }
069        return headers.getHeaders();
070    }
071
072    /* @Override */
073    public final boolean isRedirected() {
074        return (status.getStatusCode() >= 300) && (status.getStatusCode() <= 399);
075    }
076    
077    /* @Override */
078    public byte[] getResponseBodyAsBytes() throws IOException {
079        return AsyncHttpProviderUtils.contentToBytes(bodyParts);
080    }
081
082    /* @Override */
083    public String getResponseBody() throws IOException {
084        return getResponseBody(DEFAULT_CHARSET);
085    }
086
087    public String getResponseBody(String charset) throws IOException {
088        String contentType = getContentType();
089        if (contentType != null && charset == null) {
090            charset = AsyncHttpProviderUtils.parseCharset(contentType);
091        }
092
093        if (charset == null) {
094            charset = DEFAULT_CHARSET;
095        }
096
097        return AsyncHttpProviderUtils.contentToString(bodyParts, charset);
098    }
099
100    /* @Override */
101    public InputStream getResponseBodyAsStream() throws IOException {
102        return AsyncHttpProviderUtils.contentAsStream(bodyParts);
103    }
104
105    protected String calculateCharset() {
106        String charset = null;
107        String contentType = getContentType();
108        if (contentType != null) {
109            charset = AsyncHttpProviderUtils.parseCharset(contentType);
110        }
111
112        if (charset == null) {
113            charset = DEFAULT_CHARSET;
114        }
115        return charset;
116    }
117
118}
