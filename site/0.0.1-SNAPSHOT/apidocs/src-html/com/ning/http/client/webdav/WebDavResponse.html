013package com.ning.http.client.webdav;
014
015import com.ning.http.client.Cookie;
016import com.ning.http.client.FluentCaseInsensitiveStringsMap;
017import com.ning.http.client.Response;
018import org.w3c.dom.Document;
019
020import java.io.IOException;
021import java.io.InputStream;
022import java.net.MalformedURLException;
023import java.net.URI;
024import java.util.List;
025
026/**
027 * Customized {@link Response} which add support for getting the response's body as an XML document (@link WebDavResponse#getBodyAsXML}
028 */
029public class WebDavResponse implements Response {
030
031    private final Response response;
032    private final Document document;
033
034    public WebDavResponse(Response response, Document document) {
035        this.response = response;
036        this.document = document;
037    }
038
039    public int getStatusCode() {
040        return response.getStatusCode();
041    }
042
043    public String getStatusText() {
044        return response.getStatusText();
045    }
046
047    /* @Override */
048    public byte[] getResponseBodyAsBytes() throws IOException {
049        return response.getResponseBodyAsBytes();
050    }
051
052    public InputStream getResponseBodyAsStream() throws IOException {
053        return response.getResponseBodyAsStream();
054    }
055
056    public String getResponseBodyExcerpt(int maxLength) throws IOException {
057        return response.getResponseBodyExcerpt(maxLength);
058    }
059
060    public String getResponseBodyExcerpt(int maxLength, String charset) throws IOException {
061        return response.getResponseBodyExcerpt(maxLength, charset);
062    }
063
064    public String getResponseBody() throws IOException {
065        return response.getResponseBody();
066    }
067
068    public String getResponseBody(String charset) throws IOException {
069        return response.getResponseBody(charset);
070    }
071
072    public URI getUri() throws MalformedURLException {
073        return response.getUri();
074    }
075
076    public String getContentType() {
077        return response.getContentType();
078    }
079
080    public String getHeader(String name) {
081        return response.getHeader(name);
082    }
083
084    public List<String> getHeaders(String name) {
085        return response.getHeaders(name);
086    }
087
088    public FluentCaseInsensitiveStringsMap getHeaders() {
089        return response.getHeaders();
090    }
091
092    public boolean isRedirected() {
093        return response.isRedirected();
094    }
095
096    public List<Cookie> getCookies() {
097        return response.getCookies();
098    }
099
100    public boolean hasResponseStatus() {
101        return response.hasResponseStatus();
102    }
103
104    public boolean hasResponseHeaders() {
105        return response.hasResponseHeaders();
106    }
107
108    public boolean hasResponseBody() {
109        return response.hasResponseBody();
110    }
111
112    public Document getBodyAsXML() {
113        return document;
114    }
115}
